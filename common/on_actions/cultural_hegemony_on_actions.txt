# ============================================================================
# CULTURAL HEGEMONY — On-Actions
# ============================================================================

# ---- War end: force leaderboard recalculation ----
# When a war ends, countries may be annexed or borders may change.
# Remove the leaderboard lock so the next monthly pulse forces a full
# recalculation, ensuring dead/annexed countries are removed from rankings.
on_war_end = {
	on_actions = {
		ch_war_end_recalc_on_action
	}
}
ch_war_end_recalc_on_action = {
	effect = {
		if = {
			limit = {
				has_global_variable = ch_ldr_lock
				has_game_rule = cultural_hegemony_enabled
			}
			remove_global_variable = ch_ldr_lock
		}
	}
}

# ---- Monthly pulse: update per-country CH values and modifiers ----
# Runs from game start for all qualifying countries (even before JE activates).
# This ensures scores are stable by the time the JE appears in 1845.
on_monthly_pulse_country = {
	on_actions = {
		ch_monthly_pulse_on_action
	}
}
ch_monthly_pulse_on_action = {
	effect = {
		if = {
			limit = {
				has_game_rule = cultural_hegemony_enabled
			}
			ch_monthly_country_update = yes
			# If leaderboard lock was cleared (e.g. by war end), rebuild immediately
			if = {
				limit = { NOT = { has_global_variable = ch_ldr_lock } }
				ch_yearly_global_update = yes
			}
		}
	}
}

# ---- Yearly pulse: refresh global caches + build leaderboard ----
# Runs from game start. Lock guard inside the effect ensures only one country
# per year does the expensive global computation.
on_yearly_pulse_country = {
	on_actions = {
		ch_yearly_pulse_on_action
	}
}
ch_yearly_pulse_on_action = {
	effect = {
		if = {
			limit = {
				has_game_rule = cultural_hegemony_enabled
			}
			ch_yearly_global_update = yes
		}
	}
}

# Track "World First" technology achievements for Cultural Pull
# Root scope = Country that just acquired the technology
# scope:technology = the technology just researched (from on_acquired_technology)
# If no other country has this tech yet, this is a world first.
cultural_hegemony_tech_first_on_action = {
	effect = {
		if = {
			limit = { NOT = { has_game_rule = cultural_hegemony_enabled } }
			return = yes
		}
		if = {
			limit = {
				# True world first: no other country has researched this tech yet
				NOT = {
					any_country = {
						NOT = { this = ROOT }
						has_technology_researched = scope:technology
					}
				}
			}
			if = {
				limit = { NOT = { has_variable = ch_tech_firsts_recent } }
				set_variable = { name = ch_tech_firsts_recent value = 0 }
			}
			change_variable = { name = ch_tech_firsts_recent add = cultural_pull_tech_first_bonus }
		}
	}
}
