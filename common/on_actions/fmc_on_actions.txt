# ==============================================================================
# FMC (Free Market Construction) — On Actions
# ==============================================================================

# Weekly update: refresh price stabiliser and private/public allocation ratio.
on_weekly_pulse_country = {
	on_actions = {
		fmc_weekly_update_on_action
	}
}

fmc_weekly_update_on_action = {
	effect = {
		if = {
			limit = { NOT = { is_country_type = decentralized } }
			fmc_update_price_stabiliser = yes
			fmc_update_construction_allocation = yes
			if = {
				limit = { is_ai = yes }
				fmc_ai_set_public_target = yes
			}
		}
	}
}

# Construction site placement — add/remove sites as queues change.
on_start_expanding_building = {
	on_actions = {
		fmc_on_start_expanding_building
	}
}

fmc_on_start_expanding_building = {
	effect = {
		state = {
			fmc_on_construction_start = yes
		}
	}
}

on_building_expanded = {
	on_actions = {
		fmc_on_building_finished
	}
}

on_building_built = {
	on_actions = {
		fmc_on_building_finished
	}
}

fmc_on_building_finished = {
	effect = {
		state = {
			fmc_on_construction_end = yes
		}
	}
}

# Territory / sovereignty changes: rebuild construction site placement.
# Root = dissolving market, scope:market = absorbing market.
on_merge_markets = {
	on_actions = {
		fmc_on_merge_markets
	}
}

fmc_on_merge_markets = {
	effect = {
		if = {
			limit = { exists = owner }
			owner = { fmc_rebuild_construction_sites = yes }
		}
		if = {
			limit = { exists = scope:market.owner }
			scope:market.owner = { fmc_rebuild_construction_sites = yes }
		}
	}
}

on_create_market = {
	on_actions = {
		fmc_on_create_market
	}
}

fmc_on_create_market = {
	effect = {
		owner = { fmc_rebuild_construction_sites = yes }
	}
}

# Root = formed country.
on_country_formed = {
	effect = {
		set_variable = {
			name = recently_formed_country
			days = 7300
		}
	}

		fmc_on_country_formed
	}
}
