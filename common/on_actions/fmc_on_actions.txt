# Root = From Market
# scope:market = Market merging into
# scope:trade_center = Trade Center state of root
on_merge_markets = {
    on_actions = {
        fmc_on_merge_markets
    }
}

fmc_on_merge_markets = {
    effect = {
        if = {
            limit = {
                exists = owner
            }
            owner = {
                trigger_event = {
                    id = fmc_update_events.4
                }
            }
        }
        if = {
            limit = {
                exists = scope:market.owner
            }
            scope:market.owner = {
                trigger_event = {
                    id = fmc_update_events.4
                }
            }    
        }
    }
}
