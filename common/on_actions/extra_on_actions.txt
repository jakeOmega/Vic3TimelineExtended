# ============================================================================
# LAW ENACTMENT PASS: banking cycle modifier cleanup
# When an economy-type law changes, remove modifiers that belonged to the old
# economy type so stale planning/council/CB-tool modifiers don't linger.
# Root = country that enacted the law. Scope is country.
# ============================================================================
on_law_enactment_pass = {
	on_actions = {
		te_banking_law_change_cleanup
	}
}

te_banking_law_change_cleanup = {
	effect = {
		# Only run if the banking cycle JE is active
		if = {
			limit = { has_journal_entry = je_banking_cycle }
			# Switched AWAY from command economy → remove all planning_* modifiers
			if = {
				limit = { NOT = { has_law = law_type:law_command_economy } }
				remove_modifier = planning_plan_revision
				remove_modifier = planning_emergency_allocation
				remove_modifier = planning_target_reduction
				remove_modifier = planning_strategic_stockpile
				remove_modifier = planning_distribution_upgrade
				remove_modifier = planning_admin_campaign
				remove_modifier = planning_coordination_protocol
				remove_modifier = planning_consolidation_order
			}
			# Switched AWAY from cooperative ownership → remove all cooperative_* modifiers
			if = {
				limit = { NOT = { has_law = law_type:law_cooperative_ownership } }
				remove_modifier = cooperative_dividend_restraint
				remove_modifier = cooperative_mutual_aid
				remove_modifier = cooperative_capital_plan
				remove_modifier = cooperative_solidarity_campaign
				remove_modifier = cooperative_credit_expansion
				remove_modifier = cooperative_consumption_ceiling
				remove_modifier = cooperative_council_directive
				remove_modifier = cooperative_worker_buyout
			}
			# Switched TO command economy or cooperative → remove all market CB-tool modifiers
			if = {
				limit = {
					OR = {
						has_law = law_type:law_command_economy
						has_law = law_type:law_cooperative_ownership
					}
				}
				remove_modifier = banking_policy_rate_hike
				remove_modifier = banking_open_market_ops
				remove_modifier = banking_countercyclical_capital_buffer
				remove_modifier = banking_expand_deposit_guarantee
				remove_modifier = banking_directed_credit_infrastructure
				remove_modifier = banking_emergency_liquidity_program
				remove_modifier = banking_moral_suasion
				remove_modifier = banking_raise_margin_requirements
				remove_modifier = banking_fx_devaluation
				remove_modifier = banking_fx_support
				remove_modifier = banking_capital_controls_out
				remove_modifier = banking_fx_swap_lines
				remove_modifier = banking_export_credit_facility
				remove_modifier = banking_asset_relief_program
			}
		}
	}
}

# NOTE: Building and investment hooks for te_update_state_modifiers were removed.
# The add_modifier = { multiplier = script_value } evaluates the script value
# through the parent scope chain. Building and institution scopes don't support
# variables/modifiers, causing cascading errors. The yearly state pulse and
# law/technology hooks are sufficient for these slow-changing modifiers.

# ============================================================================
# BUILDING BUILT (Root = Building)
# When a barracks is built, a new combat unit type enters a formation.
# Recalculate combined arms bonuses immediately for the owning country.
# ============================================================================
on_building_built = {
	on_actions = {
		te_on_building_built
	}
}
te_on_building_built = {
	effect = {
		if = {
			limit = {
				is_building_type = building_barrack
				exists = owner
				owner = { modifier:country_combined_arms_bonus_enabled_bool = yes }
			}
			owner = {
				apply_combined_arms_bonuses = yes
			}
		}
	}
}

	}
}

language_reform_law_on_action = {
	effect = {
		owner = {
			# Fire founding event immediately when language reform is activated
			if = {
				limit = {
					has_law = law_type:law_state_led_language_reform
					NOT = { has_variable = language_reform_type }
				}
				trigger_event = { id = extra_law_events.25 }
			}
			# Clean up all language reform modifiers when law changes away
			if = {
				limit = {
					NOT = { has_law = law_type:law_state_led_language_reform }
					has_any_langreform_modifier = yes
				}
				remove_all_langreform_modifiers = yes
			}
		}
	}
}

# Fires from on_law_activated (ROOT = law scope).
# Checks if the activated law is radical and the government is not exempt.
# Identifies the specific radical law via a variable so the event can
# react to exactly what was enacted (not what the country already has).
radical_law_backlash_on_action = {
	effect = {
		if = {
			limit = {
				is_radical_law_activated = yes
				owner = {
					is_exempt_from_radical_backlash = no
				}
			}
			# Identify which radical law was activated (ROOT = law scope)
			if = {
				limit = { law_type = law_type:law_communal_child_rearing }
				owner = { set_variable = { name = radical_law_backlash_type value = 1 } }
			}
			else_if = {
				limit = { law_type = law_type:law_state_atheism }
				owner = { set_variable = { name = radical_law_backlash_type value = 2 } }
			}
			else_if = {
				limit = { law_type = law_type:law_cooperative_ownership }
				owner = { set_variable = { name = radical_law_backlash_type value = 3 } }
			}
			else_if = {
				limit = { law_type = law_type:law_command_economy }
				owner = { set_variable = { name = radical_law_backlash_type value = 4 } }
			}
			else_if = {
				limit = { law_type = law_type:law_collectivized_agriculture }
				owner = { set_variable = { name = radical_law_backlash_type value = 5 } }
			}
			owner = {
				trigger_event = { id = extra_law_events.36 }
			}
		}
	}
}

# Root = the dissolving (joining) market, scope:market = the absorbing market.
# owner = the country whose market is dissolving (new market member).
# scope:market.owner = the country that owns the absorbing market (the market leader).
gw_market_join_on_action = {
	effect = {
		if = {
			limit = {
				exists = owner
				exists = scope:market.owner
			}
			owner = {
				# Apply any active market-leader policies from the absorbing market's leader.
				gw_copy_market_policy_modifiers = yes
			}
		}
	}
}

global_warming_events_on_action = {
	effect = {
		# Fire threshold events exactly once globally when temperature crosses each threshold.
		# Runs in the global (world) scope of on_monthly_pulse - fires once per month,
		# so every_country here reaches all countries without per-country duplication.
		if = {
			limit = {
				temperature_anomaly_display >= 0.5
				NOT = { has_global_variable = gw_mild_event_fired }
			}
			set_global_variable = gw_mild_event_fired
			every_country = {
				limit = { NOT = { is_country_type = decentralized } }
				trigger_event = { id = environmentalism_events.1 }
			}
		}
		if = {
			limit = {
				temperature_anomaly_display >= 1.0
				NOT = { has_global_variable = gw_moderate_event_fired }
			}
			set_global_variable = gw_moderate_event_fired
			every_country = {
				limit = { NOT = { is_country_type = decentralized } }
				trigger_event = { id = environmentalism_events.2 }
			}
		}
		if = {
			limit = {
				temperature_anomaly_display >= 2.0
				NOT = { has_global_variable = gw_severe_event_fired }
			}
			set_global_variable = gw_severe_event_fired
			every_country = {
				limit = { NOT = { is_country_type = decentralized } }
				trigger_event = { id = environmentalism_events.3 }
			}
		}
		if = {
			limit = {
				temperature_anomaly_display >= 3.0
				NOT = { has_global_variable = gw_catastrophic_event_fired }
			}
			set_global_variable = gw_catastrophic_event_fired
			every_country = {
				limit = { NOT = { is_country_type = decentralized } }
				trigger_event = { id = environmentalism_events.4 }
			}
		}

		# Cooling / reversal threshold events.
		# These fire once globally when temperatures move back below major thresholds.
		if = {
			limit = {
				temperature_anomaly_display < 3.0
				has_global_variable = gw_catastrophic_event_fired
				NOT = { has_global_variable = gw_cooling_below_3_event_fired }
			}
			set_global_variable = gw_cooling_below_3_event_fired
			every_country = {
				limit = { NOT = { is_country_type = decentralized } }
				trigger_event = { id = environmentalism_events.17 }
			}
		}
		if = {
			limit = {
				temperature_anomaly_display < 2.0
				has_global_variable = gw_severe_event_fired
				NOT = { has_global_variable = gw_cooling_below_2_event_fired }
			}
			set_global_variable = gw_cooling_below_2_event_fired
			every_country = {
				limit = { NOT = { is_country_type = decentralized } }
				trigger_event = { id = environmentalism_events.18 }
			}
		}
		if = {
			limit = {
				temperature_anomaly_display < 1.0
				has_global_variable = gw_moderate_event_fired
				NOT = { has_global_variable = gw_cooling_below_1_event_fired }
			}
			set_global_variable = gw_cooling_below_1_event_fired
			every_country = {
				limit = { NOT = { is_country_type = decentralized } }
				trigger_event = { id = environmentalism_events.19 }
			}
		}
		if = {
			limit = {
				temperature_anomaly_display < 0.5
				has_global_variable = gw_mild_event_fired
				NOT = { has_global_variable = gw_cooling_below_0_5_event_fired }
			}
			set_global_variable = gw_cooling_below_0_5_event_fired
			every_country = {
				limit = { NOT = { is_country_type = decentralized } }
				trigger_event = { id = environmentalism_events.20 }
			}
		}
		if = {
			limit = {
				temperature_anomaly_display < 0.1
				has_global_variable = gw_mild_event_fired
				NOT = { has_global_variable = gw_cooling_near_baseline_event_fired }
			}
			set_global_variable = gw_cooling_near_baseline_event_fired
			every_country = {
				limit = { NOT = { is_country_type = decentralized } }
				trigger_event = { id = environmentalism_events.21 }
			}
		}

		# Recurring random events for countries experiencing global warming.
		# Each country has a small monthly chance of getting one event.
		# Event triggers are checked inside the event definitions themselves.
		every_country = {
			limit = {
				NOT = { is_country_type = decentralized }
				has_global_variable = greenhouse_gas_emissions
			}
			random_list = {
				1800 = { } # ~90% chance of nothing happening this month
				# Crop failures (severe+ warming, farming country)
				15 = {
					trigger = {
						temperature_anomaly_display >= 2.0
						any_scope_state = {
							is_incorporated = yes
							any_scope_building = { is_building_type = building_wheat_farm }
						}
					}
					trigger_event = { id = environmentalism_events.5 }
				}
				# Climate refugees (moderate+ warming)
				15 = {
					trigger = {
						has_modifier = global_warming
						temperature_anomaly_display >= 1.0
					}
					trigger_event = { id = environmentalism_events.6 }
				}
				# Pollution scandal (has coal/oil industry)
				15 = {
					trigger = {
						any_scope_state = {
							is_incorporated = yes
							any_scope_building = {
								OR = {
									is_building_type = building_coal_mine
									is_building_type = building_oil_rig
								}
								level >= 3
							}
						}
					}
					trigger_event = { id = environmentalism_events.7 }
				}
				# Environmentalist movement (mild+ warming, has pollution_control)
				15 = {
					trigger = {
						has_modifier = global_warming
						temperature_anomaly_display >= 0.5
						has_technology_researched = pollution_control
					}
					trigger_event = { id = environmentalism_events.8 }
				}
				# Coastal flooding (moderate+ warming, coastal ports)
				10 = {
					trigger = {
						has_modifier = global_warming
						temperature_anomaly_display >= 1.0
						any_scope_state = {
							is_incorporated = yes
							is_coastal = yes
							has_building = building_port
						}
					}
					trigger_event = { id = environmentalism_events.9 }
				}
				# Water shortage (severe+ warming)
				10 = {
					trigger = {
						has_modifier = global_warming
						temperature_anomaly_display >= 2.0
					}
					trigger_event = { id = environmentalism_events.10 }
				}
				# Green tech boom (environmental_movement tech + renewables)
				10 = {
					trigger = {
						has_modifier = global_warming
						has_technology_researched = environmental_movement
						any_scope_state = {
							has_building = building_renewable_energy_plant
						}
					}
					trigger_event = { id = environmentalism_events.11 }
				}
				# Fossil fuel lobby (has emission policies + large fossil industry)
				10 = {
					trigger = {
						OR = {
							has_modifier = carbon_tax_modifier
							has_modifier = emission_standards_modifier
							has_modifier = fossil_fuel_divestment_modifier
						}
						any_scope_state = {

# INTELLIGENT MODIFIER UPDATE ON_ACTIONS
# These fire immediately when technologies or laws change, so that
# migration crowding modifiers respond quickly rather than waiting for
# the next periodic pulse.
# NOTE: Pollution and free-port modifiers use has_variable in their
# script values. The engine traces the parent scope chain, so calling
# them from law scope (root = law) triggers "has_variable: scope doesn't
# support variables" errors. Only migration crowding (which has no
# has_variable) is safe to call from the law hook. The full set of
# state modifiers is refreshed on the periodic state pulses.
# ============================================================================

# Country scope. Updates all states owned by the country when a technology
# is researched, since techs can unlock production methods that carry
# modifiers such as state_migration_crowding_density_mult.
# NOTE: Do NOT call te_update_pollution_modifier here. pollution_generation
# is state-scoped and the engine traces the on_acquired_technology country
# root scope, producing hundreds of "Wrong scope: country, expected state"
# errors. Pollution is caught by the periodic state pulse instead.
te_modifier_update_on_technology_on_action = {
	effect = {
		every_scope_state = {
			te_update_migration_crowding_modifier = yes
			te_update_free_port_tariff = yes
		}
	}
}

#==================================================
# Overbuild Protection - removes excess sub-buildings
# when slot modifier goes negative (race condition fix).
# Grace period: buildings are only removed after ~6 consecutive
# months over the limit. The counter resets if the modifier
# returns to zero or positive.
#==================================================

overbuild_protection_on_action = {
	effect = {
		# --- Solar receivers ---
		if = {
			limit = { modifier:country_solar_receiver_max_level_add < 0 }
			# Increment grace counter
			if = {
				limit = { NOT = { has_variable = overbuild_solar_receiver_months } }
				set_variable = { name = overbuild_solar_receiver_months value = 1 }
			}
			else = {
				change_variable = { name = overbuild_solar_receiver_months add = 1 }
			}
			# After 6 months, start removing 1 excess level per month
			if = {
				limit = { var:overbuild_solar_receiver_months >= 6 }
				ordered_scope_state = {
					limit = { has_building = building_solar_receiver }
					order_by = market.mg:electricity.market_goods_cheaper
					max = 1
					set_variable = {
						name = lvl_tmp
						value = b:building_solar_receiver.level
					}
					change_variable = {
						name = lvl_tmp
						add = -1
					}
					remove_building = building_solar_receiver
					if = {
						limit = { var:lvl_tmp > 0 }
						extended_timeline_add_specified_building_level = {
							ADD_LEVEL = var:lvl_tmp
							SPEC_TYPE = building_solar_receiver
						}
					}
					remove_variable = lvl_tmp
				}
			}
		}
		else = {
			remove_variable = overbuild_solar_receiver_months
		}

		# --- Antimatter sub-buildings ---
		if = {
			limit = { modifier:country_antimatter_facility_max_level_add < 0 }
			# Increment grace counter
			if = {
				limit = { NOT = { has_variable = overbuild_antimatter_months } }
				set_variable = { name = overbuild_antimatter_months value = 1 }
			}
			else = {
				change_variable = { name = overbuild_antimatter_months add = 1 }
			}
			# After 6 months, start removing 1 excess level per month
			if = {
				limit = { var:overbuild_antimatter_months >= 6 }
				random_scope_state = {
					limit = { has_building = building_antimatter_engine }
					set_variable = {
						name = lvl_tmp
						value = b:building_antimatter_engine.level
					}
					change_variable = {
						name = lvl_tmp
						add = -1
					}
					remove_building = building_antimatter_engine
					if = {
						limit = { var:lvl_tmp > 0 }
						extended_timeline_add_specified_building_level = {
							ADD_LEVEL = var:lvl_tmp
							SPEC_TYPE = building_antimatter_engine
						}
					}
					remove_variable = lvl_tmp
				}
				random_scope_state = {
					limit = { has_building = building_antimatter_warhead_plant }
					set_variable = {
						name = lvl_tmp
						value = b:building_antimatter_warhead_plant.level
					}
					change_variable = {
						name = lvl_tmp
						add = -1
					}
					remove_building = building_antimatter_warhead_plant
					if = {
						limit = { var:lvl_tmp > 0 }
						extended_timeline_add_specified_building_level = {
							ADD_LEVEL = var:lvl_tmp
							SPEC_TYPE = building_antimatter_warhead_plant
						}
					}
					remove_variable = lvl_tmp
				}
			}
		}
		else = {
			remove_variable = overbuild_antimatter_months
		}
	}
}
