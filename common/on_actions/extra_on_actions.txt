# Root = the dissolving (joining) market, scope:market = the absorbing market.
# owner = the country whose market is dissolving (new market member).
# scope:market.owner = the country that owns the absorbing market (the market leader).
gw_market_join_on_action = {
	effect = {
		if = {
			limit = {
				exists = owner
				exists = scope:market.owner
			}
			owner = {
				# Apply any active market-leader policies from the absorbing market's leader.
				if = {
					limit = {
						NOT = { has_modifier = carbon_tax_modifier }
						scope:market.owner = { has_modifier = carbon_tax_modifier }
					}
					add_modifier = { name = carbon_tax_modifier }
				}
				if = {
					limit = {
						NOT = { has_modifier = renewable_investment_modifier }
						scope:market.owner = { has_modifier = renewable_investment_modifier }
					}
					add_modifier = { name = renewable_investment_modifier }
				}
				if = {
					limit = {
						NOT = { has_modifier = emission_standards_modifier }
						scope:market.owner = { has_modifier = emission_standards_modifier }
					}
					add_modifier = { name = emission_standards_modifier }
				}
			}
		}
	}
}

global_warming_update_on_action = {
	effect = {
		if = {
			limit = {
				NOT = { has_global_variable = greenhouse_gas_emissions }
			}
			set_global_variable = {
				name = greenhouse_gas_emissions
				value = 0
			}
		}
		if = {
			limit = {
				owner.market_capital = THIS
			}
			market = {
				# Apply market emissions scaled by the owner country's net emission multiplier.
				# market_capital.owner evaluates gw_emission_multiplier_script_value in country scope,
				# picking up carbon_tax_modifier (-20%), renewable_investment_modifier (-15%),
				# and any future country_greenhouse_gas_emissions_mult modifiers automatically.
				change_global_variable = {
					name = greenhouse_gas_emissions
					add = {
						value = market_greenhouse_gas_emissions_script_value
						multiply = market_capital.owner.gw_emission_multiplier_script_value
					}
				}
			}
		}
		# Carbon capture: subtract synthetic oil/coal production (not fossil extraction)
		every_scope_building = {
			limit = { is_building_type = building_synthetics_plant_oil }
			if = {
				limit = { has_active_production_method = pm_synthetic_oil_2 }
				change_global_variable = {
					name = greenhouse_gas_emissions
					add = {
						value = level
						multiply = -0.06  # 300 oil Ã— 2 weight / 10000
					}
				}
			}
			else = {
				change_global_variable = {
					name = greenhouse_gas_emissions
					add = {
						value = level
						multiply = -0.036  # 180 oil Ã— 2 weight / 10000
					}
				}
			}
		}
		every_scope_building = {
			limit = { is_building_type = building_synthetics_plant_coal }
			change_global_variable = {
				name = greenhouse_gas_emissions
				add = {
					value = level
					multiply = -0.168  # 840 coal Ã— 2 weight / 10000
				}
			}
		}
	}
}

global_warming_events_on_action = {
	effect = {
		# Fire threshold events exactly once globally when temperature crosses each threshold.
		# Runs in the global (world) scope of on_monthly_pulse — fires once per month,
		# so every_country here reaches all countries without per-country duplication.
		if = {
			limit = {
				temperature_anomaly_display >= 0.5
				NOT = { has_global_variable = gw_mild_event_fired }
			}
			set_global_variable = gw_mild_event_fired
			every_country = {
				limit = { NOT = { is_country_type = decentralized } }
				trigger_event = { id = environmentalism_events.1 }
			}
		}
		if = {
			limit = {
				temperature_anomaly_display >= 1.0
				NOT = { has_global_variable = gw_moderate_event_fired }
			}
			set_global_variable = gw_moderate_event_fired
			every_country = {
				limit = { NOT = { is_country_type = decentralized } }
				trigger_event = { id = environmentalism_events.2 }
			}
		}
		if = {
			limit = {
				temperature_anomaly_display >= 2.0
				NOT = { has_global_variable = gw_severe_event_fired }
			}
			set_global_variable = gw_severe_event_fired
			every_country = {
				limit = { NOT = { is_country_type = decentralized } }
				trigger_event = { id = environmentalism_events.3 }
			}
		}
		if = {
			limit = {
				temperature_anomaly_display >= 3.0
				NOT = { has_global_variable = gw_catastrophic_event_fired }
			}
			set_global_variable = gw_catastrophic_event_fired
			every_country = {
				limit = { NOT = { is_country_type = decentralized } }
				trigger_event = { id = environmentalism_events.4 }
			}
		}

		# Recurring random events for countries experiencing global warming.
		# Each country has a small monthly chance of getting one event.
		# Event triggers are checked inside the event definitions themselves.
		every_country = {
			limit = {
				NOT = { is_country_type = decentralized }
				has_global_variable = greenhouse_gas_emissions
			}
			random_list = {
				900 = { } # ~90% chance of nothing happening this month
				# Crop failures (severe+ warming, farming country)
				15 = {
					trigger = {
						temperature_anomaly_display >= 2.0
						any_scope_state = {
							is_incorporated = yes
							any_scope_building = { is_building_type = building_wheat_farm }
						}
					}
					trigger_event = { id = environmentalism_events.5 }
				}
				# Climate refugees (moderate+ warming)
				15 = {
					trigger = {
						has_modifier = global_warming
						temperature_anomaly_display >= 1.0
					}
					trigger_event = { id = environmentalism_events.6 }
				}
				# Pollution scandal (has coal/oil industry)
				15 = {
					trigger = {
						any_scope_state = {
							is_incorporated = yes
							any_scope_building = {
								OR = {
									is_building_type = building_coal_mine
									is_building_type = building_oil_rig
								}
								level >= 3
							}
						}
					}
					trigger_event = { id = environmentalism_events.7 }
				}
				# Environmentalist movement (mild+ warming, has pollution_control)
				15 = {
					trigger = {
						has_modifier = global_warming
						temperature_anomaly_display >= 0.5
						has_technology_researched = pollution_control
					}
					trigger_event = { id = environmentalism_events.8 }
				}
				# Coastal flooding (moderate+ warming, coastal ports)
				10 = {
					trigger = {
						has_modifier = global_warming
						temperature_anomaly_display >= 1.0
						any_scope_state = {
							is_incorporated = yes
							is_coastal = yes
							has_building = building_port
						}
					}
					trigger_event = { id = environmentalism_events.9 }
				}
				# Water shortage (severe+ warming)
				10 = {
					trigger = {
						has_modifier = global_warming
						temperature_anomaly_display >= 2.0
					}
					trigger_event = { id = environmentalism_events.10 }
				}
				# Green tech boom (environmental_movement tech + renewables)
				10 = {
					trigger = {
						has_modifier = global_warming
						has_technology_researched = environmental_movement
						any_scope_state = {
							has_building = building_renewable_energy_plant
						}
					}
					trigger_event = { id = environmentalism_events.11 }
				}
				# Fossil fuel lobby (has emission policies + large fossil industry)
				10 = {
					trigger = {
						OR = {
							has_modifier = carbon_tax_modifier
							has_modifier = emission_standards_modifier
							has_modifier = fossil_fuel_divestment_modifier
						}
						any_scope_state = {
							any_scope_building = {
								OR = {
									is_building_type = building_coal_mine
									is_building_type = building_oil_rig
								}
								level >= 5
							}
						}
					}
					trigger_event = { id = environmentalism_events.12 }
				}
				# Climate summit (major+ power, moderate+ warming)
				10 = {
					trigger = {
						has_modifier = global_warming
						temperature_anomaly_display >= 1.0
						country_rank >= rank_value:major_power
						has_technology_researched = pollution_control
					}
					trigger_event = { id = environmentalism_events.13 }
				}
				# Disease outbreak (severe+ warming)
				10 = {
					trigger = {
						has_modifier = global_warming
						temperature_anomaly_display >= 2.0
					}
					trigger_event = { id = environmentalism_events.14 }
				}
				# Political clash (clean_energy_technologies + industrialists in gov + catastrophic warming)
				10 = {
					trigger = {
						has_modifier = global_warming
						temperature_anomaly_display >= 3.0
						has_technology_researched = clean_energy_technologies
						ig:ig_industrialists ?= { is_in_government = yes }
					}
					trigger_event = { id = environmentalism_events.15 }
				}
				# Melting ice caps (high warming, coastal)
				10 = {
					trigger = {
						has_modifier = global_warming
						temperature_anomaly_display >= 2.5
						any_scope_state = { is_coastal = yes }
					}
					trigger_event = { id = environmentalism_events.16 }
				}
			}
		}
	}
}

excess_private_construction_on_action = {
	effect = {
		if = {
			limit = {
				OR = {
					AND = {
						construction_queue_num_queued_private_levels > target_queue_length
						investment_pool_net_income > 0
					}
					has_modifier = too_much_private_construction
				}
			}
			if = {
				limit = {
					OR = {
						has_modifier = too_much_private_construction
						too_much_private_construction_script_value > 0
					}
				}
				remove_modifier = too_much_private_construction
				if = {
					limit = {
						too_much_private_construction_script_value > 0
					}
					add_modifier = {
						name = too_much_private_construction
						multiplier = too_much_private_construction_script_value
	
					}
				}
			}
			set_variable = {
				name = too_much_private_construction_cached
				value = too_much_private_construction_script_value
			}
		}
		remove_modifier = overinvestment_modifier
		if = {
			limit = {
				has_variable = too_much_private_construction_cached
				var:too_much_private_construction_cached > 1000
				investment_pool > gdp
			}
			add_modifier = {
				name = overinvestment_modifier
			}
		}
	}
}

state_yearly_cultural_acceptance_add_on_action = {
	effect = {
		if = {
			limit = {
				AND = {
					OR = {
						modifier:state_yearly_cultural_acceptance_add > 0
						modifier:state_min_cultural_acceptance_delta_add > 0
					}
					any_scope_culture = {
						ROOT = {
							"cultural_acceptance_delta(PREV)" < -1
						}
					}
				}
			}
			every_scope_culture = {
				limit = {
					ROOT = {
						"cultural_acceptance_delta(PREV)" < -1
					}
				}
				ROOT = {
					if = {
						limit = {
							"cultural_acceptance_delta(PREV)" < negative_state_min_cultural_acceptance_delta_add
						}
						add_acceptance = {
							target = PREV
							value = {
								value = negative_state_min_cultural_acceptance_delta_add
								subtract = "cultural_acceptance_delta(PREV)"
							}
						}
					}
					add_acceptance = {
						target = PREV
						value = {
							add = modifier:state_yearly_cultural_acceptance_add
							divide = 12 # monthly on_action
							max = {
								add = "cultural_acceptance_delta(PREV)"
								multiply = -1
								subtract = 1 # cannot add more delta than what gets you to -1
							}
							min = 0
						}
					}
				}
			}
		}
	}
}

international_relations_events_on_action = {
	effect = {
		if = {
			limit = {
				NOT = { is_country_type = decentralized }
				country_rank >= rank_value:major_power
			}
			random_list = {
				900 = { } # ~90% chance of nothing
				# Arms Race Escalation
				10 = {
					trigger = {
						has_technology_researched = radar
						any_rival_country = { country_rank >= rank_value:major_power }
					}
					trigger_event = { id = international_relations_events.1 }
				}
				# Proxy Conflict
				10 = {
					trigger = {
						has_technology_researched = intergovernmental_organizations
						any_rival_country = {
							country_rank >= rank_value:major_power
							any_subject_or_below = { always = yes }
						}
					}
					trigger_event = { id = international_relations_events.2 }
				}
				# International Summit
				10 = {
					trigger = {
						has_technology_researched = intergovernmental_organizations
						country_rank >= rank_value:great_power
					}
					trigger_event = { id = international_relations_events.3 }
				}
				# Propaganda War
				15 = {
					trigger = {
						has_technology_researched = mass_media
						any_rival_country = { always = yes }
					}
					trigger_event = { id = international_relations_events.4 }
				}
				# Trade Embargo
				10 = {
					trigger = {
						has_technology_researched = keynesian_economics
						any_rival_country = { country_rank >= rank_value:major_power }
					}
					trigger_event = { id = international_relations_events.5 }
				}
				# Espionage Scandal
				15 = {
					trigger = {
						has_technology_researched = cryptography
						any_rival_country = { always = yes }
					}
					trigger_event = { id = international_relations_events.6 }
				}
				# Space Race Achievement
				5 = {
					trigger = {
						has_technology_researched = space_exploration
						country_rank >= rank_value:great_power
					}
					trigger_event = { id = international_relations_events.7 }
				}
				# Nuclear Brinkmanship
				5 = {
					trigger = {
						has_modifier = nuclear_power
						any_rival_country = { has_modifier = nuclear_power }

decolonization_events_on_action = {
	effect = {
		if = {
			limit = {
				NOT = { is_country_type = decentralized }
				has_technology_researched = decolonization
			}
			random_list = {
				900 = { } # ~90% chance of nothing
				# Colonial Independence Movement
				15 = {
					trigger = {
						any_subject_or_below = { liberty_desire >= 50 }
					}
					trigger_event = { id = decolonization_events.1 }
				}
				# International Pressure to Decolonize
				10 = {
					trigger = {
						has_technology_researched = intergovernmental_organizations
						any_subject_or_below = { always = yes }
						any_country = {
							NOT = { this = root }
							country_rank >= rank_value:great_power
							has_technology_researched = decolonization
						}
					}
					trigger_event = { id = decolonization_events.2 }
				}
				# Violent Independence Struggle
				10 = {
					trigger = {
						has_technology_researched = civil_rights_movement
						any_subject_or_below = { liberty_desire >= 75 }
					}
					trigger_event = { id = decolonization_events.3 }
				}
				# Cultural Reclamation (for former colonies)
				10 = {
					trigger = {
						is_subject = no
						NOT = { any_scope_state = { is_homeland_of_country_cultures = ROOT } }
					}
					trigger_event = { id = decolonization_events.4 }
				}
				# Former Colony Seeks Closer Ties
				10 = {
					trigger = {
						country_rank >= rank_value:major_power
						NOT = { any_subject_or_below = { always = yes } }
						any_country = {
							NOT = { this = root }
							was_formed_from = ROOT
							is_subject = no
						}
					}
					trigger_event = { id = decolonization_events.5 }
				}
				# Colonial Reparations Demanded
				10 = {
					trigger = {
						country_rank >= rank_value:major_power
						any_country = {
							NOT = { this = root }
							was_formed_from = ROOT
							is_subject = no
						}
					}
					trigger_event = { id = decolonization_events.6 }
				}
				# Post-Colonial Nation Building (for new countries)
				10 = {
					trigger = {
						country_rank <= rank_value:minor_power
						has_variable = recently_formed_country
					}
					trigger_event = { id = decolonization_events.7 }
				}
				# Rise of Anti-Colonial Ideology
				10 = {
					trigger = {
						any_subject_or_below = { always = yes }
						ig:ig_intelligentsia ?= {
							leader = { has_ideology = ideology:ideology_anti_colonialist }
						}
					}
					trigger_event = { id = decolonization_events.8 }
				}
				# Overseas Colonial Unrest (directly-owned non-homeland territory)
				15 = {
					trigger = {
						any_scope_state = {
							is_overseas_colonial_state = yes
							any_scope_pop = {
								pop_acceptance < 20
							}
						}
					}
					trigger_event = { id = decolonization_events.9 }
				}
				# Colonial Integration Milestone
				10 = {
					trigger = {
						any_scope_state = {
							NOT = { is_homeland_of_country_cultures = ROOT }
							state_population >= 100000
							is_overseas_from_capital = yes
							NOT = {
								any_scope_pop = {
									pop_acceptance < acceptance_status_4
								}
							}
						}
					}
					trigger_event = { id = decolonization_events.10 }
				}
				# Great Power Condemns Colonial Empire
				10 = {
					trigger = {
						any_scope_state = {
							is_overseas_colonial_state = yes
						}
						any_country = {
							NOT = { this = root }
							country_rank >= rank_value:great_power
							has_variable = gp_anti_colonial_stance
						}
					}
					trigger_event = { id = decolonization_events.11 }
				}
				# Great Power Supports Colonial Regime
				10 = {
					trigger = {
						any_scope_state = {
							is_overseas_colonial_state = yes
						}
						any_country = {
							NOT = { this = root }
							country_rank >= rank_value:great_power
							has_variable = gp_pro_colonial_stance
						}
					}
					trigger_event = { id = decolonization_events.12 }
				}
				# Colonial Uprising (high unrest)
				10 = {
					trigger = {
						colonial_low_acceptance_count >= 2
						any_scope_state = {
							is_overseas_colonial_state = yes
							any_scope_pop = {
								pop_acceptance < 20
							}
						}
					}
					trigger_event = { id = decolonization_events.13 }
				}
				# Great Power Colonial Policy Review (fires for GPs)
				15 = {
					trigger = {
						country_rank >= rank_value:great_power
						NOT = { has_variable = gp_colonial_policy_cooldown }
						any_country = {
							NOT = { this = root }
							any_scope_state = {
								is_overseas_colonial_state = yes
							}
						}
					}
					trigger_event = { id = decolonization_events.14 }
				}
			}
		}
	}
