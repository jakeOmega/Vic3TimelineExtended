# Root = From Market
# scope:market = Market merging into
# scope:trade_center = Trade Center state of root
on_merge_markets = {
	on_actions = {
		te_construction_market_on_merge_markets
	}
}

te_construction_market_on_merge_markets = {
	effect = {
		if = {
			limit = {
				exists = owner
			}
			owner = {
				trigger_event = {
					id = te_construction_market_events.4
				}
			}
		}
		if = {
			limit = {
				exists = scope:market.owner
			}
			scope:market.owner = {
				trigger_event = {
					id = te_construction_market_events.4
				}
			}
		}
	}
}
