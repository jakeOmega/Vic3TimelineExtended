########################################
# LEGACY MODIFIER CLEANUP - On Action Hook
# Runs every month to remove country-scoped modifiers that are now
# JE-scoped. Cheap (just if/remove checks) and harmlessly no-ops
# when modifiers aren't present.
# Safe to delete this file once no saves from before the JE modifier
# migration exist.
# Also delete: common/scripted_effects/legacy_modifier_cleanup.txt
########################################

# Custom on_action with the cleanup effect — runs unconditionally
legacy_je_modifier_cleanup_on_action = {
	effect = {
		legacy_je_modifier_cleanup_effect = yes
	}
}

# Hook into monthly pulse so it runs quickly after save load
on_monthly_pulse_country = {
	on_actions = {
		legacy_je_modifier_cleanup_on_action
	}
}
