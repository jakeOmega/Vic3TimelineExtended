# Construction-market subsystem.
# Based on the third-party "Free Market Construction" mod. See README.md.
#
# Bound from gui/construction_panel.gui to the +/- buttons in the construction
# queue panel. Each handler nudges the country's public-construction buy target
# (te_construction_market_public_target) by a fixed amount: 1 (default click),
# 10 (shift), 100 (ctrl), 1000 (alt), 10000 (right-click). On the decrease
# side, right-click resets to 0 instead of stepping by 10000.

te_construction_market_increase_target_1 = {

	scope = country

	is_shown = {
		always = yes
	}

	ai_is_valid = {
		always = no
	}

	is_valid = {
		always = yes
	}

	effect = {
		#custom_tooltip = ""
		te_construction_market_init_target = yes
		change_variable = {
			name = te_construction_market_public_target
			add = 1
		}
		te_construction_market_country_update_private_share = yes
		te_construction_market_apply_per_site_modifier = yes
	}
}
