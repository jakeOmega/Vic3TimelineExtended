# =================================================================
# TREATY ARTICLE: seize_company
# A punitive article that forces a country to surrender one of its companies
# to another country. The source country is the one whose company is seized.
# The target country receives the company.
# =================================================================
seize_company = {
	icon = "gfx/interface/icons/law_icons/command_economy.dds" # Placeholder Icon
	kind = directed
	maintenance_paid_by = source_country
	cost = 200 # More expensive than disband (150) since the winner gains the company.

	flags = {
		can_be_enforced
		can_be_renegotiated
	}

	usage_limit = once_per_side_with_same_inputs

	# This article requires you to specify which company you are targeting.
	required_inputs = {
		company
	}
		
	visible = {
		always = yes
	}

	possible = {
		scope:other_country = {
			any_company = {
				owner = prev
			}
		}
	}

	can_ratify = {
		# Prevents adding the same demand twice in one treaty.
		custom_tooltip = {
			text = DUPLICATE_ARTICLE_SAME_INPUTS
			scope:treaty ?= {
				NOT = {
					any_scope_article_option = {
						count > 1
						has_type = seize_company
						input_company = root.input_company
					}
				}
			}
		}
		# The target country must not already have a company of this type.
		custom_tooltip = {
			text = SEIZE_COMPANY_ALREADY_HAS
			trigger_if = {
				limit = { exists = root.input_company }
				scope:target_country = {
					NOT = {
						has_company = root.input_company.type
					}
				}
			}
		}
	}

	# The source country's company is removed, the target country gains it.
	on_entry_into_force = {
		scope:article_options = {
			if = {
				limit = { exists = input_company }
				input_company = {
					save_scope_as = target_company
				}
				source_country = {
					remove_company = scope:target_company.type
				}
				target_country = {
					add_company = scope:target_company.type
				}
			}
		}
	}

	on_withdrawal = {}
	on_break = {}

	company_valid_trigger = {
		always = yes
	}

	# This section governs AI behavior regarding the treaty article.
	ai = {
		treaty_categories = { economy }
		# The AI will demand this via "request": forcing the other side (source)
		# to surrender their company to the requesting country (target).
		article_ai_usage = { request }

		# AI will only consider this when significantly more powerful.
		evaluation_chance = {
			value = 0
			if = {
				limit = {
					exists = scope:other_country
					country_rank > scope:other_country.country_rank
				}
				add = 0.05
			}
		}

		# AI decision-making on whether to ACCEPT the article.
		inherent_accept_score = {
			value = 0

			# --- SOURCE COUNTRY EVALUATION (I must surrender my company) ---
			if = {
				limit = {
					exists = scope:article.input_company
					scope:source_country = this
				}

				scope:source_country = {
					save_temporary_scope_as = target
				}
				scope:target_country = {
					save_temporary_scope_as = actor
				}

				# Base reluctance: losing a company to a rival is worse than disbanding.
				add = {
					desc = "SEIZE_COMPANY_BASE_RELUCTANCE"
					value = -200
				}

				# Penalty scales with the company's economic contribution.
				subtract = {
					desc = "SEIZE_COMPANY_ECONOMIC_SIGNIFICANCE"
					scope:article.input_company ?= {
						every_scope_regional_hqs = {
							add = this.level
							multiply = {
								value = weekly_profit
								divide = 250
								add = 1
								max = 2
							}
						}
					}
					max = 300
					min = 30
				}

				# Great powers are extremely reluctant.
				if = {
					limit = { country_rank >= rank_value:great_power }
					add = {
						desc = "great_power"
						value = -300
					}
				}
				# Major powers are also quite reluctant.
				else_if = {
					limit = { country_rank >= rank_value:major_power }
					add = {
						desc = "major_power"
						value = -200
					}
				}

				# Relative rank: stronger countries resist more.
				if = {
					limit = { country_rank >= scope:target_country.country_rank }
					add = {
						desc = "DIPLOMATIC_ACCEPTANCE_RANK"
						value = scope:target_country.country_rank
						subtract = 2
						subtract = country_rank
						multiply = 10
					}
				}
			}

			# --- TARGET COUNTRY EVALUATION (I gain the company) ---
			else_if = {
				limit = {
					exists = scope:article.input_company
					scope:target_country = this
				}

				scope:target_country = {
					save_temporary_scope_as = target
				}
				scope:source_country = {
					save_temporary_scope_as = actor
				}

				# Gaining a company is highly desirable.
				add = {
					desc = "SEIZE_COMPANY_ACQUISITION_BENEFIT"
					scope:article.input_company ?= {
						every_scope_regional_hqs = {
							limit = {
								weekly_profit > 5
							}
							add = this.level
							multiply = {
								value = weekly_profit
								divide = 250
								add = 1
								max = 2
							}
						}
					}
					max = 75
					min = 10
				}
			}
		}
	}

	# This defines the wargoal if this article is pushed in a diplomatic play.
	wargoal = {
		execution_priority = 75 # Higher than disband (70): more aggressive demand.
		contestion_type = control_target_country_capital
		
		# Infamy is higher than disband since you're taking, not just destroying.
		infamy = {
			value = 15
			scope:target_country = {
				multiply = {
					value = gdp
					divide = 10000000
					add = 1
					max = 2.5
				}
			}
		}
	}
}
