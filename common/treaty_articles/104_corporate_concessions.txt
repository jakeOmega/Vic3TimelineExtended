# =================================================================
# TREATY ARTICLE: corporate_concessions
# Forces a country to give preferential regulatory treatment to
# corporate interests, boosting company throughput at the expense of
# general industrial output and political stability.
# Historical: East India Company, United Fruit, Belgian Congo
# concessions, modern Special Economic Zones.
#
# The company input selects a target company for diplomatic flavor
# and usage_limit tracking. The mechanical effects are country-wide
# (engine has no per-company modifier).
# =================================================================
corporate_concessions = {
	icon = "gfx/interface/icons/law_icons/laissez_faire.dds"
	kind = directed
	cost = 100

	flags = {
		can_be_enforced
		can_be_renegotiated
		hostile
	}

	usage_limit = once_per_side_with_same_inputs

	maintenance_paid_by = target_country

	required_inputs = {
		company
	}

	unlocked_by_technologies = {
		corporate_charters
	}

	source_modifier = {
		country_company_throughput_bonus_add = 0.25
		state_radicals_from_political_movements_mult = 0.25
		country_authority_add = -200
	}

	target_modifier = {
		country_prestige_mult = 0.03
		country_company_construction_efficiency_bonus_add = 0.1
	}

	visible = {
		always = no # WIP
	}

	possible = {
		has_diplomatic_relevance = scope:other_country
		NOT = { has_war_with = scope:other_country }
		NOT = { is_country_type = decentralized }
		scope:other_country = {
			NOT = { is_country_type = decentralized }
			NOR = {
				has_law = law_type:law_command_economy
				has_law = law_type:law_cooperative_ownership
			}
		}
	}

	can_ratify = {
		custom_tooltip = {
			text = DUPLICATE_ARTICLE_SAME_INPUTS
			scope:treaty ?= {
				NOT = {
					any_scope_article_option = {
						count > 1
						has_type = corporate_concessions
						input_company = root.input_company
					}
				}
			}
		}
	}

	non_fulfillment = {
		consequences = freeze
		conditions = {
			monthly = {
				scope:article = {
					source_country = {
						NOR = {
							has_law = law_type:law_command_economy
							has_law = law_type:law_cooperative_ownership
						}
					}
				}
			}
		}
	}

	ai = {
		treaty_categories = { other }
		article_ai_usage = { request }

		evaluation_chance = {
			value = 0.05
			if = {
				limit = {
					has_attitude = { who = scope:other_country attitude = domineering }
				}
				add = 0.15
			}
			if = {
				limit = {
					country_rank >= rank_value:great_power
				}
				add = 0.1
			}
			if = {
				limit = {
					OR = {
						has_law = law_type:law_laissez_faire
						has_law = law_type:law_interventionism
					}
				}
				add = 0.05
			}
		}

		inherent_accept_score = {
			value = 0

			# Source (granting concessions) is reluctant
			if = {
				limit = {
					scope:source_country = this
				}
				add = {
					desc = "DIPLOMATIC_ACCEPTANCE_BASE"
					value = -60
				}
				# Trade unions in government make it worse
				if = {
					limit = {
						ig:ig_trade_unions = {
							is_in_government = yes
						}
					}
					add = {
						desc = "AI_TRADE_UNIONS_IN_GOVERNMENT"
						value = -20
					}
				}
				# Industrialists in power make it less bad
				if = {
					limit = {
						ig:ig_industrialists = {
							is_in_government = yes
						}
					}
					add = {
						desc = "AI_INDUSTRIALISTS_IN_GOVERNMENT"
						value = 15
					}
				}
				# Hostile to the demander: extra resistance
				if = {
					limit = {
						OR = {
							has_attitude = { who = scope:target_country attitude = hostile }
							has_attitude = { who = scope:target_country attitude = belligerent }
						}
					}
					add = {
						desc = "AI_HOSTILE_ATTITUDE"
						value = -30
					}
				}
			}
			# Target (imperial power) wants this
			else = {
				# Non-capitalist countries have no use for corporate concessions
				if = {
					limit = {
						NOR = {
							has_law = law_type:law_laissez_faire
							has_law = law_type:law_interventionism
						}
					}
					add = {
						desc = "AI_NOT_CAPITALIST_ECONOMY"
						value = -1000
					}
				}
				else = {
					add = {
						desc = "DIPLOMATIC_ACCEPTANCE_BASE"
						value = 15
					}
					if = {
						limit = {
							has_attitude = { who = scope:source_country attitude = domineering }
						}
						add = {
							desc = "AI_DOMINEERING_ATTITUDE"
							value = 15
						}
					}
					# Industrialists in target benefit from concessions
					if = {
						limit = {
							ig:ig_industrialists = {
								is_in_government = yes
							}
						}
						add = {
							desc = "AI_INDUSTRIALISTS_IN_GOVERNMENT"
							value = 10
						}
					}
					# Laissez faire: ideologically aligned
					if = {
						limit = {
							has_law = law_type:law_laissez_faire
						}
						add = {
							desc = "AI_LAISSEZ_FAIRE_ECONOMY"
							value = 10
						}
					}
					# Friendly nations don't impose
					if = {
						limit = {
							OR = {
								has_attitude = { who = scope:source_country attitude = friendly }
								has_attitude = { who = scope:source_country attitude = cooperative }
							}
						}
						add = {
							desc = "AI_FRIENDLY_ATTITUDE"
							value = -20
						}
					}
				}
			}
		}
	}
}
