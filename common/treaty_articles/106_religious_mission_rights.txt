# =================================================================
# TREATY ARTICLE: religious_mission_rights
# A foreign power secures the right to send missionaries, converting
# the population of a target state toward the missionary power's
# religion. Historical: Jesuits in Japan/China, Christian missions
# in colonial Africa, French protectorate over Catholics in the Levant.
#
# Uses a periodic on_yearly_pulse_state on_action to apply
# convert_population toward the target country's religion.
# =================================================================
religious_mission_rights = {
	icon = "gfx/interface/icons/law_icons/state_religion.dds"
	kind = directed
	cost = 50

	relations_progress_per_day = -0.25
	relations_improvement_min = 0

	flags = {
		can_be_enforced
		can_be_renegotiated
		hostile
	}

	usage_limit = once_per_side_with_same_inputs

	maintenance_paid_by = target_country

	required_inputs = {
		state
	}

	unlocked_by_technologies = {
		colonization
	}

	source_modifier = {
		country_authority_add = -100
		country_legitimacy_base_add = -5
	}

	target_modifier = {
		country_prestige_mult = 0.03
	}

	visible = {
		always = yes
	}

	possible = {
		has_diplomatic_relevance = scope:other_country
		NOT = { has_war_with = scope:other_country }
		NOT = { is_country_type = decentralized }
		scope:other_country = {
			NOT = { is_country_type = decentralized }
			NOT = { has_law = law_type:law_state_atheism }
		}
	}

	can_ratify = {
		custom_tooltip = {
			text = DUPLICATE_ARTICLE_SAME_INPUTS
			NOT = {
				scope:source_country = {
					any_scope_treaty = {
						binds = scope:target_country
						hidden_trigger = {
							OR = {
								scope:treaty = {
									is_renegotiation = no
								}
								scope:treaty.amended_treaty != this
							}
						}
						any_scope_article = {
							has_type = religious_mission_rights
							source_country = scope:source_country
						}
					}
				}
			}
		}
	}

	state_valid_trigger = {
		scope:input = {
			owner = root
		}
	}

	requirement_to_maintain = {
		trigger = {
			exists = scope:article.input_state
		}
	}

	non_fulfillment = {
		consequences = freeze
		conditions = {
			monthly = {
				scope:article = {
					source_country = {
						NOT = { has_law = law_type:law_state_atheism }
					}
				}
			}
		}
	}

	on_entry_into_force = {
		scope:article_options = {
			target_country = {
				save_scope_as = mission_country
			}
			if = {
				limit = { exists = input_state }
				input_state = {
					add_modifier = {
						name = religious_mission_state_modifier
					}
					set_variable = {
						name = religious_mission_country
						value = scope:mission_country
					}
				}
			}
		}
	}

	on_break = {
		scope:article_options.input_state ?= {
			remove_modifier = religious_mission_state_modifier
			remove_variable = religious_mission_country
		}
	}

	on_withdrawal = {
		scope:article_options.input_state ?= {
			remove_modifier = religious_mission_state_modifier
			remove_variable = religious_mission_country
		}
	}

	ai = {
		treaty_categories = { other }
		article_ai_usage = { request }

		evaluation_chance = {
			value = 0.05
			# Don't bother if same religion — no need for missionaries
			if = {
				limit = {
					religion = scope:other_country.religion
				}
				add = -1.0
			}
			# Secular states don't care about proselytizing
			if = {
				limit = {
					OR = {
						has_law = law_type:law_total_separation
						has_law = law_type:law_freedom_of_conscience
					}
				}
				add = -1.0
			}
			if = {
				limit = {
					has_law = law_type:law_state_religion
				}
				add = 0.15
			}
			if = {
				limit = {
					ig:ig_devout = {
						is_in_government = yes
					}
				}
				add = 0.1
			}
			if = {
				limit = {
					has_attitude = { who = scope:other_country attitude = domineering }
				}
				add = 0.1
			}
		}

		inherent_accept_score = {
			value = 0

			# Source (allowing missionaries) is reluctant
			if = {
				limit = {
					scope:source_country = this
				}
				add = {
					desc = "DIPLOMATIC_ACCEPTANCE_BASE"
					value = -40
				}
				# Same religion: missionaries are harmless
				if = {
					limit = {
						religion = scope:target_country.religion
					}
					add = {
						desc = "AI_SAME_RELIGION"
						value = 35
					}
				}
				else = {
					# Different religion with state religion: faith is directly threatened
					if = {
						limit = {
							OR = {
								has_law = law_type:law_state_religion
								has_law = law_type:law_millet_system
							}
						}
						add = {
							desc = "AI_STATE_RELIGION_THREATENED"
							value = -25
						}
					}
					# Devout IG in government resists foreign missionaries
					if = {
						limit = {
							ig:ig_devout = {
								is_in_government = yes
							}
						}
						add = {
							desc = "AI_DEVOUT_IN_GOVERNMENT"
							value = -20
						}
					}
				}
				# Already tolerant: somewhat less reluctant
				if = {
					limit = {
						OR = {
							has_law = law_type:law_freedom_of_conscience
							has_law = law_type:law_total_separation
						}
					}
					add = {
						desc = "AI_TOLERANT_LAWS"
						value = 15
					}
				}
			}
			# Target (missionary power) wants this
			else = {
				# Same religion: no point sending missionaries
				if = {
					limit = {
						religion = scope:source_country.religion
					}
					add = {
						desc = "AI_SAME_RELIGION_NO_NEED"
						value = -1000
					}
				}
				else = {
					# Secular states have no interest in proselytizing
					if = {
						limit = {
							has_law = law_type:law_total_separation
						}
						add = {
							desc = "AI_SECULAR_STATE"
							value = -1000
						}
					}
					else_if = {
						limit = {
							has_law = law_type:law_freedom_of_conscience
						}
						add = {
							desc = "AI_SECULAR_STATE"
							value = -40
						}
					}
					else_if = {
						limit = {
							has_law = law_type:law_state_religion
						}
						# State religion actively wants to spread faith
						add = {
							desc = "AI_STATE_RELIGION"
							value = 25
						}
					}
					else_if = {
						limit = {
							has_law = law_type:law_people_of_the_book
						}
						add = {
							desc = "AI_PEOPLE_OF_THE_BOOK"
							value = 10
						}
					}
					else_if = {
						limit = {
							has_law = law_type:law_millet_system
						}
						add = {
							desc = "AI_MILLET_SYSTEM"
							value = 5
						}
					}
					# Devout IG in government wants to spread faith
					if = {
						limit = {
							ig:ig_devout = {
								is_in_government = yes
							}
						}
						add = {
							desc = "AI_DEVOUT_IN_GOVERNMENT"
							value = 15
						}
					}
					# Domineering attitude: wants to impose
					if = {
						limit = {
							has_attitude = { who = scope:source_country attitude = domineering }
						}
						add = {
							desc = "AI_DOMINEERING_ATTITUDE"
							value = 10
						}
					}
				}
			}
		}
	}
}
