# =================================================================
# TREATY ARTICLE: disband_company
# A punitive article that forces a country to dissolve one of its major companies.
# The source country (the side whose column the article appears in) is the one
# whose company will be disbanded. The target country uses "request" to demand
# the source disband it.
# =================================================================
disband_company = {
	icon = "gfx/interface/icons/law_icons/command_economy.dds" # Placeholder Icon
	kind = directed
	maintenance_paid_by = source_country
	cost = 150 # A significant influence cost to demand this.

	flags = {
		can_be_enforced
		can_be_renegotiated
	}

	usage_limit = once_per_side_with_same_inputs

	# This article requires you to specify which company you are targeting.
	required_inputs = {
		company
	}
		
	visible = {
		always = yes
	}

	possible = {
		scope:other_country = {
			any_company = {
				owner = prev
			}
		}
	}

	can_ratify = {
		# Prevents adding the same demand twice in one treaty.
		custom_tooltip = {
			text = DUPLICATE_ARTICLE_SAME_INPUTS
			scope:treaty ?= {
				NOT = {
					any_scope_article_option = {
						count > 1
						has_type = disband_company
						input_company = root.input_company
					}
				}
			}
		}
	}

	# The source country's company is disbanded.
	on_entry_into_force = {
		scope:article_options = {
			if = {
				limit = { exists = input_company }
				input_company = {
					save_scope_as = target_company
				}
				source_country = {
					remove_company = scope:target_company.type
				}
			}
		}
	}

	on_withdrawal = {}
	on_break = {}

	company_valid_trigger = {
		always = yes
    }

	# This section governs AI behavior regarding the treaty article.
	ai = {
		treaty_categories = { economy }
		# The AI will demand this from others via "request" - the other side becomes
		# source_country and must disband their own company.
		article_ai_usage = { request }

		# AI will only consider this when significantly more powerful.
		evaluation_chance = {
			value = 0
			if = {
				limit = {
					exists = scope:other_country
					country_rank > scope:other_country.country_rank
				}
				add = 0.1
			}
		}

		# AI decision-making on whether to ACCEPT the article.
		# When evaluated, "this" is the country deciding. The article forces
		# source_country to disband their company.
		inherent_accept_score = {
			value = 0

			# --- SOURCE COUNTRY EVALUATION (I must disband my company) ---
			if = {
				limit = {
					exists = scope:article.input_company
					scope:source_country = this
				}

				# We are doing this to be able to reuse loc across treaties and diplomatic actions
				scope:source_country = {
					save_temporary_scope_as = target
				}
				scope:target_country = {
					save_temporary_scope_as = actor
				}

				# Base reluctance: nobody wants to lose a company.
				add = {
					desc = "DISBAND_COMPANY_BASE_RELUCTANCE"
					value = -100
				}

				# Penalty scales with the company's economic contribution.
				# Count the levels of buildings owned by the targeted company.
				subtract = {
					desc = "DISBAND_COMPANY_ECONOMIC_SIGNIFICANCE"
					scope:article.input_company ?= {
						every_scope_regional_hqs = {
							add = this.level
							multiply = {
								value = weekly_profit
								divide = 250
								add = 1
								max = 2
							}
						}
					}
					max = 200
					min = 25
				}

				# Great powers are extremely reluctant to surrender companies via treaty.
				if = {
					limit = { country_rank >= rank_value:great_power }
					add = {
						desc = "great_power"
						value = -200
					}
				}
				# Major powers are also quite reluctant.
				else_if = {
					limit = { country_rank >= rank_value:major_power }
					add = {
						desc = "major_power"
						value = -100
					}
				}

				# Relative rank: stronger countries resist more.
				if = {
					limit = { country_rank >= scope:target_country.country_rank }
					add = {
						desc = "DIPLOMATIC_ACCEPTANCE_RANK"
						value = scope:target_country.country_rank
						subtract = 2
						subtract = country_rank
						multiply = 10
					}
				}
			}

			# --- TARGET COUNTRY EVALUATION (the other side disbands a company) ---
			else_if = {
				limit = {
					exists = scope:article.input_company
					scope:target_country = this
				}

				# We are doing this to be able to reuse loc across treaties and diplomatic actions
				scope:target_country = {
					save_temporary_scope_as = target
				}
				scope:source_country = {
					save_temporary_scope_as = actor
				}

				# Benefit from the source losing economic capacity.
				# The more profitable the company's buildings, the more we gain.
				add = {
					desc = "DISBAND_COMPANY_COMPETITIVE_BENEFIT"
					scope:article.input_company ?= {
						every_scope_regional_hqs = {
							limit = {
								weekly_profit > 5
							}
							add = this.level
							multiply = {
								value = weekly_profit
								divide = 250
								add = 1
								max = 2
							}
						}
					}
					max = 50
					min = 0
				}
			}
		}
	}

	# This defines the wargoal if this article is pushed in a diplomatic play.
	wargoal = {
		execution_priority = 70 # High priority, as it's a major concession.
		contestion_type = control_target_country_capital
		
		# Infamy is calculated based on the economic power of the target country.
		infamy = {
			value = 10
			scope:target_country = {
				multiply = {
					value = gdp
					divide = 10000000 # Scale based on GDP
					add = 1
					max = 2.5
				}
			}
		}
	}
}
