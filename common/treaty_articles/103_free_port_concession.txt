# =================================================================
# TREATY ARTICLE: free_port_concession
# Forces a country to open a specific coastal state as a free port,
# eliminating tariffs and boosting trade capacity. Historical: Treaty
# of Nanking opening Chinese ports, Hamburg as a free port, Singapore.
# =================================================================
free_port_concession = {
	icon = "gfx/interface/icons/law_icons/free_trade.dds"
	kind = directed
	cost = 75

	flags = {
		can_be_enforced
		can_be_renegotiated
		hostile
		giftable
	}

	usage_limit = once_per_side_with_same_inputs

	maintenance_paid_by = target_country

	required_inputs = {
		state
	}

	unlocked_by_technologies = {
		international_trade
	}

	source_modifier = {
		country_authority_add = -100
	}

	target_modifier = {
		country_prestige_mult = 0.03
	}

	visible = {
		always = yes
	}

	possible = {
		has_diplomatic_relevance = scope:other_country
		NOT = { has_war_with = scope:other_country }
		NOT = { is_country_type = decentralized }
		NOT = { has_law = law_type:law_isolationism }
		scope:other_country = {
			NOT = { is_country_type = decentralized }
			NOT = { has_law = law_type:law_isolationism }
		}
	}

	can_ratify = {
		custom_tooltip = {
			text = DUPLICATE_ARTICLE_SAME_INPUTS
			NOT = {
				scope:source_country = {
					any_scope_treaty = {
						binds = scope:target_country
						hidden_trigger = {
							OR = {
								scope:treaty = {
									is_renegotiation = no
								}
								scope:treaty.amended_treaty != this
							}
						}
						any_scope_article = {
							has_type = free_port_concession
							source_country = scope:source_country
						}
					}
				}
			}
		}
	}

	state_valid_trigger = {
		scope:input = {
			owner = root
			is_coastal = yes
		}
	}

	requirement_to_maintain = {
		trigger = {
			exists = scope:article.input_state
		}
	}

	non_fulfillment = {
		consequences = freeze
		conditions = {
			monthly = {
				scope:article = {
					source_country = {
						has_law = law_type:law_isolationism
					}
				}
			}
		}
	}

	on_entry_into_force = {
		scope:article_options = {
			if = {
				limit = { exists = input_state }
				input_state = {
					add_modifier = {
						name = free_port_state_modifier
					}
					set_variable = has_free_port
					te_update_free_port_tariff = yes
				}
			}
		}
	}

	on_break = {
		scope:article_options.input_state ?= {
			remove_modifier = free_port_state_modifier
			remove_modifier = free_port_tariff_import_modifier
			remove_modifier = free_port_tariff_export_modifier
			remove_variable = has_free_port
		}
	}

	on_withdrawal = {
		scope:article_options.input_state ?= {
			remove_modifier = free_port_state_modifier
			remove_modifier = free_port_tariff_import_modifier
			remove_modifier = free_port_tariff_export_modifier
			remove_variable = has_free_port
		}
	}

	ai = {
		treaty_categories = { other }
		article_ai_usage = { request offer }

		evaluation_chance = {
			value = 0
			if = {
				limit = {
					OR = {
						has_law = law_type:law_laissez_faire
						has_law = law_type:law_free_trade
					}
				}
				add = 0.05
			}
			if = {
				limit = {
					has_attitude = { who = scope:other_country attitude = domineering }
				}
				add = 0.1
			}
			if = {
				limit = {
					country_rank >= rank_value:great_power
				}
				add = 0.05
			}
			# Friendly nations don't impose this on friends
			if = {
				limit = {
					OR = {
						has_attitude = { who = scope:other_country attitude = friendly }
						has_attitude = { who = scope:other_country attitude = cooperative }
					}
				}
				add = -0.15
			}
		}

		inherent_accept_score = {
			value = 0

			# Source (losing tariff sovereignty) is reluctant
			if = {
				limit = {
					scope:source_country = this
				}
				add = {
					desc = "DIPLOMATIC_ACCEPTANCE_BASE"
					value = -40
				}
				# Less reluctant if already free trade
				if = {
					limit = {
						has_law = law_type:law_free_trade
					}
					add = {
						desc = "AI_FREE_TRADE_POLICY"
						value = 30
					}
				}
				# More reluctant if mercantilist
				if = {
					limit = {
						has_law = law_type:law_mercantilism
					}
					add = {
						desc = "AI_MERCANTILIST_POLICY"
						value = -20
					}
				}
				# Somewhat reluctant if protectionist
				if = {
					limit = {
						has_law = law_type:law_protectionism
					}
					add = {
						desc = "AI_PROTECTIONIST_POLICY"
						value = -10
					}
				}
				# Hostile to the demander: extra resistance
				if = {
					limit = {
						OR = {
							has_attitude = { who = scope:target_country attitude = hostile }
							has_attitude = { who = scope:target_country attitude = belligerent }
							has_attitude = { who = scope:target_country attitude = antagonistic }
						}
					}
					add = {
						desc = "AI_HOSTILE_ATTITUDE"
						value = -20
					}
				}
			}
			# Target (benefiting from open port)
			else = {
				add = {
					desc = "DIPLOMATIC_ACCEPTANCE_BASE"
					value = -15
				}
				# Mutual trade value makes this more worthwhile (vanilla GDP-normalized pattern)
				add = {
					desc = "AI_MUTUAL_TRADE_VALUE"
					value = "mutual_trade_value_with_country(scope:source_country)"
					multiply = 50000
					divide = {
						value = gdp
						max = scope:source_country.gdp
						min = 1000000
					}
					max = 15
				}
				# Free trade policy: aligned with open markets
				if = {
					limit = {
						has_law = law_type:law_free_trade
					}
					add = {
						desc = "AI_FREE_TRADE_POLICY"
						value = 10
					}
				}
				# Mercantilists want OTHER countries' ports open
				if = {
					limit = {
						has_law = law_type:law_mercantilism
					}
					add = {
						desc = "AI_MERCANTILIST_IMPERIALIST"
						value = 10
					}
				}
				# Domineering: wants to impose
				if = {
					limit = {
						has_attitude = { who = scope:source_country attitude = domineering }
					}
					add = {
						desc = "AI_DOMINEERING_ATTITUDE"
						value = 10
					}
				}
				# Friendly nations don't impose hostile articles
				if = {
					limit = {
						OR = {
							has_attitude = { who = scope:source_country attitude = friendly }
							has_attitude = { who = scope:source_country attitude = cooperative }
							has_attitude = { who = scope:source_country attitude = conciliatory }
						}
					}
					add = {
						desc = "AI_FRIENDLY_ATTITUDE"
						value = -20
					}
				}
			}
		}
	}

	wargoal = {
		execution_priority = 40
		contestion_type = control_target_state

		infamy = {
			value = 5
			scope:target_country = {
				multiply = {
					value = total_population
					divide = 20000000
					add = 1
					max = 2.0
				}
			}
		}
	}
}
