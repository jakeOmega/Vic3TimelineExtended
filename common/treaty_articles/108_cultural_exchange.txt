# =================================================================
# TREATY ARTICLE: cultural_exchange_program
# Mutual agreement boosting cultural acceptance between both
# countries. Historical: Cultural exchange during detente,
# Fulbright program, Erasmus, artistic/cultural diplomacy.
# =================================================================
cultural_exchange_program = {
	icon = "gfx/interface/icons/law_icons/freedom_of_conscience.dds"
	kind = mutual
	cost = 100

	relations_progress_per_day = 0.5
	relations_improvement_max = 30

	flags = {
		can_be_renegotiated
		friendly
	}

	usage_limit = once_per_treaty

	mutual_exclusions = {
		cultural_exchange_program
	}

	unlocked_by_technologies = {
		pan-nationalism
	}

	mutual_modifier = {
		state_yearly_cultural_acceptance_add = 1
		country_prestige_mult = 0.02
		# Integration with cultural hegemony system — boosts soft power
		country_cultural_pull_add = 1
		# Cultural openness creates ideological vulnerability
		country_covert_defense_ideological_add = -1
	}

	visible = {
		always = yes
	}

	possible = {
		has_diplomatic_relevance = scope:other_country
		NOT = { has_war_with = scope:other_country }
		NOT = { is_country_type = decentralized }
		scope:other_country = {
			NOT = { is_country_type = decentralized }
		}
	}

	can_ratify = {
		custom_tooltip = {
			text = DUPLICATE_ARTICLE_SAME_INPUTS
			NOT = {
				scope:first_country = {
					any_scope_treaty = {
						binds = scope:second_country
						hidden_trigger = {
							OR = {
								scope:treaty = {
									is_renegotiation = no
								}
								scope:treaty.amended_treaty != this
							}
						}
						any_scope_article = {
							has_type = cultural_exchange_program
						}
					}
				}
			}
		}
	}

	ai = {
		treaty_categories = { other }
		article_ai_usage = { offer request }

		evaluation_chance = {
			value = 0.08
			if = {
				limit = {
					has_attitude = { who = scope:other_country attitude = cooperative }
				}
				add = 0.12
			}
			if = {
				limit = {
					has_attitude = { who = scope:other_country attitude = friendly }
				}
				add = 0.08
			}
			if = {
				limit = {
					has_attitude = { who = scope:other_country attitude = conciliatory }
				}
				add = 0.04
			}
			# Don't consider for hostile nations
			if = {
				limit = {
					OR = {
						has_attitude = { who = scope:other_country attitude = antagonistic }
						has_attitude = { who = scope:other_country attitude = hostile }
						has_attitude = { who = scope:other_country attitude = belligerent }
					}
				}
				add = -1.0
			}
			# Ethnostates don't do cultural exchange
			if = {
				limit = {
					has_law = law_type:law_ethnostate
				}
				add = -1.0
			}
		}

		inherent_accept_score = {
			value = -10

			# Ethnostate: hard refusal
			if = {
				limit = {
					has_law = law_type:law_ethnostate
				}
				add = {
					desc = "AI_ETHNOSTATE"
					value = -1000
				}
			}
			# Hostile nations refuse cultural exchange
			if = {
				limit = {
					OR = {
						has_attitude = { who = scope:other_country attitude = antagonistic }
						has_attitude = { who = scope:other_country attitude = hostile }
						has_attitude = { who = scope:other_country attitude = belligerent }
					}
				}
				add = {
					desc = "AI_HOSTILE_ATTITUDE"
					value = -1000
				}
			}
			# Countries sharing heritage group have less need for exchange
			if = {
				limit = {
					any_primary_culture = {
						shares_heritage_trait_group_with_any_primary_culture = scope:other_country
					}
				}
				add = {
					desc = "AI_SHARED_HERITAGE"
					value = -25
				}
			}
			# National supremacy strongly dislikes it
			if = {
				limit = {
					has_law = law_type:law_national_supremacy
				}
				add = {
					desc = "AI_NATIONAL_SUPREMACY"
					value = -30
				}
			}
			# Multicultural nations value this
			if = {
				limit = {
					has_law = law_type:law_multicultural
				}
				add = {
					desc = "AI_MULTICULTURAL_LAWS"
					value = 15
				}
			}
			# Cooperative attitude
			if = {
				limit = {
					has_attitude = { who = scope:other_country attitude = cooperative }
				}
				add = {
					desc = "AI_COOPERATIVE_ATTITUDE"
					value = 15
				}
			}
			# Friendly attitude
			if = {
				limit = {
					has_attitude = { who = scope:other_country attitude = friendly }
				}
				add = {
					desc = "AI_FRIENDLY_ATTITUDE"
					value = 10
				}
			}
			# Same power bloc
			if = {
				limit = {
					is_in_same_power_bloc = scope:other_country
				}
				add = {
					desc = "AI_SAME_POWER_BLOC"
					value = 10
				}
			}
		}
	}
}
