# =================================================================
# TREATY ARTICLE: enforce_privatization
# Forces a country to privatize state-owned enterprises.
# Historical: IMF structural adjustment, post-Soviet privatization,
# Washington Consensus conditionality.
# Uses the engine's built-in country_force_privatization_bool.
# =================================================================
enforce_privatization = {
	icon = "gfx/interface/icons/law_icons/interventionism.dds"
	kind = directed
	cost = 150

	flags = {
		can_be_enforced
		hostile
	}

	usage_limit = once_per_side

	maintenance_paid_by = target_country

	unlocked_by_technologies = {
		stock_exchange
	}

	source_modifier = {
		country_force_privatization_bool = yes
		country_authority_add = -200
	}

	target_modifier = {
		country_prestige_mult = 0.05
	}

	visible = {
		always = yes
	}

	possible = {
		has_diplomatic_relevance = scope:other_country
		NOT = { has_war_with = scope:other_country }
		NOT = { is_country_type = decentralized }
		scope:other_country = {
			NOT = { is_country_type = decentralized }
			NOR = {
				has_law = law_type:law_laissez_faire
				has_law = law_type:law_command_economy
			}
		}
	}

	can_ratify = {
		custom_tooltip = {
			text = DUPLICATE_ARTICLE_SAME_INPUTS
			NOT = {
				scope:source_country = {
					any_scope_treaty = {
						binds = scope:target_country
						hidden_trigger = {
							OR = {
								scope:treaty = {
									is_renegotiation = no
								}
								scope:treaty.amended_treaty != this
							}
						}
						any_scope_article = {
							has_type = enforce_privatization
							source_country = scope:source_country
						}
					}
				}
			}
		}
		scope:source_country = {
			NOR = {
				has_law = law_type:law_laissez_faire
				has_law = law_type:law_command_economy
			}
		}
	}

	ai = {
		treaty_categories = { other }
		article_ai_usage = { request }

		evaluation_chance = {
			value = 0.05
			# Non-capitalist countries have no interest in demanding privatization
			if = {
				limit = {
					NOR = {
						has_law = law_type:law_laissez_faire
						has_law = law_type:law_interventionism
					}
				}
				add = -1.0
			}
			if = {
				limit = {
					OR = {
						has_law = law_type:law_laissez_faire
						has_law = law_type:law_interventionism
					}
				}
				add = 0.1
			}
			if = {
				limit = {
					ig:ig_industrialists = {
						is_in_government = yes
					}
				}
				add = 0.1
			}
			if = {
				limit = {
					has_attitude = { who = scope:other_country attitude = domineering }
				}
				add = 0.1
			}
		}

		inherent_accept_score = {
			value = 0

			# Source (being forced to privatize) is very reluctant
			if = {
				limit = {
					scope:source_country = this
				}
				add = {
					desc = "DIPLOMATIC_ACCEPTANCE_BASE"
					value = -70
				}
				# Command economy countries refuse
				if = {
					limit = {
						has_law = law_type:law_command_economy
					}
					add = {
						desc = "AI_COMMAND_ECONOMY"
						value = -100
					}
				}
				# Cooperative ownership countries strongly resist
				if = {
					limit = {
						has_law = law_type:law_cooperative_ownership
					}
					add = {
						desc = "AI_COOPERATIVE_ECONOMY"
						value = -50
					}
				}
				# Already trending toward privatization
				if = {
					limit = {
						has_law = law_type:law_interventionism
					}
					add = {
						desc = "AI_INTERVENTIONIST_ECONOMY"
						value = 20
					}
				}
				# Hostile to the one demanding: extra refusal
				if = {
					limit = {
						OR = {
							has_attitude = { who = scope:target_country attitude = hostile }
							has_attitude = { who = scope:target_country attitude = belligerent }
						}
					}
					add = {
						desc = "AI_HOSTILE_ATTITUDE"
						value = -30
					}
				}
			}
			# Target (demanding privatization)
			else = {
				# Non-capitalist countries have no interest in demanding privatization
				if = {
					limit = {
						NOR = {
							has_law = law_type:law_laissez_faire
							has_law = law_type:law_interventionism
						}
					}
					add = {
						desc = "AI_NOT_CAPITALIST_ECONOMY"
						value = -1000
					}
				}
				else = {
					add = {
						desc = "DIPLOMATIC_ACCEPTANCE_BASE"
						value = 5
					}
					# Laissez faire countries are ideologically motivated
					if = {
						limit = {
							has_law = law_type:law_laissez_faire
						}
						add = {
							desc = "AI_LAISSEZ_FAIRE_ECONOMY"
							value = 15
						}
					}
					# Source has cooperative ownership: more to privatize
					if = {
						limit = {
							scope:source_country = {
								has_law = law_type:law_cooperative_ownership
							}
						}
						add = {
							desc = "AI_SOURCE_COOPERATIVE_ECONOMY"
							value = 10
						}
					}
					# Domineering
					if = {
						limit = {
							has_attitude = { who = scope:source_country attitude = domineering }
						}
						add = {
							desc = "AI_DOMINEERING_ATTITUDE"
							value = 10
						}
					}
					# Industrialists want this
					if = {
						limit = {
							ig:ig_industrialists = {
								is_in_government = yes
							}
						}
						add = {
							desc = "AI_INDUSTRIALISTS_IN_GOVERNMENT"
							value = 10
						}
					}
					# Friendly nations don't impose
					if = {
						limit = {
							OR = {
								has_attitude = { who = scope:source_country attitude = friendly }
								has_attitude = { who = scope:source_country attitude = cooperative }
							}
						}
						add = {
							desc = "AI_FRIENDLY_ATTITUDE"
							value = -20
						}
					}
				}
			}
		}
	}

	wargoal = {
		execution_priority = 60
		contestion_type = control_target_country_capital

		infamy = {
			value = 10
			scope:target_country = {
				multiply = {
					value = gdp
					divide = 10000000
					add = 1
					max = 2.5
				}
			}
		}
	}
}
