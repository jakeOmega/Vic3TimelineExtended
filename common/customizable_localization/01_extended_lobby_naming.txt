############################
# Custom lobby name flavoring for extended lobby types
# Resolves $CUSTOM_NAME$ in lobby_<type>_name localization
############################

flavored_name_lobby_colonial = {
	type = political_lobby
	random_valid = yes

	text = {
		trigger = { has_variable = lobby_custom_name_var_3 }
		localization_key = custom_lobby_society
	}
	text = {
		trigger = { has_variable = lobby_custom_name_var_21 }
		localization_key = custom_lobby_committee
	}
	text = {
		trigger = { has_variable = lobby_custom_name_var_22 }
		localization_key = custom_lobby_council
	}
	text = {
		trigger = {
			NOR = {
				has_variable = lobby_custom_name_var_3
				has_variable = lobby_custom_name_var_21
				has_variable = lobby_custom_name_var_22
			}
		}
		localization_key = custom_lobby_lobby
	}
}
