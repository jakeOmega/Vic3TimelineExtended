sr_grain_mode_text = {
	type = country
	random_valid = no

	text = {
		trigger = {
			var:sr_grain_rate > 0
			var:sr_grain_stored < sr_grain_capacity
		}
		localization_key = sr_mode_storing
	}
	text = {
		trigger = {
			var:sr_grain_rate < 0
			var:sr_grain_stored > 0
		}
		localization_key = sr_mode_withdrawing
	}
	text = {
		trigger = { always = yes }
		localization_key = sr_mode_idle
	}
}
