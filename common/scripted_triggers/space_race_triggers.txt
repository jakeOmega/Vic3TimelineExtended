# Space Race Scripted Triggers

sr_has_space_program = {
	any_scope_building = {
		is_building_type = building_space_program
	}
}

sr_is_pursuing_milestone = {
	has_variable = sr_active_milestone
}

# AI participation gate: only the top ~3 countries by prestige should pursue space race
# Players can always participate regardless
sr_ai_should_participate = {
	OR = {
		is_player = yes
		global_country_ranking <= 3
	}
}

sr_can_start_space_race = {
	has_technology_researched = rocketry
	OR = {
		country_rank = rank_value:great_power
		country_rank = rank_value:major_power
	}
	NOT = { has_variable = sr_active_milestone }
}

sr_has_failure_cooldown = {
	has_variable = sr_failure_cooldown
	var:sr_failure_cooldown > 0
}

has_space_company = {
	OR = {
		has_company = company_type:company_spacex
		has_company = company_type:company_lockheed_martin
		has_company = company_type:company_roscosmos
	}
}
