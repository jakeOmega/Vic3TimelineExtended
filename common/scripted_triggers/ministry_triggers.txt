# ======================================================================
#  Ministry Triggers
#  Reusable triggers for ministry law enactment checks and
#  institution investment level gates.
# ======================================================================

# True if the country is enacting ANY ministry creation law
is_enacting_any_ministry_law = {
	OR = {
		is_enacting_law = law_type:law_ministry_of_war
		is_enacting_law = law_type:law_ministry_of_foreign_affairs
		is_enacting_law = law_type:law_ministry_of_commerce
		is_enacting_law = law_type:law_ministry_of_culture
		is_enacting_law = law_type:law_pro_capital_ministry_of_labor
		is_enacting_law = law_type:law_pro_labor_ministry_of_labor
		is_enacting_law = law_type:law_ministry_of_the_environment
		is_enacting_law = law_type:law_ministry_of_intelligence_and_security
		is_enacting_law = law_type:law_ministry_of_refugee_affairs
		is_enacting_law = law_type:law_ministry_of_propaganda
		is_enacting_law = law_type:law_ministry_of_science
		is_enacting_law = law_type:law_ministry_of_thought_control
		is_enacting_law = law_type:law_ministry_of_consumer_protection
		is_enacting_law = law_type:law_ministry_of_urban_planning
		is_enacting_law = law_type:law_ministry_of_religion
		is_enacting_law = law_type:law_ministry_of_international_aid
	}
}

# True if the country is enacting ANY ministry dissolution law
is_dissolving_any_ministry_law = {
	OR = {
		is_enacting_law = law_type:law_no_ministry_of_war
		is_enacting_law = law_type:law_no_ministry_of_foreign_affairs
		is_enacting_law = law_type:law_no_ministry_of_commerce
		is_enacting_law = law_type:law_no_ministry_of_culture
		is_enacting_law = law_type:law_no_ministry_of_labor
		is_enacting_law = law_type:law_no_ministry_of_the_environment
		is_enacting_law = law_type:law_no_ministry_of_intelligence_and_security
		is_enacting_law = law_type:law_no_ministry_of_refugee_affairs
		is_enacting_law = law_type:law_no_ministry_of_propaganda
		is_enacting_law = law_type:law_no_ministry_of_science
		is_enacting_law = law_type:law_no_ministry_of_thought_control
		is_enacting_law = law_type:law_no_ministry_of_consumer_protection
		is_enacting_law = law_type:law_no_ministry_of_urban_planning
		is_enacting_law = law_type:law_no_ministry_of_religion
		is_enacting_law = law_type:law_no_ministry_of_international_aid
	}
}
