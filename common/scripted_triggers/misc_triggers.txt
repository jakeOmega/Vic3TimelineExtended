# True if the activated law (law scope) is a radical law that triggers backlash
is_radical_law_activated = {
	OR = {
		law_type = law_type:law_communal_child_rearing
		law_type = law_type:law_state_atheism
		law_type = law_type:law_cooperative_ownership
		law_type = law_type:law_command_economy
		law_type = law_type:law_collectivized_agriculture
	}
}

# True if the country (country scope) has a governance type that exempts it
# from radical law backlash: revolutionary, authoritarian, or anarchist regimes
is_exempt_from_radical_backlash = {
	OR = {
		has_law = law_type:law_council_republic
		has_law = law_type:law_anarchy
		has_law = law_type:law_single_party_state
	}
}
