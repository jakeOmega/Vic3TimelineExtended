# True if the country has any language reform modifier active
has_any_langreform_modifier = {
	OR = {
		has_modifier = langreform_revival
		has_modifier = langreform_constructed
		has_modifier = langreform_simplification
		has_modifier = langreform_powerful_academy
		has_modifier = langreform_advisory_panel
		has_modifier = langreform_minority_protection
		has_modifier = langreform_national_focus
	}
}

# True if the country has adopted any custom religion as state religion
country_has_any_custom_religion = {
	OR = {
		country_has_state_religion = rel:custom_religion_christian
		country_has_state_religion = rel:custom_religion_muslim
		country_has_state_religion = rel:custom_religion_jewish
		country_has_state_religion = rel:custom_religion_eastern
		country_has_state_religion = rel:custom_religion_animist
		country_has_state_religion = rel:custom_religion_buddhist
		country_has_state_religion = rel:custom_religion_custom
	}
}
