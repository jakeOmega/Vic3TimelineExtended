############################
# COLONIAL / ANTI-COLONIAL LOBBIES
# Form in overlord countries, targeting their subjects
# Represents the metropolitan political debate over empire
############################

lobby_colonial = {
	category = foreign_anti_country

	texture = "gfx/interface/icons/political_lobby_icons/lobby_anti.dds"

	can_create = {
		has_dlc_feature = lobbies
		NOT = { is_country_type = decentralized }
		is_revolutionary = no
		scope:target_country = {
			is_revolutionary = no
			is_subject_of = root
		}
	}

	on_created = {
		save_temporary_scope_as = lobby_country
		scope:political_lobby ?= {
			change_appeasement = {
				amount = 2
				factor = appeasement_diplomatic_status_quo
			}
			extended_initialize_lobby_custom_name = yes
		}
	}

	# Target must remain a subject; if independent - swap to anti-country
	requirement_to_maintain = {
		trigger = { scope:target_country = { is_subject_of = root } }
		swap_type_on_failed = lobby_anti_country
		on_failed = {
			scope:political_lobby = { initialize_lobby_custom_name = yes }
		}
	}

	# Disband if all members are marginal
	requirement_to_maintain = {
		trigger = {
			OR = {
				NOT = { exists = scope:political_lobby }
				scope:political_lobby ?= {
					any_lobby_member = { ig_counts_as_marginal = no }
				}
			}
		}
		on_failed = { }
	}

	appeasement_factors_pro = {
		appeasement_autonomy_decreased
		appeasement_diplomatic_demand_made
		appeasement_colonial_crackdown
		appeasement_special_events_positive
	}

	appeasement_factors_anti = {
		appeasement_autonomy_increased
		appeasement_gained_independence
		appeasement_independence_supported
		appeasement_colonial_concession
		appeasement_special_events_negative
	}

	available_for_interest_group = {
		NOT = {
			any_political_lobby = {
				target = scope:target_country
			}
		}
	}

	join_weight = {
		### Stickiness factor if already in lobby
		if = {
			limit = {
				exists = scope:political_lobby
				scope:political_lobby = { any_lobby_member = { this = root } }
			}
			add = {
				value = 50
				desc = "member_of_lobby"
			}
		}

		### Formation reasons
		add = {
			if = {
				limit = { is_lobby_formation_reason_ideology = yes }
				leader ?= {
					if = {
						limit = {
							OR = {
								has_ideology = ideology:ideology_jingoist_leader
								has_ideology = ideology:ideology_fascist
							}
						}
						add = 25
					}
					else_if = {
						limit = { has_ideology = ideology:ideology_pacifist }
						add = -25
					}
				}
				if = {
					limit = {
						OR = {
							has_ideology = ideology:ideology_jingoist
							has_ideology = ideology:ideology_patriotic
							has_ideology = ideology:ideology_military_absolutist
						}
					}
					add = 25
				}
				if = {
					limit = { has_ideology = ideology:ideology_liberal }
					add = -25
				}
			}
			desc = "formation_reason_ideology_desc"
		}

		add = {
			if = {
				limit = { is_lobby_formation_reason_economy = yes }
				leader ?= {
					if = {
						limit = {
							OR = {
								has_ideology = ideology:ideology_market_liberal
								has_ideology = ideology:ideology_protectionist
							}
						}
						add = 15
					}
				}
				if = {
					limit = { has_ideology = ideology:ideology_laissez_faire }
					add = 15
				}
			}
			desc = "formation_reason_economy_desc"
		}

		add = {
			if = {
				limit = { is_lobby_formation_reason_aggression = yes }
				leader ?= {
					if = {
						limit = {
							OR = {
								has_ideology = ideology:ideology_jingoist_leader
								has_ideology = ideology:ideology_fascist
							}
						}
						add = 25
					}
					else_if = {
						limit = { has_ideology = ideology:ideology_pacifist }
						add = -25
					}
				}
				if = {
					limit = {
						OR = {
							has_ideology = ideology:ideology_jingoist
							has_ideology = ideology:ideology_patriotic
						}
					}
					add = 25
				}
			}
			desc = "formation_reason_aggression_desc"
		}

		### IG type tendencies
		# Military: colonial garrisons, strategic bases, prestige
		if = {
			limit = { is_interest_group_type = ig_armed_forces }
			add = {
				value = 20
				desc = "military_colonial_interest"
			}
		}
		# Landowners: plantations, resource extraction
		if = {
			limit = { is_interest_group_type = ig_landowners }
			add = {
				value = 15
				desc = "colonial_economic_interest"
			}
		}
		# Industrialists: colonial markets, raw materials
		if = {
			limit = { is_interest_group_type = ig_industrialists }
			add = {
				value = 10
				desc = "colonial_resource_interest"
			}
		}
		# Devout: missionary activity
		if = {
			limit = { is_interest_group_type = ig_devout }
			add = {
				value = 10
				desc = "missionary_interest"
			}
		}
		# Petty bourgeoisie: colonial trade
		if = {
			limit = { is_interest_group_type = ig_petty_bourgeoisie }
			add = {
				value = 5
				desc = "colonial_trade_interest"
			}
		}
		# Intelligentsia: anti-imperialist
		if = {
			limit = { is_interest_group_type = ig_intelligentsia }
			add = {
				value = -15
				desc = "anti_imperialist_sentiment"
			}
		}
		# Trade unions: anti-exploitation
		if = {
			limit = { is_interest_group_type = ig_trade_unions }
			add = {
				value = -10
				desc = "anti_exploitation_sentiment"
			}
		}
		# Rural folk: sons serving overseas
		if = {
			limit = { is_interest_group_type = ig_rural_folk }
			add = {
				value = -5
				desc = "anti_conscription_sentiment"
			}
		}

		### Ideology bonuses (cumulative with IG type)
		if = {
			limit = {
				OR = {
					has_ideology = ideology:ideology_jingoist
					has_ideology = ideology:ideology_patriotic
					has_ideology = ideology:ideology_military_absolutist
				}
			}
			add = {
				value = 15
				desc = "imperialist_ideology"
			}
		}
		if = {
			limit = { has_ideology = ideology:ideology_anti_colonialist }
			add = {
				value = -50
				desc = "anti_colonial_ideology"
			}
		}
		if = {
			limit = { has_ideology = ideology:ideology_liberal }
			add = {
				value = -10
				desc = "liberal_anti_colonialism"
			}
		}

		### Leader ideologies
		leader ?= {
			if = {
				limit = {
					OR = {
						has_ideology = ideology:ideology_jingoist_leader
						has_ideology = ideology:ideology_fascist
					}
				}
				add = {
					value = 15
					desc = "jingoist_leader"
				}
			}
			if = {
				limit = { has_ideology = ideology:ideology_pacifist }
				add = {
					value = -15
					desc = "pacifist_leader"
				}
			}
		}
	}
}

lobby_fossil_fuel = {
	category = foreign_anti_country

	texture = "gfx/interface/icons/political_lobby_icons/lobby_anti.dds"

	can_create = {
		has_dlc_feature = lobbies
		NOT = { is_country_type = decentralized }

		}
	}

	join_weight = {
		### Stickiness factor if already in lobby
		if = {
			limit = {
				exists = scope:political_lobby
				scope:political_lobby = { any_lobby_member = { this = root } }
			}
			add = {
				value = 50
				desc = "member_of_lobby"
			}
		}

		### Formation reasons
		add = {
			if = {
				limit = { is_lobby_formation_reason_economy = yes }
				leader ?= {
					if = {
						limit = { has_ideology = ideology:ideology_positivist }
						add = 20
					}
					else_if = {
						limit = { has_ideology = ideology:ideology_luddite }
						add = -20
					}
				}
				if = {
					limit = {
						OR = {
							has_ideology = ideology:ideology_meritocratic
							has_ideology = ideology:ideology_modernizer
						}
					}
					add = 15
				}
			}
			desc = "formation_reason_economy_desc"
		}

		add = {
			if = {
				limit = { is_lobby_formation_reason_technology = yes }
				leader ?= {
					if = {
						limit = { has_ideology = ideology:ideology_positivist }
						add = 25
					}
					else_if = {
						limit = { has_ideology = ideology:ideology_luddite }
						add = -25
					}
				}
				if = {
					limit = {
						OR = {
							has_ideology = ideology:ideology_meritocratic
							has_ideology = ideology:ideology_modernizer
						}
					}
					add = 20
				}
			}
			desc = "formation_reason_technology_desc"
		}

		### IG type tendencies
		# Intelligentsia: scientific consensus
		if = {
			limit = { is_interest_group_type = ig_intelligentsia }
			add = {
				value = 25
				desc = "scientific_consensus"
			}
		}
		# Trade unions: green jobs, worker safety
		if = {
			limit = { is_interest_group_type = ig_trade_unions }
			add = {
				value = 10
				desc = "green_jobs_interest"
			}
		}
		# Rural folk: affected by climate degradation
		if = {
			limit = { is_interest_group_type = ig_rural_folk }
			add = {
				value = 5
				desc = "climate_impact_concern"
			}
		}
		# Industrialists: regulatory burden
		if = {
			limit = { is_interest_group_type = ig_industrialists }
			add = {
				value = -20
				desc = "regulatory_burden"
			}
		}
		# Petty bourgeoisie: compliance costs
		if = {
			limit = { is_interest_group_type = ig_petty_bourgeoisie }
			add = {
				value = -10
				desc = "compliance_costs"
			}
		}
		# Landowners: resource extraction threatened
		if = {
			limit = { is_interest_group_type = ig_landowners }
			add = {
				value = -10
				desc = "resource_interests_threatened"
			}
		}

		### Ideology bonuses
		if = {
			limit = {
				OR = {
					has_ideology = ideology:ideology_meritocratic
					has_ideology = ideology:ideology_modernizer
				}
			}
			add = {
				value = 10
				desc = "modernizer_green_support"
			}
		}

		### Environmental law checks
		# Environmentalist leaders gravitate to this lobby
		leader ?= {
			if = {
				limit = { has_ideology = ideology:ideology_environmentalists }
				add = {
					value = 50
					desc = "environmentalist_leader"
				}
			}
			if = {
				limit = { has_ideology = ideology:ideology_positivist }
				add = {
					value = 15
					desc = "scientific_leader"
				}
			}
		}

		# Country with environmental ministry: env lobby is the supporter
		if = {
			limit = {
				owner = { has_law = law_type:law_ministry_of_the_environment }
			}
			add = {
				value = 10
				desc = "supporting_environmental_policy"
			}
		}
	}
}
