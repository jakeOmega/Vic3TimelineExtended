# Formation rules for the formable countries added by this mod.
#
# All major formations: candidates compete via leadership/unification
# diplomatic plays. AFU/BHA/NUS/EUN/UNA/UNE/ZHO/INM/EAF/DAR.
#
# The seven pan-region political unions (AFU/EUN/UNA/UNE/INM/EAF/DAR) are
# `is_major_formation = yes` so they bypass vanilla's downward-only rank
# gate — thematically they exist to be formed by countries already at or
# above their tier (a hegemony forming the EU, etc.), which a minor
# formation makes impossible.
#
# To compensate for losing the rank gate, each carries sophisticated
# world-state gating in `possible`: external rank pressure (a top-3 GP
# exists outside the region), bloc / treaty alignment with every major
# regional constituent, and no rivalry / active war with any of them.
# UNE is gated on a power-bloc supermajority instead (no external bloc
# applies for a global unification). All gating triggers live in
# common/scripted_triggers/formable_gating_triggers.txt.
#
# Mirror the load-bearing conditions in `ai_will_do` so the AI doesn't
# pursue candidacy under conditions that would just block the play.
#
# All require the era_6 `decolonization` mod tech (era_6.txt:785).
# UNE additionally gates on era-11 `quantum_communications` and era-12
# `space_colonization`.
#
# India and Indonesia each have TWO formations:
#   - Vanilla BHT / IDN (minor, state-list, pan-nationalism gate) — vanilla
#     entries are gated behind NOT={has_technology_researched=decolonization}
#     via small REPLACE: directives below, so they remain available pre-
#     decolonization but step aside in the post-decolonization era.
#   - Mod BHA (Bharat) / NUS (Nusantara) (major, use_culture_states,
#     leadership/unification plays) — the late-game flavor.
# BHA/NUS coexistence: their unification plays guard NOT={exists=c:BHT/IDN},
# so if a player formed vanilla BHT/IDN early-game it blocks the major
# formation. ZHO is a NEW tag (Greater China); vanilla CHI is left intact
# for its early-game minor-formation role (Qing remnants / Taiping etc.).

# === Pan-region political unions ===

AFU = {
	geographic_region = geographic_region_united_africa

	is_major_formation = yes

	unification_play = dp_unify_african_union
	leadership_play = dp_leadership_african_union

	required_states_fraction = 0.5

	ai_will_do = {
		has_technology_researched = decolonization
		has_technology_researched = pan-nationalism
		afu_formation_external_pressure = yes
		afu_formation_bloc_aligned = yes
		afu_formation_no_internal_rivalry = yes
	}

	possible = {
		has_technology_researched = decolonization
		has_technology_researched = pan-nationalism
		afu_formation_external_pressure = yes
		afu_formation_bloc_aligned = yes
		afu_formation_no_internal_rivalry = yes
	}

	max_num_formation_candidates = 3
	# Candidacy is engine-gated on "has a primary culture in AFU's culture list"
	# alone — which leaks across regions (e.g. a Misri-primary Arabia, capital in
	# Hedjaz, qualifies via misri ∈ AFU.cultures). Bind candidacy to an
	# African-capital power so the African Union is led from Africa.
	can_be_formation_candidate = {
		is_major_african_power = yes
	}
}

# Pan-Islamic Caliphate. Like the other pan-region unions it carries the
# external-pressure / bloc-aligned / no-internal-rivalry triple (DAR-scoped,
# in formable_gating_triggers.txt). What makes it distinct is the religion
# gate: only a country under an Islamic state religion (sunni/shiite/ibadi)
# may proclaim the Caliphate. That is a requirement on the FORMER, so it sits
# in `possible` — NOT a `can_be_unification_target` guard (which would instead
# decide which states get swept up, the wrong axis for "who may form it").
#
# The Islamic-state-religion check alone is insufficient: a country keeps its
# (culture/ruler-derived) state-religion value even under secular church-state
# law, so e.g. a Total Separation state still reads as "Sunni". A Caliphate
# must actually establish OR at least tolerate that religion, so we also forbid
# the two genuinely secular church-state laws via NOR: law_total_separation and
# law_state_atheism. The four permitted laws — state_religion, people_of_the_book,
# millet_system, freedom_of_conscience — all describe an officially-Islamic state
# (freedom_of_conscience = officially Sunni but at ease with Shia / People of the
# Book, the historical millet/dhimmi norm).
DAR = {
	geographic_region = geographic_region_dar_al_islam

	is_major_formation = yes

	unification_play = dp_unify_dar_al_islam
	leadership_play = dp_leadership_dar_al_islam

	required_states_fraction = 0.5

	ai_will_do = {
		has_technology_researched = decolonization
		has_technology_researched = pan-nationalism
		dar_formation_external_pressure = yes
		dar_formation_bloc_aligned = yes
		dar_formation_no_internal_rivalry = yes
		OR = {
			country_has_state_religion = rel:sunni
			country_has_state_religion = rel:shiite
			country_has_state_religion = rel:ibadi
		}
		NOR = {
			has_law_or_variant = law_type:law_total_separation
			has_law_or_variant = law_type:law_state_atheism
		}
	}

	possible = {
		has_technology_researched = decolonization
		has_technology_researched = pan-nationalism
		dar_formation_external_pressure = yes
		dar_formation_bloc_aligned = yes
		dar_formation_no_internal_rivalry = yes
		OR = {
			country_has_state_religion = rel:sunni
			country_has_state_religion = rel:shiite
			country_has_state_religion = rel:ibadi
		}
		NOR = {
			has_law_or_variant = law_type:law_total_separation
			has_law_or_variant = law_type:law_state_atheism
		}
	}

	max_num_formation_candidates = 3
	# Bind candidacy to a power whose capital is in the Islamic world (see AFU) —
	# otherwise an Indian (bengali/panjabi/sindi/baluchi) or Malay candidate leaks
	# in via shared cultures, even though it could never lead the Caliphate.
	can_be_formation_candidate = {
		is_major_dar_al_islam_power = yes
	}
}

# === Mod replacements for vanilla minor formations ===

# India forms from its modern core only. Pre-1.x this used `use_culture_states`,
# which demanded ~75% of every homeland state of all 23 Indian cultures — i.e.
# control of the whole subcontinent from Baluchistan to Bangladesh before India
# could form at all. The explicit core list below scopes the *requirement* to
# the modern-India interior; the Pakistan / Bangladesh / Nepal-Bhutan / Ceylon
# periphery is granted as claims by formation_te.bharat instead (events/
# te_formable_formation_events.txt), and Muslim-ruled in-core holdouts (chiefly
# Hyderabad) are handled by the religion guard + Operation Polo chain below.
BHA = {
	states = {
		STATE_ASSAM
		STATE_AWADH
		STATE_BIHAR
		STATE_BOMBAY
		STATE_MALWA
		STATE_CIRCARS
		STATE_DELHI
		STATE_GUJARAT
		STATE_HYDERABAD
		STATE_MADRAS
		STATE_MYSORE
		STATE_CENTRAL_PROVINCES
		STATE_WEST_BENGAL
		STATE_ORISSA
		STATE_RAJPUTANA
		STATE_TRAVANCORE
		STATE_KURNOOL
		STATE_BUNDELKHAND
		STATE_AGRA
	}

	is_major_formation = yes

	unification_play = dp_unify_india
	leadership_play = dp_leadership_india

	required_states_fraction = 0.65

	ai_will_do = { has_technology_researched = decolonization }

	possible = {
		has_technology_researched = decolonization
	}

	max_num_formation_candidates = 3
	# Bind candidacy to an India-capital power. Engine candidacy is culture-only,
	# and baluchi/bengali/panjabi/sindi are shared with DAR, so an out-of-region
	# power (e.g. Persia) could otherwise leak in. A Pakistan-capital state stays
	# a valid candidate — its capital is inside geographic_region_india.
	can_be_formation_candidate = {
		country_rank >= rank_value:major_power
		capital = { is_in_geographic_region = geographic_region_india }
	}

	# Emergent partition: a Muslim-ruled state inside the core (e.g. Hyderabad
	# under the Nizam — culture deccani, state religion sunni) cannot be swept up
	# by Indian unification, mirroring how a dual-monarchy Austria is exempted
	# from German unification. It survives as an independent holdout that a
	# formed India can later absorb via the Operation Polo event chain. A Hindu
	# owner of the same state is absorbed normally — ownership decides the fate.
	can_be_unification_target = {
		NOT = {
			OR = {
				country_has_state_religion = rel:sunni
				country_has_state_religion = rel:shiite
				country_has_state_religion = rel:ibadi
			}
		}
	}
}

# === Vanilla minor-formation hide gate ===
#
# Hide vanilla BHT / IDN entirely once decolonization is researched so the
# formable list cleanly transitions from minor (pre-decolonization) to
# major BHA / NUS (post-decolonization) without leaving an obsolete greyed-
# out entry behind. `potential = { ... }` is the engine's "show in UI"
# trigger (vs `possible`, which only enables/disables); when potential is
# false the formation is hidden from the country_formation panel entirely.
# Vanilla BHT / IDN don't declare `potential`, so INJECT is a clean field-
# add (no collision with vanilla's existing `possible` block).

INJECT:BHT = {
	potential = {
		NOT = { has_technology_researched = decolonization }
	}
}

INJECT:IDN = {
	potential = {
		NOT = { has_technology_researched = decolonization }
	}
}
