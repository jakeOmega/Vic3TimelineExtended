# Formation rules for the seven new formable countries.
#
# AFU/BHT/IDN/EUN/UNA are major formations (candidates compete via
# leadership/unification plays); INM/EAF are minor formations (whoever
# directly controls enough of the required states/region).
#
# All require the era_6 `decolonization` mod tech (era_6.txt:785).
# BHT and IDN re-declare the vanilla tags so this file's entries override
# vanilla `00_formable_countries.txt` (last-wins by file load order).

# === Major formations ===

AFU = {
	geographic_region = geographic_region_africa

	is_major_formation = yes

	unification_play = dp_unify_african_union
	leadership_play = dp_leadership_african_union

	required_states_fraction = 0.5

	ai_will_do = { has_technology_researched = decolonization }

	possible = {
		has_technology_researched = decolonization
		has_technology_researched = pan-nationalism
	}

	max_num_formation_candidates = 3
	can_be_formation_candidate = {
		country_rank >= rank_value:minor_power
	}
}
