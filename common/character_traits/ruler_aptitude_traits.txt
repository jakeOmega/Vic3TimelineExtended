### Ruler Aptitude Traits - Education-based skill traits inspired by EU4/CK3 monarch skills
### Applied primarily through the heir education JE, but also available via natural weight system

# ──────────────────────── ADMINISTRATIVE ────────────────────────

ruler_terrible_administrator = {
	type = condition
	texture = "gfx/interface/icons/character_trait_icons/ruler_terrible_administrator.dds"

	character_modifier = {
		character_popularity_add = -20
	}

	command_modifier = {
	}

	country_modifier = {
		country_bureaucracy_mult = -0.20
		state_tax_capacity_mult = -0.20
		country_legitimacy_base_add = -10
		country_bubble_pressure_monthly_add = 0.2
	}

	agitator_modifier = {
	}

	interest_group_modifier = {
		interest_group_pol_str_mult = -0.10
	}

	executive_modifier = {
		building_throughput_add = -0.05
	}

	possible = {
		NOT = { age < define:NCharacters|ADULT_AGE }
		OR = {
			has_role = politician
			has_role = agitator
			is_ruler = yes
			is_heir = yes
		}
	}

	weight = {
		value = 0
	}

	replace = {
		ruler_poor_administrator
		ruler_average_administrator
		ruler_skilled_administrator
		ruler_exceptional_administrator
	}
}
ruler_poor_administrator = {
	type = condition

		country_bubble_pressure_monthly_add = 0.1
	}

	agitator_modifier = {
	}

	interest_group_modifier = {
		interest_group_pol_str_mult = -0.05
	}

	executive_modifier = {
		building_throughput_add = -0.03
	}

	possible = {
		NOT = { age < define:NCharacters|ADULT_AGE }
		OR = {
			has_role = politician
			has_role = agitator
			is_ruler = yes
			is_heir = yes
		}
	}

	weight = {

		}
	}

	replace = {
		ruler_average_administrator

	}
