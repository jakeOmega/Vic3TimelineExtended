### Ruler Aptitude Traits - Education-based skill traits inspired by EU4/CK3 monarch skills
### Applied primarily through the heir education JE, but also available via natural weight system

# ──────────────────────── ADMINISTRATIVE ────────────────────────

ruler_poor_administrator = {
	type = condition
	texture = "gfx/interface/icons/character_trait_icons/political_appointee.dds"

	character_modifier = {
		character_popularity_add = -10
	}

	command_modifier = {
	}

	country_modifier = {
		country_bureaucracy_mult = -0.10
		country_legitimacy_base_add = -5
	}

	agitator_modifier = {
	}

	interest_group_modifier = {
		interest_group_pol_str_mult = -0.05
	}

	executive_modifier = {
		building_throughput_add = -0.03
	}

	possible = {
		NOT = { age < define:NCharacters|ADULT_AGE }
		OR = {
			has_role = politician
			has_role = agitator
			is_ruler = yes
			is_heir = yes
		}
	}

	weight = {
		value = 0
		if = {
			limit = {
				is_ruler = yes
				NOR = {
					has_trait = ruler_skilled_administrator
					has_trait = ruler_exceptional_administrator
				}
			}
			add = 1
		}
	}

	replace = {
		ruler_skilled_administrator
		ruler_exceptional_administrator
	}
}
