# Vanilla Company Updates
# Adds mod-exclusive buildings to vanilla flavored and generic companies via INJECT
# INJECT appends new entries to existing fields without replacing the entire definition

#============================================================
# Phase 1: Flavored Company Buildings
#============================================================

# Krupp (Germany) - Adds: Essen Works building, aerospace extension
INJECT:company_krupp = {
	building_types = {
		building_krupp_essen_works
	}
	extension_building_types = {
		building_aerospace_industry
	}
	prosperity_modifier = {
		state_building_krupp_essen_works_max_level_add = 1
	}
