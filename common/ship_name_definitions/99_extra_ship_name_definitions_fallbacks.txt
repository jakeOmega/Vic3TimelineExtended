# =============================================================================
# Vic3TimelineExtended — fallback name lists for mod-added ship types
# =============================================================================
# Vanilla's 99_ship_name_definitions_fallbacks.txt only lists vanilla
# ship_type_* values in its allowed_ship_types blocks, so mod-added ships
# (asw_destroyer, nuclear_submarine, guided_missile_ship, nuclear_supercarrier,
# arsenal_ship, expeditionary_sea_base, hypersonic_missile_platform,
# pulsed_laser_escort, swarm_coordination_vessel) match no fallback and ship
# unnamed.
#
# These entries reuse vanilla loc keys (shipname_default_format,
# shipname_fallback_ordinal_generic, shipname_fallback_submarine) and the
# shipname_prefixes_clone_template defined in vanilla — no new loc required.
#
# Selection-weight tiering mirrors vanilla:
#   capital ships    high=states  med=hubs  low=surnames
#   cruisers         high=hubs    med=states low=surnames
#   torpedo craft    high=surnames med=hubs  low=states
# =============================================================================


### Ordinal fallbacks (always usable, infinite supply)

shipname_fallback_ordinal_te_capital_ships = {
	name = "shipname_fallback_ordinal_generic" # SHIP_TYPE No. 1
	selection_weight = 1
	fallback = yes

	allowed_ship_types = {
		ship_type_arsenal_ship
		ship_type_hypersonic_missile_platform
	}

	clone_properties = shipname_prefixes_clone_template

	properties = {
		{
			type = ship_type
			key = "SHIP_TYPE_NAME"
		}
		{
			type = ordinal_number
			key = "NUM"
		}
	}
}
