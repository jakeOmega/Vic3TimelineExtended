# ============================================================================
# Strategic Reserve Journal Entry
# ============================================================================
# Activates automatically when a country builds a Strategic Reserve Hub.
# Displays current stockpile vs capacity for each good and exposes signed rate
# controls plus a shared step-size selector. Positive rates store goods,
# negative rates withdraw them, and the hub follows the chosen weekly rate for
# each good. All stockpile bookkeeping runs in on_weekly_pulse_country
# (sr_on_actions).
# ============================================================================

je_strategic_reserve = {
	icon = "gfx/interface/icons/building_icons/building_government_administration.dds"
	group = je_group_internal_affairs

	is_shown_when_inactive = {
		has_technology_researched = logistics
	}

	possible = {
		any_scope_building = {
			is_building_type = building_strategic_reserve_hub
		}
	}

	immediate = {
		sr_init_effect = yes
		sr_refresh_hub_flow_effect = yes
		sr_apply_modes_effect = yes
	}

	invalid = {
		NOT = {
			any_scope_building = {
				is_building_type = building_strategic_reserve_hub
			}
		}
	}

	on_invalid = {
		sr_reset_vars_effect = yes
	}

	scripted_button = sr_cycle_step_size_button
	scripted_button = sr_decrease_grain_rate_button
	scripted_button = sr_increase_grain_rate_button
	scripted_button = sr_decrease_ammunition_rate_button
	scripted_button = sr_increase_ammunition_rate_button
	scripted_button = sr_decrease_oil_rate_button
	scripted_button = sr_increase_oil_rate_button

	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_strategic_reserve_status_no_hub
				trigger = {
					NOT = {
						any_scope_building = { is_building_type = building_strategic_reserve_hub }
					}
				}
			}
			triggered_desc = {
				desc = je_strategic_reserve_status_active
				trigger = { always = yes }
			}
		}
		triggered_desc = {
			desc = je_strategic_reserve_rate_line
			trigger = { always = yes }
		}
		triggered_desc = {
			desc = je_strategic_reserve_step_line
			trigger = { always = yes }
		}
		triggered_desc = {
			desc = je_strategic_reserve_grain_line
			trigger = { always = yes }
		}
		triggered_desc = {
			desc = je_strategic_reserve_ammunition_line
			trigger = { has_technology_researched = nitroglycerin }
		}
		triggered_desc = {
			desc = je_strategic_reserve_oil_line
			trigger = { has_technology_researched = oil_rig }
		}
	}

	weight = 500
}
