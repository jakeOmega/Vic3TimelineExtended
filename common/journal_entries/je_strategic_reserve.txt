# ============================================================================
# Strategic Reserve Journal Entry
# ============================================================================
# Activates automatically when a country builds a Strategic Reserve Hub.
# Displays current stockpile vs capacity for each good and exposes signed rate
# controls plus a shared step-size selector. Positive rates store goods,
# negative rates withdraw them, and the hub follows the chosen weekly rate for
# each good. Persistent reserve state lives on country vars; this JE only owns
# visibility, buttons, and lifecycle entry points into the SR wrapper effects.
# ============================================================================

je_strategic_reserve = {
	icon = "gfx/interface/icons/building_icons/building_government_administration.dds"
	group = je_group_internal_affairs

	is_shown_when_inactive = {
		has_technology_researched = logistics
	}

	possible = {
		any_scope_building = {
			is_building_type = building_strategic_reserve_hub
		}
	}

	immediate = {
		st_res_je_immediate_effect = yes
	}

	invalid = {
		NOT = {
			any_scope_building = {
				is_building_type = building_strategic_reserve_hub
			}
		}
	}

	on_invalid = {
		st_res_je_invalid_effect = yes
	}

	widget = {
		gui = "gui/journal_entry_widgets/strategic_reserve_widget.gui"
		name = "widget_je_strategic_reserve_step_size"
		container = "custom_widget_container_1"
	}

	scripted_progress_bar = st_res_grain_fill_bar
	scripted_progress_bar = st_res_ammunition_fill_bar
	scripted_progress_bar = st_res_oil_fill_bar
	scripted_progress_bar = st_res_small_arms_fill_bar
	scripted_progress_bar = st_res_artillery_fill_bar
	scripted_progress_bar = st_res_aeroplanes_fill_bar
	scripted_progress_bar = st_res_tanks_fill_bar

	scripted_button = st_res_cycle_step_size_button
	scripted_button = st_res_reset_rates_button
	scripted_button = st_res_decrease_grain_rate_button
	scripted_button = st_res_increase_grain_rate_button
	scripted_button = st_res_decrease_ammunition_rate_button
	scripted_button = st_res_increase_ammunition_rate_button
	scripted_button = st_res_decrease_oil_rate_button
	scripted_button = st_res_increase_oil_rate_button
	scripted_button = st_res_decrease_small_arms_rate_button
	scripted_button = st_res_increase_small_arms_rate_button
	scripted_button = st_res_decrease_artillery_rate_button
	scripted_button = st_res_increase_artillery_rate_button
	scripted_button = st_res_decrease_aeroplanes_rate_button
	scripted_button = st_res_increase_aeroplanes_rate_button
	scripted_button = st_res_decrease_tanks_rate_button
	scripted_button = st_res_increase_tanks_rate_button

	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_strategic_reserve_status_no_hub
				trigger = {
					NOT = {
						any_scope_building = { is_building_type = building_strategic_reserve_hub }
					}
				}
			}
			triggered_desc = {
				desc = je_strategic_reserve_status_deactivated
				trigger = {
					var:st_res_hub_workforce_cached <= 0.99
				}
			}
			triggered_desc = {
				desc = je_strategic_reserve_status_active
				trigger = { always = yes }
			}
		}
		triggered_desc = {
			desc = je_strategic_reserve_profit_line
			trigger = { always = yes }
		}
		triggered_desc = {
			desc = je_strategic_reserve_rate_line
			trigger = { always = yes }
		}
		triggered_desc = {
			desc = je_strategic_reserve_step_line
			trigger = { always = yes }
		}
		triggered_desc = {
			desc = je_strategic_reserve_grain_line
			trigger = { always = yes }
		}
		triggered_desc = {
			desc = je_strategic_reserve_ammunition_line
			trigger = { has_technology_researched = percussion_cap }
		}
		triggered_desc = {
			desc = je_strategic_reserve_oil_line
			trigger = { has_technology_researched = fractional_distillation }
		}
		triggered_desc = {
			desc = je_strategic_reserve_small_arms_line
			trigger = { always = yes }
		}
		triggered_desc = {
			desc = je_strategic_reserve_artillery_line
			trigger = { always = yes }
		}
		triggered_desc = {
			desc = je_strategic_reserve_aeroplanes_line
			trigger = { has_technology_researched = military_aviation }
		}
		triggered_desc = {
			desc = je_strategic_reserve_tanks_line
			trigger = { has_technology_researched = mobile_armor }
		}
	}

	# Weekly bookkeeping lives on country vars; the JE only provides the pulse.
	on_weekly_pulse = {
		effect = {
			st_res_je_weekly_pulse_effect = yes
		}
	}

	weight = 500
	should_be_pinned_by_default_uninvolved_or_context = yes
	should_update_on_player_command = yes
}
