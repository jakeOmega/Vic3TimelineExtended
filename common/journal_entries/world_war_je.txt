
# ============================================================================
# WORLD WAR JOURNAL ENTRY
# Tracks ideological polarization building toward a global conflict
# Phases: Leadup (rising tensions) > Active War > Peace Conference > Post-War
#
# This JE is the SOLE driver of all world war events. The on_action is empty.
# The JE activates for any great power when opposing ideology GPs exist,
# and remains active through war and post-war phases.
# ============================================================================

je_world_war = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"
	group = je_group_foreign_affairs

	is_shown_when_inactive = {
		has_game_rule = world_war_enabled
		has_technology_researched = combined_arms
		country_rank >= rank_value:great_power
		NOT = { has_variable = ww_fully_resolved }
	}

	possible = {
		has_technology_researched = combined_arms
		country_rank >= rank_value:great_power
		OR = {
			# Leadup: opposing ideology great powers exist anywhere
			any_country = {
				country_rank >= rank_value:great_power
				NOT = { this = ROOT }
				ROOT = { country_has_opposed_ideology = { TARGET = prev } }
			}
			# Active participant in world war
			has_variable = ww_active_phase
			# Post-war phase (not yet fully resolved)
			AND = {
				has_variable = ww_peace_concluded
				NOT = { has_variable = ww_fully_resolved }
			}
		}
	}

	status_desc = {
		first_valid = {
			# Post-war phase
			triggered_desc = {
				desc = je_world_war_status_post_war
				trigger = { has_variable = ww_peace_concluded }
			}
			# Active war: total war (3+ years, combatant)
			triggered_desc = {
				desc = je_world_war_status_total_war
				trigger = {
					country_is_ww_belligerent = yes
					is_at_war = yes
					has_variable = ww_years_elapsed
					var:ww_years_elapsed >= 3
				}
			}
			# Active war: combatant
			triggered_desc = {
				desc = je_world_war_status_active_war
				trigger = {
					country_is_ww_belligerent = yes
					is_at_war = yes
				}
			}
			# Active war: non-combatant observer (a world war is happening but we're not in it)
			triggered_desc = {
				desc = je_world_war_status_observing
				trigger = {
					any_country = {
						country_is_ww_belligerent = yes
						is_at_war = yes
					}
				}
			}
			# War declared but no one fighting yet (brief transitional state)
			triggered_desc = {
				desc = je_world_war_status_war_declared
				trigger = {
					any_country = { country_is_ww_belligerent = yes }
				}
			}
			# Leadup phase descriptions
			triggered_desc = {
				desc = je_world_war_status_crisis
				trigger = { ww_ideological_tension >= 70 }
			}
			triggered_desc = {
				desc = je_world_war_status_high_tension
				trigger = { ww_ideological_tension >= 50 }
			}
			triggered_desc = {
				desc = je_world_war_status_rising_tension
				trigger = { ww_ideological_tension >= 30 }
			}
			triggered_desc = {
				desc = je_world_war_status_simmering
				trigger = { always = yes }
			}
		}
	}

	progressbar = no

	# --- Leadup Phase Buttons ---
	scripted_button = ww_rearm_button
	scripted_button = ww_stop_rearm_button
	scripted_button = ww_appease_button
	scripted_button = ww_stop_appease_button
	scripted_button = ww_lend_lease_button
	scripted_button = ww_stop_lend_lease_button

	# --- Active War Phase Buttons ---
	scripted_button = ww_total_war_economy_button
	scripted_button = ww_stop_total_war_economy_button
	scripted_button = ww_war_propaganda_button
	scripted_button = ww_stop_war_propaganda_button
	scripted_button = ww_wartime_rationing_button
	scripted_button = ww_stop_wartime_rationing_button

	# --- Late Entry Button ---
	scripted_button = ww_join_war_button

	immediate = { }

	complete = {
		has_variable = ww_fully_resolved
	}

	on_complete = {
		# Clean up all world war variables
		remove_variable = ww_active_phase
		remove_variable = ww_aggressor_side
		remove_variable = ww_defender_side
		remove_variable = ww_years_elapsed
		remove_variable = ww_months_counter
		remove_variable = ww_peace_concluded
		remove_variable = ww_peace_months
		remove_variable = ww_fully_resolved
		# Clean up any lingering modifiers
		if = { limit = { je:je_world_war = { has_modifier = ww_rising_tensions_modifier } } je:je_world_war = { remove_modifier = ww_rising_tensions_modifier } }
		if = { limit = { je:je_world_war = { has_modifier = ww_home_front_strain_modifier } } je:je_world_war = { remove_modifier = ww_home_front_strain_modifier } }
		if = { limit = { je:je_world_war = { has_modifier = ww_prolonged_war_exhaustion_modifier } } je:je_world_war = { remove_modifier = ww_prolonged_war_exhaustion_modifier } }
	}

	fail = {
		# Failed if no longer a great power
		country_rank < rank_value:great_power
	}

	on_fail = {
		remove_variable = ww_active_phase
		remove_variable = ww_aggressor_side
		remove_variable = ww_defender_side
		remove_variable = ww_years_elapsed
		remove_variable = ww_months_counter
		remove_variable = ww_peace_concluded
		remove_variable = ww_peace_months
		remove_variable = ww_fully_resolved
	}

	on_monthly_pulse = {
		effect = {
			# ---- LEADUP PHASE: Apply/remove rising tensions modifier ----
			if = {
				limit = {
					NOT = { has_variable = ww_peace_concluded }
					NOT = { country_is_ww_belligerent = yes }
					NOT = { je:je_world_war = { has_modifier = ww_rising_tensions_modifier } }
					ww_ideological_tension >= 30
				}
				je:je_world_war = { add_modifier = { name = ww_rising_tensions_modifier } }
			}
			if = {
				limit = {
					je:je_world_war = { has_modifier = ww_rising_tensions_modifier }
					OR = {
						ww_ideological_tension < 30
						country_is_ww_belligerent = yes
						has_variable = ww_peace_concluded
					}
				}
				je:je_world_war = { remove_modifier = ww_rising_tensions_modifier }
			}
			# ---- ACTIVE WAR PHASE: Track war duration for belligerents ----
			if = {
				limit = {
					country_is_ww_belligerent = yes
					is_at_war = yes
					NOT = { has_variable = ww_peace_concluded }
				}
				if = {
					limit = { NOT = { has_variable = ww_months_counter } }
					set_variable = { name = ww_months_counter value = 0 }
				}
				change_variable = { name = ww_months_counter add = 1 }
				# Every 12 months, increment years
				if = {
					limit = { var:ww_months_counter >= 12 }
					change_variable = { name = ww_months_counter add = -12 }
					if = {
						limit = { NOT = { has_variable = ww_years_elapsed } }
						set_variable = { name = ww_years_elapsed value = 1 }
					}
					if = {
						limit = { has_variable = ww_years_elapsed }
						change_variable = { name = ww_years_elapsed add = 1 }
					}
				}
			}
			# Home front strain after 2+ years
			if = {
				limit = {
					country_is_ww_belligerent = yes
					is_at_war = yes
					has_variable = ww_years_elapsed
					var:ww_years_elapsed >= 2
					NOT = { je:je_world_war = { has_modifier = ww_home_front_strain_modifier } }
				}
				je:je_world_war = { add_modifier = { name = ww_home_front_strain_modifier } }
			}
			# Prolonged war exhaustion after 4+ years
			if = {
				limit = {
					country_is_ww_belligerent = yes
					is_at_war = yes
					has_variable = ww_years_elapsed
					var:ww_years_elapsed >= 4
					NOT = { je:je_world_war = { has_modifier = ww_prolonged_war_exhaustion_modifier } }
				}
				je:je_world_war = { add_modifier = { name = ww_prolonged_war_exhaustion_modifier } }
			}
			# ---- PEACE DETECTION ----
			# Belligerents: war is over when they stop fighting
			if = {
				limit = {
					country_is_ww_belligerent = yes
					is_at_war = no
					NOT = { has_variable = ww_peace_concluded }
				}
				set_variable = ww_peace_concluded
				# Clean up war-phase modifiers
				if = { limit = { je:je_world_war = { has_modifier = ww_total_war_economy_modifier } } je:je_world_war = { remove_modifier = ww_total_war_economy_modifier } }
				if = { limit = { je:je_world_war = { has_modifier = ww_war_propaganda_modifier } } je:je_world_war = { remove_modifier = ww_war_propaganda_modifier } }
				if = { limit = { je:je_world_war = { has_modifier = ww_wartime_rationing_modifier } } je:je_world_war = { remove_modifier = ww_wartime_rationing_modifier } }
				if = { limit = { je:je_world_war = { has_modifier = ww_home_front_strain_modifier } } je:je_world_war = { remove_modifier = ww_home_front_strain_modifier } }
				if = { limit = { je:je_world_war = { has_modifier = ww_prolonged_war_exhaustion_modifier } } je:je_world_war = { remove_modifier = ww_prolonged_war_exhaustion_modifier } }
				if = { limit = { has_modifier = ww_fresh_forces_modifier } remove_modifier = ww_fresh_forces_modifier }
				if = { limit = { has_modifier = ww_arsenal_of_democracy_modifier } remove_modifier = ww_arsenal_of_democracy_modifier }
				# Fire the peace conference event directly for belligerents
				trigger_event = { id = world_war_events.100 }
			}
			# Non-combatant observers: peace when no belligerent is still fighting
			if = {
				limit = {
					NOT = { country_is_ww_belligerent = yes }
					NOT = { has_variable = ww_peace_concluded }
					# At least one country was in the war (it happened)
					any_country = { has_variable = ww_peace_concluded }
					# No belligerents still fighting
					NOT = {
						any_country = {
							country_is_ww_belligerent = yes
							is_at_war = yes
						}
					}
				}
				set_variable = ww_peace_concluded
				# Clean up leadup modifiers for observers
				if = { limit = { je:je_world_war = { has_modifier = ww_rising_tensions_modifier } } je:je_world_war = { remove_modifier = ww_rising_tensions_modifier } }
				if = { limit = { je:je_world_war = { has_modifier = ww_rearmament_modifier } } je:je_world_war = { remove_modifier = ww_rearmament_modifier } }
				if = { limit = { je:je_world_war = { has_modifier = ww_appeasement_modifier } } je:je_world_war = { remove_modifier = ww_appeasement_modifier } }
				if = { limit = { je:je_world_war = { has_modifier = ww_lend_lease_modifier } } je:je_world_war = { remove_modifier = ww_lend_lease_modifier } }
			}
			# ---- POST-WAR TIMER ----
			# After peace, count months. Resolve after 3 years (36 months).
			if = {
				limit = {
					has_variable = ww_peace_concluded
					NOT = { has_variable = ww_fully_resolved }
				}
				if = {
					limit = { NOT = { has_variable = ww_peace_months } }
					set_variable = { name = ww_peace_months value = 0 }
				}
				change_variable = { name = ww_peace_months add = 1 }
				if = {
					limit = { var:ww_peace_months >= 36 }
					set_variable = ww_fully_resolved
				}
			}
		}
		# ---- RANDOM EVENTS ----
		# All world war events fire from here. Each event has its own trigger block.
		random_events = {
			200 = 0 # mostly nothing
			# Leadup events (fire when not yet at war, tension rising)
			10 = world_war_events.1   # ideological confrontation
			10 = world_war_events.2   # border incident
			10 = world_war_events.3   # diplomatic crisis
			# War outbreak (fire when tension is critical)
			5 = world_war_events.5    # war breaks out
			# Active war events (fire during the war for belligerents)
			10 = world_war_events.10  # home front rally
			10 = world_war_events.11  # strategic bombing
			10 = world_war_events.12  # resistance movement
			# Late entry prompt (non-combatants during war)
			10 = world_war_events.20  # opportunity to join
			# War weariness (prolonged war)
			5 = world_war_events.30   # war weariness grows
			# Post-war events
			5 = world_war_events.103  # war crimes tribunal
			5 = world_war_events.104  # post-war order
			5 = world_war_events.105  # the new rivalry
		}
	}

	weight = 500
	should_be_pinned_by_default = yes
	can_revolution_inherit = yes
	can_deactivate = no
}
