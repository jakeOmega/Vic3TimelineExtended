# ============================================================================
# CULTURAL HEGEMONY — Journal Entry (Display Only)
# ============================================================================
# This JE is the player's window into the Cultural Hegemony system.
# It shows cultural pull breakdown, global leaderboard, and tier status.
# All computation runs in on_actions (hegemony_cyber_on_actions.txt) via
# scripted effects (cultural_hegemony_effects.txt) from game start.
# The JE only activates after 1845 so scores are stable when first shown.
# ============================================================================

je_cultural_hegemony = {
	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	group = je_group_soft_power

	is_shown_when_inactive = {
		has_game_rule = cultural_hegemony_enabled
	}

	possible = {
		any_country = {
            has_technology_researched = mass_media
        }
		has_technology_researched = romanticism
	}

	scripted_button = ch_increase_program_funding_button
	scripted_button = ch_decrease_program_funding_button
	scripted_button = ch_world_exposition_button
	scripted_button = ch_cultural_institutes_button
	scripted_button = ch_disable_cultural_institutes_button
	scripted_button = ch_global_media_campaign_button
	scripted_button = ch_disable_global_media_campaign_button
	scripted_button = ch_cultural_protectionism_button
	scripted_button = ch_disable_cultural_protectionism_button

	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_cultural_hegemony_status_dominant
				trigger = { has_variable = ch_total var:ch_total >= 25 }
			}
			triggered_desc = {
				desc = je_cultural_hegemony_status_major
				trigger = { has_variable = ch_total var:ch_total >= 15 }
			}
			triggered_desc = {
				desc = je_cultural_hegemony_status_significant
				trigger = { has_variable = ch_total var:ch_total >= 10 }
			}
			triggered_desc = {
				desc = je_cultural_hegemony_status_moderate
				trigger = { has_variable = ch_total var:ch_total >= 5 }
			}
			triggered_desc = {
				desc = je_cultural_hegemony_status_minor
				trigger = { has_variable = ch_total var:ch_total >= 2 }
			}
			triggered_desc = {
				desc = je_cultural_hegemony_status_negligible
				trigger = { always = yes }
			}
		}
		# Share info
		triggered_desc = {
			desc = je_ch_share_info
			trigger = { always = yes }
		}
		triggered_desc = {
			desc = je_ch_policy_header
			trigger = {
				OR = {
					has_law = law_type:law_ministry_of_culture
					je:je_cultural_hegemony = { has_modifier = ch_cultural_institutes_program }
					je:je_cultural_hegemony = { has_modifier = ch_global_media_campaign }
					je:je_cultural_hegemony = { has_modifier = ch_cultural_protectionism_program }
				}
			}
		}
		triggered_desc = {
			desc = je_ch_policy_program_funding
			trigger = { has_law = law_type:law_ministry_of_culture }
		}
		triggered_desc = {
			desc = je_ch_policy_cultural_institutes
			trigger = { je:je_cultural_hegemony = { has_modifier = ch_cultural_institutes_program } }
		}
		triggered_desc = {
			desc = je_ch_policy_media_campaign
			trigger = { je:je_cultural_hegemony = { has_modifier = ch_global_media_campaign } }
		}
		triggered_desc = {
			desc = je_ch_policy_cultural_protectionism
			trigger = { je:je_cultural_hegemony = { has_modifier = ch_cultural_protectionism_program } }
		}
		# Raw score breakdown
		triggered_desc = {
			desc = je_ch_score_header
			trigger = { always = yes }
		}
		triggered_desc = {
			desc = je_ch_breakdown_prestige
			trigger = { prestige > 0 }
		}
		triggered_desc = {
			desc = je_ch_breakdown_art
			trigger = { has_variable = ch_art var:ch_art > 0 }
		}
		triggered_desc = {
			desc = je_ch_breakdown_sol
			trigger = { has_variable = ch_sol var:ch_sol > 0 }
		}
		triggered_desc = {
			desc = je_ch_breakdown_tech
			trigger = {
				has_variable = ch_tech_firsts_recent
				var:ch_tech_firsts_recent > 0
			}
		}
		triggered_desc = {
			desc = je_ch_breakdown_monuments
			trigger = { has_variable = ch_monuments var:ch_monuments >= 1 }
		}
		triggered_desc = {
			desc = je_ch_breakdown_megaprojects
			trigger = { has_variable = ch_megaprojects var:ch_megaprojects >= 1 }
		}
		triggered_desc = {
			desc = je_ch_breakdown_modifiers
			trigger = { modifier:country_cultural_pull_add > 0 }
		}
		triggered_desc = {
			desc = je_ch_breakdown_infamy
			trigger = { infamy > 0 }
		}
		triggered_desc = {
			desc = je_ch_breakdown_stability
			trigger = {
				OR = {
					country_turmoil > 0.01
					any_civil_war = { }
				}
			}
		}
        desc = separator
		triggered_desc = {
			desc = je_ch_breakdown_mult
			trigger = { modifier:country_cultural_pull_mult > 0 }
		}
        desc = separator
        desc = je_ce_raw_score
        desc = je_ce_global_raw_score
        desc = je_ch_leaderboard_delimiter
		# Top 10 Cultural Powers leaderboard
		triggered_desc = {
			desc = je_ch_leaderboard_header
			trigger = { has_global_variable = ch_rank_1 }
		}
		triggered_desc = {
			desc = je_ch_rank_1_entry
			trigger = { has_global_variable = ch_rank_1 }
		}
		triggered_desc = {
			desc = je_ch_rank_2_entry
			trigger = { has_global_variable = ch_rank_2 }
		}
		triggered_desc = {
			desc = je_ch_rank_3_entry
			trigger = { has_global_variable = ch_rank_3 }
		}
		triggered_desc = {
			desc = je_ch_rank_4_entry
			trigger = { has_global_variable = ch_rank_4 }
		}
		triggered_desc = {
			desc = je_ch_rank_5_entry
			trigger = { has_global_variable = ch_rank_5 }
		}
		triggered_desc = {
			desc = je_ch_rank_6_entry
			trigger = { has_global_variable = ch_rank_6 }
		}
		triggered_desc = {
			desc = je_ch_rank_7_entry
			trigger = { has_global_variable = ch_rank_7 }
		}
		triggered_desc = {
			desc = je_ch_rank_8_entry
			trigger = { has_global_variable = ch_rank_8 }
		}
		triggered_desc = {
			desc = je_ch_rank_9_entry
			trigger = { has_global_variable = ch_rank_9 }
		}
		triggered_desc = {
			desc = je_ch_rank_10_entry
			trigger = { has_global_variable = ch_rank_10 }
		}
	}

	immediate = {
		if = {
			limit = { NOT = { has_variable = ch_program_funding_level } }
			set_variable = { name = ch_program_funding_level value = 0 }
		}

		# Variables are already populated by on_actions before JE activates.
		# Initialize any that might not exist (safety fallback).
		if = {
			limit = { NOT = { has_variable = ch_total } }
			set_variable = { name = ch_total value = 0 }
			set_variable = { name = ch_art value = 0 }
			set_variable = { name = ch_sol value = 0 }
			set_variable = { name = ch_monuments value = 0 }
			set_variable = { name = ch_megaprojects value = 0 }
			set_variable = { name = ch_disp_total value = 0 }
			set_variable = { name = ch_disp_raw value = 0 }
			set_variable = { name = ch_disp_art value = 0 }
			set_variable = { name = ch_disp_prestige value = 0 }
			set_variable = { name = ch_disp_sol value = 0 }
			set_variable = { name = ch_disp_tech value = 0 }
			set_variable = { name = ch_disp_monuments value = 0 }
			set_variable = { name = ch_disp_megaprojects value = 0 }
			set_variable = { name = ch_disp_rank value = 0 }
		}
	}

	complete = {
		always = no
	}

	on_monthly_pulse = {
        effect = {
			if = {
				limit = { NOT = { has_law = law_type:law_ministry_of_culture } }
				set_variable = { name = ch_program_funding_level value = 0 }
				je:je_cultural_hegemony ?= { remove_modifier = ch_cultural_institutes_program }
				je:je_cultural_hegemony ?= { remove_modifier = ch_global_media_campaign }
				je:je_cultural_hegemony ?= { remove_modifier = ch_cultural_protectionism_program }
			}

			je:je_cultural_hegemony ?= { remove_modifier = ch_cultural_program_funding }
			je:je_cultural_hegemony ?= { remove_modifier = ch_cultural_program_funding_cost }
			if = {
				limit = {
					has_law = law_type:law_ministry_of_culture
					ch_program_funding_level_value >= 1
				}
				je:je_cultural_hegemony ?= { add_modifier = { name = ch_cultural_program_funding multiplier = ch_program_funding_level_value } }
				je:je_cultural_hegemony ?= { add_modifier = { name = ch_cultural_program_funding_cost multiplier = ch_program_funding_cost_mult } }
			}

            # Apply/remove cultural hegemony modifier (hegemon bonus)
            je:je_cultural_hegemony ?= { remove_modifier = cultural_hegemony_effect }
            if = {
                limit = { var:ch_total >= 1 }
                je:je_cultural_hegemony ?= { add_modifier = { name = cultural_hegemony_effect multiplier = cultural_hegemony_modifier_mult } }
            }

            # Apply/remove foreign benchmark (SoL expectations from hegemon)
            # Only applies to countries with this JE active
            if = {
                limit = { cultural_hegemony_benchmark_mult >= 0.1 }
                remove_modifier = cultural_hegemony_foreign_benchmark
                add_modifier = { name = cultural_hegemony_foreign_benchmark multiplier = cultural_hegemony_benchmark_mult }
            }
            else_if = {
                limit = { has_modifier = cultural_hegemony_foreign_benchmark }
                remove_modifier = cultural_hegemony_foreign_benchmark
            }
        }
	}

	weight = 900
	should_be_pinned_by_default = yes
	can_revolution_inherit = yes
	can_deactivate = no
}
