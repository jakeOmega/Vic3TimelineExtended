### Heir Education Journal Entry
### Reworked: 3% base monthly pulse (down from 5%), intelligence-modified gains,
### adult heir pre-initialization, 5-tier trait resolution, rebel child mechanic.

je_heir_education = {
	icon = "gfx/interface/icons/event_icons/event_portrait.dds"
	group = je_group_internal_affairs

	is_shown_when_inactive = {
		has_law = law_type:law_monarchy
		has_game_rule = heir_education_enabled
	}

	possible = {
		has_law = law_type:law_monarchy
		exists = heir
		heir ?= {
			NOT = { has_variable = received_education }
		}
	}

	scripted_progress_bar = heir_education_progress_bar

	# Toggle buttons - enable/disable pairs (only one of each pair visible at a time)
	scripted_button = he_enable_admin_focus
	scripted_button = he_disable_admin_focus
	scripted_button = he_enable_diplo_focus
	scripted_button = he_disable_diplo_focus
	scripted_button = he_enable_military_focus
	scripted_button = he_disable_military_focus
	scripted_button = he_enable_progressive
	scripted_button = he_disable_progressive
	scripted_button = he_enable_conservative
	scripted_button = he_disable_conservative
	scripted_button = he_enable_radical
	scripted_button = he_disable_radical
	scripted_button = he_enable_moderate
	scripted_button = he_disable_moderate
	scripted_button = he_enable_regressive
	scripted_button = he_disable_regressive

	immediate = {
		set_variable = { name = heir_ed_admin value = 0 }
		set_variable = { name = heir_ed_diplo value = 0 }
		set_variable = { name = heir_ed_military value = 0 }
		set_variable = { name = heir_ed_ideology value = 0 }
		set_variable = { name = heir_ed_ig_radical value = 0 }
		set_variable = { name = heir_ed_ig_moderate value = 0 }
		set_variable = { name = heir_ed_ig_regressive value = 0 }
		set_variable = { name = heir_ed_total value = 0 }
		# Hidden intelligence score (1-5): affects gain amount per pulse tick
		# Higher = faster learning, but player never sees exact value
		random_list = {
			20 = { set_variable = { name = heir_ed_intelligence value = 1 } }
			20 = { set_variable = { name = heir_ed_intelligence value = 2 } }
			20 = { set_variable = { name = heir_ed_intelligence value = 3 } }
			20 = { set_variable = { name = heir_ed_intelligence value = 4 } }
			20 = { set_variable = { name = heir_ed_intelligence value = 5 } }
		}
		heir ?= {
			set_variable = being_educated
		}
		# Pre-initialize adult heirs based on their age
		heir_education_init_adult = yes
		# Update progress bar with any pre-initialized total
		scope:journal_entry = {
			set_bar_progress = { name = heir_education_progress_bar value = root.var:heir_ed_total }
		}
		# Grace period prevents instant completion for adult heirs
		add_modifier = { name = heir_education_grace_period days = 365 }
	}

	complete = {
		exists = heir
		heir ?= {
			has_variable = being_educated
			NOT = { age < define:NCharacters|ADULT_AGE }
		}
		NOT = { has_modifier = heir_education_grace_period }
	}

	on_complete = {
		trigger_event = { id = heir_education_events.200 }
	}

	# If the heir changes (dies, replaced), detect and reset
	on_monthly_pulse = {
		effect = {
			# Backwards compatibility: initialize each variable separately if missing
			if = { limit = { NOT = { has_variable = heir_ed_admin } } set_variable = { name = heir_ed_admin value = 0 } }
			if = { limit = { NOT = { has_variable = heir_ed_diplo } } set_variable = { name = heir_ed_diplo value = 0 } }
			if = { limit = { NOT = { has_variable = heir_ed_military } } set_variable = { name = heir_ed_military value = 0 } }
			if = { limit = { NOT = { has_variable = heir_ed_ideology } } set_variable = { name = heir_ed_ideology value = 0 } }
			if = { limit = { NOT = { has_variable = heir_ed_ig_radical } } set_variable = { name = heir_ed_ig_radical value = 0 } }
			if = { limit = { NOT = { has_variable = heir_ed_ig_moderate } } set_variable = { name = heir_ed_ig_moderate value = 0 } }
			if = { limit = { NOT = { has_variable = heir_ed_ig_regressive } } set_variable = { name = heir_ed_ig_regressive value = 0 } }
			# Ensure total exists; also perform the adult pre-init and grace-period when total was missing
			if = {
				limit = { NOT = { has_variable = heir_ed_total } }
				set_variable = { name = heir_ed_total value = 0 }
				heir ?= { set_variable = being_educated }
				heir_education_init_adult = yes
				scope:journal_entry = { set_bar_progress = { name = heir_education_progress_bar value = root.var:heir_ed_total } }
				add_modifier = { name = heir_education_grace_period days = 365 }
			}
			# Initialize newly added intelligence value if missing
			if = {
				limit = { NOT = { has_variable = heir_ed_intelligence } }
				random_list = {
					20 = { set_variable = { name = heir_ed_intelligence value = 1 } }
					20 = { set_variable = { name = heir_ed_intelligence value = 2 } }
					20 = { set_variable = { name = heir_ed_intelligence value = 3 } }
					20 = { set_variable = { name = heir_ed_intelligence value = 4 } }
					20 = { set_variable = { name = heir_ed_intelligence value = 5 } }
				}
			}
			# ─── Heir change detection ───
			if = {
				limit = {
					OR = {
						NOT = { exists = heir }
						heir ?= { NOT = { has_variable = being_educated } }
					}
					has_variable = heir_ed_total
				}
				set_variable = { name = heir_ed_admin value = 0 }
				set_variable = { name = heir_ed_diplo value = 0 }
				set_variable = { name = heir_ed_military value = 0 }
				set_variable = { name = heir_ed_ideology value = 0 }
				set_variable = { name = heir_ed_ig_radical value = 0 }
				set_variable = { name = heir_ed_ig_moderate value = 0 }
				set_variable = { name = heir_ed_ig_regressive value = 0 }
				set_variable = { name = heir_ed_total value = 0 }
				# Re-roll intelligence for new heir
				random_list = {
					20 = { set_variable = { name = heir_ed_intelligence value = 1 } }
					20 = { set_variable = { name = heir_ed_intelligence value = 2 } }
					20 = { set_variable = { name = heir_ed_intelligence value = 3 } }
					20 = { set_variable = { name = heir_ed_intelligence value = 4 } }
					20 = { set_variable = { name = heir_ed_intelligence value = 5 } }
				}
				# Clear focus variables
				if = { limit = { has_variable = heir_ed_focus_admin } remove_variable = heir_ed_focus_admin }
				if = { limit = { has_variable = heir_ed_focus_diplo } remove_variable = heir_ed_focus_diplo }
				if = { limit = { has_variable = heir_ed_focus_military } remove_variable = heir_ed_focus_military }
				if = { limit = { has_variable = heir_ed_focus_progressive } remove_variable = heir_ed_focus_progressive }
				if = { limit = { has_variable = heir_ed_focus_conservative } remove_variable = heir_ed_focus_conservative }
				if = { limit = { has_variable = heir_ed_focus_radical } remove_variable = heir_ed_focus_radical }
				if = { limit = { has_variable = heir_ed_focus_moderate } remove_variable = heir_ed_focus_moderate }
				if = { limit = { has_variable = heir_ed_focus_regressive } remove_variable = heir_ed_focus_regressive }
				scope:journal_entry = {
					set_bar_progress = { name = heir_education_progress_bar value = 0 }
				}
				if = {
					limit = { exists = heir }
					heir ?= {
						set_variable = being_educated
					}
					# Pre-initialize for new adult heir
					heir_education_init_adult = yes
					scope:journal_entry = {
						set_bar_progress = { name = heir_education_progress_bar value = root.var:heir_ed_total }
					}
				}
			}

			# ─── Monthly investment processing ───
			# Each active focus has a 3% chance per month to trigger a gain.
			# Gain amount is modified by hidden intelligence score.
			# Admin focus
			if = {
				limit = { has_variable = heir_ed_focus_admin }
				random_list = {
					3 = { heir_ed_gain_admin = yes }
					97 = { }
				}
			}
			# Diplo focus
			if = {
				limit = { has_variable = heir_ed_focus_diplo }
				random_list = {
					3 = { heir_ed_gain_diplo = yes }
					97 = { }
				}
			}
			# Military focus
			if = {
				limit = { has_variable = heir_ed_focus_military }
				random_list = {
					3 = { heir_ed_gain_military = yes }
					97 = { }
				}
			}
			# Progressive focus
			if = {
				limit = { has_variable = heir_ed_focus_progressive }
				random_list = {
					3 = { heir_ed_gain_progressive = yes }
					97 = { }
				}
			}
			# Conservative focus
			if = {
				limit = { has_variable = heir_ed_focus_conservative }
				random_list = {
					3 = { heir_ed_gain_conservative = yes }
					97 = { }
				}
			}
			# Radical focus
			if = {
				limit = { has_variable = heir_ed_focus_radical }
				random_list = {
					3 = { heir_ed_gain_radical = yes }
					97 = { }
				}
			}
			# Moderate focus
			if = {
				limit = { has_variable = heir_ed_focus_moderate }
				random_list = {
					3 = { heir_ed_gain_moderate = yes }
					97 = { }
				}
			}
			# Regressive focus
			if = {
				limit = { has_variable = heir_ed_focus_regressive }
				random_list = {
					3 = { heir_ed_gain_regressive = yes }
					97 = { }
				}
			}

			# ─── Random education events ───
			random_list = {
				2 = {
					trigger = {
						NOT = { has_variable = heir_education_event_cooldown }
						has_variable = heir_ed_total
						var:heir_ed_total >= 2
					}
					trigger_event = { id = heir_education_events.1 }
					set_variable = { name = heir_education_event_cooldown days = 365 }
				}
				2 = {
					trigger = {
						NOT = { has_variable = heir_education_event_cooldown }
						has_variable = heir_ed_total
						var:heir_ed_total >= 1
					}
					trigger_event = { id = heir_education_events.2 }
					set_variable = { name = heir_education_event_cooldown days = 365 }
				}
				2 = {
					trigger = {
						NOT = { has_variable = heir_education_event_cooldown }
						has_variable = heir_ed_total
						var:heir_ed_total >= 3
						any_country = {
							has_treaty_alliance_with = { TARGET = ROOT }
						}
					}
					trigger_event = { id = heir_education_events.3 }
					set_variable = { name = heir_education_event_cooldown days = 365 }
				}
				94 = { }
			}
		}
	}

	invalid = {
		NOT = { has_law = law_type:law_monarchy }
	}

	on_invalid = {
		# Clean up all heir education variables and modifiers when the JE is invalidated
		# (e.g. government type switched away from monarchy). Without this, authority
		# cost modifiers like heir_ed_trait_focus_active would persist indefinitely.
		heir_education_cleanup_effect = yes
	}

	progressbar = no
	should_be_pinned_by_default = yes

	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_heir_education_status_no_focus
				trigger = {
					NOR = {
						has_variable = heir_ed_focus_admin
						has_variable = heir_ed_focus_diplo
						has_variable = heir_ed_focus_military
						has_variable = heir_ed_focus_progressive
						has_variable = heir_ed_focus_conservative
						has_variable = heir_ed_focus_radical
						has_variable = heir_ed_focus_moderate
						has_variable = heir_ed_focus_regressive
					}
				}
			}
			triggered_desc = {
				desc = je_heir_education_status_bright
				trigger = {
					var:heir_ed_intelligence >= 4
				}
			}
			triggered_desc = {
				desc = je_heir_education_status_slow
				trigger = {
					var:heir_ed_intelligence <= 2
				}
			}
			triggered_desc = {
				desc = je_heir_education_status_active
				trigger = { always = yes }
			}
		}
	}

	weight = 100

	timeout = 6570 # 18 years - safety fallback
	on_timeout = {
		heir_education_resolve_effect = yes
		heir_education_cleanup_effect = yes
	}

	current_value = {
		value = var:heir_ed_total
	}

	goal_add_value = {
		value = 20
	}
}
