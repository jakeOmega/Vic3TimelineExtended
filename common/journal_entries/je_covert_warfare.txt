# ============================================================================
# COVERT WARFARE — Journal Entry (Command Center)
# ============================================================================
# Displays Intelligence Capacity, active operations (from pacts), slot
# breakdown, funding level, detection risk, and covert defense info.
# Operations are toggled via diplomatic actions (pact-based).
# ============================================================================

je_covert_warfare = {
	icon = "gfx/interface/icons/event_icons/event_default.dds"

	group = je_group_covert_warfare

	is_shown_when_inactive = {
		has_game_rule = covert_warfare_enabled
	}

	possible = {
		has_technology_researched = television
	}

	status_desc = {
		# ---- Intelligence Status Tier ----
		first_valid = {
			triggered_desc = {
				desc = je_iw_status_fortress
				trigger = { intelligence_capacity_total >= 80 }
			}
			triggered_desc = {
				desc = je_iw_status_hardened
				trigger = { intelligence_capacity_total >= 60 }
			}
			triggered_desc = {
				desc = je_iw_status_defended
				trigger = { intelligence_capacity_total >= 40 }
			}
			triggered_desc = {
				desc = je_iw_status_exposed
				trigger = { intelligence_capacity_total >= 20 }
			}
			triggered_desc = {
				desc = je_iw_status_vulnerable
				trigger = { always = yes }
			}
		}

		# ---- Intelligence Capacity Breakdown ----
		triggered_desc = {
			desc = je_iw_capacity_header
			trigger = { always = yes }
		}
		triggered_desc = {
			desc = je_iw_capacity_modifiers
			trigger = { always = yes }
		}
		triggered_desc = {
			desc = je_iw_capacity_literacy
			trigger = { intelligence_capacity_from_literacy >= 1 }
		}
		triggered_desc = {
			desc = je_iw_capacity_gdp
			trigger = { intelligence_capacity_from_gdp >= 1 }
		}

		# ---- Operation Slots Breakdown ----
		triggered_desc = {
			desc = je_iw_slots_header
			trigger = { always = yes }
		}
		triggered_desc = {
			desc = je_iw_slots_base
			trigger = { always = yes }
		}
		triggered_desc = {
			desc = je_iw_slots_rank
			trigger = { covert_ops_slots_from_rank >= 1 }
		}
		triggered_desc = {
			desc = je_iw_slots_modifiers
			trigger = { covert_ops_slots_from_modifiers >= 1 }
		}

		# ---- Funding & Cost Info ----
		triggered_desc = {
			desc = je_iw_funding_header
			trigger = { always = yes }
		}

		# ---- Detection Risk ----
		triggered_desc = {
			desc = je_iw_detection_info
			trigger = { covert_operations_active >= 1 }
		}

		# ---- Covert Defense Info ----
		triggered_desc = {
			desc = je_iw_defense_header
			trigger = { always = yes }
		}

		# ---- Active Operations ----
		triggered_desc = {
			desc = je_iw_ops_header
			trigger = { always = yes }
		}
		triggered_desc = {
			desc = je_iw_op_election_interference_active
			trigger = { any_scope_diplomatic_pact = { is_diplomatic_action_type = covert_election_interference_action } }
		}
		triggered_desc = {
			desc = je_iw_op_financial_subversion_active
			trigger = { any_scope_diplomatic_pact = { is_diplomatic_action_type = covert_financial_subversion_action } }
		}
		triggered_desc = {
			desc = je_iw_op_infrastructure_sabotage_active
			trigger = { any_scope_diplomatic_pact = { is_diplomatic_action_type = covert_infrastructure_sabotage_action } }
		}
		triggered_desc = {
			desc = je_iw_op_comms_disruption_active
			trigger = { any_scope_diplomatic_pact = { is_diplomatic_action_type = covert_comms_disruption_action } }
		}
		triggered_desc = {
			desc = je_iw_op_industrial_espionage_active
			trigger = { any_scope_diplomatic_pact = { is_diplomatic_action_type = covert_industrial_espionage_action } }
		}
		triggered_desc = {
			desc = je_iw_op_military_espionage_active
			trigger = { any_scope_diplomatic_pact = { is_diplomatic_action_type = covert_military_espionage_action } }
		}
		triggered_desc = {
			desc = je_iw_op_influence_campaign_active
			trigger = { any_scope_diplomatic_pact = { is_diplomatic_action_type = covert_influence_campaign_action } }
		}
		triggered_desc = {
			desc = je_iw_op_ideological_subversion_active
			trigger = { any_scope_diplomatic_pact = { is_diplomatic_action_type = covert_ideological_subversion_action } }
		}
		triggered_desc = {
			desc = je_iw_op_destabilization_active
			trigger = { any_scope_diplomatic_pact = { is_diplomatic_action_type = covert_destabilization_action } }
		}
		triggered_desc = {
			desc = je_iw_no_operations
			trigger = {
				covert_operations_active < 1
			}
		}
	}

	# Funding management buttons (operations launched via diplomatic actions)
	scripted_button = iw_increase_funding_button
	scripted_button = iw_decrease_funding_button

	immediate = {
		# Initialize funding variable if missing
		if = {
			limit = { NOT = { has_variable = iw_funding_level } }
			set_variable = { name = iw_funding_level value = 0 }
		}
	}

	complete = {
		always = no
	}

	on_monthly_pulse = {
		effect = {
			# ---- Update intelligence capacity defense modifier ----
			if = {
				limit = { intelligence_capacity_total >= 1 }
				remove_modifier = intelligence_capacity_defense
				add_modifier = { name = intelligence_capacity_defense multiplier = intelligence_capacity_modifier_mult }
			}
			else_if = {
				limit = { has_modifier = intelligence_capacity_defense }
				remove_modifier = intelligence_capacity_defense
			}

			# ---- Domestic defense from unused operation slots ----
			if = {
				limit = { covert_operations_available_slots >= 1 }
				remove_modifier = iw_domestic_defense
				add_modifier = { name = iw_domestic_defense multiplier = covert_operations_available_slots }
			}
			else_if = {
				limit = { has_modifier = iw_domestic_defense }
				remove_modifier = iw_domestic_defense
			}

			# ---- Apply operation funding cost ----
			if = {
				limit = {
					covert_operations_active >= 1
					has_variable = iw_funding_level
					var:iw_funding_level >= 1
				}
				remove_modifier = covert_operation_funding_cost
				add_modifier = { name = covert_operation_funding_cost multiplier = covert_operation_cost_mult }
			}
			else_if = {
				limit = { has_modifier = covert_operation_funding_cost }
				remove_modifier = covert_operation_funding_cost
			}

			# ---- Apply operation effects (only if funding > 0) ----
			if = {
				limit = {
					has_variable = iw_funding_level
					var:iw_funding_level >= 1
				}

				# Election Interference: apply to target
				every_scope_diplomatic_pact = {
					limit = { is_diplomatic_action_type = covert_election_interference_action }
					every_participant = {
						limit = { NOT = { this = ROOT } }
						remove_modifier = covert_election_interference
						add_modifier = { name = covert_election_interference months = 3 is_decaying = yes }
					}
				}

				# Financial Subversion: apply to target
				every_scope_diplomatic_pact = {
					limit = { is_diplomatic_action_type = covert_financial_subversion_action }
					every_participant = {
						limit = { NOT = { this = ROOT } }
						remove_modifier = covert_financial_subversion
						add_modifier = { name = covert_financial_subversion months = 3 is_decaying = yes }
					}
				}

				# Infrastructure Sabotage: apply to random state in target
				every_scope_diplomatic_pact = {
					limit = { is_diplomatic_action_type = covert_infrastructure_sabotage_action }
					every_participant = {
						limit = { NOT = { this = ROOT } }
						random_scope_state = {
							limit = { any_scope_building = { level >= 1 } }
							remove_modifier = covert_infrastructure_sabotage
							add_modifier = { name = covert_infrastructure_sabotage months = 3 is_decaying = yes }
						}
					}
				}

				# Communications Disruption: apply to target
				every_scope_diplomatic_pact = {
					limit = { is_diplomatic_action_type = covert_comms_disruption_action }
					every_participant = {
						limit = { NOT = { this = ROOT } }
						remove_modifier = covert_comms_disruption
						add_modifier = { name = covert_comms_disruption months = 3 is_decaying = yes }
					}
				}

				# Industrial Espionage: apply to SELF (tech spread gated by tech gap)
				every_scope_diplomatic_pact = {
					limit = { is_diplomatic_action_type = covert_industrial_espionage_action }
					every_participant = {
						limit = { NOT = { this = ROOT } }
						save_scope_as = espionage_target
					}
				}
				if = {
					limit = {
						any_scope_diplomatic_pact = { is_diplomatic_action_type = covert_industrial_espionage_action }
						exists = scope:espionage_target
						espionage_target_tech_advantage > 0
					}
					remove_modifier = covert_industrial_espionage
					add_modifier = { name = covert_industrial_espionage months = 3 is_decaying = yes }
				}
				else_if = {
					limit = {
						any_scope_diplomatic_pact = { is_diplomatic_action_type = covert_industrial_espionage_action }
					}
					# Target has no tech advantage — apply base bonus only (innovation, no tech spread)
					remove_modifier = covert_industrial_espionage
					remove_modifier = covert_espionage_base
					add_modifier = { name = covert_espionage_base months = 3 is_decaying = yes }
				}

				# Military Espionage: apply to SELF (tech spread gated by tech gap)
				every_scope_diplomatic_pact = {
					limit = { is_diplomatic_action_type = covert_military_espionage_action }
					every_participant = {
						limit = { NOT = { this = ROOT } }
						save_scope_as = espionage_target
					}
				}
				if = {
					limit = {
						any_scope_diplomatic_pact = { is_diplomatic_action_type = covert_military_espionage_action }
						exists = scope:espionage_target
						espionage_target_tech_advantage > 0
					}
					remove_modifier = covert_military_espionage
					add_modifier = { name = covert_military_espionage months = 3 is_decaying = yes }
				}
				else_if = {
					limit = {
						any_scope_diplomatic_pact = { is_diplomatic_action_type = covert_military_espionage_action }
					}
					# Target has no tech advantage — apply base bonus only (unit stats, no tech spread)
					remove_modifier = covert_military_espionage
					remove_modifier = covert_espionage_base
					add_modifier = { name = covert_espionage_base months = 3 is_decaying = yes }
				}

				# Influence Campaign: reduce target's leverage resistance
				every_scope_diplomatic_pact = {
					limit = { is_diplomatic_action_type = covert_influence_campaign_action }
					every_participant = {
						limit = { NOT = { this = ROOT } }
						remove_modifier = covert_influence_campaign
						add_modifier = { name = covert_influence_campaign months = 3 is_decaying = yes }
					}
				}

				# Ideological Subversion: boost movements matching YOUR ideology in target
				if = {
					limit = {
						any_scope_diplomatic_pact = { is_diplomatic_action_type = covert_ideological_subversion_action }
					}
					save_scope_as = cultural_hegemon
					every_scope_diplomatic_pact = {
						limit = { is_diplomatic_action_type = covert_ideological_subversion_action }
						every_participant = {
							limit = { NOT = { this = ROOT } }
							ch_apply_hegemon_movement_pressure = { MODIFIER = covert_ideological_subversion MONTHS = 3 }
						}
					}
				}

				# Destabilization: boost separatist movements + general radicalism in target
				every_scope_diplomatic_pact = {
					limit = { is_diplomatic_action_type = covert_destabilization_action }
					every_participant = {
						limit = { NOT = { this = ROOT } }
						# Boost separatist movements strongly
						every_political_movement = {
							limit = { is_political_movement_type = movement_secession }
							remove_modifier = covert_destabilization_separatist
							add_modifier = { name = covert_destabilization_separatist months = 3 is_decaying = yes }
						}
						# General radicalism boost to all movements
						every_political_movement = {
							remove_modifier = covert_destabilization_general
							add_modifier = { name = covert_destabilization_general months = 3 is_decaying = yes }
						}
					}
				}
			}

			# ---- Detection check for active operations (only if funding > 0) ----
			if = {
				limit = {
					covert_operations_active >= 1
					has_variable = iw_funding_level
					var:iw_funding_level >= 1
				}
				random_list = {
					1 = {
						modifier = {
							trigger = { always = yes }
							add = covert_operation_detection_chance
						}
						# Detected! Fire the detection event
						trigger_event = { id = covert_warfare.1 }
					}
					99 = {
						# Not detected this month
					}
				}
			}
		}
	}

	weight = 800
	should_be_pinned_by_default = yes
	can_revolution_inherit = yes
	can_deactivate = no
}
