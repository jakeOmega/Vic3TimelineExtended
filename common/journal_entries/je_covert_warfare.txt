# ============================================================================
# COVERT WARFARE — Journal Entry (Command Center)
# ============================================================================
# Displays Intelligence Capacity, active operations (from pacts), slot
# breakdown, funding level, detection risk, and covert defense info.
# Operations are toggled via diplomatic actions (pact-based).
# ============================================================================

je_covert_warfare = {
	icon = "gfx/interface/icons/event_icons/event_default.dds"

	group = je_group_covert_warfare

	is_shown_when_inactive = {
		has_game_rule = covert_warfare_enabled
	}

	possible = {
		has_technology_researched = television
	}

	status_desc = {
		# ---- Intelligence Status Tier (relative to global best) ----
		first_valid = {
			triggered_desc = {
				desc = je_iw_status_fortress
				trigger = { intelligence_capacity_relative >= 80 }
			}
			triggered_desc = {
				desc = je_iw_status_hardened
				trigger = { intelligence_capacity_relative >= 60 }
			}
			triggered_desc = {
				desc = je_iw_status_defended
				trigger = { intelligence_capacity_relative >= 40 }
			}
			triggered_desc = {
				desc = je_iw_status_exposed
				trigger = { intelligence_capacity_relative >= 20 }
			}
			triggered_desc = {
				desc = je_iw_status_vulnerable
				trigger = { always = yes }
			}
		}

		# ---- Intelligence Capacity Breakdown ----
		triggered_desc = {
			desc = je_iw_capacity_header
			trigger = { always = yes }
		}
		triggered_desc = {
			desc = je_iw_capacity_modifiers
			trigger = { always = yes }
		}
		triggered_desc = {
			desc = je_iw_capacity_literacy
			trigger = { intelligence_capacity_from_literacy >= 1 }
		}
		triggered_desc = {
			desc = je_iw_capacity_gdp
			trigger = { intelligence_capacity_from_gdp >= 1 }
		}

		# ---- Operation Slots Breakdown ----
		triggered_desc = {
			desc = je_iw_slots_header
			trigger = { always = yes }
		}

		# ---- Funding & Cost Info ----
		triggered_desc = {
			desc = je_iw_funding_header
			trigger = { always = yes }
		}
		triggered_desc = {
			desc = je_iw_cost_info
			trigger = {
				covert_operations_active >= 1
				has_variable = iw_funding_level
				var:iw_funding_level >= 1
			}
		}

		# ---- Detection Risk (shown per-operation in Active Ops section) ----
		triggered_desc = {
			desc = je_iw_detection_factors
			trigger = { covert_operations_active >= 1 }
		}

		# ---- Covert Defense Info ----
		triggered_desc = {
			desc = je_iw_defense_header
			trigger = { always = yes }
		}

		# ---- Active Operations ----
		triggered_desc = {
			desc = je_iw_ops_header
			trigger = { always = yes }
		}
		triggered_desc = {
			desc = je_iw_op_election_interference_active
			trigger = { any_scope_diplomatic_pact = { is_diplomatic_action_type = covert_election_interference_action } }
		}
		triggered_desc = {
			desc = je_iw_op_financial_subversion_active
			trigger = { any_scope_diplomatic_pact = { is_diplomatic_action_type = covert_financial_subversion_action } }
		}
		triggered_desc = {
			desc = je_iw_op_infrastructure_sabotage_active
			trigger = { any_scope_diplomatic_pact = { is_diplomatic_action_type = covert_infrastructure_sabotage_action } }
		}
		triggered_desc = {
			desc = je_iw_op_comms_disruption_active
			trigger = { any_scope_diplomatic_pact = { is_diplomatic_action_type = covert_comms_disruption_action } }
		}
		triggered_desc = {
			desc = je_iw_op_industrial_espionage_active
			trigger = { any_scope_diplomatic_pact = { is_diplomatic_action_type = covert_industrial_espionage_action } }
		}
		triggered_desc = {
			desc = je_iw_op_military_espionage_active
			trigger = { any_scope_diplomatic_pact = { is_diplomatic_action_type = covert_military_espionage_action } }
		}
		triggered_desc = {
			desc = je_iw_op_influence_campaign_active
			trigger = { any_scope_diplomatic_pact = { is_diplomatic_action_type = covert_influence_campaign_action } }
		}
		triggered_desc = {
			desc = je_iw_op_ideological_subversion_active
			trigger = { any_scope_diplomatic_pact = { is_diplomatic_action_type = covert_ideological_subversion_action } }
		}
		triggered_desc = {
			desc = je_iw_op_destabilization_active
			trigger = { any_scope_diplomatic_pact = { is_diplomatic_action_type = covert_destabilization_action } }
		}
		triggered_desc = {
			desc = je_iw_no_operations
			trigger = {
				covert_operations_active < 1
			}
		}
	}

	# Funding management buttons (operations launched via diplomatic actions)
	scripted_button = iw_increase_funding_button
	scripted_button = iw_decrease_funding_button

	immediate = {
		# Initialize funding variable if missing
		if = {
			limit = { NOT = { has_variable = iw_funding_level } }
			set_variable = { name = iw_funding_level value = 0 }
		}
	}

	complete = {
		always = no
	}

	on_monthly_pulse = {
		effect = {
			# ---- AI Funding Management ----
			# AI automatically manages funding level based on financial health
			# and strategic need. Player uses scripted buttons instead.
			if = {
				limit = {
					is_player = no
					has_technology_researched = television
				}
				# In default or severe financial trouble: cut funding to zero
				if = {
					limit = {
						OR = {
							in_default = yes
							has_modifier = declared_bankruptcy
						}
					}
					set_variable = { name = iw_funding_level value = 0 }
				}
				# Has rivals/enemies and is financially healthy: fund operations
				else_if = {
					limit = {
						country_rank >= rank_value:major_power
						OR = {
							any_country = {
								has_diplomatic_pact = { who = ROOT type = rivalry }
							}
							is_at_war = yes
						}
					}
					# Great powers with strong finances get high funding
					if = {
						limit = {
							country_rank = rank_value:great_power
							NOT = { in_default = yes }
						}
						set_variable = { name = iw_funding_level value = 2 }
					}
					# Major powers or cautious rulers get low funding
					else = {
						set_variable = { name = iw_funding_level value = 1 }
					}
				}
				# No rivals and not at war: no need to fund operations
				else = {
					set_variable = { name = iw_funding_level value = 0 }
				}
			}

			# ---- Track global max IC for relative tier display ----
			ordered_country = {
				limit = { has_technology_researched = television }
				order_by = intelligence_capacity_total
				max = 1
				position = 0
				set_global_variable = { name = global_max_ic value = intelligence_capacity_total }
			}

			# ---- Update intelligence capacity defense modifier ----
			if = {
				limit = { intelligence_capacity_total >= 1 }
				remove_modifier = intelligence_capacity_defense
				add_modifier = { name = intelligence_capacity_defense multiplier = intelligence_capacity_modifier_mult }
			}
			else_if = {
				limit = { has_modifier = intelligence_capacity_defense }
				remove_modifier = intelligence_capacity_defense
			}

			# ---- Domestic defense from unused operation slots ----
			if = {
				limit = { covert_operations_available_slots >= 1 }
				remove_modifier = iw_domestic_defense
				add_modifier = { name = iw_domestic_defense multiplier = covert_operations_available_slots }
			}
			else_if = {
				limit = { has_modifier = iw_domestic_defense }
				remove_modifier = iw_domestic_defense
			}

			# ---- Apply operation funding cost ----
			if = {
				limit = {
					covert_operations_active >= 1
					has_variable = iw_funding_level
					var:iw_funding_level >= 1
				}
				remove_modifier = covert_operation_funding_cost
				add_modifier = { name = covert_operation_funding_cost multiplier = covert_operation_cost_mult }
			}
			else_if = {
				limit = { has_modifier = covert_operation_funding_cost }
				remove_modifier = covert_operation_funding_cost
			}

			# ---- Apply operation effects (only if funding > 0) ----
			if = {
				limit = {
					has_variable = iw_funding_level
					var:iw_funding_level >= 1
				}

				# Infrastructure Sabotage: apply to random state in target (state-scoped, not in pact)
				every_scope_diplomatic_pact = {
					limit = { is_diplomatic_action_type = covert_infrastructure_sabotage_action }
					every_participant = {
						limit = { NOT = { this = ROOT } }
						random_scope_state = {
							limit = { any_scope_building = { level >= 1 } }
							remove_modifier = covert_infrastructure_sabotage
							add_modifier = { name = covert_infrastructure_sabotage months = 3 is_decaying = yes }
						}
					}
				}

				# Communications Disruption: effects handled by pact second_modifier

				# Industrial Espionage: conditional tech spread bonus (base innovation via pact)
				every_scope_diplomatic_pact = {
					limit = { is_diplomatic_action_type = covert_industrial_espionage_action }
					every_participant = {
						limit = { NOT = { this = ROOT } }
						save_scope_as = espionage_target
					}
				}
				if = {
					limit = {
						any_scope_diplomatic_pact = { is_diplomatic_action_type = covert_industrial_espionage_action }
						exists = scope:espionage_target
						espionage_target_tech_advantage > 0
					}
					remove_modifier = covert_industrial_espionage
					add_modifier = { name = covert_industrial_espionage months = 3 is_decaying = yes }
				}

				# Military Espionage: conditional tech spread bonus (base unit stats via pact)
				every_scope_diplomatic_pact = {
					limit = { is_diplomatic_action_type = covert_military_espionage_action }
					every_participant = {
						limit = { NOT = { this = ROOT } }
						save_scope_as = espionage_target
					}
				}
				if = {
					limit = {
						any_scope_diplomatic_pact = { is_diplomatic_action_type = covert_military_espionage_action }
						exists = scope:espionage_target
						espionage_target_tech_advantage > 0
					}
					remove_modifier = covert_military_espionage
					add_modifier = { name = covert_military_espionage months = 3 is_decaying = yes }
				}

				# Influence Campaign: effects handled by pact second_modifier

				# Ideological Subversion: boost movements matching YOUR ideology in target (movement-scoped)
				if = {
					limit = {
						any_scope_diplomatic_pact = { is_diplomatic_action_type = covert_ideological_subversion_action }
					}
					save_scope_as = cultural_hegemon
					every_scope_diplomatic_pact = {
						limit = { is_diplomatic_action_type = covert_ideological_subversion_action }
						every_participant = {
							limit = { NOT = { this = ROOT } }
							ch_apply_hegemon_movement_pressure = { MODIFIER = covert_ideological_subversion MONTHS = 3 }
						}
					}
				}

				# Destabilization: boost all political movements + general radicalism in target
				every_scope_diplomatic_pact = {
					limit = { is_diplomatic_action_type = covert_destabilization_action }
					every_participant = {
						limit = { NOT = { this = ROOT } }
						# Boost all political movements strongly
						every_political_movement = {
							remove_modifier = covert_destabilization_separatist
							add_modifier = { name = covert_destabilization_separatist months = 3 is_decaying = yes }
						}
						# General radicalism boost
						remove_modifier = covert_destabilization_general
						add_modifier = { name = covert_destabilization_general months = 3 is_decaying = yes }
					}
				}
			}

			# ---- Per-operation target tracking & detection ----
			# For each op: save target country, set target_max_ic to their IC,
			# then compute & save per-op detection chance for JE display.
			# Election Interference
			every_scope_diplomatic_pact = {
				limit = { is_diplomatic_action_type = covert_election_interference_action }
				every_participant = {
					limit = { NOT = { this = ROOT } }
					ROOT = {
						set_variable = { name = iw_target_election_interference value = PREV }
						set_variable = { name = target_max_ic value = PREV.intelligence_capacity_total }
					}
				}
			}
			if = {
				limit = { any_scope_diplomatic_pact = { is_diplomatic_action_type = covert_election_interference_action } }
				set_variable = { name = iw_detect_election_interference value = covert_operation_detection_chance }
			}
			# Financial Subversion
			every_scope_diplomatic_pact = {
				limit = { is_diplomatic_action_type = covert_financial_subversion_action }
				every_participant = {
					limit = { NOT = { this = ROOT } }
					ROOT = {
						set_variable = { name = iw_target_financial_subversion value = PREV }
						set_variable = { name = target_max_ic value = PREV.intelligence_capacity_total }
					}
				}
			}
			if = {
				limit = { any_scope_diplomatic_pact = { is_diplomatic_action_type = covert_financial_subversion_action } }
				set_variable = { name = iw_detect_financial_subversion value = covert_operation_detection_chance }
			}
			# Infrastructure Sabotage
			every_scope_diplomatic_pact = {
				limit = { is_diplomatic_action_type = covert_infrastructure_sabotage_action }
				every_participant = {
					limit = { NOT = { this = ROOT } }
					ROOT = {
						set_variable = { name = iw_target_infrastructure_sabotage value = PREV }
						set_variable = { name = target_max_ic value = PREV.intelligence_capacity_total }
					}
				}
			}
			if = {
				limit = { any_scope_diplomatic_pact = { is_diplomatic_action_type = covert_infrastructure_sabotage_action } }
				set_variable = { name = iw_detect_infrastructure_sabotage value = covert_operation_detection_chance }
			}
			# Communications Disruption
			every_scope_diplomatic_pact = {
				limit = { is_diplomatic_action_type = covert_comms_disruption_action }
				every_participant = {
					limit = { NOT = { this = ROOT } }
					ROOT = {
						set_variable = { name = iw_target_comms_disruption value = PREV }
						set_variable = { name = target_max_ic value = PREV.intelligence_capacity_total }
					}
				}
			}
			if = {
				limit = { any_scope_diplomatic_pact = { is_diplomatic_action_type = covert_comms_disruption_action } }
				set_variable = { name = iw_detect_comms_disruption value = covert_operation_detection_chance }
			}
			# Industrial Espionage
			every_scope_diplomatic_pact = {
				limit = { is_diplomatic_action_type = covert_industrial_espionage_action }
				every_participant = {
					limit = { NOT = { this = ROOT } }
					ROOT = {
						set_variable = { name = iw_target_industrial_espionage value = PREV }
						set_variable = { name = target_max_ic value = PREV.intelligence_capacity_total }
					}
				}
			}
			if = {
				limit = { any_scope_diplomatic_pact = { is_diplomatic_action_type = covert_industrial_espionage_action } }
				set_variable = { name = iw_detect_industrial_espionage value = covert_operation_detection_chance }
			}
			# Military Espionage
			every_scope_diplomatic_pact = {
				limit = { is_diplomatic_action_type = covert_military_espionage_action }
				every_participant = {
					limit = { NOT = { this = ROOT } }
					ROOT = {
						set_variable = { name = iw_target_military_espionage value = PREV }
						set_variable = { name = target_max_ic value = PREV.intelligence_capacity_total }
					}
				}
			}
			if = {
				limit = { any_scope_diplomatic_pact = { is_diplomatic_action_type = covert_military_espionage_action } }
				set_variable = { name = iw_detect_military_espionage value = covert_operation_detection_chance }
			}
			# Influence Campaign
			every_scope_diplomatic_pact = {
				limit = { is_diplomatic_action_type = covert_influence_campaign_action }
				every_participant = {
					limit = { NOT = { this = ROOT } }
					ROOT = {
						set_variable = { name = iw_target_influence_campaign value = PREV }
						set_variable = { name = target_max_ic value = PREV.intelligence_capacity_total }
					}
				}
			}
			if = {
				limit = { any_scope_diplomatic_pact = { is_diplomatic_action_type = covert_influence_campaign_action } }
				set_variable = { name = iw_detect_influence_campaign value = covert_operation_detection_chance }
			}
			# Ideological Subversion
			every_scope_diplomatic_pact = {
				limit = { is_diplomatic_action_type = covert_ideological_subversion_action }
				every_participant = {
					limit = { NOT = { this = ROOT } }
					ROOT = {
						set_variable = { name = iw_target_ideological_subversion value = PREV }
						set_variable = { name = target_max_ic value = PREV.intelligence_capacity_total }
					}
				}
			}
			if = {
				limit = { any_scope_diplomatic_pact = { is_diplomatic_action_type = covert_ideological_subversion_action } }
				set_variable = { name = iw_detect_ideological_subversion value = covert_operation_detection_chance }
			}
			# Destabilization
			every_scope_diplomatic_pact = {
				limit = { is_diplomatic_action_type = covert_destabilization_action }
				every_participant = {
					limit = { NOT = { this = ROOT } }
					ROOT = {
						set_variable = { name = iw_target_destabilization value = PREV }
						set_variable = { name = target_max_ic value = PREV.intelligence_capacity_total }
					}
				}
			}
			if = {
				limit = { any_scope_diplomatic_pact = { is_diplomatic_action_type = covert_destabilization_action } }
				set_variable = { name = iw_detect_destabilization value = covert_operation_detection_chance }
			}

			# ---- Restore target_max_ic to actual maximum for detection roll ----
			set_variable = { name = target_max_ic value = 0 }
			every_scope_diplomatic_pact = {
				limit = {
					OR = {
						is_diplomatic_action_type = covert_election_interference_action
						is_diplomatic_action_type = covert_financial_subversion_action
						is_diplomatic_action_type = covert_infrastructure_sabotage_action
						is_diplomatic_action_type = covert_comms_disruption_action
						is_diplomatic_action_type = covert_industrial_espionage_action
						is_diplomatic_action_type = covert_military_espionage_action
						is_diplomatic_action_type = covert_influence_campaign_action
						is_diplomatic_action_type = covert_ideological_subversion_action
						is_diplomatic_action_type = covert_destabilization_action
					}
				}
				every_participant = {
					limit = {
						NOT = { this = ROOT }
						intelligence_capacity_total > ROOT.var:target_max_ic
					}
					ROOT = { set_variable = { name = target_max_ic value = PREV.intelligence_capacity_total } }
				}
			}

			# ---- Detection check for active operations (only if funding > 0) ----
			if = {
				limit = {
					covert_operations_active >= 1
					has_variable = iw_funding_level
					var:iw_funding_level >= 1
				}
				random_list = {
					1 = {
						modifier = {
							add = covert_operation_detection_chance
						}
						# Detected! Fire the detection event
						trigger_event = { id = covert_warfare.1 }
					}
					99 = {
						# Not detected this month
					}
				}
			}
		}
	}

	weight = 800
	should_be_pinned_by_default = yes
	can_revolution_inherit = yes
	can_deactivate = no
}
