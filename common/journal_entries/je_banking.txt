        first_valid = {

            triggered_desc = { desc = banking_cycle_status_bubble_frenzy          trigger = { var:bubble_pressure >= 65  var:bubble_pressure < 85 } }

    on_monthly_pulse = {
        effect = {
            # Update fiscal policy effect based on spending relative to GDP
            remove_modifier = financial_cycle_government_fiscal_policy_effect
            add_modifier = {
                name = financial_cycle_government_fiscal_policy_effect
                multiplier = financial_cycle_government_fiscal_policy_effect_size
            }

            # --- Cycle update ---
            # Momentum decays:
            change_variable = { name = finance_cycle_momentum multiply = 0.9 }

            change_variable = {
                name = finance_cycle_momentum
                add = "modifier:country_finance_momentum_monthly_add"
            }

            change_variable = {
                name = bubble_pressure
                add = "modifier:country_bubble_pressure_monthly_add"
            }

            change_variable = {
                name = finance_cycle_value
                add = "modifier:country_finance_value_monthly_add"
            }

            add_investment_pool = investment_pool_banking_cycle_add

            # Random nudge: -1 / 0 / +1
            random_list = {
                25 = {
                    modifier = { # Biased towards lower momentum if economy is hot
                        value = var:finance_cycle_value
                    }
                    change_variable = { name = finance_cycle_momentum add = -1 }
                }
                50 = { }
                25 = {
                    modifier = { # Biased towards higher momentum if economy is doing poorly
                        value = 100
                        subtract = var:finance_cycle_value
                    }
                    change_variable = { name = finance_cycle_momentum add =  1 }
                }
            }

            # Apply momentum to value and clamp 0 to 100
            change_variable = {
                name = finance_cycle_value
                add = var:finance_cycle_momentum
                }
            if = { limit = { var:finance_cycle_value < 0 }     set_variable = { name = finance_cycle_value value = 0 } }
            if = { limit = { var:finance_cycle_value > 100 }   set_variable = { name = finance_cycle_value value = 100 } }

            # clamp
            if = { limit = { var:bubble_pressure < 0 }   set_variable = { name = bubble_pressure value = 0 } }
            if = { limit = { var:bubble_pressure > 100 } set_variable = { name = bubble_pressure value = 100 } }

            # --- sudden crash check (asymmetric) ---
            # Base monthly crash chances by phase (tuneable):
            # frenzy  = (bubble-15)*0.35; boom = (bubble-25)*0.15; expansion = (bubble-50)*0.05; stable = (bubble-75)*0.02
            set_variable = { name = crash value = 0 }
            random_list = {
                0 = { 
                    modifier = {
                        add = var:bubble_pressure
                        if = {
                            limit = { var:finance_cycle_value >= 88 } # frenzy
                            subtract = 15
                            multiply = 0.35
                        }
                        else_if = {
                            limit = { var:finance_cycle_value >= 75 } # boom
                            subtract = 25
                            multiply = 0.15
                        }
                        else_if = {
                            limit = { var:finance_cycle_value >= 60 } # expansion
                            subtract = 50
                            multiply = 0.05
                        }
                        else_if = {
                            limit = { var:finance_cycle_value >= 40 } # stable
                            subtract = 75
                            multiply = 0.02
                        }
                        min = 0
                    }
                    set_variable = { name = crash value = 1 }
                }
                100 = { }
            }

            # If we set crash, execute crash:
            if = {
                limit = { var:crash > 0 }
                save_scope_as = banking_crash_origin_country
                # Severity scales with bubble (more bubble -> likely PANIC; else DOWNTURN)
                set_variable = { name = crash_severity value = var:bubble_pressure }
                random_list = {
                    10 = {
                        change_variable = { name = crash_severity multiply = 0.5 }
                    }
                    20 = {
                        change_variable = { name = crash_severity multiply = 0.75 }
                    }
                    40 = { }
                    20 = {
                        change_variable = { name = crash_severity multiply = 1.5 }
                    }
                    10 = {
                        change_variable = { name = crash_severity multiply = 2 }
                    }
                }
                # Crash effects (value/momentum/bubble reset) now applied in event .6 options
                # via apply_banking_crash_origin_effects, tiered by crash_severity.

                trigger_event = { id = minor_events_timelineextended.6 popup = yes }

                every_country = {
                    limit = {
                        NOT = { THIS = scope:banking_crash_origin_country }
                        has_journal_entry = je_banking_cycle
                        # Same market OR trade-reachable (same-market countries don't need trade routes)
                        OR = {
                            this.market = scope:banking_crash_origin_country.market
                            is_market_reachable_for_trade = scope:banking_crash_origin_country.market
                        }
                        OR = {
                            this.market = scope:banking_crash_origin_country.market  # same market = auto-qualify
                            is_in_customs_union_with = scope:banking_crash_origin_country
                            is_adjacent_to_country = scope:banking_crash_origin_country
                            economic_dependence = {
                                target = scope:banking_crash_origin_country
                                value > 0.05
                            }
                            market = {
                                market_trade_reliance = {
                                    target = scope:banking_crash_origin_country.market
                                    value > 0.01
                                }
                            }
                            market = {
                                market_production_share = {
                                    target = scope:banking_crash_origin_country
                                    value > 0.01
                                }
                            }
                            market = {
                                market_consumption_share = {
                                    target = scope:banking_crash_origin_country
                                    value > 0.01
                                }
                            }
                        }
                    }

                    random = {
                        chance = 25
                        # Same market = strongest contagion channel
                        modifier = {
                            trigger = { this.market = scope:banking_crash_origin_country.market }
                            add = 10
                        }
                        # Hard crash spreads more easily
                        modifier = {
                            trigger = { scope:banking_crash_origin_country = { var:crash_severity >= 60 } }
                            add = 10
                        }
                        modifier = {
                            trigger = { scope:banking_crash_origin_country = { var:crash_severity >= 30 var:crash_severity < 60 } }
                            add = 5
                        }
                        modifier = {
                            trigger = { is_in_customs_union_with = scope:banking_crash_origin_country }
                            add = 10
                        }
                        modifier = {
                            trigger = { is_adjacent_to_country = scope:banking_crash_origin_country }
                            add = 5
                        }
                        modifier = {
                            trigger = { economic_dependence = { target = scope:banking_crash_origin_country value > 0.10 } }
                            add = 5
                        }
                        modifier = {
                            trigger = { economic_dependence = { target = scope:banking_crash_origin_country value > 0.25 } }
                            add = 10
                        }
                        modifier = {
                            trigger = { market = { market_trade_reliance = { target = scope:banking_crash_origin_country.market value > 0.05 } } }
                            add = 5
                        }
                        modifier = {
                            trigger = { market = { market_trade_reliance = { target = scope:banking_crash_origin_country.market value > 0.1 } } }
                            add = 10
                        }
                        modifier = {
                            trigger = { scope:banking_crash_origin_country.relations:THIS >= relations_threshold:cordial }
                            add = 1.5
                        }
                        modifier = {
                            trigger = { has_law_or_variant = law_type:law_free_trade }
                            add = 1.5
                        }

                        scope:banking_crash_origin_country = {
                            save_scope_as = banking_crash_origin_country
                        }

                        # Contagion effects (value/momentum/bubble) now applied in event .7 options
                        # via apply_banking_contagion_effects, tiered by origin crash_severity.

                        trigger_event = { id = minor_events_timelineextended.7 popup = yes days = 7 }
                    }
                }
            }

            # Sync a human-readable "max" for UI text:
            set_variable = { name = banking_points_max_from_law value = banking_law_base_points_value }

            # --- Banking cycle random events (cooldown-gated) ---
            banking_cycle_random_event_effect = yes

            # --- Phase modifiers (remove previous; add current) ---
            remove_all_banking_phase_modifiers = yes

            # --- Bubble inertia: high speculation sustains expansion (positive feedback) ---
            # Applied as modifiers so players can see the effect in the modifier stack.
            if = {
                limit = { var:bubble_pressure >= 75 }
                add_modifier = { name = bubble_inertia_extreme }
            }
            else_if = {
                limit = { var:bubble_pressure >= 50 }
                add_modifier = { name = bubble_inertia_high }
            }
            else_if = {
                limit = { var:bubble_pressure >= 25 }
                add_modifier = { name = bubble_inertia_moderate }
            }

            if = {
                limit = { has_law = law_type:law_command_economy }
                if = { limit = { var:finance_cycle_value < 10 }  add_modifier = { name = financial_cycle_phase_panic_cmd } }
                else_if = { limit = { var:finance_cycle_value < 25 } add_modifier = { name = financial_cycle_phase_downturn_cmd } }
                else_if = { limit = { var:finance_cycle_value < 40 } add_modifier = { name = financial_cycle_phase_stagnation_cmd } }
                else_if = { limit = { var:finance_cycle_value < 60 } } # Stable = no national mod
                else_if = { limit = { var:finance_cycle_value < 75 } add_modifier = { name = financial_cycle_phase_expansion_cmd } }
                else_if = { limit = { var:finance_cycle_value < 88 } add_modifier = { name = financial_cycle_phase_boom_cmd } }
                else = { add_modifier = { name = financial_cycle_phase_frenzy_cmd } }
            }
            else_if = {
                limit = { has_law = law_type:law_cooperative_ownership }
                if = { limit = { var:finance_cycle_value < 10 }  add_modifier = { name = financial_cycle_phase_panic_coop } }
                else_if = { limit = { var:finance_cycle_value < 25 } add_modifier = { name = financial_cycle_phase_downturn_coop } }
                else_if = { limit = { var:finance_cycle_value < 40 } add_modifier = { name = financial_cycle_phase_stagnation_coop } }
                else_if = { limit = { var:finance_cycle_value < 60 } } # Stable = no national mod
                else_if = { limit = { var:finance_cycle_value < 75 } add_modifier = { name = financial_cycle_phase_expansion_coop } }
                else_if = { limit = { var:finance_cycle_value < 88 } add_modifier = { name = financial_cycle_phase_boom_coop } }
                else = { add_modifier = { name = financial_cycle_phase_frenzy_coop } }
            }
            else = {
                if = { limit = { var:finance_cycle_value < 10 }  add_modifier = { name = financial_cycle_phase_panic } }
                else_if = { limit = { var:finance_cycle_value < 25 } add_modifier = { name = financial_cycle_phase_downturn } }
                else_if = { limit = { var:finance_cycle_value < 40 } add_modifier = { name = financial_cycle_phase_stagnation } }
                else_if = { limit = { var:finance_cycle_value < 60 } } # Stable = no national mod
                else_if = { limit = { var:finance_cycle_value < 75 } add_modifier = { name = financial_cycle_phase_expansion } }
                else_if = { limit = { var:finance_cycle_value < 88 } add_modifier = { name = financial_cycle_phase_boom } }
                else = { add_modifier = { name = financial_cycle_phase_frenzy } }
            }

            scope:journal_entry = {
                set_bar_progress = { name = banking_cycle_value_bar value = ROOT.var:finance_cycle_value }
                set_bar_progress = { name = banking_bubble_pressure_bar value = ROOT.var:bubble_pressure }
                if = {
                    limit = { var:finance_cycle_momentum < -5 }
                    set_bar_progress = { name = banking_cycle_momentum_bar value = -5 }
                }
                else_if = {
                    limit = { var:finance_cycle_momentum > 5 }
                    set_bar_progress = { name = banking_cycle_momentum_bar value = 5 }
                }
                else = {
                    set_bar_progress = { name = banking_cycle_momentum_bar value = ROOT.var:finance_cycle_momentum }
                }
            }

            if = {
                limit = {
                    has_modifier = banking_capital_controls_out
                    NOT = {
                        OR = {
                            has_law = law_type:law_prudential_narrow_banking
                            has_law = law_type:law_directed_credit_development_banks
                            is_at_war = yes
                        }
                    }
                }
                remove_modifier = banking_capital_controls_out
            }
        }
    }
