je_environmental_crisis = {
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"
	group = je_group_environment

	modifiers_while_active = {
		environmental_crisis_active_modifier
	}

	is_shown_when_inactive = {
		has_game_rule = social_movements_enabled
		has_technology_researched = pollution_control
		NOT = { is_country_type = decentralized }
		NOT = {
			has_law = law_type:law_ministry_of_the_environment
		}
	}

	possible = {
		has_technology_researched = pollution_control
		NOT = {
			has_law = law_type:law_ministry_of_the_environment
		}
	}

	immediate = {
		# Grace period: industrialist failure condition cannot trigger for 5 years
		add_modifier = {
			name = env_crisis_reform_window
			days = 1825
		}
	}

	on_monthly_pulse = {
		random_events = {
			200 = 0
			10 = environmental_events.1
			10 = environmental_events.2
			10 = environmental_events.3
			10 = environmental_events.4
			10 = environmental_events.5
		}
	}

	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_environmental_status_no_ministry
				trigger = {
					has_law = law_type:law_no_ministry_of_the_environment
				}
			}
		}
	}

	complete = {
		has_law = law_type:law_ministry_of_the_environment
		institution_investment_level = {
			institution = institution_ministry_of_the_environment
			value >= 3
		}
	}

	on_complete = {
		trigger_event = { id = environmental_events.200 }
	}

	invalid = {
		is_country_type = decentralized
	}

	fail = {
		has_law = law_type:law_no_ministry_of_the_environment
		# Grace period must have expired (5 years after JE activation)
		NOT = { has_modifier = env_crisis_reform_window }
		# Industrialists have consolidated power and blocked reform
		ig:ig_industrialists = {
			is_in_government = yes
			ig_clout >= 0.25
		}
	}

	on_fail = {
		trigger_event = { id = environmental_events.100 }
	}

	timeout = 5475

	on_timeout = {
		add_radicals = {
			value = small_radicals
			pop_type = academics
		}
		add_modifier = {
			name = environmental_paralysis_modifier
			days = long_modifier_time
			is_decaying = yes
		}
	}

	weight = 100
	should_be_pinned_by_default = yes
}
