je_colonial_empire = {
	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	group = je_group_foreign_affairs

	is_shown_when_inactive = {
		has_game_rule = decolonization_enabled
		has_technology_researched = decolonization
	}

	possible = {
		# Has at least one overseas colonial state with unintegrated population
		any_scope_state = {
			is_overseas_colonial_state = yes
		}
	}

	should_be_pinned_by_default_uninvolved_or_context = yes

	scripted_progress_bar = colonial_stability_bar

	# Scripted buttons - policies the player/AI can toggle
	scripted_button = ce_invest_in_development
	scripted_button = ce_remove_invest_in_development
	scripted_button = ce_military_garrison
	scripted_button = ce_remove_military_garrison
	scripted_button = ce_cultural_assimilation
	scripted_button = ce_remove_cultural_assimilation
	scripted_button = ce_release_colonial_territory
	scripted_button = ce_round_table_conference
	scripted_button = ce_planned_decolonization

	immediate = {
	}

	progressbar = no

	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_colonial_empire_status_solidified
				trigger = {
					scope:journal_entry = { "scripted_bar_progress(colonial_stability_bar)" >= 90 }
				}
			}
			triggered_desc = {
				desc = je_colonial_empire_status_stable
				trigger = {
					scope:journal_entry = { "scripted_bar_progress(colonial_stability_bar)" >= 65 }
				}
			}
			triggered_desc = {
				desc = je_colonial_empire_status_strained
				trigger = {
					scope:journal_entry = { "scripted_bar_progress(colonial_stability_bar)" >= 40 }
				}
			}
			triggered_desc = {
				desc = je_colonial_empire_status_crumbling
				trigger = {
					scope:journal_entry = { "scripted_bar_progress(colonial_stability_bar)" >= 20 }
				}
			}
			triggered_desc = {
				desc = je_colonial_empire_status_collapsing
				trigger = { always = yes }
			}
		}
		first_valid = {
			triggered_desc = {
				desc = je_colonial_empire_pressure_none
				trigger = { colonial_gp_condemners_count = 0 }
			}
			triggered_desc = {
				desc = je_colonial_empire_pressure_low
				trigger = { colonial_gp_condemners_count <= 1 }
			}
			triggered_desc = {
				desc = je_colonial_empire_pressure_moderate
				trigger = { colonial_gp_condemners_count <= 3 }
			}
			triggered_desc = {
				desc = je_colonial_empire_pressure_high
				trigger = { colonial_gp_condemners_count <= 5 }
			}
			triggered_desc = {
				desc = je_colonial_empire_pressure_extreme
				trigger = { always = yes }
			}
		}
	}

	complete = {
		custom_tooltip = {
			text = je_colonial_empire_complete_tt
			scope:journal_entry = {
				"scripted_bar_progress(colonial_stability_bar)" >= 100
			}
		}
	}

	# Path-dependent resolution: which policy did the empire lean on?
	# Months tracked in vars colonial_invest_months / colonial_garrison_months /
	# colonial_assimilate_months by the monthly pulse below. The dispatch
	# picks the dominant path and triggers a flavor-matched event.
	on_complete = {
		if = {
			limit = {
				var:colonial_garrison_months >= var:colonial_invest_months
				var:colonial_garrison_months >= var:colonial_assimilate_months
				var:colonial_garrison_months > 24
			}
			trigger_event = { id = decolonization_events.203 }   # Iron Fist
		}
		else_if = {
			limit = {
				var:colonial_invest_months >= var:colonial_assimilate_months
				var:colonial_invest_months > 24
			}
			trigger_event = { id = decolonization_events.202 }   # Commonwealth Path
		}
		else_if = {
			limit = { var:colonial_assimilate_months > 24 }
			trigger_event = { id = decolonization_events.205 }   # Quiet Assimilation
		}
		else = {
			trigger_event = { id = decolonization_events.200 }   # generic Empire Endures
		}
	}

	fail = {
		scope:journal_entry = {
			"scripted_bar_progress(colonial_stability_bar)" <= 0
		}
	}

	# Path-dependent failure: did the empire bleed out (garrison-only) or
	# negotiate its way out (invest + assimilate)?
	on_fail = {
		if = {
			limit = {
				NAND = {
					var:colonial_garrison_months >= var:colonial_invest_months
					var:colonial_garrison_months >= var:colonial_assimilate_months
				}
				OR = {
					var:colonial_invest_months > 24
					var:colonial_assimilate_months > 24
				}
			}
			trigger_event = { id = decolonization_events.204 }   # Negotiated Withdrawal
		}
		else = {
			trigger_event = { id = decolonization_events.201 }   # The Empire Crumbles (bloody)
		}
	}

	on_monthly_pulse = {
		effect = {
			# --- Path-dependence tracking ---
			# Count months each policy is active. Read at on_complete / on_fail
			# to choose the appropriate resolution event.
			if = {
				limit = { NOT = { has_variable = colonial_invest_months } }
				set_variable = { name = colonial_invest_months value = 0 }
			}
			if = {
				limit = { NOT = { has_variable = colonial_garrison_months } }
				set_variable = { name = colonial_garrison_months value = 0 }
			}
			if = {
				limit = { NOT = { has_variable = colonial_assimilate_months } }
				set_variable = { name = colonial_assimilate_months value = 0 }
			}
			if = {
				limit = { has_modifier = colonial_development_investment_modifier }
				change_variable = { name = colonial_invest_months add = 1 }
			}
			if = {
				limit = { has_modifier = colonial_military_garrison_modifier }
				change_variable = { name = colonial_garrison_months add = 1 }
			}
			if = {
				limit = { has_modifier = colonial_cultural_assimilation_modifier }
				change_variable = { name = colonial_assimilate_months add = 1 }
			}

			# Phase modifiers based on current bar value (five status bands)
			je:je_colonial_empire = { remove_modifier = colonial_empire_solidified_modifier }
			je:je_colonial_empire = { remove_modifier = colonial_empire_stable_modifier }
			je:je_colonial_empire = { remove_modifier = colonial_empire_strained_modifier }
			je:je_colonial_empire = { remove_modifier = colonial_empire_under_pressure_modifier }
			je:je_colonial_empire = { remove_modifier = colonial_empire_crumbling_modifier }

			if = {
				limit = {
					scope:journal_entry = { "scripted_bar_progress(colonial_stability_bar)" < 20 }
				}
				je:je_colonial_empire = { add_modifier = { name = colonial_empire_crumbling_modifier } }
			}
			else_if = {
				limit = {
					scope:journal_entry = { "scripted_bar_progress(colonial_stability_bar)" < 40 }
				}
				je:je_colonial_empire = { add_modifier = { name = colonial_empire_under_pressure_modifier } }
			}
			else_if = {
				limit = {
					scope:journal_entry = { "scripted_bar_progress(colonial_stability_bar)" < 65 }
				}
				je:je_colonial_empire = { add_modifier = { name = colonial_empire_strained_modifier } }
			}
			else_if = {
				limit = {
					scope:journal_entry = { "scripted_bar_progress(colonial_stability_bar)" < 90 }
				}
				je:je_colonial_empire = { add_modifier = { name = colonial_empire_stable_modifier } }
			}
			else = {
				je:je_colonial_empire = { add_modifier = { name = colonial_empire_solidified_modifier } }
			}

			# Apply radicalism to civil rights movement specifically (not all movements)
			every_political_movement = {
				limit = { is_political_movement_type = movement_civil_rights }
				remove_modifier = colonial_pressure_movement_radicalism
				remove_modifier = colonial_crumbling_movement_radicalism
			}
			if = {
				limit = {
					scope:journal_entry = { "scripted_bar_progress(colonial_stability_bar)" < 20 }
				}
				every_political_movement = {
					limit = { is_political_movement_type = movement_civil_rights }
					add_modifier = { name = colonial_crumbling_movement_radicalism }
				}
			}
			else_if = {
				limit = {
					scope:journal_entry = { "scripted_bar_progress(colonial_stability_bar)" < 40 }
				}
				every_political_movement = {
					limit = { is_political_movement_type = movement_civil_rights }
					add_modifier = { name = colonial_pressure_movement_radicalism }
				}
			}
		}
	}

	weight = 500
	can_revolution_inherit = yes
	can_deactivate = no
}
