je_colonial_empire = {
	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	group = je_group_foreign_affairs

	is_shown_when_inactive = {
		has_game_rule = decolonization_enabled
		has_technology_researched = decolonization
	}

	possible = {
		# Direct overseas-state holders OR holders of qualifying colonial
		# subjects (Colonial Administration Colonies, Chartered Companies, and
		# overseas Protectorates of a different heritage group from the
		# metropole). The subject branch catches modern-era colonial relations
		# where the metropole owns no overseas states directly -- e.g. UK over
		# India, Netherlands over the DEIC, France over Tunisia/Morocco.
		OR = {
			any_scope_state = {
				is_overseas_colonial_state = yes
			}
			country_has_qualifying_colonial_subject = yes
		}
	}

	should_be_pinned_by_default_uninvolved_or_context = yes

	scripted_progress_bar = colonial_stability_bar

	# Scripted buttons - policies the player/AI can toggle
	scripted_button = ce_invest_in_development
	scripted_button = ce_remove_invest_in_development
	scripted_button = ce_military_garrison
	scripted_button = ce_remove_military_garrison
	scripted_button = ce_cultural_assimilation
	scripted_button = ce_remove_cultural_assimilation
	scripted_button = ce_release_colonial_territory
	scripted_button = ce_round_table_conference
	scripted_button = ce_planned_decolonization

	immediate = {
	}

	progressbar = no

	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_colonial_empire_status_solidified
				trigger = {
					scope:journal_entry = { "scripted_bar_progress(colonial_stability_bar)" >= 90 }
				}
			}
			triggered_desc = {
				desc = je_colonial_empire_status_stable
				trigger = {
					scope:journal_entry = { "scripted_bar_progress(colonial_stability_bar)" >= 65 }
				}
			}
			triggered_desc = {
				desc = je_colonial_empire_status_strained
				trigger = {
					scope:journal_entry = { "scripted_bar_progress(colonial_stability_bar)" >= 40 }
				}
			}
			triggered_desc = {
				desc = je_colonial_empire_status_crumbling
				trigger = {
					scope:journal_entry = { "scripted_bar_progress(colonial_stability_bar)" >= 20 }
				}
			}
			triggered_desc = {
				desc = je_colonial_empire_status_collapsing
				trigger = { always = yes }
			}
		}
		first_valid = {
			triggered_desc = {
				desc = je_colonial_empire_pressure_none
				trigger = { colonial_gp_condemners_count = 0 }
			}
			triggered_desc = {
				desc = je_colonial_empire_pressure_low
				trigger = { colonial_gp_condemners_count <= 1 }
			}
			triggered_desc = {
				desc = je_colonial_empire_pressure_moderate
				trigger = { colonial_gp_condemners_count <= 3 }
			}
			triggered_desc = {
				desc = je_colonial_empire_pressure_high
				trigger = { colonial_gp_condemners_count <= 5 }
			}
			triggered_desc = {
				desc = je_colonial_empire_pressure_extreme
				trigger = { always = yes }
			}
		}
		# Named-condemner roster. Slot variables are populated each pulse;
		# entries surface only the slots currently holding a country, so the
		# list grows and shrinks as great powers shift stance.
		triggered_desc = {
			desc = je_colonial_empire_condemners_header
			trigger = { has_variable = colonial_condemner_rank_1 }
		}
		triggered_desc = {
			desc = je_colonial_empire_condemner_rank_1_entry
			trigger = { has_variable = colonial_condemner_rank_1 }
		}
		triggered_desc = {
			desc = je_colonial_empire_condemner_rank_2_entry
			trigger = { has_variable = colonial_condemner_rank_2 }
		}
		triggered_desc = {
			desc = je_colonial_empire_condemner_rank_3_entry
			trigger = { has_variable = colonial_condemner_rank_3 }
		}
		triggered_desc = {
			desc = je_colonial_empire_condemner_rank_4_entry
			trigger = { has_variable = colonial_condemner_rank_4 }
		}
		triggered_desc = {
			desc = je_colonial_empire_condemner_rank_5_entry
			trigger = { has_variable = colonial_condemner_rank_5 }
		}
		triggered_desc = {
			desc = je_colonial_empire_condemner_rank_6_entry
			trigger = { has_variable = colonial_condemner_rank_6 }
		}
		triggered_desc = {
			desc = je_colonial_empire_condemner_rank_7_entry
			trigger = { has_variable = colonial_condemner_rank_7 }
		}
	}

	complete = {
		custom_tooltip = {
			text = je_colonial_empire_complete_tt
			OR = {
				# Basic completion: 60 months (5 years) sustained at the cap.
				# Reaching 100 is not enough on its own — the bar must hold.
				# Longer than the IF Act's 36-mo ≥90 gate so the decision path
				# stays meaningfully faster for countries that qualify.
				root = {
					has_variable = colonial_at_100_months
					var:colonial_at_100_months >= 60
				}
				# IF Act capstone — set by the Imperial Federation Act decision
				root = {
					has_variable = imperial_federation_taken
				}
			}
		}
	}

	# Path-dependent resolution: which policy did the empire lean on?
	# Months tracked in vars colonial_invest_months / colonial_garrison_months /
	# colonial_assimilate_months by the monthly pulse below. The dispatch
	# picks the dominant path and triggers a flavor-matched event.
	#
	# IF Act capstone routes (300/301) take priority — reaching the Solidified
	# capstone is the explicit "victory" path; it bypasses the bar-100 routing.
	on_complete = {
		if = {
			limit = { has_variable = imperial_federation_iron_fist }
			trigger_event = { id = decolonization_events.300 }   # Iron Fist Federation capstone
		}
		else_if = {
			limit = { has_variable = imperial_federation_civilizing }
			trigger_event = { id = decolonization_events.301 }   # Civilizing Mission capstone
		}
		else_if = {
			limit = {
				var:colonial_garrison_months >= var:colonial_invest_months
				var:colonial_garrison_months >= var:colonial_assimilate_months
				var:colonial_garrison_months > 24
			}
			trigger_event = { id = decolonization_events.203 }   # Iron Fist
		}
		else_if = {
			limit = {
				var:colonial_invest_months >= var:colonial_assimilate_months
				var:colonial_invest_months > 24
			}
			trigger_event = { id = decolonization_events.202 }   # Commonwealth Path
		}
		else_if = {
			limit = { var:colonial_assimilate_months > 24 }
			trigger_event = { id = decolonization_events.205 }   # Quiet Assimilation
		}
		else = {
			trigger_event = { id = decolonization_events.200 }   # generic Empire Endures
		}

		# Clean up persistent state from this JE run. Path-tracking variables,
		# country-scope policy modifiers, civil-rights movement modifiers, and
		# subject-side LD modifiers all need explicit removal -- they are
		# attached to entities that survive the JE's end and would otherwise
		# stick around indefinitely.
		colonial_empire_je_cleanup_effect = yes
	}

	fail = {
		custom_tooltip = {
			text = je_colonial_empire_fail_tt
			scope:journal_entry = {
				"scripted_bar_progress(colonial_stability_bar)" <= 0
			}
		}
	}

	# Third ending: player voluntarily released all colonies before bar hit zero.
	# Guard bar > 0 so this can't fire on the same tick as the collapse fail.
	fail = {
		custom_tooltip = {
			text = je_colonial_empire_voluntary_fail_tt
			NOT = {
				OR = {
					any_scope_state = { is_overseas_colonial_state = yes }
					country_has_qualifying_colonial_subject = yes
				}
			}
			scope:journal_entry = {
				"scripted_bar_progress(colonial_stability_bar)" > 0
			}
		}
	}

	# Path-dependent failure: did the empire bleed out (garrison-only),
	# negotiate its way out (invest + assimilate), or voluntarily decolonize?
	on_fail = {
		if = {
			limit = {
				scope:journal_entry = {
					"scripted_bar_progress(colonial_stability_bar)" > 0
				}
			}
			trigger_event = { id = decolonization_events.206 }   # Voluntary Decolonization
		}
		else_if = {
			limit = {
				NAND = {
					var:colonial_garrison_months >= var:colonial_invest_months
					var:colonial_garrison_months >= var:colonial_assimilate_months
				}
				OR = {
					var:colonial_invest_months > 24
					var:colonial_assimilate_months > 24
				}
			}
			trigger_event = { id = decolonization_events.204 }   # Negotiated Withdrawal
		}
		else = {
			trigger_event = { id = decolonization_events.201 }   # The Empire Crumbles (bloody)
		}

		# Subject-holders: instead of forced release, every qualifying
		# colonial subject takes a strong LD spike + relations hit. Subjects
		# don't auto-secede but become much harder to hold; the player has to
		# negotiate releases or weather the LD-driven independence pressure.
		# Direct-state release (where applicable) already happens via the
		# event-204/201 flavor branches and the existing form_decolonized
		# pathway.
		every_subject_or_below = {
			limit = { is_qualifying_colonial_subject = yes }
			add_modifier = {
				name = colonial_collapse_subject_unrest_modifier
				days = very_long_modifier_time
				is_decaying = no
			}
			change_relations = { country = ROOT value = -50 }
		}

		# Clean up persistent JE state. The collapse_subject_unrest_modifier
		# applied just above is intentionally preserved (failure consequence)
		# -- the cleanup helper removes the bar-band LD modifiers but not the
		# failure aftershock.
		colonial_empire_je_cleanup_effect = yes
	}

	on_monthly_pulse = {
		effect = {
			# --- Path-dependence tracking ---
			# Count months each policy is active. Read at on_complete / on_fail
			# to choose the appropriate resolution event.
			if = {
				limit = { NOT = { has_variable = colonial_invest_months } }
				set_variable = { name = colonial_invest_months value = 0 }
			}
			if = {
				limit = { NOT = { has_variable = colonial_garrison_months } }
				set_variable = { name = colonial_garrison_months value = 0 }
			}
			if = {
				limit = { NOT = { has_variable = colonial_assimilate_months } }
				set_variable = { name = colonial_assimilate_months value = 0 }
			}
			if = {
				limit = { NOT = { has_variable = colonial_solidified_months } }
				set_variable = { name = colonial_solidified_months value = 0 }
			}
			# JE total duration counter — read by long-duration home-front events
			# (e.g. "The Universities Speak" at >10 years).
			if = {
				limit = { NOT = { has_variable = colonial_je_total_months } }
				set_variable = { name = colonial_je_total_months value = 0 }
			}
			change_variable = { name = colonial_je_total_months add = 1 }
			if = {
				limit = { has_modifier = colonial_development_investment_modifier }
				change_variable = { name = colonial_invest_months add = 1 }
			}
			if = {
				limit = { has_modifier = colonial_military_garrison_modifier }
				change_variable = { name = colonial_garrison_months add = 1 }
			}
			if = {
				limit = { has_modifier = colonial_cultural_assimilation_modifier }
				change_variable = { name = colonial_assimilate_months add = 1 }
			}

			# --- Solidified-band sustained-time counter ---
			# Increments while bar is in Solidified band; resets if it drops.
			# The Imperial Federation Act capstone requires 36 sustained months.
			if = {
				limit = {
					scope:journal_entry = { "scripted_bar_progress(colonial_stability_bar)" >= 90 }
				}
				change_variable = { name = colonial_solidified_months add = 1 }
			}
			else = {
				set_variable = { name = colonial_solidified_months value = 0 }
			}

			# --- Bar-at-100 sustained-time counter ---
			# Increments while bar is at the cap; resets if it drops. The basic
			# completion path requires 36 months at 100 (mirrors the IF Act's
			# 36-mo ≥90 gate) so reaching the cap isn't an instant win and
			# leaves room to take the Imperial Federation Act decision.
			if = {
				limit = { NOT = { has_variable = colonial_at_100_months } }
				set_variable = { name = colonial_at_100_months value = 0 }
			}
			if = {
				limit = {
					scope:journal_entry = { "scripted_bar_progress(colonial_stability_bar)" >= 100 }
				}
				change_variable = { name = colonial_at_100_months add = 1 }
			}
			else = {
				set_variable = { name = colonial_at_100_months value = 0 }
			}

			# --- GDP-scaled monthly expense while Invest is active ---
			# Refreshes the dedicated expense modifier (country_expenses_add
			# = 1, scaled here by ~0.5% of GDP) so it shows up as a budget
			# line item rather than a silent treasury debit. A 1960 UK GDP
			# of ~£700M means roughly £3.5M/month; substantial but not
			# bankrupting. The modifier is removed by the policy's remove
			# button (ce_remove_invest_in_development); the per-pulse
			# re-application keeps the multiplier in sync with current GDP.
			if = {
				limit = { has_modifier = colonial_development_investment_modifier }
				remove_modifier = colonial_development_investment_expense_modifier
				add_modifier = {
					name = colonial_development_investment_expense_modifier
					multiplier = colonial_invest_monthly_expense
				}
			}

			# --- Pressuring great powers leaderboard ---
			# Populates colonial_condemner_rank_N slot variables so the
			# status_desc can show *who* is applying decolonization pressure
			# (not just the count). Order is by country_rank — the larger
			# the condemner, the more prominent in the list.
			remove_variable = colonial_condemner_rank_1
			remove_variable = colonial_condemner_rank_2
			remove_variable = colonial_condemner_rank_3
			remove_variable = colonial_condemner_rank_4
			remove_variable = colonial_condemner_rank_5
			remove_variable = colonial_condemner_rank_6
			remove_variable = colonial_condemner_rank_7
			set_variable = { name = colonial_condemner_idx value = 0 }
			ordered_country = {
				limit = {
					NOT = { THIS = ROOT }
					country_rank >= rank_value:great_power
					has_variable = gp_anti_colonial_stance
				}
				order_by = country_rank
				check_range_bounds = no
				save_temporary_scope_as = current_condemner
				if = {
					limit = { ROOT.var:colonial_condemner_idx < 1 }
					ROOT = { set_variable = { name = colonial_condemner_rank_1 value = scope:current_condemner } }
				}
				else_if = {
					limit = { ROOT.var:colonial_condemner_idx < 2 }
					ROOT = { set_variable = { name = colonial_condemner_rank_2 value = scope:current_condemner } }
				}
				else_if = {
					limit = { ROOT.var:colonial_condemner_idx < 3 }
					ROOT = { set_variable = { name = colonial_condemner_rank_3 value = scope:current_condemner } }
				}
				else_if = {
					limit = { ROOT.var:colonial_condemner_idx < 4 }
					ROOT = { set_variable = { name = colonial_condemner_rank_4 value = scope:current_condemner } }
				}
				else_if = {
					limit = { ROOT.var:colonial_condemner_idx < 5 }
					ROOT = { set_variable = { name = colonial_condemner_rank_5 value = scope:current_condemner } }
				}
				else_if = {
					limit = { ROOT.var:colonial_condemner_idx < 6 }
					ROOT = { set_variable = { name = colonial_condemner_rank_6 value = scope:current_condemner } }
				}
				else_if = {
					limit = { ROOT.var:colonial_condemner_idx < 7 }
					ROOT = { set_variable = { name = colonial_condemner_rank_7 value = scope:current_condemner } }
				}
				ROOT = { change_variable = { name = colonial_condemner_idx add = 1 } }
			}

			# --- State-side assimilation disruption reapplication ---
			# While the country has the assimilation modifier, every overseas
			# colonial state takes a recurring -5% goods output / -10% pop
			# qualifications hit. Forced cultural conversion stalls the local
			# economy. 2-month duration; reapplied each pulse so it sticks
			# while the policy is on, decays naturally when it ends.
			if = {
				limit = { has_modifier = colonial_cultural_assimilation_modifier }
				every_scope_state = {
					limit = { is_overseas_colonial_state = yes }
					add_modifier = {
						name = colonial_assimilation_disruption_modifier
						months = 2
					}
				}
				# Subject-state branch: cultural assimilation policy also
				# disrupts qualifying subjects' colonial states.
				every_subject_or_below = {
					limit = { is_qualifying_colonial_subject = yes }
					every_scope_state = {
						limit = { is_unintegrated_culturally_distinct_state = yes }
						add_modifier = {
							name = colonial_assimilation_disruption_modifier
							months = 2
						}
					}
				}
			}

			# Phase modifiers based on current bar value (five status bands)
			je:je_colonial_empire = { remove_modifier = colonial_empire_solidified_modifier }
			je:je_colonial_empire = { remove_modifier = colonial_empire_stable_modifier }
			je:je_colonial_empire = { remove_modifier = colonial_empire_strained_modifier }
			je:je_colonial_empire = { remove_modifier = colonial_empire_under_pressure_modifier }
			je:je_colonial_empire = { remove_modifier = colonial_empire_crumbling_modifier }

			if = {
				limit = {
					scope:journal_entry = { "scripted_bar_progress(colonial_stability_bar)" < 20 }
				}
				je:je_colonial_empire = { add_modifier = { name = colonial_empire_crumbling_modifier } }
			}
			else_if = {
				limit = {
					scope:journal_entry = { "scripted_bar_progress(colonial_stability_bar)" < 40 }
				}
				je:je_colonial_empire = { add_modifier = { name = colonial_empire_under_pressure_modifier } }
			}
			else_if = {
				limit = {
					scope:journal_entry = { "scripted_bar_progress(colonial_stability_bar)" < 65 }
				}
				je:je_colonial_empire = { add_modifier = { name = colonial_empire_strained_modifier } }
			}
			else_if = {
				limit = {
					scope:journal_entry = { "scripted_bar_progress(colonial_stability_bar)" < 90 }
				}
				je:je_colonial_empire = { add_modifier = { name = colonial_empire_stable_modifier } }
			}
			else = {
				je:je_colonial_empire = { add_modifier = { name = colonial_empire_solidified_modifier } }
			}

			# --- Subject-side LD pressure ---
			# Bar status pushes Liberty Desire onto qualifying colonial
			# subjects. Strained band -> mild LD bump; Crumbling -> strong;
			# Collapsing -> severe. Stable / Solidified -> no LD pressure.
			# Pressure is one-way: subjects' LD does not feed back into the
			# bar (the bar is driven by GP condemnation, governance, and
			# overreach terms in the bar's monthly_progress script).
			every_subject_or_below = {
				limit = { is_qualifying_colonial_subject = yes }
				remove_modifier = colonial_overlord_strained_subject_ld_modifier
				remove_modifier = colonial_overlord_crumbling_subject_ld_modifier
				remove_modifier = colonial_overlord_collapsing_subject_ld_modifier
			}
			if = {
				limit = {
					scope:journal_entry = { "scripted_bar_progress(colonial_stability_bar)" < 20 }
				}
				every_subject_or_below = {
					limit = { is_qualifying_colonial_subject = yes }
					add_modifier = { name = colonial_overlord_collapsing_subject_ld_modifier }
				}
			}
			else_if = {
				limit = {
					scope:journal_entry = { "scripted_bar_progress(colonial_stability_bar)" < 40 }
				}
				every_subject_or_below = {
					limit = { is_qualifying_colonial_subject = yes }
					add_modifier = { name = colonial_overlord_crumbling_subject_ld_modifier }
				}
			}
			else_if = {
				limit = {
					scope:journal_entry = { "scripted_bar_progress(colonial_stability_bar)" < 65 }
				}
				every_subject_or_below = {
					limit = { is_qualifying_colonial_subject = yes }
					add_modifier = { name = colonial_overlord_strained_subject_ld_modifier }
				}
			}
			# Stable (>=65) and Solidified (>=90): no LD pressure (modifiers
			# already removed at the top of this block).

			# Apply radicalism to civil rights movement specifically (not all movements)
			every_political_movement = {
				limit = { is_political_movement_type = movement_civil_rights }
				remove_modifier = colonial_pressure_movement_radicalism
				remove_modifier = colonial_crumbling_movement_radicalism
			}
			if = {
				limit = {
					scope:journal_entry = { "scripted_bar_progress(colonial_stability_bar)" < 20 }
				}
				every_political_movement = {
					limit = { is_political_movement_type = movement_civil_rights }
					add_modifier = { name = colonial_crumbling_movement_radicalism }
				}
			}
			else_if = {
				limit = {
					scope:journal_entry = { "scripted_bar_progress(colonial_stability_bar)" < 40 }
				}
				every_political_movement = {
					limit = { is_political_movement_type = movement_civil_rights }
					add_modifier = { name = colonial_pressure_movement_radicalism }
				}
			}
		}
	}

	weight = 500
	can_revolution_inherit = yes
	can_deactivate = no
}
