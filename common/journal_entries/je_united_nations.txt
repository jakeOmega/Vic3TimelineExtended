je_united_nations = {
	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	group = je_group_foreign_affairs

	is_shown_when_inactive = {
		has_game_rule = united_nations_enabled
		OR = {
			has_technology_researched = intergovernmental_organizations
			has_global_variable = un_founded
		}
	}

	possible = {
		OR = {
			has_technology_researched = intergovernmental_organizations
			has_global_variable = un_founded
		}
	}

	scripted_progress_bar = un_authority_bar

	# ---- Founding button (GP with tech, before UN exists) ----
	scripted_button = un_found_button

	# ---- Membership buttons ----
	scripted_button = un_join_button
	scripted_button = un_leave_button

	# ---- Request buttons (ask UN for help) ----
	scripted_button = un_request_peacekeepers_button
	scripted_button = un_request_humanitarian_aid_button
	scripted_button = un_propose_condemn_button

	# ---- Policy buttons (members only) ----
	scripted_button = un_propose_sanctions_button
	scripted_button = un_lift_sanctions_button
	scripted_button = un_peacekeeping_mission_button
	scripted_button = un_end_peacekeeping_button
	scripted_button = un_fund_development_button
	scripted_button = un_defund_development_button
	scripted_button = un_human_rights_resolution_button
	scripted_button = un_withdraw_human_rights_button
	scripted_button = un_arms_control_button
	scripted_button = un_withdraw_arms_control_button

	# ---- Great Power influence buttons ----
	scripted_button = un_champion_order_button
	scripted_button = un_stop_championing_button
	scripted_button = un_undermine_order_button
	scripted_button = un_stop_undermining_button

	immediate = {
		# Set initial UN authority to 50 (moderate legitimacy)
		set_variable = { name = un_authority value = 50 }
		scope:journal_entry = {
			set_bar_progress = { name = un_authority_bar value = 50 }
		}
	}

	complete = {
		always = no
	}

	progressbar = no
	progress_desc = un_authority_progress_desc

	status_desc = {
		# 1. Personal membership status
		first_valid = {
			triggered_desc = {
				desc = je_un_status_not_member
				trigger = { NOT = { je:je_united_nations ?= { has_modifier = un_member_modifier } } }
			}
			triggered_desc = {
				desc = je_un_status_member
				trigger = { je:je_united_nations ?= { has_modifier = un_member_modifier } }
			}
		}

		# 2. Authority tier
		first_valid = {
			triggered_desc = {
				desc = je_un_authority_collapsed
				trigger = { var:un_authority < 10 }
			}
			triggered_desc = {
				desc = je_un_authority_weak
				trigger = { var:un_authority >= 10 var:un_authority < 30 }
			}
			triggered_desc = {
				desc = je_un_authority_moderate
				trigger = { var:un_authority >= 30 var:un_authority < 60 }
			}
			triggered_desc = {
				desc = je_un_authority_strong
				trigger = { var:un_authority >= 60 var:un_authority < 85 }
			}
			triggered_desc = {
				desc = je_un_authority_dominant
				trigger = { var:un_authority >= 85 }
			}
		}

		# 2b. Authority drift details (members only)
		first_valid = {
			triggered_desc = {
				desc = je_un_authority_drift
				trigger = { je:je_united_nations ?= { has_modifier = un_member_modifier } }
			}
		}

		# 3. UN Headquarters
		first_valid = {
			triggered_desc = {
				desc = je_un_hq_host
				trigger = { je:je_united_nations ?= { has_modifier = un_headquarters_modifier } }
			}
			triggered_desc = {
				desc = je_un_hq_established
				trigger = { has_global_variable = un_hq_country }
			}
		}

		# 4. General Assembly statistics
		first_valid = {
			triggered_desc = {
				desc = je_un_statistics
				trigger = {
					any_country = {
						je:je_united_nations ?= { has_modifier = un_member_modifier }
						count >= 1
					}
				}
			}
		}

		# 5. Security Council
		first_valid = {
			triggered_desc = {
				desc = je_un_security_council_member
				trigger = { je:je_united_nations ?= { has_modifier = un_security_council_modifier } }
			}
			triggered_desc = {
				desc = je_un_security_council_nonmember
				trigger = {
					je:je_united_nations ?= { has_modifier = un_member_modifier }
					any_country = {
						je:je_united_nations ?= { has_modifier = un_security_council_modifier }
						count >= 1
					}
				}
			}
		}

		# 6. Specialized Agencies header
		first_valid = {
			triggered_desc = {
				desc = je_un_agencies_header
				trigger = { je:je_united_nations ?= { has_modifier = un_member_modifier } }
			}
		}

		# 6a-6g. Individual agency status (only shown when established)
		triggered_desc = {
			desc = je_un_agency_who_active
			trigger = { has_global_variable = un_agency_who }
		}
		triggered_desc = {
			desc = je_un_agency_unesco_active
			trigger = { has_global_variable = un_agency_unesco }
		}
		triggered_desc = {
			desc = je_un_agency_icj_active
			trigger = { has_global_variable = un_agency_icj }
		}
		triggered_desc = {
			desc = je_un_agency_unhrc_active
			trigger = { has_global_variable = un_agency_unhrc }
		}
		triggered_desc = {
			desc = je_un_agency_iaea_active
			trigger = { has_global_variable = un_agency_iaea }
		}
		triggered_desc = {
			desc = je_un_agency_unep_active
			trigger = { has_global_variable = un_agency_unep }
		}
		triggered_desc = {
			desc = je_un_agency_unhcr_active
			trigger = { has_global_variable = un_agency_unhcr }
		}
		triggered_desc = {
			desc = je_un_agency_unoosa_active
			trigger = { has_global_variable = un_agency_unoosa }
		}

		# 7. Active resolutions & program participation (members) / join prompt (non-members)
		first_valid = {
			triggered_desc = {
				desc = je_un_active_programs
				trigger = { je:je_united_nations ?= { has_modifier = un_member_modifier } }
			}
			triggered_desc = {
				desc = je_un_active_policies
				trigger = { always = yes }
			}
		}
	}

	on_monthly_pulse = {
		effect = {
			# ---- Security Council cleanup (all countries with JE) ----
			if = {
				limit = {
					je:je_united_nations ?= { has_modifier = un_security_council_modifier }
					OR = {
						NOT = { je:je_united_nations ?= { has_modifier = un_member_modifier } }
						NOT = { country_rank >= rank_value:great_power }
					}
				}
				je:je_united_nations ?= { remove_modifier = un_security_council_modifier }
			}

			# ---- Treaty obligation auto-enrollment ----
			if = {
				limit = {
					NOT = { je:je_united_nations ?= { has_modifier = un_member_modifier } }
					modifier:un_membership_obligation = yes
				}
				je:je_united_nations ?= { add_modifier = { name = un_member_modifier } }
				if = {
					limit = {
						country_rank >= rank_value:great_power
						NOT = {
							any_country = {
								je:je_united_nations ?= { has_modifier = un_security_council_modifier }
								count >= 5
							}
						}
					}
					je:je_united_nations ?= { add_modifier = { name = un_security_council_modifier } }
				}
			}

			# ---- Member-only logic ----
			if = {
				limit = { je:je_united_nations ?= { has_modifier = un_member_modifier } }

				# Base monthly drift: toward 50 (slow)
				if = {
					limit = { var:un_authority > 55 }
					change_variable = { name = un_authority add = -0.25 }
				}
				else_if = {
					limit = { var:un_authority < 45 }
					change_variable = { name = un_authority add = 0.25 }
				}

				# Boost authority if country has strong diplomacy
				if = {
					limit = {
						country_rank >= rank_value:great_power
						has_law = law_type:law_limited_war
					}
					change_variable = { name = un_authority add = 0.2 }
				}
				if = {
					limit = {
						country_rank >= rank_value:great_power
						has_law = law_type:law_humanitarian_regulations
					}
					change_variable = { name = un_authority add = 0.1 }
				}

				# Great Power toggle: championing adds authority monthly, scaled by rank
				if = {
					limit = { je:je_united_nations = { has_modifier = un_champion_order_cost } }
					change_variable = { name = un_authority add = 0.3 }
					if = {
						limit = { global_country_ranking < 3 }
						change_variable = { name = un_authority add = 0.2 }
					}
				}

				# Great Power toggle: undermining reduces authority monthly, scaled by rank
				if = {
					limit = { je:je_united_nations = { has_modifier = un_undermine_order_cost } }
					change_variable = { name = un_authority add = -0.3 }
					if = {
						limit = { global_country_ranking < 3 }
						change_variable = { name = un_authority add = -0.2 }
					}
				}

				# Wars involving members hurt authority
				if = {
					limit = {
						is_at_war = yes
						any_country = {
							has_war_with = ROOT
							je:je_united_nations ?= { has_modifier = un_member_modifier }
						}
					}
					change_variable = { name = un_authority add = -1 }
				}

				# Clamp 0-100
				if = {
					limit = { var:un_authority > 100 }
					set_variable = { name = un_authority value = 100 }
				}
				else_if = {
					limit = { var:un_authority < 0 }
					set_variable = { name = un_authority value = 0 }
				}

				# Sync progress bar
				scope:journal_entry = {
					set_bar_progress = { name = un_authority_bar value = root.var:un_authority }
				}

				# ---- Apply scaled UN membership modifier ----
				if = {
					limit = { var:un_authority >= 30 }
					je:je_united_nations = { remove_modifier = un_membership_benefits_modifier }
					je:je_united_nations = { add_modifier = {
						name = un_membership_benefits_modifier
						multiplier = {
							value = var:un_authority
							divide = 50
						} }
					}
				}
				else = {
					je:je_united_nations = { remove_modifier = un_membership_benefits_modifier }
				}

				# ---- High authority: infamy generation scales for everyone ----
				if = {
					limit = { var:un_authority >= 50 }
					je:je_united_nations = { remove_modifier = un_high_authority_infamy_modifier }
					je:je_united_nations = { add_modifier = {
						name = un_high_authority_infamy_modifier
						multiplier = {
							value = var:un_authority
							subtract = 40
							divide = 30
						} }
					}
				}
				else = {
					je:je_united_nations = { remove_modifier = un_high_authority_infamy_modifier }
				}

				# ---- NPT enforcement tiers (requires IAEA agency) ----
				# Authority 80+: Non-nuclear member powers face disarmament pressure
				if = {
					limit = {
						has_global_variable = un_agency_iaea
						var:un_authority >= 80
						NOT = { has_modifier = nuclear_power }
						NOT = { je:je_united_nations = { has_modifier = un_npt_disarmament_modifier } }
					}
					je:je_united_nations = { add_modifier = { name = un_npt_disarmament_modifier } }
				}
				# Remove if authority drops or they become nuclear
				if = {
					limit = {
						je:je_united_nations = { has_modifier = un_npt_disarmament_modifier }
						OR = {
							var:un_authority < 75
							has_modifier = nuclear_power
							NOT = { has_global_variable = un_agency_iaea }
						}
					}
					je:je_united_nations = { remove_modifier = un_npt_disarmament_modifier }
				}

				# ---- Security Council seat award ----
				if = {
					limit = {
						country_rank >= rank_value:great_power
						NOT = { je:je_united_nations ?= { has_modifier = un_security_council_modifier } }
						NOT = {
							any_country = {
								je:je_united_nations ?= { has_modifier = un_security_council_modifier }
								count >= 5
							}
						}
					}
					je:je_united_nations ?= { add_modifier = { name = un_security_council_modifier } }
				}
			}

			# ---- Non-member penalties at high authority (all countries with JE) ----
			if = {
				limit = {
					NOT = { je:je_united_nations ?= { has_modifier = un_member_modifier } }
					has_global_variable = un_founded
					# Check authority from any member (use global estimate)
					any_country = {
						je:je_united_nations ?= { has_modifier = un_member_modifier }
						has_variable = un_authority
						var:un_authority >= 60
					}
				}
				# Apply pariah status: non-members face diplomatic pressure
				if = {
					limit = { NOT = { je:je_united_nations = { has_modifier = un_nonmember_pariah_modifier } } }
					je:je_united_nations = { add_modifier = { name = un_nonmember_pariah_modifier } }
				}
			}
			# Remove if authority drops or UN not founded or they join
			if = {
				limit = {
					je:je_united_nations = { has_modifier = un_nonmember_pariah_modifier }
					OR = {
						je:je_united_nations ?= { has_modifier = un_member_modifier }
						NOT = { has_global_variable = un_founded }
						NOT = {
							any_country = {
								je:je_united_nations ?= { has_modifier = un_member_modifier }
								has_variable = un_authority
								var:un_authority >= 55
							}
						}
					}
				}
				je:je_united_nations = { remove_modifier = un_nonmember_pariah_modifier }
			}
		}
	}

	weight = 500
	should_be_pinned_by_default = yes
	can_revolution_inherit = yes
	can_deactivate = no
}
