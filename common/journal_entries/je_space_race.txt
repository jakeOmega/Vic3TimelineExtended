# ===========================================
# 1. SUBORBITAL FLIGHT
# Prereq: rocketry tech, GP/MP rank
# ===========================================
je_space_race_suborbital = {
	icon = "gfx/interface/icons/event_icons/event_industry.dds"
	group = je_group_foreign_affairs

	is_shown_when_inactive = {
		has_technology_researched = rocketry
		OR = {
			country_rank = rank_value:great_power
			country_rank = rank_value:major_power
		}
	}

	possible = {
		has_technology_researched = rocketry
		OR = {
			country_rank = rank_value:great_power
			country_rank = rank_value:major_power
		}
		NOT = { has_variable = sr_completed_suborbital }
		NOT = { has_variable = sr_active_suborbital }
	}

	immediate = {
		set_variable = { name = sr_active_suborbital value = yes }
		set_variable = { name = sr_active_milestone value = yes }
		set_variable = { name = sr_milestone_progress value = 0 }
		set_variable = { name = sr_funding_level value = 1 }
		add_modifier = {
			name = sr_space_program_funding
			multiplier = 1
		}
	}

	scripted_button = sr_select_safe_approach
	scripted_button = sr_select_ambitious_approach
	scripted_button = sr_increase_funding
	scripted_button = sr_decrease_funding

	current_value = {
		value = sr_current_milestone_progress
	}

	goal_add_value = {
		value = sr_suborbital_goal
	}

	progressbar = yes

	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_space_race_suborbital_status_ambitious
				trigger = { has_variable = sr_approach_ambitious }
			}
			triggered_desc = {
				desc = je_space_race_suborbital_status_safe
				trigger = { has_variable = sr_approach_safe }
			}
			triggered_desc = {
				desc = je_space_race_suborbital_status_default
				trigger = { always = yes }
			}
		}
	}

	complete = {
		var:sr_milestone_progress >= sr_suborbital_goal
	}

	on_complete = {
		trigger_event = { id = space_race_events.1 }
		sr_complete_suborbital_effect = yes
		remove_variable = sr_active_suborbital
	}

	on_monthly_pulse = {
		effect = {
			# Tick down failure cooldown
			if = {
				limit = { sr_has_failure_cooldown = yes }
				change_variable = { name = sr_failure_cooldown subtract = 1 }
			}
			# Progress based on approach
			if = {
				limit = { has_variable = sr_approach_ambitious }
				change_variable = {
					name = sr_milestone_progress
					add = sr_ambitious_progress
				}
				# Failure check for ambitious approach
				if = {
					limit = {
						NOT = { sr_has_failure_cooldown = yes }
					}
					random_list = {
						90 = { }
						10 = {
							trigger_event = { id = space_race_events.10 }
						}
					}
				}
			}
			else_if = {
				limit = { has_variable = sr_approach_safe }
				change_variable = {
					name = sr_milestone_progress
					add = sr_safe_progress
				}
				# Very small failure chance even for safe
				if = {
					limit = {
						NOT = { sr_has_failure_cooldown = yes }
					}
					random_list = {
						98 = { }
						2 = {
							trigger_event = { id = space_race_events.11 }
						}
					}
				}
			}
			else = {
				# Default (no approach selected) - minimal progress
				change_variable = {
					name = sr_milestone_progress
					add = 0.5
				}
			}
		}
	}

	weight = 100
	should_be_pinned_by_default = yes
	can_revolution_inherit = yes
	can_deactivate = yes
}

	icon = "gfx/interface/icons/event_icons/event_industry.dds"
	group = je_group_foreign_affairs

	is_shown_when_inactive = {
		has_variable = sr_completed_probe
		has_variable = sr_completed_mars_landing
		has_technology_researched = space_colonization
	}

	possible = {
		has_variable = sr_completed_probe
		has_variable = sr_completed_mars_landing
		has_technology_researched = space_colonization
		NOT = { has_variable = sr_completed_interstellar_probe }
		NOT = { has_variable = sr_active_interstellar_probe }
		OR = {
			country_rank = rank_value:great_power
			country_rank = rank_value:major_power
		}
	}

	immediate = {
		set_variable = { name = sr_active_interstellar_probe value = yes }
		set_variable = { name = sr_active_milestone value = yes }
		set_variable = { name = sr_milestone_progress value = 0 }
		if = {
			limit = { NOT = { has_variable = sr_funding_level } }
			set_variable = { name = sr_funding_level value = 1 }
		}
	}

	scripted_button = sr_select_safe_approach
	scripted_button = sr_select_ambitious_approach
	scripted_button = sr_increase_funding
	scripted_button = sr_decrease_funding

	current_value = {
		value = sr_current_milestone_progress
	}

	goal_add_value = {
		value = sr_interstellar_probe_goal
	}

	progressbar = yes

	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_space_race_interstellar_probe_status_ambitious
				trigger = { has_variable = sr_approach_ambitious }
			}
			triggered_desc = {
				desc = je_space_race_interstellar_probe_status_safe
				trigger = { has_variable = sr_approach_safe }
			}
			triggered_desc = {
				desc = je_space_race_interstellar_probe_status_default
				trigger = { always = yes }
			}
		}
	}

	complete = {
		var:sr_milestone_progress >= sr_interstellar_probe_goal
	}

	on_complete = {
		trigger_event = { id = space_race_events.8 }
		sr_complete_interstellar_probe_effect = yes
		remove_variable = sr_active_interstellar_probe
	}

	on_monthly_pulse = {
		effect = {
			if = {
				limit = { sr_has_failure_cooldown = yes }
				change_variable = { name = sr_failure_cooldown subtract = 1 }
			}
			if = {
				limit = { has_variable = sr_approach_ambitious }
				change_variable = {
					name = sr_milestone_progress
					add = sr_ambitious_progress
				}
				if = {
					limit = { NOT = { sr_has_failure_cooldown = yes } }
					random_list = {
						82 = { }
						18 = {
							trigger_event = { id = space_race_events.10 }
						}
					}
				}
			}
			else_if = {
				limit = { has_variable = sr_approach_safe }
				change_variable = {
					name = sr_milestone_progress
					add = sr_safe_progress
				}
				if = {
					limit = { NOT = { sr_has_failure_cooldown = yes } }
					random_list = {
						94 = { }
						6 = {
							trigger_event = { id = space_race_events.11 }
						}
					}
				}
			}
			else = {
				change_variable = {
					name = sr_milestone_progress
					add = 0.5
				}
			}
		}
	}

	weight = 90
	should_be_pinned_by_default = yes
	can_revolution_inherit = yes
	can_deactivate = yes
}

# 7b. INTERSTELLAR PROBE - AWAITING RESULTS
# Passive JE tracking probe transit to Alpha Centauri
# ~11 years (132 months) transit + return signal
# ===========================================
je_space_race_interstellar_results = {
	icon = "gfx/interface/icons/event_icons/event_industry.dds"
	group = je_group_foreign_affairs

	is_shown_when_inactive = {
		has_variable = sr_probe_launched
	}

	possible = {
		has_variable = sr_probe_launched
		NOT = { has_variable = sr_interstellar_results_received }
		NOT = { has_variable = sr_active_interstellar_results }
	}

	immediate = {
		set_variable = { name = sr_active_interstellar_results value = yes }
		set_variable = { name = sr_interstellar_transit_progress value = 0 }
	}

	current_value = {
		value = sr_interstellar_transit_value
	}

	goal_add_value = {
		value = 132
	}

	progressbar = yes

	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_space_race_interstellar_results_status
				trigger = { always = yes }
			}
		}
	}

	complete = {
		var:sr_interstellar_transit_progress >= 132
	}

	on_complete = {
		trigger_event = { id = space_race_events.60 }
		set_variable = { name = sr_interstellar_results_received value = yes }
		remove_variable = sr_active_interstellar_results
		add_modifier = { name = sr_interstellar_data }
	}

	on_monthly_pulse = {
		effect = {
			change_variable = { name = sr_interstellar_transit_progress add = 1 }
		}
	}

	weight = 100
	should_be_pinned_by_default = yes
	can_revolution_inherit = yes
	can_deactivate = yes
}

# ===========================================
# 8. SOLAR SYSTEM COLONIZATION (Staged, Repeatable)
# Prereq: Moon Base + Mars Landing + space_colonization tech
# 5 stages of colony establishment across the solar system
# Stage 1: Mars & Asteroids
# Stage 2: Jupiter System & Venus
# Stage 3: Mercury & Saturn System
# Stage 4: Uranus & Neptune Systems
# Stage 5: Kuiper Belt & Beyond
# ===========================================
je_space_race_solar_colonization = {
	icon = "gfx/interface/icons/event_icons/event_industry.dds"
	group = je_group_foreign_affairs

	is_shown_when_inactive = {
		has_variable = sr_completed_moon_base
		has_variable = sr_completed_mars_landing
		has_technology_researched = space_colonization
	}

	possible = {
		has_variable = sr_completed_moon_base
		has_variable = sr_completed_mars_landing
		has_technology_researched = space_colonization
		NOT = { has_variable = sr_active_solar_colonization }
		NOT = { has_global_variable = sr_all_colonies_complete }
		OR = {
			country_rank = rank_value:great_power
			country_rank = rank_value:major_power
		}
	}

	immediate = {
		set_variable = { name = sr_active_solar_colonization value = yes }
		set_variable = { name = sr_active_milestone value = yes }
		set_variable = { name = sr_milestone_progress value = 0 }
		if = {
			limit = { NOT = { has_global_variable = sr_colonization_stage } }
			set_global_variable = { name = sr_colonization_stage value = 1 }
		}
		if = {
			limit = { NOT = { has_variable = sr_colony_count } }
			set_variable = { name = sr_colony_count value = 0 }
		}
		if = {
			limit = { NOT = { has_variable = sr_funding_level } }
			set_variable = { name = sr_funding_level value = 1 }
		}
	}

	scripted_button = sr_select_safe_approach
	scripted_button = sr_select_ambitious_approach
	scripted_button = sr_increase_funding
	scripted_button = sr_decrease_funding

	current_value = {
		value = sr_current_milestone_progress
	}

	goal_add_value = {
		value = sr_solar_colonization_goal
	}

	progressbar = yes

	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_space_race_solar_colonization_status_stage1
				trigger = {
					OR = {
						NOT = { has_global_variable = sr_colonization_stage }
						global_var:sr_colonization_stage = 1
					}
				}
			}
			triggered_desc = {
				desc = je_space_race_solar_colonization_status_stage2
				trigger = { global_var:sr_colonization_stage = 2 }
			}
			triggered_desc = {
				desc = je_space_race_solar_colonization_status_stage3
				trigger = { global_var:sr_colonization_stage = 3 }
			}
			triggered_desc = {
				desc = je_space_race_solar_colonization_status_stage4
				trigger = { global_var:sr_colonization_stage = 4 }
			}
			triggered_desc = {
				desc = je_space_race_solar_colonization_status_stage5
				trigger = { global_var:sr_colonization_stage = 5 }
			}
		}
	}

	complete = {
		var:sr_milestone_progress >= sr_solar_colonization_goal
	}

	on_complete = {
		# Establish a colony at a random unclaimed location in the current stage
		sr_establish_colony_effect = yes
		remove_variable = sr_active_solar_colonization
		sr_complete_milestone_effect = yes
		# Notify with generic completion event
		trigger_event = { id = space_race_events.9 }
		# If all 5 stages exhausted, mark fully complete
		if = {
			limit = { has_global_variable = sr_all_colonies_complete }
			set_variable = { name = sr_completed_solar_colonization value = yes }
			add_modifier = { name = sr_solar_system_trade }
		}
	}

	on_monthly_pulse = {
		effect = {
			if = {
				limit = { sr_has_failure_cooldown = yes }
				change_variable = { name = sr_failure_cooldown subtract = 1 }
			}
			if = {
				limit = { has_variable = sr_approach_ambitious }
				change_variable = {
					name = sr_milestone_progress
					add = sr_ambitious_progress
				}
				if = {
					limit = { NOT = { sr_has_failure_cooldown = yes } }
					random_list = {
						80 = { }
						20 = {
							trigger_event = { id = space_race_events.10 }
						}
					}
				}
			}
			else_if = {
				limit = { has_variable = sr_approach_safe }
				change_variable = {
					name = sr_milestone_progress
					add = sr_safe_progress
				}
				if = {
					limit = { NOT = { sr_has_failure_cooldown = yes } }
					random_list = {
						94 = { }
						6 = {
							trigger_event = { id = space_race_events.11 }
						}
					}
				}
			}
			else = {
				change_variable = {
					name = sr_milestone_progress
					add = 0.5
				}
			}
		}
	}

	weight = 85
	should_be_pinned_by_default = yes
	can_revolution_inherit = yes
	can_deactivate = yes
}
