# ===========================================
# 1. SUBORBITAL FLIGHT
# Prereq: rocketry tech, GP/MP rank
# ===========================================
je_space_race_suborbital = {
	icon = "gfx/interface/icons/event_icons/event_industry.dds"
	group = je_group_foreign_affairs

	is_shown_when_inactive = {
		has_game_rule = space_race_enabled
		has_technology_researched = rocketry
		OR = {
			country_rank = rank_value:great_power
			country_rank = rank_value:major_power
		}
	}

	possible = {
		has_technology_researched = rocketry
		modifier:country_sr_earth_orbit_program_bool = yes
		OR = {
			country_rank = rank_value:great_power
			country_rank = rank_value:major_power
		}
		NOT = { has_variable = sr_completed_suborbital }
	}

	immediate = {
		set_variable = { name = sr_active_suborbital value = yes }
		set_variable = { name = sr_active_milestone value = yes }
		set_variable = { name = sr_progress_suborbital value = 0 }
		set_variable = { name = sr_funding_suborbital value = 1 }
		sr_recalculate_cost = yes
	}

	scripted_button = sr_btn_safe_suborbital
	scripted_button = sr_btn_ambitious_suborbital
	scripted_button = sr_btn_fund_up_suborbital
	scripted_button = sr_btn_fund_down_suborbital

	current_value = {
		value = sr_progress_suborbital_value
	}

	goal_add_value = {
		value = sr_suborbital_goal
	}

	progressbar = yes

	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_space_race_suborbital_status_ambitious
				trigger = { has_variable = sr_ambitious_suborbital }
			}
			triggered_desc = {
				desc = je_space_race_suborbital_status_safe
				trigger = { has_variable = sr_safe_suborbital }
			}
			triggered_desc = {
				desc = je_space_race_suborbital_status_default
				trigger = { always = yes }
			}
		}
	}

	complete = {
		has_variable = sr_progress_suborbital
		var:sr_progress_suborbital >= sr_suborbital_goal
	}

	on_complete = {
		trigger_event = { id = space_race_events.1 }
		sr_complete_suborbital_effect = yes
		remove_variable = sr_active_suborbital
	}

	fail = {
		custom_tooltip = {
			text = sr_space_program_incompatible_tt
			NOT = { modifier:country_sr_earth_orbit_program_bool = yes }
		}
	}

	on_fail = {
		if = { limit = { has_variable = sr_active_suborbital } remove_variable = sr_active_suborbital }
		if = { limit = { has_variable = sr_progress_suborbital } remove_variable = sr_progress_suborbital }
		if = { limit = { has_variable = sr_funding_suborbital } remove_variable = sr_funding_suborbital }
		if = { limit = { has_variable = sr_safe_suborbital } remove_variable = sr_safe_suborbital }
		if = { limit = { has_variable = sr_ambitious_suborbital } remove_variable = sr_ambitious_suborbital }
		sr_recalculate_cost = yes
	}

	on_monthly_pulse = {
		effect = {
			# Set funding proxy for script value calculation
			set_variable = { name = sr_funding_level value = var:sr_funding_suborbital }
			# Progress based on approach
			if = {
				limit = { has_variable = sr_ambitious_suborbital }
				change_variable = {
					name = sr_progress_suborbital
					add = sr_ambitious_progress
				}
				# Failure check for ambitious approach
				if = {
					limit = {
						NOT = { sr_has_failure_cooldown = yes }
					}
					random_list = {
						95 = { }
						5 = {
							change_variable = { name = sr_progress_suborbital multiply = 0.75 }
							set_variable = { name = sr_failed_suborbital value = yes }
							sr_fire_ambitious_failure_event = yes
						}
					}
				}
			}
			else_if = {
				limit = { has_variable = sr_safe_suborbital }
				change_variable = {
					name = sr_progress_suborbital
					add = sr_safe_progress
				}
				# Very small failure chance even for safe
				if = {
					limit = {
						NOT = { sr_has_failure_cooldown = yes }
					}
					random_list = {
						98 = { }
						2 = {
							change_variable = { name = sr_progress_suborbital multiply = 0.85 }
							set_variable = { name = sr_failed_suborbital value = yes }
							sr_fire_safe_setback_event = yes
						}
					}
				}
			}
			else = {
				# Default (no approach selected) - minimal progress
				change_variable = {
					name = sr_progress_suborbital
					add = 0.5
				}
			}
		}
	}

	weight = 100
	should_be_pinned_by_default = yes
	can_revolution_inherit = yes
	can_deactivate = yes
}
