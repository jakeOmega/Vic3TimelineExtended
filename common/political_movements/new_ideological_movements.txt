movement_environmental = {
    category = movement_category_ideological

    ideology = ideology_environmentalists
    character_ideologies = {
        ideology_environmentalists
    }
    
    disband_trigger = {
        # Conditions under which the movement disbands
        always = no
    }
    on_disbanded = {
        # Actions to perform when the movement disbands
    }
    
    creation_trigger = {
        # Conditions under which the movement is created
        # For example, when environmental degradation reaches a certain level
        NOT = { has_law = law_type:law_ministry_of_the_environment }
        has_technology_researched = pollution_control
    }
    creation_weight = {
        value = 100
    }    
    on_created = {
        add_movement_enthusiasm_modifier = yes
    }
    
    character_support_trigger = {
        OR = {
            has_ideology = ideology:ideology_environmentalists
            OR = {
                is_interest_group_type = ig_intelligentsia
            }
        }
    }
    character_support_weight = {
        value = 200
        if = {
            limit = {
                has_ideology = ideology:ideology_environmentalists
            }
            multiply = {
                value = 5
            }
        }
    }
    
    pop_support_factors = {
        # Factors influencing popular support
        movement_support_high_pollution
		movement_support_high_literacy
        movement_support_academics
    }    

    pop_support_weight = {
		add = {
			value = 3
			desc = "POP_BASE_SUPPORT"
		}

        add = {
			desc = "POLLUTED_STATE"
            value = state.state_region.pollution_amount
			divide = 255 # max pollution
			multiply = 20
            if = {
                limit = { state.state_region.pollution_amount > 200 }
                multiply = 2
            }
        }

        if = {
            limit = {
                strata = upper
            }
            if = {
                limit = {
                    is_pop_type = capitalists
                }
                multiply = {
                    value = 0.1
                    desc = "POP_CAPITALISTS"
                }
            }
            else = {
                multiply = {
                    value = 0.2
                    desc = "UPPER_CLASS"
                }
            }
        }
        else_if = {
            limit = {
                strata = middle
            }
            if = {
                limit = {
                    is_pop_type = academics
                }
                multiply = {
                    value = 5
                    desc = "POP_ACADEMICS"
                }
            }
            else_if = {
                limit = {
                    is_pop_type = clergymen
                }
                multiply = {
                    value = 0.5
                    desc = "POP_CLERGYMEN"
                }
            }
            else = {
                multiply = {
                    value = 2
                    desc = "MIDDLE_CLASS"
                }
            }
        }
        else_if = {
            limit = {
                strata = lower
            }
			multiply = {
				value = 0.5
				desc = "LOWER_CLASS"
			}
        }

        multiply = { # Illiterate pops participate at 1/10 the rate of fully literate pops
			desc = "POP_LITERACY"
			value = literacy_rate
			multiply = 0.9
			add = 0.10
		}

		if = {
			limit = { owner = { has_technology_researched = environmental_movement } }
			multiply = {
				value = 2
				desc = "TECHNOLOGY_RESEARCHED_environmental_movement"
			}
		}
		if = {
			limit = { owner = { has_technology_researched = genetic_engineering } }
			multiply = {
				value = 1.25
				desc = "TECHNOLOGY_RESEARCHED_genetic_engineering"
			}
		}
        if = {
            limit = { temperature_anomaly_display >= 2 }
            add = {
                value = 50
                desc = "TEMPERATURE_ANOMALY_2"
            }
        }
        else_if = {
            limit = { temperature_anomaly_display >= 1.5 }
            add = {
                value = 25
                desc = "TEMPERATURE_ANOMALY_1.5"
            }
        }
        else_if = {
            limit = { temperature_anomaly_display >= 1.0 }
            add = {
                value = 10
                desc = "TEMPERATURE_ANOMALY_1.0"
            }
        }
        else_if = {
            limit = { temperature_anomaly_display >= 0.5 }
            add = {
                value = 5
                desc = "TEMPERATURE_ANOMALY_0.5"
            }
        }
    }

    revolution = {
        possible = {
            political_movement_support >= 0.05
        }
        
        weight = {
            value = 100
        }
    
        state_weight = {
            value = 1
        }
    }

    secession = {
        possible = {
            political_movement_support >= 0.05
        }    
    
        weight = {
            value = 1
        }  

        state_weight = {
            value = 1          
        }        
    }    
    
    additional_radicalism_factors = {
        owner = {
			if = {
				limit = { has_institution = institution_ministry_of_the_environment }
				
				add = {
					value = modifier:country_institution_ministry_of_the_environment_max_investment_add
					subtract = "institution_investment_level(institution_ministry_of_the_environment)"
					multiply = 0.05
					desc = "INSTITUTION_FUNDING_LEVEL_ministry_of_the_environment"
				}
			}
			else = {
				add = {
					value = modifier:country_institution_ministry_of_the_environment_max_investment_add
                    multiply = 0.05
					desc = "NO_INSTITUTION_ministry_of_the_environment"
				}		
			}

            # Radicalism adjustments based on climate policy availability/activation
            # For each policy: -5 if active, +5 if the matching button is possible but not active
            if = { # Carbon tax
                limit = { has_journal_entry = je_global_warming je:je_global_warming = { has_modifier = carbon_tax_modifier } }
                add = { value = -5 desc = "POLICY_ACTIVE_CARBON_TAX" }
            }
            else_if = {
                limit = {
                    temperature_anomaly_display >= 0.5
                    market_capital.owner = THIS
                }
                add = { value = 5 desc = "POLICY_POSSIBLE_CARBON_TAX" }
            }

            if = { # Renewable investment
                limit = { has_journal_entry = je_global_warming je:je_global_warming = { has_modifier = renewable_investment_modifier } }
                add = { value = -5 desc = "POLICY_ACTIVE_RENEWABLE_INVESTMENT" }
            }
            else_if = {
                limit = {
                    temperature_anomaly_display >= 0.5
                    market_capital.owner = THIS
                }
                add = { value = 5 desc = "POLICY_POSSIBLE_RENEWABLE_INVESTMENT" }
            }

            if = { # Climate adaptation
                limit = { has_journal_entry = je_global_warming je:je_global_warming = { has_modifier = climate_adaptation_modifier } }
                add = { value = -5 desc = "POLICY_ACTIVE_CLIMATE_ADAPTATION" }
            }
            else_if = {
                limit = {
                    adjusted_produced_authority > 250
                    temperature_anomaly_display >= 0.5
                }
                add = { value = 5 desc = "POLICY_POSSIBLE_CLIMATE_ADAPTATION" }
            }

            if = { # Emission standards
                limit = { has_journal_entry = je_global_warming je:je_global_warming = { has_modifier = emission_standards_modifier } }
                add = { value = -5 desc = "POLICY_ACTIVE_EMISSION_STANDARDS" }
            }
            else_if = {
                limit = {
                    temperature_anomaly_display >= 0.5
                    market_capital.owner = THIS
                    adjusted_produced_authority > 200
                }
                add = { value = 5 desc = "POLICY_POSSIBLE_EMISSION_STANDARDS" }
            }

            if = { # Reforestation subsidies
                limit = { has_journal_entry = je_global_warming je:je_global_warming = { has_modifier = reforestation_subsidies_modifier } }
                add = { value = -5 desc = "POLICY_ACTIVE_REFORESTATION_SUBSIDIES" }
            }
            else_if = {
                limit = {
                    temperature_anomaly_display >= 0.5
                    adjusted_produced_authority > 100
                }
                add = { value = 5 desc = "POLICY_POSSIBLE_REFORESTATION_SUBSIDIES" }
            }

            if = { # Public transit
                limit = { has_journal_entry = je_global_warming je:je_global_warming = { has_modifier = public_transit_modifier } }
                add = { value = -5 desc = "POLICY_ACTIVE_PUBLIC_TRANSIT" }
            }
            else_if = {
                limit = {
                    temperature_anomaly_display >= 0.5
                    adjusted_produced_authority > 150
                }
                add = { value = 5 desc = "POLICY_POSSIBLE_PUBLIC_TRANSIT" }
            }

            if = { # Fossil fuel divestment
                limit = { has_journal_entry = je_global_warming je:je_global_warming = { has_modifier = fossil_fuel_divestment_modifier } }
                add = { value = -5 desc = "POLICY_ACTIVE_FOSSIL_FUEL_DIVESTMENT" }
            }
            else_if = {
                limit = {
                    temperature_anomaly_display >= 1.0
                    adjusted_produced_authority > 200
                }
                add = { value = 5 desc = "POLICY_POSSIBLE_FOSSIL_FUEL_DIVESTMENT" }
            }
		}
    }
}
