# =================================================================
# Resettlement Production Methods
# =================================================================
# Two-building system:
#   SOURCE (Resettlement Camp) - employs administrative staff (bureaucrats,
#     soldiers, officers) who are NOT moved, plus settlers (laborers, farmers,
#     peasants) who ARE moved. The on-action uses state_resettlement_transfer_add
#     to determine monthly transfer volume and filters to movable pop types.
#   DESTINATION (Resettlement Colony) - administrators and infrastructure
#     that receive incoming settlers and provide migration pull.
#
# Transfer volume is shown via the state_resettlement_transfer_add modifier
# so the player can see exactly how many people move per month.
# =================================================================

# =====================================================================
# SOURCE BUILDING PMs: Resettlement Camp (pmg_resettlement_method)
# =====================================================================
# Tradeoff summary:
#   Homesteading:        Cheap, slow (50/lvl), no negatives. Early game.
#   Organized Colonization: Moderate cost, moderate speed (100/lvl). Era 3.
#   Forced Resettlement: Expensive (arms), fast (200/lvl), radicals + mortality. Era 5.
#   Penal Transportation: Cheap, moderate (75/lvl), high mortality. Requires auth laws.
#   Incentivized Relocation: Expensive (services), fast (150/lvl), no negatives. Era 6.
# =====================================================================

# Voluntary: government offers free land and subsidies to recruit settlers
# Cheap and conflict-free, but slow throughput
pm_homesteading_program = {
	texture = "gfx/interface/icons/production_method_icons/homesteading.dds"

	country_modifiers = {
		workforce_scaled = {
			country_bureaucracy_add = -5
		}
	}

	state_modifiers = {
		workforce_scaled = {
			# 50 people moved per building level per month (~600/year)
			state_resettlement_transfer_add = 50
		}
	}

	building_modifiers = {
		workforce_scaled = {
			goods_input_paper_add = 10
			goods_input_services_add = 5
		}
		level_scaled = {
			# Administrative staff (not moved)
			building_employment_bureaucrats_add = 100
			# Settlers being processed (moved to destination)
			building_employment_laborers_add = 50
		}
	}

	ai_value = 100
}

# Coercive: state-directed population transfers enforced by military/police
# The fastest method but generates radicals and increases mortality
pm_forced_resettlement = {
	texture = "gfx/interface/icons/production_method_icons/ownership_bureacrats.dds"

	unlocking_technologies = {
		mass_propaganda
	}

	# Cannot use forced methods under liberal security laws
	disallowing_laws = {
		law_guaranteed_liberties
	}

	country_modifiers = {
		workforce_scaled = {
			country_bureaucracy_add = -10
		}
	}

	state_modifiers = {
		workforce_scaled = {
			# 200 people moved per building level per month (~2400/year)
			state_resettlement_transfer_add = 200
			# Forced roundups cause social disruption in the source state
			state_radicals_from_political_movements_mult = 0.25
			state_mortality_mult = 0.02
		}
	}

	building_modifiers = {
		workforce_scaled = {
			goods_input_paper_add = 10
			goods_input_small_arms_add = 10
			goods_input_ammunition_add = 5
		}
		level_scaled = {
			# Military staff running the operation (not moved)
			building_employment_bureaucrats_add = 100
			building_employment_soldiers_add = 200
			building_employment_officers_add = 50
			# People being forcibly relocated (moved to destination)
			building_employment_laborers_add = 200
		}
	}

	# ai_value blocks don't support script values in PMs - use flat number
	# AI should rarely pick forced resettlement over peaceful alternatives
	ai_value = 50
}
