# ------ 1. Emergency Plan Revision ------
# Clear bottlenecks by forcing an emergency revision of industrial targets.
ce_plan_revision = {
	name = "CE_PLAN_REVISION"
	desc = "CE_PLAN_REVISION_DESC"
	visible = {
		banking_is_command_economy = yes
		NOT = { je:je_banking_cycle = { has_modifier = planning_plan_revision } }
	}
	ai_chance = {
		value = 0
		# Core: high friction (bubble) calls for revision
		if = { limit = { var:bubble_pressure >= 50 } add = 30 }
		if = { limit = { var:bubble_pressure >= 70 } add = 25 }
		if = { limit = { banking_cycle_is_frenzy = yes } add = 20 }
		if = { limit = { banking_cycle_is_boom = yes } add = 15 }
		# Avoid when already in supply-chain crisis (panic → emergency allocation instead)
		if = { limit = { banking_cycle_is_panic = yes } add = -30 }
		# Points awareness
		if = { limit = { banking_points_high = yes } add = 10 }
		if = { limit = { banking_points_low = yes } add = -10 }
		# Conflicting tools
		if = { limit = { banking_tool_ce_target_cut_active = yes } add = -20 }
	}
	possible = {
		custom_tooltip = {
			text = PLANNING_POINTS_3
			modifier:country_banking_intervention_max_add >= 3
		}
	}
	effect = {
		je:je_banking_cycle = { add_modifier = { name = planning_plan_revision } }
	}
}

ce_disable_plan_revision = {
	name = "CE_DISABLE_PLAN_REVISION"

	ai_chance = {

		if = { limit = { banking_cycle_is_panic = yes } add = 55 }
		if = { limit = { banking_cycle_is_downturn = yes } add = 30 }
		if = { limit = { banking_cycle_is_stable = yes banking_cycle_bubble_risk_rising = no } add = 20 }
		if = { limit = { banking_tool_ce_allocation_active = yes } add = 25 }
	}
	possible = { je:je_banking_cycle = { has_modifier = planning_plan_revision } }
	effect = { je:je_banking_cycle = { remove_modifier = planning_plan_revision } }

# ------ 9. Capital Injection (Treasury → Investment Pool) ------
# Redirect state treasury funds into the national investment pool.
# One-shot effect with a 90-day cooldown. Costs 2% of GDP from treasury.
ce_invest_pool_inject = {
	name = "CE_INVEST_POOL_INJECT"
	desc = "CE_INVEST_POOL_INJECT_DESC"
	visible = {
		banking_is_command_economy = yes
	}
	ai_chance = {
		value = 0
		# AI injects when investment pool is very low and treasury is comfortable
		if = { limit = { investment_pool < 50000 gold_reserves > 100000 } add = 30 }
		if = { limit = { investment_pool < 20000 gold_reserves > 50000 } add = 50 }
		# Don't inject when treasury is tight
		if = { limit = { gold_reserves < 50000 } add = -50 }
		if = { limit = { in_default = yes } add = -100 }
	}
	possible = {
		always = yes
	}
	effect = {
		add_treasury = ce_pool_inject_treasury_cost
		add_investment_pool = ce_pool_transfer_amount
		banking_cycle_update_ce_pool_balance = yes
	}
}

# ------ 10. Capital Withdrawal (Investment Pool → Treasury) ------
# Redirect investment pool funds back to the state treasury.
# One-shot effect with a 90-day cooldown. Withdraws up to 2% of GDP (or all, if pool < 2% GDP).
ce_invest_pool_withdraw = {
	name = "CE_INVEST_POOL_WITHDRAW"
	desc = "CE_INVEST_POOL_WITHDRAW_DESC"
	visible = {
		banking_is_command_economy = yes
	}
	ai_chance = {
		value = 0
		# AI withdraws when treasury is critically low
		if = { limit = { gold_reserves < 10000 investment_pool > 20000 } add = 60 }
		if = { limit = { in_default = yes investment_pool > 5000 } add = 80 }
		# Avoid draining pool when construction is needed
		if = { limit = { investment_pool < 10000 } add = -30 }
	}
	possible = {
		custom_tooltip = {
			text = CE_WITHDRAW_POSSIBLE_TT
			investment_pool > 0
		}
	}
	effect = {
		add_investment_pool = ce_pool_withdraw_pool_cost
		add_treasury = ce_pool_withdraw_amount
		banking_cycle_update_ce_pool_balance = yes
	}
}
