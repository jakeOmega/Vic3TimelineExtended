# ============================================================================
# CIVIL RIGHTS SCRIPTED BUTTONS
# Used by je_civil_rights. 6 toggle pairs (12 buttons) representing player
# stance toward the movement. Pro buttons stack with each other; anti buttons
# stack with each other; cooptation is compatible with anti buttons (the
# historical "coopt moderates, jail radicals" stance) but no other pro/anti
# pair coexists. Each button increments a months-tracker variable read by
# je_civil_rights' path-dependent on_complete / on_fail.
# ============================================================================

# ------ Grassroots Organizing (pro) ------
cr_grassroots_organizing = {
	name = "CR_GRASSROOTS_ORGANIZING"
	desc = "CR_GRASSROOTS_ORGANIZING_DESC"

	visible = {
		NOT = { has_modifier = cr_grassroots_modifier }
		NOT = { has_modifier = cr_suppression_modifier }
		NOT = { has_modifier = cr_segregationist_modifier }
	}

	ai_chance = {
		value = 0
		if = {
			limit = {
				OR = {
					has_law = law_type:law_minority_rights_indifference
					has_law = law_type:law_minority_rights_protection
					has_law = law_type:law_minority_rights_affirmative_action
				}
			}
			add = 30
		}
		if = {
			limit = { ig:ig_intelligentsia ?= { is_in_government = yes } }
			add = 15
		}
	}

	possible = {
		NOT = { has_modifier = cr_grassroots_modifier }
		NOT = { has_modifier = cr_suppression_modifier }
		NOT = { has_modifier = cr_segregationist_modifier }
	}

	effect = {
		custom_tooltip = {
			text = "CR_GRASSROOTS_ORGANIZING_DESC"
			add_modifier = { name = cr_grassroots_modifier }
		}
	}
}

cr_remove_grassroots_organizing = {
	name = "CR_REMOVE_GRASSROOTS_ORGANIZING"
	desc = "CR_REMOVE_GRASSROOTS_ORGANIZING_DESC"

	visible = { has_modifier = cr_grassroots_modifier }

	ai_chance = { value = 0 }

	possible = { has_modifier = cr_grassroots_modifier }

	effect = {
		custom_tooltip = {
			text = "CR_REMOVE_GRASSROOTS_ORGANIZING_DESC"
			remove_modifier = cr_grassroots_modifier
		}
	}
}
