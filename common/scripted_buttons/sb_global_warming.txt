# Global warming climate policy scripted buttons.
gw_carbon_tax_button = {
    name = "GW_CARBON_TAX"
    desc = "GW_CARBON_TAX_DESC"

    visible = {
        NOT = { has_modifier = carbon_tax_modifier }
        # Emissions reduction only applies for the market leader; restrict button accordingly
        market_capital.owner = THIS
    }

    ai_chance = {
        value = 0
        if = {
            limit = { temperature_anomaly_display >= 1.0 }
            add = 30
        }
        if = {
            limit = { temperature_anomaly_display >= 2.0 }
            add = 30
        }
        if = {
            limit = {
                has_law = law_type:law_laissez_faire
            }
            add = -50
        }
    }

    possible = {
        NOT = { has_modifier = carbon_tax_modifier }
        temperature_anomaly_display >= 0.5
        market_capital.owner = THIS
    }

    effect = {
        custom_tooltip = {
            text = "GW_CARBON_TAX_DESC"
            every_country = {
                limit = {
                    market_capital = ROOT.market_capital
                    NOT = { has_modifier = carbon_tax_modifier }
                }
                add_modifier = { name = carbon_tax_modifier }
            }
        }
    }
}

    visible = {
        has_modifier = renewable_investment_modifier
        market_capital.owner = THIS
    }

    ai_chance = {
        value = 0
        # AI removes renewable investment only when warming is negligible
        # (warming below threshold = less urgency to maintain green policies)
        if = {
            limit = { temperature_anomaly_display < 0.3 }
            add = 60
        }
        # AI also more likely to remove if warming isn't worsening significantly
        if = {
            limit = {
                has_law = law_type:law_laissez_faire
                temperature_anomaly_display < 1.0
            }
            add = 30
        }
    }

    possible = {
        has_modifier = renewable_investment_modifier

    }

        custom_tooltip = {
            text = "GW_REMOVE_RENEWABLE_INVESTMENT_DESC"

                    market_capital = ROOT.market_capital
                    has_modifier = renewable_investment_modifier

            }
        }
    }
}
