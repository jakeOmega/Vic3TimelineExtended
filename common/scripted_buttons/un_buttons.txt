# ---- 0. FOUND THE UNITED NATIONS ----
# A Great Power with intergovernmental_organizations tech creates the UN.
# Any recognized independent country can then join.

un_found_button = {
	name = "UN_FOUND"
	desc = "UN_FOUND_DESC"

	visible = {
		NOT = { has_global_variable = un_founded }
		has_technology_researched = intergovernmental_organizations
	}

	possible = {
		NOT = { has_global_variable = un_founded }
		has_technology_researched = intergovernmental_organizations
		country_rank >= rank_value:great_power
		is_subject = no
	}

	ai_chance = {
		value = 20
		if = { limit = { has_law = law_type:law_isolationism } add = -60 }
		if = { limit = { has_law = law_type:law_humanitarian_regulations } add = 20 }
		if = { limit = { has_law = law_type:law_limited_war } add = 40 }
	}

	effect = {
		custom_tooltip = {
			text = "UN_FOUND_DESC"
			# Cost: prestige and bureaucracy
			add_modifier = {
				name = un_founding_cost_modifier
				days = normal_modifier_time
				is_decaying = yes
			}
			# Establish the UN
			set_global_variable = un_founded
			# Founder becomes first member, HQ host, and founding member
			add_modifier = { name = un_member_modifier }
			add_modifier = { name = un_founding_member_modifier }
			add_modifier = { name = un_headquarters_modifier }
			set_global_variable = { name = un_hq_country value = ROOT }
			# GP founder gets Security Council seat
			if = {
				limit = { country_rank >= rank_value:great_power }
				add_modifier = { name = un_security_council_modifier }
			}
			set_variable = { name = can_build_un_hq value = 1 }
			# Fire charter event for all other recognized independent countries
			every_country = {
				limit = {
					NOT = { this = ROOT }
					is_subject = no
					NOT = { is_country_type = decentralized }
				}
				trigger_event = { id = un_events.1 }
			}
		}
	}
}

# ---- 1. JOIN / LEAVE UN ----

un_join_button = {
	name = "UN_JOIN"
	desc = "UN_JOIN_DESC"

	visible = {
		has_global_variable = un_founded
		NOT = { has_modifier = un_member_modifier }
	}

	possible = {
		has_global_variable = un_founded
		NOT = { has_modifier = un_member_modifier }
		is_subject = no
		NOT = { is_country_type = decentralized }
		NOT = { has_modifier = un_expelled_cooldown }
		NOT = { has_modifier = un_withdrawal_penalty_modifier }
	}

	ai_chance = {
		value = 50
		if = { limit = { country_rank >= rank_value:great_power } add = 30 }
		if = { limit = { country_rank >= rank_value:major_power } add = 15 }
		if = { limit = { has_law = law_type:law_isolationism } add = -80 }
		if = { limit = { has_law = law_type:law_humanitarian_regulations } add = 20 }
		if = { limit = { has_law = law_type:law_limited_war } add = 20 }
		if = { limit = { has_law = law_type:law_total_war } add = -40 }
	}

	effect = {
		custom_tooltip = {
			text = "UN_JOIN_DESC"
			add_modifier = { name = un_member_modifier }
			# Great powers auto-get SC seats if fewer than 5 exist
			if = {
				limit = {
					country_rank >= rank_value:great_power
					NOT = {
						any_country = {
							has_modifier = un_security_council_modifier
							count >= 5
						}
					}
				}
				add_modifier = { name = un_security_council_modifier }
			}
		}
	}
}

un_request_peacekeepers_button = {
	name = "UN_REQUEST_PEACEKEEPERS"
	desc = "UN_REQUEST_PEACEKEEPERS_DESC"

	visible = {
		has_modifier = un_member_modifier
		NOT = { has_modifier = un_peacekeeping_welcomed_modifier }
		OR = {
			is_at_war = yes
			any_scope_state = { devastation > 5 }
		}
	}

	possible = {
		has_modifier = un_member_modifier
		NOT = { has_global_variable = un_vote_active }
		NOT = { has_modifier = un_peacekeeping_welcomed_modifier }
		NOT = { has_modifier = un_request_cooldown }
		OR = {
			is_at_war = yes
			any_scope_state = { devastation > 5 }
		}
	}

	ai_chance = {
		value = 0
		if = { limit = { is_at_war = yes } add = 30 }
		if = {
			limit = { any_scope_state = { devastation > 10 } }
			add = 20
		}
		if = {
			limit = { NOT = { country_rank >= rank_value:major_power } }
			add = 15
		}
	}

	effect = {
		custom_tooltip = {
			text = "UN_REQUEST_PEACEKEEPERS_TT"
			# Set up vote
			set_global_variable = un_vote_active
			set_global_variable = un_vote_topic_peacekeeping_request
			set_global_variable = { name = un_vote_support value = 1 }
			set_global_variable = { name = un_vote_oppose value = 0 }
			set_global_variable = { name = un_vote_net value = 1 }
			set_global_variable = { name = un_vote_proposer_country value = ROOT }
			set_global_variable = { name = un_vote_target_country value = ROOT }
			add_modifier = {
				name = un_request_cooldown
				days = long_modifier_time
				is_decaying = yes
			}
			# Trigger vote for all other members
			every_country = {
				limit = {
					has_modifier = un_member_modifier
					NOT = { this = ROOT }
				}
				trigger_event = { id = un_vote.1 days = 30 }
			}
			# Schedule results
			trigger_event = { id = un_vote.2 days = 90 }
		}
	}
}

		OR = {
			has_law = law_type:law_war_crimes_forbidden
			has_law = law_type:law_humanitarian_regulations
			has_law = law_type:law_limited_war
		}
	}

	ai_chance = {
		value = 0
		if = { limit = { has_law = law_type:law_limited_war } add = 25 }
		if = { limit = { has_law = law_type:law_humanitarian_regulations } add = 20 }
		if = { limit = { var:un_authority >= 50 } add = 10 }
		if = {
			limit = {
				any_country = {
					has_diplomatic_pact = { who = ROOT type = rivalry }
					has_modifier = un_arms_control_participant_modifier
				}
			}
			add = 15
		}
	}

	effect = {
		custom_tooltip = {
			text = "UN_ARMS_CONTROL_DESC"
			add_modifier = { name = un_arms_control_participant_modifier }
			change_variable = { name = un_authority add = 3 }
		}
	}
}
