# ---- 1. JOIN / LEAVE UN ----

un_join_button = {
	name = "UN_JOIN"
	desc = "UN_JOIN_DESC"

	visible = {
		NOT = { has_modifier = un_member_modifier }
	}

	possible = {
		NOT = { has_modifier = un_member_modifier }
		modifier:country_can_join_united_nations_bool = yes
		is_subject = no
		NOT = { has_modifier = un_expelled_cooldown }
		NOT = {
			has_modifier = un_withdrawal_penalty_modifier
		}
	}

	ai_chance = {
		value = 50
		if = { limit = { country_rank >= rank_value:great_power } add = 30 }
		if = { limit = { country_rank >= rank_value:major_power } add = 15 }
		if = { limit = { has_law = law_type:law_isolationism } add = -80 }
		if = { limit = { has_law = law_type:law_humanitarian_regulations } add = 20 }
		if = { limit = { has_law = law_type:law_limited_war } add = 20 }
		if = { limit = { has_law = law_type:law_total_war } add = -40 }
	}

	effect = {
		custom_tooltip = {
			text = "UN_JOIN_DESC"
			add_modifier = { name = un_member_modifier }
		}
	}
}

un_leave_button = {
	name = "UN_LEAVE"
	desc = "UN_LEAVE_DESC"

	visible = {
		has_modifier = un_member_modifier
	}

	possible = {
		has_modifier = un_member_modifier
	}

	ai_chance = {
		value = 0
		if = { limit = { has_law = law_type:law_isolationism } add = 20 }
		if = { limit = { has_law = law_type:law_total_war } add = 15 }
		if = { limit = { var:un_authority < 20 } add = 10 }
	}

	effect = {
		custom_tooltip = {
			text = "UN_LEAVE_DESC"
			remove_modifier = un_member_modifier
			remove_modifier = un_membership_benefits_modifier
			remove_modifier = un_security_council_modifier
			remove_modifier = un_founding_member_modifier
			remove_modifier = un_sanctions_target_modifier
			remove_modifier = un_sanctions_enforcer_modifier
			remove_modifier = un_peacekeeping_contributor_modifier
			remove_modifier = un_development_contributor_modifier
			remove_modifier = un_human_rights_champion_modifier
			remove_modifier = un_arms_control_participant_modifier
			# Remove permanent treaty commitments
			remove_modifier = un_human_rights_declaration_modifier
			remove_modifier = un_human_rights_reservations_modifier
			remove_modifier = un_nonproliferation_modifier
			remove_modifier = un_climate_binding_modifier
			remove_modifier = un_heritage_program_modifier
			remove_modifier = un_pandemic_cooperation_modifier
			remove_modifier = un_refugee_program_modifier
			remove_modifier = un_space_partnership_modifier
			# If we host the HQ, reassign it
			if = {
				limit = { has_modifier = un_headquarters_modifier }
				remove_modifier = un_headquarters_modifier
				random_country = {
					limit = {
						has_modifier = un_member_modifier
						country_rank >= rank_value:great_power
						NOT = { this = ROOT }
					}
					add_modifier = { name = un_headquarters_modifier }
					set_global_variable = { name = un_hq_country value = THIS }
				}
			}
			add_modifier = {
				name = un_withdrawal_penalty_modifier
				days = normal_modifier_time
				is_decaying = yes
			}
		}
	}
}

# ---- 2. SANCTIONS ----

un_propose_sanctions_button = {
	name = "UN_PROPOSE_SANCTIONS"
	desc = "UN_PROPOSE_SANCTIONS_DESC"

	visible = {
		has_modifier = un_member_modifier
		NOT = { has_modifier = un_sanctions_enforcer_modifier }
	}

	possible = {
		has_modifier = un_member_modifier
		country_rank >= rank_value:major_power
		any_country = {
			has_diplomatic_pact = { who = ROOT type = rivalry }
			NOT = { has_modifier = un_sanctions_target_modifier }
		}
		NOT = { has_modifier = un_sanctions_cooldown }
	}

	ai_chance = {
		value = 0
		if = { limit = { var:un_authority >= 50 } add = 20 }
		if = {
			limit = {
				any_country = {
					has_diplomatic_pact = { who = ROOT type = rivalry }
					has_law = law_type:law_total_war
				}
			}
			add = 30
		}
		if = { limit = { country_rank >= rank_value:great_power } add = 10 }
	}

	effect = {
		custom_tooltip = {
			text = "UN_PROPOSE_SANCTIONS_DESC"
			add_modifier = { name = un_sanctions_enforcer_modifier }
			add_modifier = {
				name = un_sanctions_cooldown
				days = long_modifier_time
				is_decaying = yes
			}
			random_country = {
				limit = {
					has_diplomatic_pact = { who = ROOT type = rivalry }
					NOT = { has_modifier = un_sanctions_target_modifier }
				}
				add_modifier = { name = un_sanctions_target_modifier }
			}
			change_variable = { name = un_authority add = 5 }
		}
	}
}

# ---- 7. GREAT POWER INFLUENCE ON UN ----

un_champion_order_button = {
	name = "UN_CHAMPION_ORDER"
	desc = "UN_CHAMPION_ORDER_DESC"

	visible = {
		has_modifier = un_member_modifier
		country_rank >= rank_value:great_power
	}

	possible = {
		has_modifier = un_member_modifier
		country_rank >= rank_value:great_power
		NOT = { has_modifier = un_influence_cooldown }
	}

	ai_chance = {
		value = 0
		if = { limit = { has_law = law_type:law_humanitarian_regulations } add = 30 }
		if = { limit = { has_law = law_type:law_limited_war } add = 20 }
		if = { limit = { var:un_authority < 30 } add = 20 }
	}

	effect = {
		custom_tooltip = {
			text = "UN_CHAMPION_ORDER_DESC"
			# Base authority boost
			change_variable = { name = un_authority add = 2 }
			# Extra for high rank - stronger countries have more influence
			if = {
				limit = { global_country_ranking < 3 }
				change_variable = { name = un_authority add = 1 }
			}
			if = {
				limit = { global_country_ranking < 5 }
				change_variable = { name = un_authority add = 1 }
			}
			if = {
				limit = { global_country_ranking < 7 }
				change_variable = { name = un_authority add = 1 }
			}
			# Cost: authority and influence (these don't scale with GDP, so flat values are fine)
			add_authority = -50
			add_influence = -100
			add_modifier = {
				name = un_influence_cooldown
				days = 365
				is_decaying = yes
			}
		}
	}
}

un_undermine_order_button = {
	name = "UN_UNDERMINE_ORDER"
	desc = "UN_UNDERMINE_ORDER_DESC"

	visible = {
		has_modifier = un_member_modifier
		country_rank >= rank_value:great_power
	}

	possible = {
		has_modifier = un_member_modifier
		country_rank >= rank_value:great_power
		NOT = { has_modifier = un_influence_cooldown }
	}

	ai_chance = {
		value = 0
		if = { limit = { has_law = law_type:law_total_war } add = 25 }

			}

			add_influence = -50
			add_modifier = {
				name = un_influence_cooldown

				is_decaying = yes
