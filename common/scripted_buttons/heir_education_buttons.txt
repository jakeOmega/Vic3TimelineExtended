### Heir Education - Scripted Buttons
### 7 investment buttons + 1 conclude button

# ─────────────────── TRAIT INVESTMENT BUTTONS ───────────────────

he_invest_admin = {
	name = "HE_INVEST_ADMIN"
	desc = "HE_INVEST_ADMIN_DESC"

	visible = {
		has_law = law_type:law_monarchy
		exists = heir
	}

	ai_chance = {
		value = 0
		if = {
			limit = {
				ig:ig_industrialists = { is_in_government = yes }
			}
			add = 30
		}
		if = { limit = { NOT = { has_variable = heir_ed_admin } } add = 20 }
		if = { limit = { has_variable = heir_ed_admin var:heir_ed_admin >= 2 } add = -10 }
	}

	possible = {
		custom_tooltip = {
			text = HE_INVESTMENT_CAP_TT
			OR = {
				NOT = { has_variable = heir_ed_total }
				var:heir_ed_total < 10
			}
		}
		custom_tooltip = {
			text = HE_COOLDOWN_TT
			NOT = { has_modifier = heir_education_cooldown }
		}
	}

	effect = {
		# Increment investment
		if = {
			limit = { has_variable = heir_ed_admin }
			change_variable = { name = heir_ed_admin add = 1 }
		}
		else = {
			set_variable = { name = heir_ed_admin value = 1 }
		}
		if = {
			limit = { has_variable = heir_ed_total }
			change_variable = { name = heir_ed_total add = 1 }
		}
		else = {
			set_variable = { name = heir_ed_total value = 1 }
		}

		# Update progress bar
		scope:journal_entry = {
			set_bar_progress = { name = heir_education_progress_bar value = root.var:heir_ed_total }
		}

		# Cost: temporary innovation penalty
		add_modifier = { name = heir_education_innovation_cost days = 180 is_decaying = yes }
		# Cooldown
		add_modifier = { name = heir_education_cooldown days = 180 }

		# IG reactions
		heir_education_admin_ig_reaction = yes
	}
}

# ─────────────────── CONCLUDE EDUCATION BUTTON ───────────────────

he_conclude_education = {
	name = "HE_CONCLUDE_EDUCATION"
	desc = "HE_CONCLUDE_EDUCATION_DESC"

	visible = {
		has_law = law_type:law_monarchy
		exists = heir
		has_variable = heir_ed_total
	}

	ai_chance = {
		value = 0
		# AI concludes when they've invested enough or when cap is close
		if = { limit = { has_variable = heir_ed_total var:heir_ed_total >= 5 } add = 40 }
		if = { limit = { has_variable = heir_ed_total var:heir_ed_total >= 8 } add = 40 }
		if = { limit = { has_variable = heir_ed_total var:heir_ed_total >= 10 } add = 100 }
	}

	possible = {
		custom_tooltip = {
			text = HE_MIN_INVESTMENT_TT
			has_variable = heir_ed_total
			var:heir_ed_total >= 1
		}
	}

	effect = {
		# Resolve education into actual character changes
		heir_education_resolve_effect = yes
	}
}
