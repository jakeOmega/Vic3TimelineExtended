### Heir Education - Toggle Buttons
### 7 focus categories with enable/disable toggle pairs
### Trait Focus (admin/diplo/military) - mutually exclusive
### Ideology Focus (progressive/conservative) - mutually exclusive
### IG Focus (establishment/reformist) - mutually exclusive

# ─────────── ADMINISTRATIVE FOCUS ───────────

he_enable_admin_focus = {
	name = "HE_ENABLE_ADMIN_FOCUS"
	desc = "HE_ENABLE_ADMIN_FOCUS_DESC"

	visible = {
		has_law = law_type:law_monarchy
		exists = heir
		NOT = { has_variable = heir_ed_focus_admin }
	}

	ai_chance = {
		value = 0
		if = {
			limit = {
				NOR = {
					has_variable = heir_ed_focus_admin
					has_variable = heir_ed_focus_diplo
					has_variable = heir_ed_focus_military
				}
			}
			add = 20
		}
		if = {
			limit = { ig:ig_industrialists = { is_in_government = yes } }
			add = 10
		}
	}

	possible = {
		always = yes
	}

	effect = {
		if = {
			limit = { has_variable = heir_ed_focus_diplo }
			remove_variable = heir_ed_focus_diplo
		}
		if = {
			limit = { has_variable = heir_ed_focus_military }
			remove_variable = heir_ed_focus_military
		}
		set_variable = heir_ed_focus_admin
	}
}

he_disable_admin_focus = {
	name = "HE_DISABLE_ADMIN_FOCUS"
	desc = "HE_DISABLE_ADMIN_FOCUS_DESC"

	visible = {
		has_law = law_type:law_monarchy
		exists = heir
		has_variable = heir_ed_focus_admin
	}

	ai_chance = {
		value = 0
	}

	possible = {
		always = yes
	}

	effect = {
		remove_variable = heir_ed_focus_admin
	}
}
