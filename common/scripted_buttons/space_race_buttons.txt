# Space Race Per-JE Scripted Buttons
# 4 buttons per milestone x 8 milestones = 32 buttons
# Each milestone has independent funding level and approach selection

# ======= SUBORBITAL Buttons =======

sr_btn_safe_suborbital = {
	name = "SR_SAFE_APPROACH"
	desc = "SR_SAFE_APPROACH_DESC"

	visible = {
		has_variable = sr_active_suborbital
	}

	ai_chance = {
		value = 50
		if = {
			limit = { country_rank = rank_value:great_power }
			add = -20
		}
	}

	possible = {
		custom_tooltip = {
			text = "SR_NOT_ALREADY_SAFE"
			NOT = { has_variable = sr_safe_suborbital }
		}
	}

	effect = {
		custom_tooltip = {
			text = "SR_SAFE_APPROACH_EFFECT"
			set_variable = { name = sr_safe_suborbital value = yes }
			if = {
				limit = { has_variable = sr_ambitious_suborbital }
				remove_variable = sr_ambitious_suborbital
			}
			sr_recalculate_cost = yes
		}
	}
}

sr_btn_fund_up_suborbital = {
	name = "SR_INCREASE_FUNDING"
	desc = "SR_INCREASE_FUNDING_DESC"

	visible = {
		has_variable = sr_active_suborbital
	}

	ai_chance = {
		value = 0
		if = {
			limit = { country_innovation > 50 }
			add = 30
		}
		if = {
			limit = { var:sr_funding_suborbital < 2 }
			add = 20
		}
	}

	possible = {
		custom_tooltip = {
			text = "SR_MAX_FUNDING_NOT_REACHED"
			var:sr_funding_suborbital < sr_max_funding_level
		}
	}

	effect = {
		custom_tooltip = {
			text = "SR_INCREASE_FUNDING_EFFECT"
			change_variable = { name = sr_funding_suborbital add = 1 }
			sr_recalculate_cost = yes
		}
	}
}
