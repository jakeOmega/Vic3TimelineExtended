# ------ Invest in Colonial Development ------
ce_invest_in_development = {
	name = "CE_INVEST_IN_DEVELOPMENT"
	desc = "CE_INVEST_IN_DEVELOPMENT_DESC"

	visible = {
		NOT = { has_modifier = colonial_development_investment_modifier }
	}

	ai_chance = {
		value = 0
		if = {
			limit = { colonial_state_count >= 1 }
			add = 30
		}
		if = {
			limit = { colonial_low_acceptance_count >= 2 }
			add = 20
		}
		if = {
			limit = { country_rank >= rank_value:great_power }
			add = 10
		}
	}

	possible = {
		custom_tooltip = {
			text = CE_NEEDS_COLONIAL_STATES_OR_SUBJECTS
			OR = {
				any_scope_state = {
					is_overseas_colonial_state = yes
				}
				country_has_qualifying_colonial_subject = yes
			}
		}
		NOT = { has_modifier = colonial_development_investment_modifier }
	}

	effect = {
		custom_tooltip = {
			text = "CE_INVEST_IN_DEVELOPMENT_DESC"
			add_modifier = { name = colonial_development_investment_modifier }
		}
		# Seed the GDP-scaled expense modifier so the budget line shows up
		# immediately on activation; the JE monthly pulse refreshes it from
		# here on, keeping the multiplier in sync with current GDP.
		add_modifier = {
			name = colonial_development_investment_expense_modifier
			multiplier = colonial_invest_monthly_expense
		}
		# Program startup cost — a one-shot treasury hit so enabling the
		# program isn't a free toggle. Parallels the Cultural Assimilation
		# startup cost; the two read as the same flavor of cost.
		add_treasury = colonial_invest_startup_cost
		# IG disapproval (vs. the previous mechanical IG strength erosion).
		# Industrialists wanted capital at home; the PB resents the diversion.
		ig_approval_effect = { IG = ig_industrialists MODIFIER = ig_approval_negative_modifier DAYS = short_modifier_time }
		ig_approval_effect = { IG = ig_petty_bourgeoisie MODIFIER = ig_approval_negative_modifier DAYS = short_modifier_time }
	}
}

ce_remove_invest_in_development = {
	name = "CE_REMOVE_INVEST_IN_DEVELOPMENT"
	desc = "CE_REMOVE_INVEST_IN_DEVELOPMENT_DESC"

	visible = {
		has_modifier = colonial_development_investment_modifier
	}

	ai_chance = {
		value = 0
		if = {
			limit = { colonial_state_count = 0 }
			add = 50
		}
	}

	possible = {
		has_modifier = colonial_development_investment_modifier
	}

	effect = {
		custom_tooltip = {
			text = "CE_REMOVE_INVEST_IN_DEVELOPMENT_DESC"
			remove_modifier = colonial_development_investment_modifier
		}
		# Drop the budget-line expense too; the JE pulse stops refreshing
		# it once the policy modifier is gone, but the active copy needs
		# explicit cleanup or it would linger out a full duration.
		remove_modifier = colonial_development_investment_expense_modifier
	}
}

# ------ Military Garrison ------
ce_military_garrison = {
	name = "CE_MILITARY_GARRISON"
	desc = "CE_MILITARY_GARRISON_DESC"

	visible = {
		NOT = { has_modifier = colonial_military_garrison_modifier }
	}

	ai_chance = {
		value = 0
		if = {
			limit = { colonial_low_acceptance_count >= 2 }
			add = 40
		}
		if = {
			limit = { colonial_low_acceptance_count >= 4 }
			add = 20
		}
	}

	possible = {
		custom_tooltip = {
			text = CE_NEEDS_COLONIAL_STATES_OR_SUBJECTS
			OR = {
				any_scope_state = {
					is_overseas_colonial_state = yes
				}
				country_has_qualifying_colonial_subject = yes
			}
		}
		NOT = { has_modifier = colonial_military_garrison_modifier }
		# Hard authority gate — Garrison is unsustainable for liberal AIs without
		# the authority-generating regressive law stack. The cruel/Iron-Fist
		# state has authority to spare; the AI without it can't even click.
		authority > 300
	}

	effect = {
		custom_tooltip = {
			text = "CE_MILITARY_GARRISON_DESC"
			add_modifier = { name = colonial_military_garrison_modifier }
		}
		# Infamy compounds with repeated cycles — a sustained Iron-Fist run
		# accumulates infamy that itself draws GP condemners (existing -0.6/mo
		# bar drain per condemner, escalating at tier 4 and 6).
		change_infamy = 5
	}
}

# ------ Release Colonial Territory (one region at a time) ------
ce_release_colonial_territory = {
	name = "CE_RELEASE_COLONIAL_TERRITORY"
	desc = "CE_RELEASE_COLONIAL_TERRITORY_DESC"

	ai_chance = {
		value = 0
		if = {
			limit = { colonial_gp_condemners_count >= 3 }
			add = 15
		}
		if = {
			limit = { colonial_low_acceptance_count >= 3 }
			add = 20
		}
		# Less likely for great powers
		if = {
			limit = { country_rank >= rank_value:great_power }
			add = -20
		}
	}

	possible = {
		custom_tooltip = {
			text = CE_NEEDS_COLONIAL_STATES
			any_scope_state = {
				is_decolonization_eligible_state = yes
			}
		}
	}

	effect = {
		custom_tooltip = "CE_RELEASE_COLONIAL_TERRITORY_DESC"
		# Pick the top 3 candidates by population. For each pick, the preview
		# helper computes its release region (capital + adjacent homeland
		# states owned by us) and stores the region's state count for loc
		# display. Subsequent picks exclude states already in a prior
		# candidate's region — so option 2 won't show a state that option 1
		# would have already released as part of its flood-fill.
		#
		# `max = 1` + region-marker exclusion is preferred over `position = N`:
		# `check_range_bounds = no` on `position` clamps out-of-range to the
		# nearest valid index instead of skipping, which produced duplicate
		# picks when fewer than 3 eligible exist. With max+exclusion, each
		# scope:decolonization_target_state_N just stays unset when no further
		# eligible-and-unmarked state remains.
		ordered_scope_state = {
			limit = { is_decolonization_eligible_state = yes }
			order_by = state_population
			max = 1
			save_scope_as = decolonization_target_state_1
			_decolonization_preview_region = {
				REGION_VAR = decolonization_region_1_marker
				COUNT_VAR = decolonization_region_1_count
				CULTURE_SCOPE = culture_for_region_1
			}
		}
		ordered_scope_state = {
			limit = {
				is_decolonization_eligible_state = yes
				NOT = { has_variable = decolonization_region_1_marker }
			}
			order_by = state_population
			max = 1
			save_scope_as = decolonization_target_state_2
			_decolonization_preview_region = {
				REGION_VAR = decolonization_region_2_marker
				COUNT_VAR = decolonization_region_2_count
				CULTURE_SCOPE = culture_for_region_2
			}
		}
		ordered_scope_state = {
			limit = {
				is_decolonization_eligible_state = yes
				NOT = { has_variable = decolonization_region_1_marker }
				NOT = { has_variable = decolonization_region_2_marker }
			}
			order_by = state_population
			max = 1
			save_scope_as = decolonization_target_state_3
			_decolonization_preview_region = {
				REGION_VAR = decolonization_region_3_marker
				COUNT_VAR = decolonization_region_3_count
				CULTURE_SCOPE = culture_for_region_3
			}
		}
		set_variable = decolonization_path_release_single
		debug_log = "DECOL_DEBUG: ce_release_colonial_territory clicked, candidates: 1=[SCOPE.sState('decolonization_target_state_1').GetName] 2=[SCOPE.sState('decolonization_target_state_2').GetName] 3=[SCOPE.sState('decolonization_target_state_3').GetName]"  # DEBUG_DECOLONIZATION_TRACE
		trigger_event = { id = decolonization_events.400 popup = yes }
	}
}

# ------ Planned Full Decolonization ------
ce_planned_decolonization = {
	name = "CE_PLANNED_DECOLONIZATION"
	desc = "CE_PLANNED_DECOLONIZATION_DESC"

	ai_chance = {
		value = 0
		# AI does this under moderate pressure now (lowered thresholds)
		if = {
			limit = { colonial_gp_condemners_count >= 3 }
			add = 30
		}
		if = {
			limit = { colonial_low_acceptance_count >= 3 }
			add = 30
		}
		if = {
			limit = { NOT = { country_rank >= rank_value:great_power } }
			add = 30
		}
		# Never if things are going well
		if = {
			limit = { colonial_high_acceptance_count >= 2 }
			add = -100
		}
	}

	possible = {
		custom_tooltip = {
			text = CE_NEEDS_COLONIAL_STATES
			any_scope_state = {
				is_decolonization_eligible_state = yes
			}
		}
	}

	effect = {
		custom_tooltip = "CE_PLANNED_DECOLONIZATION_DESC"
		# Mark every decolonization-eligible state; the confirmation event
		# previews the list and the player's "Proceed" option releases each one.
		# Uses the same `is_decolonization_eligible_state` trigger as Release /
		# Round Table so the three buttons agree on what counts as colonial.
		every_scope_state = {
			limit = { is_decolonization_eligible_state = yes }
			set_variable = decolonization_marker
		}
		# Count marked states for the confirmation event's loc display.
		set_variable = { name = decolonization_planned_count value = 0 }
		every_scope_state = {
			limit = { has_variable = decolonization_marker }
			prev = { change_variable = { name = decolonization_planned_count add = 1 } }
		}
		# Save the largest marked state as a named scope for event placement.
		ordered_scope_state = {
			limit = { has_variable = decolonization_marker }
			order_by = state_population
			max = 1
			save_scope_as = decolonization_target_state
		}
		set_variable = decolonization_path_planned
		debug_log = "DECOL_DEBUG: ce_planned_decolonization clicked, scope:decolonization_target_state=[SCOPE.sState('decolonization_target_state').GetName] count=[ROOT.GetVariable('decolonization_planned_count').GetValue], firing confirm_release_planned"  # DEBUG_DECOLONIZATION_TRACE
		trigger_event = { id = decolonization_events.401 popup = yes }
	}
}

# ------ Round Table Conference ------
# A negotiated transition for ONE colony at a time. Costs authority + prestige
# up front, but produces a high-acceptance independent country with the
# Commonwealth Legacy modifier — friendly diplomatic relations, technological
# cooperation, and no infamy hit. Distinct from "Release Colonial Territory"
# (which is a unilateral grant) and "Planned Full Decolonization" (which
# liquidates the entire empire). The Round Table is for the colony you want
# to leave on good terms while keeping the rest.
ce_round_table_conference = {
	name = "CE_ROUND_TABLE_CONFERENCE"
	desc = "CE_ROUND_TABLE_CONFERENCE_DESC"

	ai_chance = {
		value = 0
		if = {
			limit = {
				colonial_high_acceptance_count >= 1
				colonial_gp_condemners_count >= 2
			}
			add = 30
		}
		# Intelligentsia leadership pushes for negotiated transitions
		if = {
			limit = {
				ig:ig_intelligentsia ?= {
					leader = { has_ideology = ideology:ideology_anti_colonialist }
				}
			}
			add = 20
		}
	}

	possible = {
		custom_tooltip = {
			text = CE_ROUND_TABLE_REQUIRES_ACCEPTABLE_COLONY
			any_scope_state = {
				is_decolonization_eligible_state = yes
				NOT = {
					any_scope_pop = {
						pop_acceptance < acceptance_status_2
					}
				}
			}
		}
		authority > 200
	}

	effect = {
		custom_tooltip = "CE_ROUND_TABLE_CONFERENCE_DESC"
		# Pick the largest sufficiently-accepted colonial state — the diplomatic
		# transition has to land on a colony where there's a real negotiating
		# partner, not on a tinderbox. Save it as a named scope (vanilla pattern,
		# carries into trigger_event); the event handles the actual release.
		# Top 3 sufficiently-accepted candidates by population. Same exclusion
		# pattern as Release Colonial Territory — see the comment there for
		# why `position = N` doesn't work for distinct top-N picking. Each
		# pick triggers a preview flood-fill so option 2 / 3 won't show states
		# already inside option 1's release region.
		ordered_scope_state = {
			limit = {
				is_decolonization_eligible_state = yes
				NOT = { any_scope_pop = { pop_acceptance < acceptance_status_2 } }
			}
			order_by = state_population
			max = 1
			save_scope_as = decolonization_target_state_1
			_decolonization_preview_region = {
				REGION_VAR = decolonization_region_1_marker
				COUNT_VAR = decolonization_region_1_count
				CULTURE_SCOPE = culture_for_region_1
			}
		}
		ordered_scope_state = {
			limit = {
				is_decolonization_eligible_state = yes
				NOT = { any_scope_pop = { pop_acceptance < acceptance_status_2 } }
				NOT = { has_variable = decolonization_region_1_marker }
			}
			order_by = state_population
			max = 1
			save_scope_as = decolonization_target_state_2
			_decolonization_preview_region = {
				REGION_VAR = decolonization_region_2_marker
				COUNT_VAR = decolonization_region_2_count
				CULTURE_SCOPE = culture_for_region_2
			}
		}
		ordered_scope_state = {
			limit = {
				is_decolonization_eligible_state = yes
				NOT = { any_scope_pop = { pop_acceptance < acceptance_status_2 } }
				NOT = { has_variable = decolonization_region_1_marker }
				NOT = { has_variable = decolonization_region_2_marker }
			}
			order_by = state_population
			max = 1
			save_scope_as = decolonization_target_state_3
			_decolonization_preview_region = {
				REGION_VAR = decolonization_region_3_marker
				COUNT_VAR = decolonization_region_3_count
				CULTURE_SCOPE = culture_for_region_3
			}
		}
		set_variable = decolonization_path_round_table
		debug_log = "DECOL_DEBUG: ce_round_table_conference clicked, candidates: 1=[SCOPE.sState('decolonization_target_state_1').GetName] 2=[SCOPE.sState('decolonization_target_state_2').GetName] 3=[SCOPE.sState('decolonization_target_state_3').GetName]"  # DEBUG_DECOLONIZATION_TRACE
		trigger_event = { id = decolonization_events.400 popup = yes }
	}
}
