ch_increase_program_funding_button = {
	name = "CH_INCREASE_PROGRAM_FUNDING"
	desc = "CH_INCREASE_PROGRAM_FUNDING_DESC"

	# Stays visible while the JE is up; gating moved to `possible` so the
	# button greys out with a tooltip when the player lacks the Ministry of
	# Culture or has no funding cap yet, instead of disappearing silently.
	visible = {
		has_journal_entry = je_cultural_hegemony
	}

	ai_chance = {
		value = 0
		if = {
			limit = { has_variable = ch_total var:ch_total >= 10 }
			add = 20
		}
		if = {
			limit = { has_variable = ch_total var:ch_total < 5 }
			add = 10
		}
	}

	possible = {
		custom_tooltip = {
			text = CH_REQUIRES_MINISTRY_TT
			has_law = law_type:law_ministry_of_culture
		}
		custom_tooltip = {
			text = CH_PROGRAM_FUNDING_NOT_MAX
			OR = {
				NOT = { has_variable = ch_program_funding_level }
				var:ch_program_funding_level < ch_program_funding_max_level
			}
		}
	}

	effect = {
		hidden_effect = {
			if = {
				limit = { NOT = { has_variable = ch_program_funding_level } }
				set_variable = { name = ch_program_funding_level value = 1 }
			}
			else = {
				change_variable = { name = ch_program_funding_level add = 1 }
				clamp_variable = { name = ch_program_funding_level min = 0 max = ch_program_funding_max_level }
			}

			je:je_cultural_hegemony ?= { remove_modifier = ch_cultural_program_funding }
			je:je_cultural_hegemony ?= { remove_modifier = ch_cultural_program_funding_cost }
			if = {
				limit = { ch_program_funding_level_value >= 1 }
				je:je_cultural_hegemony ?= { add_modifier = { name = ch_cultural_program_funding multiplier = ch_program_funding_level_value_for_je } }
				je:je_cultural_hegemony ?= { add_modifier = { name = ch_cultural_program_funding_cost multiplier = ch_program_funding_cost_mult_for_je } }
			}
		}
	}
}
