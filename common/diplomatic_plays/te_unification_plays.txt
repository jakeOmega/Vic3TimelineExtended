# Unification and leadership diplomatic plays for the major formables added
# by this mod (AFU, BHA, NUS, EUN, UNA, UNE, ZHO, INM, EAF, DAR). All reuse
# vanilla war_goals (`unification` and `unification_leadership`) and the
# generic vanilla unification.dds icon.

dp_unify_african_union = {
	war_goal = unification

	requires_interest_marker = no
	blocked_by_diplomatic_status = no
	add_infamy_for_starting_initiator_wargoals = no

	texture = "gfx/interface/icons/war_goals/unification.dds"

	selectable_in_lens = {
		always = no
	}

	possible = {
		NOT = { exists = c:AFU }
		NOT = { is_country_type = decentralized }
		has_technology_researched = decolonization
	}

	on_weekly_pulse = {}
	on_war_begins = {}
	on_war_end = {}
}
