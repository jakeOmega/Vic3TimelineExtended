# specific power bloc names
atlantic_charter = {
	trigger = {
		OR = {
			capital.region = sr:region_western_europe
			capital.region = sr:region_southern_europe
			capital.region = sr:region_central_europe
			capital.region = sr:region_northern_europe
			capital.region = sr:region_canada
			capital.region = sr:region_atlantic_coast
		}
	}
}

pacific_rim_pact = {
	trigger = {
		OR = {
			capital.region = sr:region_pacific_coast
			capital.region = sr:region_northeast_asia
			capital.region = sr:region_oceania
			capital.region = sr:region_indonesia
			capital.region = sr:region_andes
		}
	}
}

REPLACE:amazon_pact = {
	trigger = {
	    OR = {
	    	capital = { state_region = s:STATE_PARA }
	    	capital = { state_region = s:STATE_AMAZONAS }
	    	capital = { state_region = s:STATE_GUAYANA }
	    }
	}
}
