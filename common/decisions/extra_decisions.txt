adopt_local_culture = {
	is_shown = {
		NOT = {
			any_scope_state = {
				is_homeland_of_country_cultures = ROOT
			}
		}
	}

	possible = {
		is_subject = no
		has_technology_researched = decolonization
		capital = {
			any_scope_culture = {
				has_homeland = PREV
			}
		}
	}

	when_taken = {
		every_primary_culture = {
			ROOT = {
				remove_primary_culture = PREV
			}
		}
		capital  = {
			every_scope_culture = {
				limit = {
					has_homeland = PREV
				}
				ROOT = {
					add_primary_culture = PREV
				}
			}
		}
	}

	ai_chance = {
		value = 100
	}
}

# Secularization Campaign
# Removes all permanent religious revival modifiers.
# Available after researching decline_of_organized_religion.
# ============================================================================
secularization_campaign = {
	is_shown = {
		has_technology_researched = decline_of_organized_religion
		OR = {
			has_modifier = rre_religious_right_modifier
			has_modifier = rre_televangelism_modifier
			has_modifier = rre_liberation_theology_modifier
			has_modifier = rre_religious_nationalism_modifier
			has_modifier = rre_digital_faith_modifier
			has_modifier = rre_culture_war_modifier
			has_modifier = rre_faith_services_modifier
		}
	}

	possible = {
		has_technology_researched = decline_of_organized_religion
		NOT = { is_country_type = decentralized }
	}

	when_taken = {
		remove_modifier_if_exists_effect = { MODIFIER = rre_religious_right_modifier }
		remove_modifier_if_exists_effect = { MODIFIER = rre_televangelism_modifier }
		remove_modifier_if_exists_effect = { MODIFIER = rre_liberation_theology_modifier }
		remove_modifier_if_exists_effect = { MODIFIER = rre_religious_nationalism_modifier }
		remove_modifier_if_exists_effect = { MODIFIER = rre_digital_faith_modifier }
		remove_modifier_if_exists_effect = { MODIFIER = rre_culture_war_modifier }
		remove_modifier_if_exists_effect = { MODIFIER = rre_faith_services_modifier }
		add_radicals = {
			value = medium_radicals
			interest_group = ig:ig_devout
		}
	}

	ai_chance = {
		value = 0
	}
}

# Open Map-Mode Survey
# Player-only meta-decision that opens te_map_modes.1, the picker for custom
# map-mode overlays (Tourism Appeal / GDP per capita / Emissions). The
# overlay applies a state_migration_pull_add correction to every state so the
# Migration Attraction map mode color-codes by the chosen metric.
#
# DISABLED — vanilla engine bug.
# The correction math depends on reading per-state migration_pull as a script
# value. According to ~/src/vic3/docs/event_targets.log:1498, migration_pull
# is documented to "Get the migration pull for a state or the mass migration
# pull for a country" (Input Scopes: country, state). In practice it always
# returns the country's mass-migration value regardless of scope, so the
# per-state correction collapses to a constant and the gradient is flat.
# Re-enable by restoring `is_shown = { is_player = yes NOT = { is_country_type
# = decentralized } }` and `possible = { is_game_paused = yes }` once the
# vanilla bug is fixed (or once we find a workable per-state pull source).
# All the supporting files (event, scripted_effects, scripted_triggers,
# script_values, static_modifier, loc) are left in place — only the entry
# point is gated off here.

# ============================================================================
# Imperial Federation Act — Iron Fist Path (capstone for Iron-Fist hold strategy)
# Available only when:
#   - colonial_solidified_months >= 36 (3 years sustained at ≥90 bar)
#   - country has explicitly committed to formal empire (Colonial Affairs)
#   - laws across 4 lawgroups all match the Iron-Fist profile
#   - country is GP and has ≥4 colonial states
# Effect: completes je_colonial_empire via flag, triggers victory event 300,
# applies permanent imperial_federation_modifier.
# ============================================================================
imperial_federation_act_iron_fist = {
	is_shown = {
		has_journal_entry = je_colonial_empire
	}

	possible = {
		# Sustained Solidified band
		var:colonial_solidified_months >= 36
		# Power and scale
		country_rank >= rank_value:great_power
		colonial_state_count >= 4
		# Colonial Affairs commitment
		OR = {
			has_law = law_type:law_colonial_exploitation
			has_law = law_type:law_colonial_resettlement
		}
		# Iron-Fist Minority Rights
		OR = {
			has_law = law_type:law_minority_rights_violent_hostility
			has_law = law_type:law_minority_rights_ghettoization
			has_law = law_type:law_minority_rights_discrimination
			has_law = law_type:law_minority_rights_cultural_assimilation
		}
		# Hard-line Citizenship
		OR = {
			has_law = law_type:law_national_supremacy
			has_law = law_type:law_ethnostate
			has_law = law_type:law_racial_segregation
		}
		# Speech suppression
		OR = {
			has_law = law_type:law_outlawed_dissent
			has_law = law_type:law_censorship
		}
		# Authoritarian DP
		OR = {
			has_law = law_type:law_autocracy
			has_law = law_type:law_oligarchy
			has_law = law_type:law_single_party_state
			has_law = law_type:law_landed_voting
		}
		# Costs (gated)
		authority > 300
		country_prestige >= 100
	}

	when_taken = {
		add_authority = -300
		add_modifier = {
			name = imperial_federation_modifier
			days = -1
		}
		set_country_flag = imperial_federation_taken
		set_country_flag = imperial_federation_iron_fist
	}

	ai_chance = {
		value = 100
	}
}

# ============================================================================
# Mandate System Decision — managed-decline option for the broad-franchise empire
# Converts the colonial JE into informal economic empire (mandate_system_modifier)
# while allowing the player to keep collecting some of the benefits.
# Available to liberal-democratic colonial empires that have already taken at
# least one Round Table Conference.
# ============================================================================
mandate_system_decision = {
	is_shown = {
		has_journal_entry = je_colonial_empire
		has_technology_researched = decolonization
	}

	possible = {
		scope:journal_entry = {
			"scripted_bar_progress(colonial_stability_bar)" >= 65
		}
		# Round Table Conference taken at least once (timed positive event modifier
		# is the cleanest signal the button has been used recently)
		# Hard requirement: neocolonialism law (the mod-added "informal empire"
		# Colonial Affairs law gated on the decolonization tech)
		has_law = law_type:law_neocolonialism
		# Broad-franchise DP — selling this requires legitimacy
		OR = {
			has_law = law_type:law_universal_suffrage
			has_law = law_type:law_census_voting
			has_law = law_type:law_wealth_voting
		}
		# Cannot have outlawed dissent — selling negotiated transition needs press
		NOT = { has_law = law_type:law_outlawed_dissent }
		authority > 100
	}

	when_taken = {
		add_authority = -100
		add_modifier = {
			name = mandate_system_modifier
			days = -1
		}
		# Push bar up — the world recognizes the negotiated mode
		add_modifier = {
			name = colonial_stability_positive_event
			months = 12
		}
	}

	ai_chance = {
		value = 50
	}
}
