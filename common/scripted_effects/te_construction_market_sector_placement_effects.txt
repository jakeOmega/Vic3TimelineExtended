te_construction_market_state_on_start = {
	if = {
		limit = {
			NOT = {
				any_scope_building = {
					is_building_type = te_construction_market_site
				}
			}
		}
		create_building = {
			building = te_construction_market_site
			level = 1
		}
	}

	owner = {
		if = {
			limit = {
				capital = {
					AND = {
						any_scope_building = {
							is_building_type = te_construction_market_site
						}
						NOT = {
							any_scope_building = {
								is_under_construction = yes
							}
						}
					}
				}
			}
			capital = {
				remove_building = te_construction_market_site
			}
		}
	}


	# owner = {
	# 	te_construction_market_apply_per_site_modifier = yes
	# }
}

te_construction_market_state_on_end = {
	if = {
		limit = {
			NOT = {
				any_scope_building = {
					is_under_construction = yes
				}
			}
			any_scope_building = {
				is_building_type = te_construction_market_site
			}
			NOT = {
				is_capital = yes
			}
		}
		remove_building = te_construction_market_site
	}
	owner = {
		if = {
			limit = {
				NOT = {
					any_scope_building = {
						is_building_type = te_construction_market_site
					}
				}
			}
			capital = {
				create_building = {
					building = te_construction_market_site
					level = 1
				}
			}

		}
	}
}
