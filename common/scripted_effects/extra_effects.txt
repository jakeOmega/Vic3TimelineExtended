space_elevator_construction = {
	if = {
		limit = {
			NOT = { has_variable = space_elevator_progress_var }
		}
		set_variable = {
			name = space_elevator_progress_var
			value = 0
		}
	}
	if = {
		limit = {
			has_variable = space_elevator_progress_var
			any_scope_building = {
				is_building_type = building_space_elevator_construction_site
				building_has_goods_shortage = no
			}
		}
		change_variable = {
			name = space_elevator_progress_var
			add = {
				value = b:building_space_elevator_construction_site.occupancy
				divide = 12
				if = {
					limit = {
						any_scope_building = { has_active_production_method = pm_space_elevator_construction_slow }
					}
					multiply = 0.25
				}
				else_if = {
					limit = {
						any_scope_building = { has_active_production_method = pm_space_elevator_construction_medium }
					}
					multiply = 0.5
				}
				else_if = {
					limit = {
						any_scope_building = { has_active_production_method = pm_space_elevator_construction_fast }
					}
					multiply = 1
				}
				else = {
					multiply = 0
				}
				multiply = 1000
				ceiling = yes
				divide = 1000
			}
		}
	}
	if = {
		limit = {
			var:space_elevator_progress_var >= 1
		}
		set_variable = {
			name = space_elevator_progress_var
			value = 0
		}
		remove_building = building_space_elevator_construction_site
		if = {
			limit = { has_building = building_space_elevator }
			switch = {
                trigger = b:building_space_elevator.level

				5 = {create_building = { building = building_space_elevator level = 6}}

				7 = {create_building = { building = building_space_elevator level = 8}}
				8 = {create_building = { building = building_space_elevator level = 9}}
				9 = {create_building = { building = building_space_elevator level = 10}}
				10 = {create_building = { building = building_space_elevator level = 11}}

# =================================================================
# Resettlement Population Transfer
# =================================================================
# State scope (SOURCE state with a resettlement camp).
# Finds the camp, identifies a destination state with a colony,
# then moves pops employed at the camp to the destination.
#
# Uses ordered_scope_pop with order_by = total_size to prefer
# moving large pops over tiny minorities.
#
# Transfer volumes per camp level per month:
#   Homesteading:            ~100  (voluntary settler recruitment)
#   Organized Colonization:  ~250  (government-organized program)
#   Forced Resettlement:     ~400  (state-directed mass transfer)
#   Penal Transportation:    ~150  (convict/undesirable deportation)
# =================================================================
resettlement_transfer_effect = {
	debug_log = "RESETTLE_DEBUG: effect entered"

	# Find a destination state (another state owned by the same country
	# that has a resettlement colony)
	owner = {
		random_scope_state = {
			limit = {
				NOT = { this = ROOT }
				any_scope_building = {
					is_building_type = building_resettlement_colony
					level > 0
				}
			}
			save_scope_as = resettlement_destination
		}
	}

	# Only proceed if we have a valid destination and a nonzero transfer modifier
	if = {
		limit = {
			exists = scope:resettlement_destination
			modifier:state_resettlement_transfer_add > 0
		}

		every_scope_pop = {
			limit = {
				pop_employment_building = building_resettlement_camp
				OR = {
					is_pop_type = laborers
					is_pop_type = slaves
				}
			}
			move_pop = scope:resettlement_destination
		}
	}
}

#==================================================

antimatter_facility_construction = {
	if = {
		limit = {
			NOT = { has_variable = antimatter_facility_progress_var }
		}
		set_variable = {
			name = antimatter_facility_progress_var
			value = 0
		}
	}
	if = {
		limit = {
			has_variable = antimatter_facility_progress_var
			any_scope_building = {
				is_building_type = building_antimatter_facility_construction_site
				building_has_goods_shortage = no
			}
		}
		change_variable = {
			name = antimatter_facility_progress_var
			add = {
				value = b:building_antimatter_facility_construction_site.occupancy
				divide = 12
				if = {
					limit = {
						any_scope_building = { has_active_production_method = pm_antimatter_facility_construction_slow }
					}
					multiply = 0.25
				}
				else_if = {
					limit = {
						any_scope_building = { has_active_production_method = pm_antimatter_facility_construction_medium }
					}
					multiply = 0.5
				}
				else_if = {
					limit = {
						any_scope_building = { has_active_production_method = pm_antimatter_facility_construction_fast }
					}
					multiply = 1
				}
				else = {
					multiply = 0
				}
				multiply = 1000
				ceiling = yes
				divide = 1000
			}
		}
	}
	if = {
		limit = {
			var:antimatter_facility_progress_var >= 1
		}
		set_variable = {
			name = antimatter_facility_progress_var
			value = 0
		}
		remove_building = building_antimatter_facility_construction_site
		if = {
			limit = { has_building = building_antimatter_facility }
			switch = {
				trigger = b:building_antimatter_facility.level
				1 = { create_building = { building = building_antimatter_facility level = 2 } }
				2 = { create_building = { building = building_antimatter_facility level = 3 } }
				3 = { create_building = { building = building_antimatter_facility level = 4 } }
				4 = { create_building = { building = building_antimatter_facility level = 5 } }
				5 = { create_building = { building = building_antimatter_facility level = 6 } }
				6 = { create_building = { building = building_antimatter_facility level = 7 } }
				7 = { create_building = { building = building_antimatter_facility level = 8 } }
				8 = { create_building = { building = building_antimatter_facility level = 9 } }
				9 = { create_building = { building = building_antimatter_facility level = 10 } }
			}
		}
		else = {
			create_building = {
				building = building_antimatter_facility
				level = 1
			}
		}
	}
}
