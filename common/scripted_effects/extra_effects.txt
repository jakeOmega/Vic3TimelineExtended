# Removes $MODIFIER$ from the current scope only if present. Safe to call
# unconditionally — emits no fail-tooltip when the modifier isn't there.
remove_modifier_if_exists_effect = {
	if = {
		limit = { has_modifier = $MODIFIER$ }
		remove_modifier = $MODIFIER$
	}
}

# Country-scope wonder construction tick. Called from each wonder's monthly
# pulse (the wonder's named wrapper). Tracks progress against the construction
# site's annual-progress modifier; on completion, removes the site, posts the
# wonder-complete notification, and either bumps the wonder one level (capped
# at $MAX_LEVEL$) or creates the level-1 wonder if none exists yet.
#
# Parameters:
#   $WONDER$    — wonder building base name (e.g. space_elevator). The
#                 progress variable, construction site, annual-progress
#                 modifier, and notification key all use this token.
#   $MAX_LEVEL$ — wonder's level cap. The single shared 19-branch chain
#                 below guards each branch on `level < $MAX_LEVEL$`, so a
#                 wonder at its cap silently no-ops on completion.
generic_wonder_construction_base = {
	if = {
		limit = { NOT = { has_variable = $WONDER$_progress_var } }
		set_variable = { name = $WONDER$_progress_var value = 0 }
	}
	if = {
		limit = {
			has_variable = $WONDER$_progress_var
			any_scope_building = {
				is_building_type = building_$WONDER$_construction_site
				building_has_goods_shortage = no
			}
		}
		change_variable = {
			name = $WONDER$_progress_var
			add = {
				value = b:building_$WONDER$_construction_site.occupancy
				divide = 12
				multiply = b:building_$WONDER$_construction_site.modifier:building_annual_$WONDER$_progress
				multiply = 1000
				ceiling = yes
				divide = 1000
			}
		}
	}
	if = {
		limit = { var:$WONDER$_progress_var >= 1 }
		set_variable = { name = $WONDER$_progress_var value = 0 }
		remove_building = building_$WONDER$_construction_site
		save_scope_as = wonder_complete_state
		owner = { post_notification = wonder_construction_complete_$WONDER$ }
		if = {
			limit = { has_building = building_$WONDER$ }
			if = { limit = { b:building_$WONDER$.level = 19 b:building_$WONDER$.level < $MAX_LEVEL$ } create_building = { building = building_$WONDER$ level = 20 } }
			else_if = { limit = { b:building_$WONDER$.level = 18 b:building_$WONDER$.level < $MAX_LEVEL$ } create_building = { building = building_$WONDER$ level = 19 } }
			else_if = { limit = { b:building_$WONDER$.level = 17 b:building_$WONDER$.level < $MAX_LEVEL$ } create_building = { building = building_$WONDER$ level = 18 } }
			else_if = { limit = { b:building_$WONDER$.level = 16 b:building_$WONDER$.level < $MAX_LEVEL$ } create_building = { building = building_$WONDER$ level = 17 } }
			else_if = { limit = { b:building_$WONDER$.level = 15 b:building_$WONDER$.level < $MAX_LEVEL$ } create_building = { building = building_$WONDER$ level = 16 } }
			else_if = { limit = { b:building_$WONDER$.level = 14 b:building_$WONDER$.level < $MAX_LEVEL$ } create_building = { building = building_$WONDER$ level = 15 } }
			else_if = { limit = { b:building_$WONDER$.level = 13 b:building_$WONDER$.level < $MAX_LEVEL$ } create_building = { building = building_$WONDER$ level = 14 } }
			else_if = { limit = { b:building_$WONDER$.level = 12 b:building_$WONDER$.level < $MAX_LEVEL$ } create_building = { building = building_$WONDER$ level = 13 } }
			else_if = { limit = { b:building_$WONDER$.level = 11 b:building_$WONDER$.level < $MAX_LEVEL$ } create_building = { building = building_$WONDER$ level = 12 } }
			else_if = { limit = { b:building_$WONDER$.level = 10 b:building_$WONDER$.level < $MAX_LEVEL$ } create_building = { building = building_$WONDER$ level = 11 } }
			else_if = { limit = { b:building_$WONDER$.level = 9 b:building_$WONDER$.level < $MAX_LEVEL$ } create_building = { building = building_$WONDER$ level = 10 } }
			else_if = { limit = { b:building_$WONDER$.level = 8 b:building_$WONDER$.level < $MAX_LEVEL$ } create_building = { building = building_$WONDER$ level = 9 } }
			else_if = { limit = { b:building_$WONDER$.level = 7 b:building_$WONDER$.level < $MAX_LEVEL$ } create_building = { building = building_$WONDER$ level = 8 } }
			else_if = { limit = { b:building_$WONDER$.level = 6 b:building_$WONDER$.level < $MAX_LEVEL$ } create_building = { building = building_$WONDER$ level = 7 } }
			else_if = { limit = { b:building_$WONDER$.level = 5 b:building_$WONDER$.level < $MAX_LEVEL$ } create_building = { building = building_$WONDER$ level = 6 } }
			else_if = { limit = { b:building_$WONDER$.level = 4 b:building_$WONDER$.level < $MAX_LEVEL$ } create_building = { building = building_$WONDER$ level = 5 } }
			else_if = { limit = { b:building_$WONDER$.level = 3 b:building_$WONDER$.level < $MAX_LEVEL$ } create_building = { building = building_$WONDER$ level = 4 } }
			else_if = { limit = { b:building_$WONDER$.level = 2 b:building_$WONDER$.level < $MAX_LEVEL$ } create_building = { building = building_$WONDER$ level = 3 } }
			else_if = { limit = { b:building_$WONDER$.level = 1 b:building_$WONDER$.level < $MAX_LEVEL$ } create_building = { building = building_$WONDER$ level = 2 } }
		}
		else = {
			create_building = { building = building_$WONDER$ level = 1 }
		}
	}
}

# Swap a country's bloc cultural-plurality/unity principles to the opposite
# tier-by-tier, called when the bloc leader's policy lever flips. Country scope.
#   $REMOVE$ — principle stem to strip (plurality | unity)
#   $ADD$    — principle stem to install (the other one)
update_countrys_bloc_principle_base = {
	if = {
		limit = { is_power_bloc_leader = yes }
		power_bloc = {
			if = {
				limit = { has_principle = principle:principle_cultural_$REMOVE$_1 }
				remove_principle = principle_cultural_$REMOVE$_1
				add_principle = principle_cultural_$ADD$_1
			}
			else_if = {
				limit = { has_principle = principle:principle_cultural_$REMOVE$_2 }
				remove_principle = principle_cultural_$REMOVE$_2
				add_principle = principle_cultural_$ADD$_2
			}
			else_if = {
				limit = { has_principle = principle:principle_cultural_$REMOVE$_3 }
				remove_principle = principle_cultural_$REMOVE$_3
				add_principle = principle_cultural_$ADD$_3
			}
			else_if = {
				limit = { has_principle = principle:principle_cultural_$REMOVE$_4 }
				remove_principle = principle_cultural_$REMOVE$_4
				add_principle = principle_cultural_$ADD$_4
			}
			else_if = {
				limit = { has_principle = principle:principle_cultural_$REMOVE$_5 }
				remove_principle = principle_cultural_$REMOVE$_5
				add_principle = principle_cultural_$ADD$_5
			}
		}
	}
}

# Called from state scope with scope:curr_culture set (the persecuted minority).
# Picks one foreign state (highest migration_attraction_script_value, owner accepts
# the culture, not closed-borders) and moves every matching pop in this state there.
# Posts the violence_against_minorities notification once per state per yearly pulse.
#
# Must be called ONCE per culture from state scope — NOT inside every_scope_pop.
# Calling from pop scope re-picks the destination on every iteration, so the
# notification announces a state different from where most pops actually arrive.
# The outer any_state check guarantees ordered_state will save migration_target,
# so no stale-scope handling is needed when called multiple times per pulse.
trigger_mass_migration = {
	if = {
		limit = {
			any_state = {
				NOT = { owner = { has_law = law_type:law_closed_borders } }
				state_cultural_acceptance = {
					target = scope:curr_culture
					value > owner.modifier:country_migration_restrictiveness_add
				}
				NOT = { owner = PREV.owner }
			}
		}
		ordered_state = {
			limit = {
				NOT = { owner = { has_law = law_type:law_closed_borders } }
				state_cultural_acceptance = {
					target = scope:curr_culture
					value > owner.modifier:country_migration_restrictiveness_add
				}
				NOT = { owner = PREV.owner }
			}
			order_by = migration_attraction_script_value
			position = 0

			save_scope_as = migration_target
		}

		every_scope_pop = {
			limit = { culture = scope:curr_culture }
			move_partial_pop = {
				state = scope:migration_target
				population_ratio = $POP_RATIO$
			}
		}

		if = {
			limit = {
				NOT = { has_variable = sent_message }
			}
			post_notification = violence_against_minorities
			set_variable = sent_message
		}
	}
}

# =================================================================
# Resettlement Population Transfer
# =================================================================
# State scope (SOURCE state with a resettlement camp).
# Finds the camp, identifies a destination state with a colony,
# then moves pops employed at the camp to the destination.
#
# Uses ordered_scope_pop with order_by = total_size to prefer
# moving large pops over tiny minorities.
#
# Transfer volumes per camp level per month:
#   Homesteading:            ~100  (voluntary settler recruitment)
#   Organized Colonization:  ~250  (government-organized program)
#   Forced Resettlement:     ~400  (state-directed mass transfer)
#   Penal Transportation:    ~150  (convict/undesirable deportation)
# =================================================================
resettlement_transfer_effect = {
	# Find a destination state (another state owned by the same country
	# that has a resettlement colony)
	owner = {
		random_scope_state = {
			limit = {
				NOT = { this = ROOT }
				any_scope_building = {
					is_building_type = building_resettlement_colony
					level > 0
				}
			}
			save_scope_as = resettlement_destination
		}
	}

	# Only proceed if we have a valid destination and a nonzero transfer modifier
	if = {
		limit = {
			exists = scope:resettlement_destination
			modifier:state_resettlement_transfer_add > 0
		}

		every_scope_pop = {
			limit = {
				pop_employment_building = building_resettlement_camp
				OR = {
					is_pop_type = laborers
					is_pop_type = slaves
				}
			}
			move_pop = scope:resettlement_destination
		}
	}
}

# Population Transfer: Moves primary culture pops of target_country from source_country to target_country
# Scope: country (either source or target, called from treaty on_entry_into_force)
# Expected saved scopes: scope:source_country, scope:target_country
population_transfer_effect = {
	# Initialize transfer tracking
	scope:source_country = {
		set_variable = { name = pop_transfer_count value = 0 }
	}
	scope:target_country = {
		set_variable = { name = pop_transfer_count value = 0 }
	}

	# For each primary culture of the receiving country (target), move matching pops from source
	scope:target_country = {
		every_primary_culture = {
			save_scope_as = transfer_culture
			scope:source_country = {
				every_scope_state = {
					every_scope_pop = {
						limit = {
							culture = { this = scope:transfer_culture }
							# Don't transfer pops whose culture is also primary in the source country
							NOT = { culture = { is_primary_culture_of = scope:source_country } }
						}
						save_scope_as = moving_pop
						# Track how many people are moving
						scope:source_country = {
							change_variable = {
								name = pop_transfer_count
								add = scope:moving_pop.total_size
							}
						}
						scope:target_country = {
							change_variable = {
								name = pop_transfer_count
								add = scope:moving_pop.total_size
							}
						}
						# Move to a random state in target, weighted by population
						scope:target_country = {
							random_scope_state = {
								weight = {
									base = 1
									modifier = {
										add = 10
										state_population >= 100000
									}
									modifier = {
										add = 50
										state_population >= 500000
									}
									modifier = {
										add = 200
										state_population >= 1000000
									}
									modifier = {
										add = 500
										state_population >= 5000000
									}
								}
								save_scope_as = destination_state
							}
						}
						if = {
							limit = { exists = scope:destination_state }
							move_pop = scope:destination_state
						}
					}
				}
			}
		}
	}

	# Apply disruption proportional to share-of-pops moved. The multiplier is
	# precomputed into a single scalar variable at apply-time and read flat by
	# `multiplier = scope:X.var:Y` — the vanilla paris-commune
	# (c:FRA.var:PRC_money_transfer) pattern, since inline `multiplier = { ... }`
	# on `add_modifier` is re-evaluated every decay tick. The strength itself
	# comes from the named script value te_pop_transfer_disruption_strength
	# (inline anonymous script-value blocks can't read total_population as a
	# bareword — see that script value's header).
	scope:source_country = {
		if = {
			limit = {
				has_variable = pop_transfer_count
				var:pop_transfer_count > 0
			}
			set_variable = {
				name = pop_transfer_disruption_strength
				value = te_pop_transfer_disruption_strength
			}
			add_modifier = {
				name = population_transfer_disruption
				years = 5
				is_decaying = yes
				multiplier = scope:source_country.var:pop_transfer_disruption_strength
			}
		}
	}

	# Apply disruption to target_country (receiving pops). Same pattern.
	scope:target_country = {
		if = {
			limit = {
				has_variable = pop_transfer_count
				var:pop_transfer_count > 0
			}
			set_variable = {
				name = pop_transfer_disruption_strength
				value = te_pop_transfer_disruption_strength
			}
			add_modifier = {
				name = population_transfer_disruption
				years = 5
				is_decaying = yes
				multiplier = scope:target_country.var:pop_transfer_disruption_strength
			}
		}
	}
}

# ============================================================================
# IRREDENTISM — third-party annexation notification
#
# Fires irredentism.6 on every country (above insignificant rank, not the
# unifier or annexee) that holds Engaged-or-higher interest in a strategic
# region containing the annexed country's former territory.
#
# Caller MUST save these scopes before invoking:
#   scope:irr_annexed_country  — the country being absorbed (must still be
#                                 valid; capture before the `annex` effect
#                                 runs in voluntary_union, or trust it
#                                 surviving briefly into a wargoal on_enforced)
#   scope:irr_unifier          — the absorbing country
#
# The variable irr_notify_peaceful is set on each notified country before
# the event fires; the event reads it to pick warlike or peaceful flavor.
# Cooldown on the event handles dedupe (the unifier may have multiple
# strategic regions, but each interested party only sees one notification).
# ============================================================================

te_irredentism_notify_third_parties_effect = {
	# Parameter: PEACEFUL = yes | no
	scope:irr_annexed_country ?= {
		every_country_strategic_region = {
			save_temporary_scope_as = irr_target_region
			every_country = {
				limit = {
					NOT = { THIS = scope:irr_unifier }
					NOT = { THIS = scope:irr_annexed_country }
					NOT = { is_country_type = decentralized }
					country_rank >= rank_value:minor_power
					has_strategic_region_interest_tier = {
						strategic_region = scope:irr_target_region
						value >= 2
					}
				}
				set_variable = {
					name = irr_notify_peaceful_flag
					value = $PEACEFUL$
					days = 7
				}
				trigger_event = { id = irredentism.6 days = 1 }
			}
		}
	}
}

# AI-side driver for decentralized absorption. Runs from on_yearly_pulse_country
# (ROOT = the pulsing country). The player uses the voluntary_union_decentralized
# diplomatic action; the engine never offers decentralized countries as autonomous
# AI diplomatic-action targets, so the AI path is driven here with the SAME
# eligibility gates the action's `possible` block uses (decolonization/globalization
# tech, irr_target_reachable, same-nation culture floor). Absorbs at most one
# reachable decentralized neighbour per year. Notifies interested third parties
# BEFORE the annex (irredentism.9) so the target scope is still alive for its name.
te_ai_decentralized_absorption_effect = {
	if = {
		limit = {
			is_ai = yes
			has_technology_researched = decolonization
			is_diplomatic_play_committed_participant = no
		}
		save_scope_as = irr_dc_annexer
		random_country = {
			limit = {
				is_country_type = decentralized
				irr_target_reachable = yes
				NOT = { has_truce_with = scope:irr_dc_annexer }
				# Era-6 floor: same-culture only; globalization (era 9) lifts it.
				# Mirrors voluntary_union_decentralized_culture_tt in the action.
				OR = {
					scope:irr_dc_annexer = { has_technology_researched = globalization }
					irr_target_same_nation = yes
				}
			}
			save_scope_as = irr_dc_target
		}
		if = {
			limit = { exists = scope:irr_dc_target }
			# Notify interested third parties while the target still exists.
			# Same audience as te_irredentism_notify_third_parties_effect:
			# non-decentralized minor+ powers with interest in the target's region.
			scope:irr_dc_target = {
				every_country_strategic_region = {
					save_temporary_scope_as = irr_dc_region
					every_country = {
						limit = {
							NOT = { THIS = scope:irr_dc_annexer }
							NOT = { THIS = scope:irr_dc_target }
							NOT = { is_country_type = decentralized }
							country_rank >= rank_value:minor_power
							has_strategic_region_interest_tier = {
								strategic_region = scope:irr_dc_region
								value >= 2
							}
						}
						trigger_event = { id = irredentism.9 days = 0 }
					}
				}
			}
			scope:irr_dc_annexer = {
				annex = scope:irr_dc_target
			}
		}
	}
}
