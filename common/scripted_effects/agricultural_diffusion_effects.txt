# Agricultural Diffusion Effects
#
# Models global propagation of transformative agricultural techniques:
# the first country to research a diffusion-eligible tech triggers a global
# event applying a permanent state_arable_land_mult modifier to every country
# (including itself), representing how Haber-Bosch / Green Revolution /
# biotech adoption spread beyond their inventors via FAO/CIMMYT/IRRI/USAID.
# The first researcher additionally gets agdiff_first_mover_prestige.
#
# Naming follows the world_first_* / is_world_first_* / first_*_country
# pattern established by nuclear weapons (common/journal_entries/
# timeline_extended_journal_entries.txt:140-164).
#
# Diffusion-eligible techs: nitrogen_fixation, modern_chemical_processes,
# green_revolution, gene_splicing, biotechnology.
#
# Wired in via on_acquired_technology in common/on_actions/extra_on_actions.txt.

# --- Dispatcher (called from agricultural_diffusion_on_action) --------------
# Guard + dispatch in one parameterized block: if $TECH$ is researched and
# no country has yet claimed the world-first, call the per-tech wrapper.
# Replaces five back-to-back inline if-blocks at the on-action call site.
# The wrapper effect name agdiff_on_$TECH$_researched is built via parse-time
# text substitution — same mechanism agdiff_handle_tech_acquired_effect uses
# for $EVENT$.
agdiff_dispatch_if_first = {
	if = {
		limit = {
			has_technology_researched = $TECH$
			NOT = { has_global_variable = world_first_$TECH$ }
		}
		agdiff_on_$TECH$_researched = yes
	}
}

# --- Per-tech entry points --------------------------------------------------
# One named helper per tech so the on_acquired_technology hook can dispatch
# with a `has_technology_researched` guard without needing $TECH$-keyed
# has_global_variable lookups (which the parser handles fine but greps poorly).

agdiff_on_nitrogen_fixation_researched = {
	agdiff_handle_tech_acquired_effect = {
		TECH = nitrogen_fixation
		EVENT = agricultural_diffusion_events.1
	}
}

agdiff_on_modern_chemical_processes_researched = {
	agdiff_handle_tech_acquired_effect = {
		TECH = modern_chemical_processes
		EVENT = agricultural_diffusion_events.2
	}
}

agdiff_on_green_revolution_researched = {
	agdiff_handle_tech_acquired_effect = {
		TECH = green_revolution
		EVENT = agricultural_diffusion_events.3
	}
}

agdiff_on_gene_splicing_researched = {
	agdiff_handle_tech_acquired_effect = {
		TECH = gene_splicing
		EVENT = agricultural_diffusion_events.4
	}
}

agdiff_on_biotechnology_researched = {
	agdiff_handle_tech_acquired_effect = {
		TECH = biotechnology
		EVENT = agricultural_diffusion_events.5
	}
}

# --- Core dispatch ----------------------------------------------------------
# Called by the per-tech helpers above. Only the first researcher trips the
# global-first branch; subsequent researchers get the local tech modifier
# (already attached by the tech entity) plus whatever diffusion modifier they
# previously received when the first researcher fired this effect.
#
# Modifier application lives in the event OPTIONS (agricultural_diffusion_events),
# not here: only option-level add_modifier renders in the event tooltip, so a
# broadcast-then-notify split would read as a no-op to the player. This effect
# only sets the world-first flags (which gate the events' triggered_desc /
# option triggers) and fires the events. The first researcher gets the event
# immediately and applies the diffusion + first-mover-prestige modifiers via its
# .a_first option; every other recognized country gets the event after 14 days
# and applies just the diffusion modifier via its .a option.
agdiff_handle_tech_acquired_effect = {
	if = {
		limit = { NOT = { has_global_variable = world_first_$TECH$ } }
		set_global_variable = { name = world_first_$TECH$ value = yes }
		set_global_variable = { name = first_$TECH$_country value = ROOT }
		set_variable = { name = is_world_first_$TECH$ value = yes }
		# Country-event narrative: same event ID to first researcher and
		# others; the event uses triggered_desc on is_world_first_$TECH$ to
		# branch between first-mover and notification flavor, and applies the
		# diffusion / prestige modifiers in its options (see above).
		trigger_event = { id = $EVENT$ }
		every_country = {
			limit = {
				NOT = { THIS = ROOT }
				is_country_type = recognized
			}
			trigger_event = { id = $EVENT$ days = 14 }
		}
	}
}

# --- Backfill for newly-formed countries -----------------------------------
# Countries that come into existence AFTER the first researcher (decolonization,
# release-as-subject, formables, civil-war secession) miss the every_country
# broadcast in agdiff_handle_tech_acquired_effect. This catch-up effect adds
# any diffusion modifier whose global-first flag is already set but which
# isn't yet on the country. Wired into on_country_formed and the three
# release on-actions in common/on_actions/extra_on_actions.txt.
#
# Without this, decolonized 1960+ African nations (the exact target of issue
# #114) would get zero benefit from the Green Revolution diffusion that fired
# years before they existed.
# Backfill one tech: if the world-first flag is set but this country lacks
# the diffusion modifier, apply it. The agdiff_$TECH$ modifier name follows
# the convention established by the five per-tech wrappers above.
agdiff_backfill_one_tech = {
	if = {
		limit = {
			has_global_variable = world_first_$TECH$
			NOT = { has_modifier = agdiff_$TECH$ }
		}
		add_modifier = { name = agdiff_$TECH$ }
	}
}

agdiff_backfill_diffusion_for_country = {
	agdiff_backfill_one_tech = { TECH = nitrogen_fixation }
	agdiff_backfill_one_tech = { TECH = modern_chemical_processes }
	agdiff_backfill_one_tech = { TECH = green_revolution }
	agdiff_backfill_one_tech = { TECH = gene_splicing }
	agdiff_backfill_one_tech = { TECH = biotechnology }
}
