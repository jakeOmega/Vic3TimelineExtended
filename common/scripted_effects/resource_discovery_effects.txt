try_discover_resource_with_access = {
	if = {
		limit = {
			has_potential_resource = $BUILDING$
			owner = { has_technology_researched = $TECH$ }
		}
		random_list = {
			10 = {
                force_resource_discovery = $BUILDING$
			}
			90 = { }
		}
	}
	if = {
		limit = {
			has_potential_resource = $BUILDING$
			any_country = {
				limit = {
					has_technology_researched = $TECH$
					country_has_resource_discovery_access = { TARGET = ROOT.owner }
				}
			}
		}
		random_list = {
			2 = {
				force_resource_discovery = $BUILDING$
			}
			98 = { }
		}
	}
}
