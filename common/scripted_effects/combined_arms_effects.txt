### Combined Arms Effects ###
# Applies or removes combined arms traits on army generals based on their formation's
# unit composition. Each army unit group provides unique combat bonuses (via command_modifier
# on the trait) when at least 5% of the formation belongs to that group AND the formation
# has 2+ qualifying groups. Traits are per-general (per-formation), so each army benefits
# independently based on its own composition.
# Called from on_monthly_pulse_country via combined_arms_update_on_action.

apply_combined_arms_bonuses = {
	# Step 1: Remove all combined arms traits from all generals (clean slate).
	# This handles generals reassigned or whose formation composition changed.
	every_scope_general = {
		remove_trait = trait_combined_arms_infantry_screen
		remove_trait = trait_combined_arms_fire_support
		remove_trait = trait_combined_arms_recon
		remove_trait = trait_combined_arms_armor
		remove_trait = trait_combined_arms_air_superiority
		remove_trait = trait_combined_arms_full_spectrum
	}

	# Step 2: For each army formation, evaluate composition and apply traits to its general.
	every_military_formation = {
		limit = { is_army = yes }

		# ---- Count units per group ----
		set_local_variable = { name = ca_total value = 0 }
		set_local_variable = { name = ca_infantry value = 0 }
		set_local_variable = { name = ca_artillery value = 0 }
		set_local_variable = { name = ca_cavalry value = 0 }
		set_local_variable = { name = ca_tanks value = 0 }
		set_local_variable = { name = ca_aircraft value = 0 }

		every_combat_unit = {
			change_local_variable = { name = ca_total add = 1 }
			if = {
				limit = { combat_unit_is_infantry_group = yes }
				change_local_variable = { name = ca_infantry add = 1 }
			}
			else_if = {
				limit = { combat_unit_is_artillery_group = yes }
				change_local_variable = { name = ca_artillery add = 1 }
			}
			else_if = {
				limit = { combat_unit_is_cavalry_group = yes }
				change_local_variable = { name = ca_cavalry add = 1 }
			}
			else_if = {
				limit = { combat_unit_is_tank_group = yes }
				change_local_variable = { name = ca_tanks add = 1 }
			}
			else_if = {
				limit = { combat_unit_is_aircraft_group = yes }
				change_local_variable = { name = ca_aircraft add = 1 }
			}
		}

		# ---- Determine which groups meet 5% threshold ----
		# Method: group_count * 20 >= total means group >= 5% of formation.
		# Guard: skip empty formations (0 * 20 - 0 = 0 would pass all checks).
		set_local_variable = { name = ca_groups value = 0 }
		set_local_variable = { name = ca_has_infantry value = 0 }
		set_local_variable = { name = ca_has_artillery value = 0 }
		set_local_variable = { name = ca_has_cavalry value = 0 }
		set_local_variable = { name = ca_has_tanks value = 0 }
		set_local_variable = { name = ca_has_aircraft value = 0 }

		if = {
			limit = { local_var:ca_total >= 1 }

			# Infantry threshold check
			set_local_variable = { name = ca_check value = local_var:ca_infantry }
			change_local_variable = { name = ca_check multiply = 20 }
			change_local_variable = { name = ca_check add = { value = local_var:ca_total multiply = -1 } }
			if = {
				limit = { local_var:ca_check >= 0 }
				set_local_variable = { name = ca_has_infantry value = 1 }
				change_local_variable = { name = ca_groups add = 1 }
			}

			# Artillery threshold check
			set_local_variable = { name = ca_check value = local_var:ca_artillery }
			change_local_variable = { name = ca_check multiply = 20 }
			change_local_variable = { name = ca_check add = { value = local_var:ca_total multiply = -1 } }
			if = {
				limit = { local_var:ca_check >= 0 }
				set_local_variable = { name = ca_has_artillery value = 1 }
				change_local_variable = { name = ca_groups add = 1 }
			}

			# Cavalry/Recon threshold check
			set_local_variable = { name = ca_check value = local_var:ca_cavalry }
			change_local_variable = { name = ca_check multiply = 20 }
			change_local_variable = { name = ca_check add = { value = local_var:ca_total multiply = -1 } }
			if = {
				limit = { local_var:ca_check >= 0 }
				set_local_variable = { name = ca_has_cavalry value = 1 }
				change_local_variable = { name = ca_groups add = 1 }
			}

			# Tanks threshold check
			set_local_variable = { name = ca_check value = local_var:ca_tanks }
			change_local_variable = { name = ca_check multiply = 20 }
			change_local_variable = { name = ca_check add = { value = local_var:ca_total multiply = -1 } }
			if = {
				limit = { local_var:ca_check >= 0 }
				set_local_variable = { name = ca_has_tanks value = 1 }
				change_local_variable = { name = ca_groups add = 1 }
			}

			# Aircraft threshold check
			set_local_variable = { name = ca_check value = local_var:ca_aircraft }
			change_local_variable = { name = ca_check multiply = 20 }
			change_local_variable = { name = ca_check add = { value = local_var:ca_total multiply = -1 } }
			if = {
				limit = { local_var:ca_check >= 0 }
				set_local_variable = { name = ca_has_aircraft value = 1 }
				change_local_variable = { name = ca_groups add = 1 }
			}
		}

		# ---- Apply traits to this formation's general ----
		every_scope_general = {
			# Infantry Screen: defense bonus
			if = {
				limit = {
					local_var:ca_groups >= 2
					local_var:ca_has_infantry >= 1
				}
				add_trait = trait_combined_arms_infantry_screen
			}
			# Fire Support: offense bonus
			if = {
				limit = {
					local_var:ca_groups >= 2
					local_var:ca_has_artillery >= 1
				}
				add_trait = trait_combined_arms_fire_support
			}
			# Recon: morale resilience bonus
			if = {
				limit = {
					local_var:ca_groups >= 2
					local_var:ca_has_cavalry >= 1
				}
				add_trait = trait_combined_arms_recon
			}
			# Armored Spearhead: offense+defense bonus
			if = {
				limit = {
					local_var:ca_groups >= 2
					local_var:ca_has_tanks >= 1
				}
				add_trait = trait_combined_arms_armor
			}
			# Air Superiority: offense+morale damage bonus
			if = {
				limit = {
					local_var:ca_groups >= 2
					local_var:ca_has_aircraft >= 1
				}
				add_trait = trait_combined_arms_air_superiority
			}
			# Full Spectrum Dominance: all 5 groups present
			if = {
				limit = { local_var:ca_groups >= 5 }
				add_trait = trait_combined_arms_full_spectrum
			}
		}
	}
}
