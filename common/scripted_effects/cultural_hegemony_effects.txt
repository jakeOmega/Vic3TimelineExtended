# ---- MONTHLY: Per-country cache + modifier application ----
# Scope: Country
# Caches all display variables and applies/removes CH modifiers.
ch_monthly_country_update = {
	# Cache variables used by JE triggers (visibility/status conditions)
	set_variable = { name = ch_total value = cultural_pull_total }
	set_variable = { name = ch_art value = cultural_pull_from_art }
	set_variable = { name = ch_sol value = cultural_pull_from_sol }
	set_variable = { name = ch_monuments value = cultural_pull_from_monuments }
	set_variable = { name = ch_megaprojects value = cultural_pull_from_megaprojects }

	# Apply/remove foreign benchmark (SoL expectations from hegemon)
	if = {
		limit = { cultural_hegemony_benchmark_mult >= 0.1 }
		remove_modifier = cultural_hegemony_foreign_benchmark
		add_modifier = { name = cultural_hegemony_foreign_benchmark multiplier = cultural_hegemony_benchmark_mult }
	}
	else_if = {
		limit = { has_modifier = cultural_hegemony_foreign_benchmark }
		remove_modifier = cultural_hegemony_foreign_benchmark
	}
}

# ---- YEARLY: Refresh global caches + build leaderboard ----
# Scope: Country (uses lock guard, only first caller runs leaderboard)
ch_yearly_global_update = {
	if = {
		limit = { NOT = { has_global_variable = ch_ldr_lock } }
		set_global_variable = { name = ch_ldr_lock value = yes days = 180 }

		set_global_variable = { name = ch_cached_prestige value = global_prestige_total }
		set_global_variable = { name = ch_cached_sol_avg value = global_average_sol }

		set_global_variable = { name = ch_cached_global_art value = global_fine_art_production }

		# Save previous rank scores for year-over-year delta
		if = {
			limit = { has_global_variable = ch_rank_1_score }
			set_global_variable = { name = ch_rank_1_prev value = global_var:ch_rank_1_score }
		}
		if = {
			limit = { has_global_variable = ch_rank_2_score }

			set_global_variable = { name = ch_rank_9_prev value = global_var:ch_rank_9_score }

		remove_global_variable = ch_rank_9
		remove_global_variable = ch_rank_10

		# Single ordered pass: top 10 countries by raw cultural pull
		set_global_variable = { name = ch_ldr_idx value = 0 }
		ordered_country = {
			limit = { cultural_pull_raw >= 0.01 }
			order_by = cultural_pull_raw
			check_range_bounds = no

			if = {
				limit = { global_var:ch_ldr_idx < 1 }
				set_global_variable = { name = ch_rank_1 value = THIS }
				set_global_variable = { name = ch_rank_1_score value = cultural_pull_display }

				set_global_variable = { name = ch_rank_1_art value = cultural_pull_from_art_display }

				set_global_variable = { name = ch_rank_1_raw value = cultural_pull_raw_display }
			}
			else_if = {
				limit = { global_var:ch_ldr_idx < 2 }
				set_global_variable = { name = ch_rank_2 value = THIS }
				set_global_variable = { name = ch_rank_2_score value = cultural_pull_display }
				set_global_variable = { name = ch_rank_2_art value = cultural_pull_from_art_display }
				set_global_variable = { name = ch_rank_2_prs value = cultural_pull_from_prestige_display }
				set_global_variable = { name = ch_rank_2_sol value = cultural_pull_from_sol_display }
				set_global_variable = { name = ch_rank_2_tech value = cultural_pull_from_tech_display }
				set_global_variable = { name = ch_rank_2_raw value = cultural_pull_raw_display }
			}
			else_if = {
				limit = { global_var:ch_ldr_idx < 3 }
				set_global_variable = { name = ch_rank_3 value = THIS }
				set_global_variable = { name = ch_rank_3_score value = cultural_pull_display }
				set_global_variable = { name = ch_rank_3_art value = cultural_pull_from_art_display }
				set_global_variable = { name = ch_rank_3_prs value = cultural_pull_from_prestige_display }
				set_global_variable = { name = ch_rank_3_sol value = cultural_pull_from_sol_display }
				set_global_variable = { name = ch_rank_3_tech value = cultural_pull_from_tech_display }

				set_global_variable = { name = ch_rank_4 value = THIS }
				set_global_variable = { name = ch_rank_4_score value = cultural_pull_display }
				set_global_variable = { name = ch_rank_4_art value = cultural_pull_from_art_display }
				set_global_variable = { name = ch_rank_4_prs value = cultural_pull_from_prestige_display }

			}
			else_if = {
				limit = { global_var:ch_ldr_idx < 5 }
				set_global_variable = { name = ch_rank_5 value = THIS }
				set_global_variable = { name = ch_rank_5_score value = cultural_pull_display }
				set_global_variable = { name = ch_rank_5_art value = cultural_pull_from_art_display }
				set_global_variable = { name = ch_rank_5_prs value = cultural_pull_from_prestige_display }
				set_global_variable = { name = ch_rank_5_sol value = cultural_pull_from_sol_display }
				set_global_variable = { name = ch_rank_5_tech value = cultural_pull_from_tech_display }
				set_global_variable = { name = ch_rank_5_raw value = cultural_pull_raw_display }
			}
			else_if = {
				limit = { global_var:ch_ldr_idx < 6 }
				set_global_variable = { name = ch_rank_6 value = THIS }
				set_global_variable = { name = ch_rank_6_score value = cultural_pull_display }
				set_global_variable = { name = ch_rank_6_art value = cultural_pull_from_art_display }
				set_global_variable = { name = ch_rank_6_prs value = cultural_pull_from_prestige_display }
				set_global_variable = { name = ch_rank_6_sol value = cultural_pull_from_sol_display }
				set_global_variable = { name = ch_rank_6_tech value = cultural_pull_from_tech_display }
				set_global_variable = { name = ch_rank_6_raw value = cultural_pull_raw_display }
			}
			else_if = {
				limit = { global_var:ch_ldr_idx < 7 }
				set_global_variable = { name = ch_rank_7 value = THIS }
				set_global_variable = { name = ch_rank_7_score value = cultural_pull_display }
				set_global_variable = { name = ch_rank_7_art value = cultural_pull_from_art_display }
				set_global_variable = { name = ch_rank_7_prs value = cultural_pull_from_prestige_display }
				set_global_variable = { name = ch_rank_7_sol value = cultural_pull_from_sol_display }
				set_global_variable = { name = ch_rank_7_tech value = cultural_pull_from_tech_display }
				set_global_variable = { name = ch_rank_7_raw value = cultural_pull_raw_display }
			}
			else_if = {
				limit = { global_var:ch_ldr_idx < 8 }
				set_global_variable = { name = ch_rank_8 value = THIS }
				set_global_variable = { name = ch_rank_8_score value = cultural_pull_display }
				set_global_variable = { name = ch_rank_8_art value = cultural_pull_from_art_display }
				set_global_variable = { name = ch_rank_8_prs value = cultural_pull_from_prestige_display }
				set_global_variable = { name = ch_rank_8_sol value = cultural_pull_from_sol_display }
				set_global_variable = { name = ch_rank_8_tech value = cultural_pull_from_tech_display }
				set_global_variable = { name = ch_rank_8_raw value = cultural_pull_raw_display }
			}
			else_if = {
				limit = { global_var:ch_ldr_idx < 9 }
				set_global_variable = { name = ch_rank_9 value = THIS }
				set_global_variable = { name = ch_rank_9_score value = cultural_pull_display }
				set_global_variable = { name = ch_rank_9_art value = cultural_pull_from_art_display }
				set_global_variable = { name = ch_rank_9_prs value = cultural_pull_from_prestige_display }
				set_global_variable = { name = ch_rank_9_sol value = cultural_pull_from_sol_display }
				set_global_variable = { name = ch_rank_9_tech value = cultural_pull_from_tech_display }
				set_global_variable = { name = ch_rank_9_raw value = cultural_pull_raw_display }
			}
			else_if = {
				limit = { global_var:ch_ldr_idx < 10 }
				set_global_variable = { name = ch_rank_10 value = THIS }
				set_global_variable = { name = ch_rank_10_score value = cultural_pull_display }
				set_global_variable = { name = ch_rank_10_art value = cultural_pull_from_art_display }
				set_global_variable = { name = ch_rank_10_prs value = cultural_pull_from_prestige_display }
				set_global_variable = { name = ch_rank_10_sol value = cultural_pull_from_sol_display }
				set_global_variable = { name = ch_rank_10_tech value = cultural_pull_from_tech_display }
				set_global_variable = { name = ch_rank_10_raw value = cultural_pull_raw_display }
			}
			change_global_variable = { name = ch_ldr_idx add = 1 }
		}

		# Compute year-over-year deltas
		if = {
			limit = { has_global_variable = ch_rank_1_prev }
			set_global_variable = { name = ch_rank_1_delta value = global_var:ch_rank_1_score }
			change_global_variable = { name = ch_rank_1_delta subtract = global_var:ch_rank_1_prev }
		}
		if = {
			limit = { has_global_variable = ch_rank_2_prev }
			set_global_variable = { name = ch_rank_2_delta value = global_var:ch_rank_2_score }
			change_global_variable = { name = ch_rank_2_delta subtract = global_var:ch_rank_2_prev }
		}
		if = {
			limit = { has_global_variable = ch_rank_3_prev }
			set_global_variable = { name = ch_rank_3_delta value = global_var:ch_rank_3_score }
			change_global_variable = { name = ch_rank_3_delta subtract = global_var:ch_rank_3_prev }
		}
		if = {
			limit = { has_global_variable = ch_rank_4_prev }
			set_global_variable = { name = ch_rank_4_delta value = global_var:ch_rank_4_score }
			change_global_variable = { name = ch_rank_4_delta subtract = global_var:ch_rank_4_prev }
		}
		if = {
			limit = { has_global_variable = ch_rank_5_prev }
			set_global_variable = { name = ch_rank_5_delta value = global_var:ch_rank_5_score }
			change_global_variable = { name = ch_rank_5_delta subtract = global_var:ch_rank_5_prev }
		}
		if = {
			limit = { has_global_variable = ch_rank_6_prev }
			set_global_variable = { name = ch_rank_6_delta value = global_var:ch_rank_6_score }
			change_global_variable = { name = ch_rank_6_delta subtract = global_var:ch_rank_6_prev }
		}
		if = {
			limit = { has_global_variable = ch_rank_7_prev }
			set_global_variable = { name = ch_rank_7_delta value = global_var:ch_rank_7_score }
			change_global_variable = { name = ch_rank_7_delta subtract = global_var:ch_rank_7_prev }
		}
		if = {
			limit = { has_global_variable = ch_rank_8_prev }
			set_global_variable = { name = ch_rank_8_delta value = global_var:ch_rank_8_score }
			change_global_variable = { name = ch_rank_8_delta subtract = global_var:ch_rank_8_prev }
		}
		if = {
			limit = { has_global_variable = ch_rank_9_prev }
			set_global_variable = { name = ch_rank_9_delta value = global_var:ch_rank_9_score }
			change_global_variable = { name = ch_rank_9_delta subtract = global_var:ch_rank_9_prev }
		}
		if = {
			limit = { has_global_variable = ch_rank_10_prev }
			set_global_variable = { name = ch_rank_10_delta value = global_var:ch_rank_10_score }
			change_global_variable = { name = ch_rank_10_delta subtract = global_var:ch_rank_10_prev }
		}

		# Decay tech firsts counter for all countries (inside lock guard so it
		# runs exactly once per cycle, even when triggered by war-end recalc)
		every_country = {
			limit = {
				has_variable = ch_tech_firsts_recent
				var:ch_tech_firsts_recent > 0
			}
			change_variable = { name = ch_tech_firsts_recent multiply = cultural_hegemony_tech_firsts_decay_mult }
			clamp_variable = { name = ch_tech_firsts_recent min = 0 max = 30 }
		}
	}
}

# ---- HEGEMON IDEOLOGY → MOVEMENT PRESSURE ----
# Scope: Country (the target being influenced)
# Requires scope:cultural_hegemon to be set.
# Detects the hegemon's ideological profile from laws and boosts
# the matching political movements in this country.
# Parameters: $MODIFIER$ (modifier name), $MONTHS$ (duration)
#
# Structure: EXCLUSIVE governance checks (else_if chain — pick one primary
# political model) PLUS NON-EXCLUSIVE social policy checks (independent if
# blocks — e.g. tolerant + feminist + environmentalist can all fire).
ch_apply_hegemon_movement_pressure = {
	if = {
		limit = { NOT = { exists = scope:cultural_hegemon } }
	}
	else = {
		# ================================================================
		# PART 1: EXCLUSIVE — Primary political model (pick one)
		# ================================================================

		# Communist/Vanguardist hegemon
		if = {
			limit = {
				scope:cultural_hegemon = {
					OR = {
						has_law = law_type:law_council_republic
						AND = {
							has_law = law_type:law_single_party_state
							has_law = law_type:law_command_economy
						}
					}
				}
			}
			every_political_movement = {
				limit = { is_political_movement_type = movement_communist }
				add_modifier = { name = $MODIFIER$ months = $MONTHS$ is_decaying = yes }
			}
		}
		# Anarchist hegemon
		else_if = {
			limit = {
				scope:cultural_hegemon = { has_law = law_type:law_anarchy }
			}
			every_political_movement = {
				limit = { is_political_movement_type = movement_anarchist }
				add_modifier = { name = $MODIFIER$ months = $MONTHS$ is_decaying = yes }
			}
		}
		# Fascist/Ethnonationalist hegemon
		else_if = {
			limit = {
				scope:cultural_hegemon = {
					OR = {
						has_law = law_type:law_ethnostate
						AND = {
							has_law = law_type:law_national_supremacy
							OR = {
								has_law = law_type:law_single_party_state
								has_law = law_type:law_autocracy
							}
						}
					}
				}
			}
			every_political_movement = {
				limit = { is_political_movement_type = movement_fascist }
				add_modifier = { name = $MODIFIER$ months = $MONTHS$ is_decaying = yes }
			}
		}
		# Corporatist hegemon
		else_if = {
			limit = {
				scope:cultural_hegemon = { has_law = law_type:law_corporate_state }
			}
			every_political_movement = {
				limit = { is_political_movement_type = movement_corporatist }
				add_modifier = { name = $MODIFIER$ months = $MONTHS$ is_decaying = yes }
			}
		}
		# Socialist hegemon (cooperative ownership, not communist/anarchist)
		else_if = {
			limit = {
				scope:cultural_hegemon = { has_law = law_type:law_cooperative_ownership }
			}
			every_political_movement = {
				limit = { is_political_movement_type = movement_socialist }
				add_modifier = { name = $MODIFIER$ months = $MONTHS$ is_decaying = yes }
			}
		}
		# Absolutist monarchist hegemon
		else_if = {
			limit = {
				scope:cultural_hegemon = {
					has_law = law_type:law_monarchy
					has_law = law_type:law_autocracy
				}
			}
			every_political_movement = {
				limit = { is_political_movement_type = movement_royalist_absolutist }
				add_modifier = { name = $MODIFIER$ months = $MONTHS$ is_decaying = yes }
			}
		}
		# Constitutional monarchist hegemon
		else_if = {
			limit = {
				scope:cultural_hegemon = {
					has_law = law_type:law_monarchy
					NOT = { has_law = law_type:law_autocracy }
				}
			}
			every_political_movement = {
				limit = { is_political_movement_type = movement_royalist_constitutional }
				add_modifier = { name = $MODIFIER$ months = $MONTHS$ is_decaying = yes }
			}
		}
		# Theocratic/religious hegemon
		else_if = {
			limit = {
				scope:cultural_hegemon = {
					has_law = law_type:law_theocracy
					has_law = law_type:law_state_religion
				}
			}
			every_political_movement = {
				limit = { is_political_movement_type = movement_religious_majority }
				add_modifier = { name = $MODIFIER$ months = $MONTHS$ is_decaying = yes }
			}
		}
		# Reactionary/authoritarian hegemon (autocracy + state religion, not monarchy/theocracy)
		else_if = {
			limit = {
				scope:cultural_hegemon = {
					has_law = law_type:law_autocracy
					has_law = law_type:law_state_religion
				}
			}
			every_political_movement = {
				limit = { is_political_movement_type = movement_reactionary }
				add_modifier = { name = $MODIFIER$ months = $MONTHS$ is_decaying = yes }
			}
		}
		# Military junta hegemon (autocracy/single-party + professional army + secret police)
		else_if = {
			limit = {
				scope:cultural_hegemon = {
					OR = {
						has_law = law_type:law_autocracy
						has_law = law_type:law_single_party_state
					}
					has_law = law_type:law_professional_army
					OR = {
						has_law = law_type:law_secret_police
						has_law = law_type:law_militarized_police
					}
				}
			}
			# Military juntas inspire reactionary movements abroad
			every_political_movement = {
				limit = { is_political_movement_type = movement_reactionary }
				add_modifier = { name = $MODIFIER$ months = $MONTHS$ is_decaying = yes }
			}
		}
		# Liberal/Progressive democratic hegemon
		else_if = {
			limit = {
				scope:cultural_hegemon = {
					has_law = law_type:law_universal_suffrage
					OR = {
						has_law = law_type:law_protected_speech
						has_law = law_type:law_right_of_assembly
					}
				}
			}
			every_political_movement = {
				limit = { is_political_movement_type = movement_liberal }
				add_modifier = { name = $MODIFIER$ months = $MONTHS$ is_decaying = yes }
			}
		}

		# ================================================================
		# PART 2: NON-EXCLUSIVE — Social policies (can stack with above)
		# ================================================================

		# Feminist hegemon → boosts feminist movements
		if = {
			limit = {
				scope:cultural_hegemon = { has_law = law_type:law_womens_suffrage }
			}
			every_political_movement = {
				limit = { is_political_movement_type = movement_feminist }
				add_modifier = { name = $MODIFIER$ months = $MONTHS$ is_decaying = yes }
			}
		}

		# Tolerant hegemon → boosts civil rights and minority rights movements
		if = {
			limit = {
				scope:cultural_hegemon = { has_law = law_type:law_multicultural }
			}
			every_political_movement = {
				limit = {
					OR = {
						is_political_movement_type = movement_civil_rights
						is_political_movement_type = movement_minority_rights
					}
				}
				add_modifier = { name = $MODIFIER$ months = $MONTHS$ is_decaying = yes }
			}
		}

		# Environmentalist hegemon → boosts environmental movements
		if = {
			limit = {
				scope:cultural_hegemon = { has_law = law_type:law_ministry_of_the_environment }
			}
			every_political_movement = {
				limit = { is_political_movement_type = movement_environmental }
				add_modifier = { name = $MODIFIER$ months = $MONTHS$ is_decaying = yes }
			}
		}

		# Peaceful hegemon → boosts anti-war movements
		if = {
			limit = {
				scope:cultural_hegemon = {
					is_at_war = no
					NOT = { has_law = law_type:law_mass_conscription }
				}
			}
			every_political_movement = {
				limit = { is_political_movement_type = movement_anti_war }
				add_modifier = { name = $MODIFIER$ months = $MONTHS$ is_decaying = yes }
			}
		}

		# Abolitionist hegemon → boosts anti-slavery movements (only if target has slavery)
		if = {
			limit = {
				scope:cultural_hegemon = { has_law = law_type:law_slavery_banned }
				NOT = { has_law = law_type:law_slavery_banned }
			}
			every_political_movement = {
				limit = { is_political_movement_type = movement_anti_slavery }
				add_modifier = { name = $MODIFIER$ months = $MONTHS$ is_decaying = yes }
			}
		}

		# Pro-labor hegemon → boosts labor movements
		if = {
			limit = {
				scope:cultural_hegemon = {
					OR = {
						has_law = law_type:law_worker_protections
						has_law = law_type:law_regulatory_bodies
					}
				}
			}
			every_political_movement = {
				limit = { is_political_movement_type = movement_labor }
				add_modifier = { name = $MODIFIER$ months = $MONTHS$ is_decaying = yes }
			}
		}

		# Land reform hegemon → boosts land reform movements
		if = {
			limit = {
				scope:cultural_hegemon = {
					OR = {
						has_law = law_type:law_homesteading
						has_law = law_type:law_collectivized_agriculture
					}
				}
			}
			every_political_movement = {
				limit = { is_political_movement_type = movement_land_reform }
				add_modifier = { name = $MODIFIER$ months = $MONTHS$ is_decaying = yes }
			}
		}
	}
}
