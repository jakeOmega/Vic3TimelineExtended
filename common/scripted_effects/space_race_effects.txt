# Space Race Scripted Effects

# Recalculate consolidated space program cost modifier
# Called whenever funding/approach changes or a milestone completes
sr_recalculate_cost = {
	# Remove old consolidated modifier
	if = {
		limit = { has_modifier = sr_space_program_cost }
		remove_modifier = sr_space_program_cost
	}
	# Remove legacy per-modifier costs (migration from shared system)
	if = {
		limit = { has_modifier = sr_space_program_funding }
		remove_modifier = sr_space_program_funding
	}
	if = {
		limit = { has_modifier = sr_safe_approach }
		remove_modifier = sr_safe_approach
	}
	if = {
		limit = { has_modifier = sr_ambitious_approach }
		remove_modifier = sr_ambitious_approach
	}
	# Apply consolidated cost if any
	if = {
		limit = { sr_total_space_cost > 0 }
		add_modifier = {
			name = sr_space_program_cost
			multiplier = sr_total_space_cost
		}
	}
}

# Switch the last-failed milestone to safe approach
# Uses sr_last_failed_milestone marker set before firing failure events
# Switch failed milestone(s) to safe approach using per-JE flags
# Each flag sr_failed_<m> is set in the monthly pulse before firing failure events
# Uses if (not else_if) so multiple simultaneous failures are all processed
sr_switch_to_safe_effect = {
	if = {
		limit = { has_variable = sr_failed_suborbital }
		if = { limit = { has_variable = sr_ambitious_suborbital } remove_variable = sr_ambitious_suborbital }
		set_variable = { name = sr_safe_suborbital value = yes }
		remove_variable = sr_failed_suborbital
	}
	if = {
		limit = { has_variable = sr_failed_orbital }
		if = { limit = { has_variable = sr_ambitious_orbital } remove_variable = sr_ambitious_orbital }
		set_variable = { name = sr_safe_orbital value = yes }
		remove_variable = sr_failed_orbital
	}
	if = {
		limit = { has_variable = sr_failed_moon_landing }
		if = { limit = { has_variable = sr_ambitious_moon_landing } remove_variable = sr_ambitious_moon_landing }
		set_variable = { name = sr_safe_moon_landing value = yes }
		remove_variable = sr_failed_moon_landing
	}
	if = {
		limit = { has_variable = sr_failed_probe }
		if = { limit = { has_variable = sr_ambitious_probe } remove_variable = sr_ambitious_probe }
		set_variable = { name = sr_safe_probe value = yes }
		remove_variable = sr_failed_probe
	}
	if = {
		limit = { has_variable = sr_failed_moon_base }
		if = { limit = { has_variable = sr_ambitious_moon_base } remove_variable = sr_ambitious_moon_base }
		set_variable = { name = sr_safe_moon_base value = yes }
		remove_variable = sr_failed_moon_base
	}
	if = {
		limit = { has_variable = sr_failed_mars_landing }
		if = { limit = { has_variable = sr_ambitious_mars_landing } remove_variable = sr_ambitious_mars_landing }
		set_variable = { name = sr_safe_mars_landing value = yes }
		remove_variable = sr_failed_mars_landing
	}
	if = {
		limit = { has_variable = sr_failed_interstellar_probe }
		if = { limit = { has_variable = sr_ambitious_interstellar_probe } remove_variable = sr_ambitious_interstellar_probe }
		set_variable = { name = sr_safe_interstellar_probe value = yes }
		remove_variable = sr_failed_interstellar_probe
	}
	if = {
		limit = { has_variable = sr_failed_solar_colonization }
		if = { limit = { has_variable = sr_ambitious_solar_colonization } remove_variable = sr_ambitious_solar_colonization }
		set_variable = { name = sr_safe_solar_colonization value = yes }
		remove_variable = sr_failed_solar_colonization
	}
	sr_recalculate_cost = yes
}


# Boost progress for all active milestones
# Set sr_progress_boost before calling (can be negative for penalties)
sr_boost_active_milestones = {
	if = {
		limit = { has_variable = sr_active_suborbital }
		change_variable = { name = sr_progress_suborbital add = var:sr_progress_boost }
	}
	if = {
		limit = { has_variable = sr_active_orbital }
		change_variable = { name = sr_progress_orbital add = var:sr_progress_boost }
	}
	if = {
		limit = { has_variable = sr_active_moon_landing }
		change_variable = { name = sr_progress_moon_landing add = var:sr_progress_boost }
	}
	if = {
		limit = { has_variable = sr_active_probe }
		change_variable = { name = sr_progress_probe add = var:sr_progress_boost }
	}
	if = {
		limit = { has_variable = sr_active_moon_base }
		change_variable = { name = sr_progress_moon_base add = var:sr_progress_boost }
	}
	if = {
		limit = { has_variable = sr_active_mars_landing }
		change_variable = { name = sr_progress_mars_landing add = var:sr_progress_boost }
	}
	if = {
		limit = { has_variable = sr_active_interstellar_probe }
		change_variable = { name = sr_progress_interstellar_probe add = var:sr_progress_boost }
	}
	if = {
		limit = { has_variable = sr_active_solar_colonization }
		change_variable = { name = sr_progress_solar_colonization add = var:sr_progress_boost }
	}
	remove_variable = sr_progress_boost
}

# Called when a milestone is completed - handles first/subsequent rewards and cleanup
sr_complete_milestone_effect = {
	# Remove legacy shared approach modifiers (migration from old system)
	if = {
		limit = { has_modifier = sr_safe_approach }
		remove_modifier = sr_safe_approach
	}
	if = {
		limit = { has_modifier = sr_ambitious_approach }
		remove_modifier = sr_ambitious_approach
	}
	# Remove legacy shared approach variables
	if = {
		limit = { has_variable = sr_approach_safe }
		remove_variable = sr_approach_safe
	}
	if = {
		limit = { has_variable = sr_approach_ambitious }
		remove_variable = sr_approach_ambitious
	}
	# Reset shared milestone progress (legacy compat)
	set_variable = { name = sr_milestone_progress value = 0 }
	# Remove active milestone flag
	if = {
		limit = { has_variable = sr_active_milestone }
		remove_variable = sr_active_milestone
	}
	# Recalculate consolidated cost (per-JE vars cleaned in per-milestone effect)
	sr_recalculate_cost = yes
}

# Suborbital completion
sr_complete_suborbital_effect = {
	set_variable = { name = sr_completed_suborbital value = yes }
	if = {
		limit = { NOT = { has_global_variable = sr_global_first_suborbital } }
		set_global_variable = { name = sr_global_first_suborbital value = yes }
		set_variable = { name = sr_was_first_suborbital value = yes }
		add_modifier = { name = sr_first_suborbital }
		# Notify other countries
		save_scope_as = sr_achiever
		every_country = {
			limit = {
				NOT = { THIS = scope:sr_achiever }
				has_technology_researched = rocketry
				OR = {
					country_rank = rank_value:great_power
					country_rank = rank_value:major_power
				}
			}
			trigger_event = { id = space_race_events.20 days = 14 }
		}
	}
	else = {
		add_modifier = { name = sr_suborbital }
	}
	# Clean up per-JE variables
	if = { limit = { has_variable = sr_funding_suborbital } remove_variable = sr_funding_suborbital }
	if = { limit = { has_variable = sr_safe_suborbital } remove_variable = sr_safe_suborbital }
	if = { limit = { has_variable = sr_ambitious_suborbital } remove_variable = sr_ambitious_suborbital }
	if = { limit = { has_variable = sr_progress_suborbital } remove_variable = sr_progress_suborbital }
	sr_complete_milestone_effect = yes
	add_loyalists = { value = small_radicals pop_type = academics }
}

# Check if current colonization stage is complete and advance
sr_check_colony_stage_effect = {
	# Stage 1 complete → Stage 2
	if = {
		limit = {
			global_var:sr_colonization_stage = 1
			has_global_variable = sr_colony_valles_marineris
			has_global_variable = sr_colony_olympus_mons
			has_global_variable = sr_colony_hellas_planitia
			has_global_variable = sr_colony_utopia_planitia
			has_global_variable = sr_colony_arcadia_planitia
			has_global_variable = sr_colony_ceres
			has_global_variable = sr_colony_vesta
			has_global_variable = sr_colony_psyche
			has_global_variable = sr_colony_pallas
			has_global_variable = sr_colony_hygiea
		}
		set_global_variable = { name = sr_colonization_stage value = 2 }
	}
	# Stage 2 complete → Stage 3
	else_if = {
		limit = {
			global_var:sr_colonization_stage = 2
			has_global_variable = sr_colony_io
			has_global_variable = sr_colony_europa
			has_global_variable = sr_colony_ganymede

			has_global_variable = sr_colony_venus
		}
		set_global_variable = { name = sr_colonization_stage value = 3 }
	}
	# Stage 3 complete → Stage 4
	else_if = {
		limit = {
			global_var:sr_colonization_stage = 3
			has_global_variable = sr_colony_mercury
			has_global_variable = sr_colony_titan
			has_global_variable = sr_colony_enceladus
			has_global_variable = sr_colony_rhea
			has_global_variable = sr_colony_mimas
			has_global_variable = sr_colony_iapetus
		}
		set_global_variable = { name = sr_colonization_stage value = 4 }
	}
	# Stage 4 complete → Stage 5
	else_if = {
		limit = {
			global_var:sr_colonization_stage = 4
			has_global_variable = sr_colony_titania
			has_global_variable = sr_colony_oberon
			has_global_variable = sr_colony_miranda
			has_global_variable = sr_colony_ariel
			has_global_variable = sr_colony_triton
			has_global_variable = sr_colony_proteus
		}
		set_global_variable = { name = sr_colonization_stage value = 5 }
	}
	# Stage 5 complete → All done
	else_if = {
		limit = {
			global_var:sr_colonization_stage = 5
			has_global_variable = sr_colony_pluto
			has_global_variable = sr_colony_eris
			has_global_variable = sr_colony_makemake
			has_global_variable = sr_colony_haumea
			has_global_variable = sr_colony_sedna
		}
		set_global_variable = { name = sr_all_colonies_complete value = yes }
	}
}
