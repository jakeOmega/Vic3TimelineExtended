# Space Race Scripted Effects

# Per-JE on_monthly_pulse body for the milestone progress tick. Country scope.
# Drives 7 of the 8 milestone JEs (solar_colonization has a structurally
# different progress loop and stays inline). The base risk literal is read
# by the `sr_failure_weight` script value via `var:sr_base_risk`.
#
# Parameters:
#   $MILESTONE$ — milestone token (suborbital, orbital, moon_landing, probe,
#                 moon_base, mars_landing, interstellar_probe). Resolves the
#                 sr_safe_/sr_ambitious_/sr_progress_/sr_failed_ variable
#                 namespace.
#   $RISK$      — base failure-roll risk for this milestone (5..10). Set on
#                 sr_base_risk before the failure_weight roll fires.
sr_monthly_progress_update_effect = {
	if = {
		limit = { sr_ai_should_participate = yes }
		if = {
			limit = {
				OR = {
					has_variable = sr_safe_$MILESTONE$
					has_variable = sr_ambitious_$MILESTONE$
				}
			}
			change_variable = { name = sr_progress_$MILESTONE$ add = sr_progress }
			if = {
				limit = { NOT = { sr_has_failure_cooldown = yes } }
				set_variable = { name = sr_base_risk value = $RISK$ }
				random = {
					chance = sr_failure_weight
					set_variable = { name = sr_failed_$MILESTONE$ value = yes }
					if = {
						limit = { has_variable = sr_ambitious_$MILESTONE$ }
						sr_fire_ambitious_failure_event = yes
					}
					else = {
						sr_fire_safe_setback_event = yes
					}
				}
			}
		}
		else = {
			# Default (no approach selected) — minimal progress drift.
			change_variable = { name = sr_progress_$MILESTONE$ add = 0.5 }
		}
	}
}

# Keeps live milestone state aligned with the JEs that own it.
# This is permanent logic for intentionally deactivatable JEs, not a save-migration cleanup.
sr_cleanup_inactive_space_race_milestones = {
	sr_cleanup_milestone_if_inactive_base = { MILESTONE = suborbital         }
	sr_cleanup_milestone_if_inactive_base = { MILESTONE = orbital            }
	sr_cleanup_milestone_if_inactive_base = { MILESTONE = moon_landing       }
	sr_cleanup_milestone_if_inactive_base = { MILESTONE = probe              }
	sr_cleanup_milestone_if_inactive_base = { MILESTONE = moon_base          }
	sr_cleanup_milestone_if_inactive_base = { MILESTONE = mars_landing       }
	sr_cleanup_milestone_if_inactive_base = { MILESTONE = interstellar_probe }
	sr_cleanup_milestone_if_inactive_base = { MILESTONE = solar_colonization }
	# interstellar_results uses a different variable set (only sr_active_interstellar_results
	# and sr_interstellar_transit_progress) — cannot share the standard 5-var base.
	if = {
		limit = { NOT = { has_journal_entry = je_space_race_interstellar_results } }
		if = {
			limit = {
				OR = {
					has_variable = sr_active_interstellar_results
					has_variable = sr_interstellar_transit_progress
				}
			}
			set_variable = { name = sr_space_race_cleanup_changed value = yes }
			if = { limit = { has_variable = sr_active_interstellar_results } remove_variable = sr_active_interstellar_results }
			if = { limit = { has_variable = sr_interstellar_transit_progress } remove_variable = sr_interstellar_transit_progress }
		}
	}
	if = {
		limit = {
			has_variable = sr_active_milestone
			NOT = {
				OR = {
					has_variable = sr_active_suborbital
					has_variable = sr_active_orbital
					has_variable = sr_active_moon_landing
					has_variable = sr_active_probe
					has_variable = sr_active_moon_base
					has_variable = sr_active_mars_landing
					has_variable = sr_active_interstellar_probe
					has_variable = sr_active_solar_colonization
				}
			}
		}
		remove_variable = sr_active_milestone
	}
	if = {
		limit = { has_variable = sr_space_race_cleanup_changed }
		remove_variable = sr_space_race_cleanup_changed
		sr_recalculate_cost = yes
	}
}

# Called when a milestone is completed - handles cleanup
sr_complete_milestone_effect = {
	# Remove active milestone flag
	if = {
		limit = { has_variable = sr_active_milestone }
		remove_variable = sr_active_milestone
	}
	# Recalculate JE cost modifiers (per-JE vars cleaned in per-milestone effect)
	sr_recalculate_cost = yes
}

# Notify all countries about a space race milestone achievement
# Scope: Country that achieved the milestone
# Set sr_is_world_first variable to yes before calling if this is a first
# $MILESTONE$ = suborbital, orbital, moon_landing, probe, moon_base, mars_landing, interstellar_probe
sr_notify_milestone_achievement = {
	save_scope_as = sr_achiever
	if = {
		limit = { has_variable = sr_is_world_first }
		save_scope_as = sr_is_first
		remove_variable = sr_is_world_first
	}
	custom_tooltip = {
		text = sr_notify_achievement_tt
		every_country = {
			limit = {
				NOT = { THIS = scope:sr_achiever }
				is_country_type = recognized
			}
			# Store achiever capital on each receiver (capital chain is save-reload safe for loc display)
			set_variable = { name = sr_notify_achiever_capital value = scope:sr_achiever.capital }
			set_variable = { name = sr_notify_$MILESTONE$ value = yes }
			if = {
				limit = { exists = scope:sr_is_first }
				set_variable = { name = sr_notify_is_first value = yes }
			}
			trigger_event = { id = space_race_events.20 days = 14 }
		}
	}
}

# --- Per-milestone completion / reward / cleanup helpers ---------------------
# Each milestone exposes three named entry points (sr_complete_<X>_effect,
# sr_apply_<X>_reward, and the JE-driven sr_cleanup_milestone_variables_effect)
# so callers in events/ and journal_entries/ can grep for the concrete name.
# All three delegate to a single $MILESTONE$-parameterized base.
#
# Supported $MILESTONE$ values (must match a corresponding sr_<X> /
# sr_first_<X> static modifier and the per-JE variable namespace):
#   suborbital | orbital | moon_landing | probe | moon_base | mars_landing |
#   interstellar_probe

sr_complete_milestone_effect_base = {
	set_variable = { name = sr_completed_$MILESTONE$ value = yes }
	if = {
		limit = { NOT = { has_global_variable = sr_global_first_$MILESTONE$ } }
		set_global_variable = { name = sr_global_first_$MILESTONE$ value = yes }
		set_variable = { name = sr_was_first_$MILESTONE$ value = yes }
		set_variable = { name = sr_is_world_first value = yes }
	}
	sr_notify_milestone_achievement = { MILESTONE = $MILESTONE$ }
	if = { limit = { has_variable = sr_funding_$MILESTONE$ } remove_variable = sr_funding_$MILESTONE$ }
	if = { limit = { has_variable = sr_safe_$MILESTONE$ } remove_variable = sr_safe_$MILESTONE$ }
	if = { limit = { has_variable = sr_ambitious_$MILESTONE$ } remove_variable = sr_ambitious_$MILESTONE$ }
	if = { limit = { has_variable = sr_progress_$MILESTONE$ } remove_variable = sr_progress_$MILESTONE$ }
	sr_complete_milestone_effect = yes
}

sr_apply_milestone_reward_base = {
	if = {
		limit = { has_variable = sr_was_first_$MILESTONE$ }
		add_modifier = { name = sr_first_$MILESTONE$ }
	}
	else = {
		add_modifier = { name = sr_$MILESTONE$ }
	}
}

# Used by JE on_fail blocks. Removes per-milestone variables (sr_active_X is
# JE-only, so it appears here but not in the completion path) and reposts cost.
sr_cleanup_milestone_variables_effect = {
	if = { limit = { has_variable = sr_active_$MILESTONE$ } remove_variable = sr_active_$MILESTONE$ }
	if = { limit = { has_variable = sr_progress_$MILESTONE$ } remove_variable = sr_progress_$MILESTONE$ }
	if = { limit = { has_variable = sr_funding_$MILESTONE$ } remove_variable = sr_funding_$MILESTONE$ }
	if = { limit = { has_variable = sr_safe_$MILESTONE$ } remove_variable = sr_safe_$MILESTONE$ }
	if = { limit = { has_variable = sr_ambitious_$MILESTONE$ } remove_variable = sr_ambitious_$MILESTONE$ }
	sr_recalculate_cost = yes
}

# --- Per-milestone iteration bases (Style B inline orchestrator pattern) -----
# Five top-level helpers (sr_temporary_safety_review_effect,
# sr_apply_ambitious_failure_progress_loss, sr_apply_safe_setback_progress_loss,
# sr_clear_failed_milestone_flags, sr_boost_active_milestones) all walk the
# same 8 milestones (suborbital..solar_colonization). Each delegates to one of
# these bases and the orchestrator lists the tokens explicitly so the support
# matrix stays grep-able at the call site.

# Apply a temporary safety review modifier to the JE owning the failed milestone.
# Country scope. Caller iterates the standard 8 milestone tokens.
sr_apply_safety_review_to_milestone_base = {
	if = {
		limit = { has_variable = sr_failed_$MILESTONE$ }
		remove_variable = sr_failed_$MILESTONE$
		if = {
			limit = { has_journal_entry = je_space_race_$MILESTONE$ }
			je:je_space_race_$MILESTONE$ = {
				if = { limit = { has_modifier = sr_temporary_safety_review } remove_modifier = sr_temporary_safety_review }
				add_modifier = { name = sr_temporary_safety_review days = long_modifier_time is_decaying = yes }
			}
		}
	}
}

# Multiplicatively reduce a milestone's progress when its failure flag is set.
# Country scope. $MULT$ is the retention factor (0.75 ambitious, 0.85 safe).
sr_apply_progress_loss_base = {
	if = {
		limit = {
			has_variable = sr_failed_$MILESTONE$
			has_variable = sr_progress_$MILESTONE$
		}
		change_variable = { name = sr_progress_$MILESTONE$ multiply = $MULT$ }
	}
}

# Clear the failure flag for one milestone if present. Country scope.
sr_clear_failed_flag_base = {
	if = { limit = { has_variable = sr_failed_$MILESTONE$ } remove_variable = sr_failed_$MILESTONE$ }
}

# Add var:sr_progress_boost (signed) to one active milestone's progress, clamped
# at zero. Country scope. Caller sets sr_progress_boost before the orchestrator
# and removes it afterwards.
sr_boost_active_milestone_base = {
	if = {
		limit = { has_variable = sr_active_$MILESTONE$ }
		change_variable = { name = sr_progress_$MILESTONE$ add = var:sr_progress_boost }
		if = {
			limit = { var:sr_progress_$MILESTONE$ < 0 }
			set_variable = { name = sr_progress_$MILESTONE$ value = 0 }
		}
	}
}

# Strip the cost and funding-progress modifiers from one milestone's JE.
# Country scope. Uses ?= so an inactive JE silently no-ops instead of erroring.
sr_clear_milestone_cost_modifiers_base = {
	je:je_space_race_$MILESTONE$ ?= {
		if = { limit = { has_modifier = sr_space_program_cost } remove_modifier = sr_space_program_cost }
		if = { limit = { has_modifier = sr_funding_progress   } remove_modifier = sr_funding_progress   }
	}
}

# Re-apply the cost and funding-progress modifiers to one milestone's JE,
# gated on the active flag. Multiplier names interpolate from $MILESTONE$
# (sr_<X>_cost and sr_funding_<X>_level — verified to exist for all 8 tokens).
# Country scope.
sr_apply_milestone_cost_modifiers_base = {
	if = {
		limit = { has_variable = sr_active_$MILESTONE$ }
		je:je_space_race_$MILESTONE$ = {
			add_modifier = { name = sr_space_program_cost multiplier = sr_$MILESTONE$_cost          }
			add_modifier = { name = sr_funding_progress   multiplier = sr_funding_$MILESTONE$_level }
		}
	}
}

# When a milestone JE is no longer active, clear its 5-var family (active /
# progress / funding / safe / ambitious) and flag a cleanup-changed signal so
# the orchestrator runs sr_recalculate_cost once at the end. Country scope.
# interstellar_results does NOT use this base — it has only 2 vars (handled
# inline in sr_cleanup_inactive_space_race_milestones).
sr_cleanup_milestone_if_inactive_base = {
	if = {
		limit = { NOT = { has_journal_entry = je_space_race_$MILESTONE$ } }
		if = {
			limit = {
				OR = {
					has_variable = sr_active_$MILESTONE$
					has_variable = sr_progress_$MILESTONE$
					has_variable = sr_funding_$MILESTONE$
					has_variable = sr_safe_$MILESTONE$
					has_variable = sr_ambitious_$MILESTONE$
				}
			}
			set_variable = { name = sr_space_race_cleanup_changed value = yes }
			if = { limit = { has_variable = sr_active_$MILESTONE$    } remove_variable = sr_active_$MILESTONE$    }
			if = { limit = { has_variable = sr_progress_$MILESTONE$  } remove_variable = sr_progress_$MILESTONE$  }
			if = { limit = { has_variable = sr_funding_$MILESTONE$   } remove_variable = sr_funding_$MILESTONE$   }
			if = { limit = { has_variable = sr_safe_$MILESTONE$      } remove_variable = sr_safe_$MILESTONE$      }
			if = { limit = { has_variable = sr_ambitious_$MILESTONE$ } remove_variable = sr_ambitious_$MILESTONE$ }
		}
	}
}

# Check if current colonization stage is complete and advance
sr_check_colony_stage_effect = {
	# Stage 1 complete → Stage 2
	if = {
		limit = {
			global_var:sr_colonization_stage = 1
			has_global_variable = sr_colony_valles_marineris
			has_global_variable = sr_colony_olympus_mons
			has_global_variable = sr_colony_hellas_planitia
			has_global_variable = sr_colony_utopia_planitia
			has_global_variable = sr_colony_arcadia_planitia
			has_global_variable = sr_colony_ceres
			has_global_variable = sr_colony_vesta
			has_global_variable = sr_colony_psyche
			has_global_variable = sr_colony_pallas
			has_global_variable = sr_colony_hygiea
		}
		set_global_variable = { name = sr_colonization_stage value = 2 }
	}
	# Stage 2 complete → Stage 3
	else_if = {
		limit = {
			global_var:sr_colonization_stage = 2
			has_global_variable = sr_colony_io
			has_global_variable = sr_colony_europa
			has_global_variable = sr_colony_ganymede
			has_global_variable = sr_colony_callisto
			has_global_variable = sr_colony_himalia
			has_global_variable = sr_colony_amalthea
			has_global_variable = sr_colony_venus
		}
		set_global_variable = { name = sr_colonization_stage value = 3 }
	}
	# Stage 3 complete → Stage 4
	else_if = {
		limit = {
			global_var:sr_colonization_stage = 3
			has_global_variable = sr_colony_mercury
			has_global_variable = sr_colony_titan
			has_global_variable = sr_colony_enceladus
			has_global_variable = sr_colony_rhea
			has_global_variable = sr_colony_mimas
			has_global_variable = sr_colony_iapetus
		}
		set_global_variable = { name = sr_colonization_stage value = 4 }
	}
	# Stage 4 complete → Stage 5
	else_if = {
		limit = {
			global_var:sr_colonization_stage = 4
			has_global_variable = sr_colony_titania
			has_global_variable = sr_colony_oberon
			has_global_variable = sr_colony_miranda
			has_global_variable = sr_colony_ariel
			has_global_variable = sr_colony_triton
			has_global_variable = sr_colony_proteus
		}
		set_global_variable = { name = sr_colonization_stage value = 5 }
	}
	# Stage 5 complete → All done
	else_if = {
		limit = {
			global_var:sr_colonization_stage = 5
			has_global_variable = sr_colony_pluto
			has_global_variable = sr_colony_eris
			has_global_variable = sr_colony_makemake
			has_global_variable = sr_colony_haumea
			has_global_variable = sr_colony_sedna
		}
		set_global_variable = { name = sr_all_colonies_complete value = yes }
	}
}
