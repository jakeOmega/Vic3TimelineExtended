# Path-dependent legacy applied to a newly decolonized country.
# Scope: the new country (called from create_dynamic_country.on_created).
# Reads the decolonizing parent's policy-time variables to decide which
# legacy modifier and starting diplomatic relations to apply.
apply_decolonization_path = {
	if = {
		limit = {
			exists = scope:decolonizing_parent
		}
		# Iron Fist legacy — parent leaned hardest on garrison
		if = {
			limit = {
				scope:decolonizing_parent = {
					var:colonial_garrison_months >= var:colonial_invest_months
					var:colonial_garrison_months >= var:colonial_assimilate_months
					var:colonial_garrison_months > 24
				}
			}
			add_modifier = {
				name = bloody_decolonization_legacy_modifier
				days = long_modifier_time
				is_decaying = yes
			}
			scope:decolonizing_parent = {
				change_relations = { country = ROOT value = -50 }
			}
		}
		# Commonwealth legacy — parent leaned hardest on investment
		else_if = {
			limit = {
				scope:decolonizing_parent = {
					var:colonial_invest_months > var:colonial_garrison_months
					var:colonial_invest_months >= var:colonial_assimilate_months
					var:colonial_invest_months > 24
				}
			}
			add_modifier = {
				name = commonwealth_legacy_modifier
				days = long_modifier_time
				is_decaying = yes
			}
			scope:decolonizing_parent = {
				change_relations = { country = ROOT value = 30 }
			}
		}
		# Quiet Assimilation legacy — parent leaned hardest on assimilation
		else_if = {
			limit = {
				scope:decolonizing_parent = {
					var:colonial_assimilate_months > var:colonial_garrison_months
					var:colonial_assimilate_months > var:colonial_invest_months
					var:colonial_assimilate_months > 24
				}
			}
			add_modifier = {
				name = assimilation_legacy_modifier
				days = long_modifier_time
				is_decaying = yes
			}
			scope:decolonizing_parent = {
				change_relations = { country = ROOT value = 15 }
			}
		}
	}
}
