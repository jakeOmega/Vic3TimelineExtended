# Path-dependent legacy applied to a newly decolonized country.
# Scope: the new country (called from create_dynamic_country.on_created).
# Reads the decolonizing parent's policy-time variables to decide which
# legacy modifier, starting laws/IG clout, and diplomatic relations to apply.
#
# Relations between the new country and the former colonizer compose two
# signals additively:
#   1. Parent policy: Iron Fist −50, Commonwealth +30, Quiet Assimilation +15
#      (set inside the per-policy if/else_if blocks below).
#   2. Release type: round table +25, single grant −10, planned liquidation
#      −25 (applied after, reading decolonization_path_* variables on the
#      parent set by colonial_empire_buttons.txt). Together: round table
#      from a Commonwealth parent → +55 (very friendly); planned from an
#      Iron Fist parent → −75 (hostile).
#
# At the end, also applies a 12-month "liberation contagion" -1.0/mo bar
# drain to every other country running the colonial-empire JE. Produces
# the historical wave pattern: India 1947 → Burma, Ceylon, Indonesia.
apply_decolonization_path = {
	if = {
		limit = {
			exists = scope:decolonizing_parent
		}
		# Iron Fist legacy — parent leaned hardest on garrison
		if = {
			limit = {
				scope:decolonizing_parent = {
					var:colonial_garrison_months >= var:colonial_invest_months
					var:colonial_garrison_months >= var:colonial_assimilate_months
					var:colonial_garrison_months > 24
				}
			}
			add_modifier = {
				name = bloody_decolonization_legacy_modifier
				days = long_modifier_time
				is_decaying = yes
			}
			scope:decolonizing_parent = {
				change_relations = { country = ROOT value = -50 }
			}
			# Successor seeded as junta state — parent's brutal exit reproduces itself
			activate_law = law_type:law_autocracy
			activate_law = law_type:law_outlawed_dissent
			activate_law = law_type:law_minority_rights_discrimination
		}
		# Commonwealth legacy — parent leaned hardest on investment
		else_if = {
			limit = {
				scope:decolonizing_parent = {
					var:colonial_invest_months > var:colonial_garrison_months
					var:colonial_invest_months >= var:colonial_assimilate_months
					var:colonial_invest_months > 24
				}
			}
			add_modifier = {
				name = commonwealth_legacy_modifier
				days = long_modifier_time
				is_decaying = yes
			}
			scope:decolonizing_parent = {
				change_relations = { country = ROOT value = 30 }
			}
			# Successor inherits parent's developmentalist institutions
			activate_law = law_type:law_parliamentary_republic
			activate_law = law_type:law_protected_speech
			activate_law = law_type:law_minority_rights_indifference
		}
		# Quiet Assimilation legacy — parent leaned hardest on assimilation
		else_if = {
			limit = {
				scope:decolonizing_parent = {
					var:colonial_assimilate_months > var:colonial_garrison_months
					var:colonial_assimilate_months > var:colonial_invest_months
					var:colonial_assimilate_months > 24
				}
			}
			add_modifier = {
				name = assimilation_legacy_modifier
				days = long_modifier_time
				is_decaying = yes
			}
			scope:decolonizing_parent = {
				change_relations = { country = ROOT value = 15 }
			}
			# Successor seeded with cultural-exclusion citizenship — parent's
			# civilizing mission left a thin elite that runs the new state
			activate_law = law_type:law_cultural_exclusion
			activate_law = law_type:law_minority_rights_indifference
		}
	}

	# Release-type relations adjustment (additive on top of the policy-based
	# change_relations inside the if/else_if blocks above). Round table is a
	# negotiated exit, so the new country starts on better terms; planned
	# liquidation dumps the whole empire and the new country resents it.
	if = {
		limit = {
			exists = scope:decolonizing_parent
			scope:decolonizing_parent = { has_variable = decolonization_path_round_table }
		}
		scope:decolonizing_parent = {
			change_relations = { country = ROOT value = 25 }
		}
	}
	else_if = {
		limit = {
			exists = scope:decolonizing_parent
			scope:decolonizing_parent = { has_variable = decolonization_path_release_single }
		}
		scope:decolonizing_parent = {
			change_relations = { country = ROOT value = -10 }
		}
	}
	else_if = {
		limit = {
			exists = scope:decolonizing_parent
			scope:decolonizing_parent = { has_variable = decolonization_path_planned }
		}
		scope:decolonizing_parent = {
			change_relations = { country = ROOT value = -25 }
		}
	}

	# Liberation contagion: every other country running the colonial-empire JE
	# feels a 12-month -1.0/mo drain on its stability bar. Independence
	# elsewhere accelerates the next wave.
	every_country = {
		limit = {
			has_journal_entry = je_colonial_empire
			NOT = { this = scope:decolonizing_parent }
			NOT = { this = ROOT }
		}
		add_modifier = {
			name = colonial_stability_negative_event
			months = 12
		}
	}
}

# Preview helper used by the three decolonization buttons. For one candidate
# state (THIS), determines the country culture that would form there, then
# flood-fills a per-call region variable across all metropole-owned states
# that are homelands of that culture and connected to the capital. Also
# stores the region's state count as a country variable for loc display.
#
# Scope (THIS): the candidate capital state.
# Params:
#   $REGION_VAR$    — per-call unique variable, marks the flood-fill region.
#                     Subsequent candidate-picks `NOT = { has_variable = X }`
#                     against this to avoid overlapping candidate regions.
#   $COUNT_VAR$     — per-call unique country variable, gets the region size
#                     (number of states that would be ceded if this candidate
#                     is released).
#   $CULTURE_SCOPE$ — per-call unique named scope holding the chosen culture.
#                     Mirrors form_decolonized_country's 3-fallback selection.
#
# After all 3 candidate-previews run, form_decolonized_country's own internal
# flood-fill (via decolonization_cede_marker) will re-mark the same states at
# release time. The region/count vars here are purely for UI preview + the
# top-3 candidate-exclusion logic; they're cleared in event 400's options.
_decolonization_preview_region = {
	# Cache the owner once per call so pop-scope culture checks below can
	# reference it. Same metropole across the 3 candidate calls — overwriting
	# the named scope is idempotent.
	owner = { save_scope_as = decolonization_helper_metropole }

	# Culture selection — same 3-fallback chain as form_decolonized_country.
	# Preferred: a culture that has the candidate state as a homeland.
	random_scope_culture = {
		limit = { has_homeland = prev }
		save_scope_as = $CULTURE_SCOPE$
	}
	# Fallback 1: largest non-metropole-primary pop's culture present here.
	if = {
		limit = { NOT = { exists = scope:$CULTURE_SCOPE$ } }
		ordered_scope_pop = {
			limit = {
				NOT = { culture = { is_primary_culture_of = scope:decolonization_helper_metropole } }
			}
			order_by = total_size
			position = 0
			culture = { save_scope_as = $CULTURE_SCOPE$ }
		}
	}
	# Fallback 2: any culture present in the state.
	if = {
		limit = { NOT = { exists = scope:$CULTURE_SCOPE$ } }
		random_scope_culture = { save_scope_as = $CULTURE_SCOPE$ }
	}

	# Seed: mark the candidate capital itself.
	set_variable = $REGION_VAR$

	# Flood-fill within the metropole's states. Each `while` iteration adds
	# one ring of homeland-of-the-new-culture neighbors; loop terminates when
	# no more unmarked metropole states qualify.
	owner = {
		while = {
			limit = {
				any_scope_state = {
					NOT = { has_variable = $REGION_VAR$ }
					state_region = { is_homeland = scope:$CULTURE_SCOPE$ }
					any_neighbouring_state = {
						has_variable = $REGION_VAR$
					}
				}
			}
			every_scope_state = {
				limit = {
					NOT = { has_variable = $REGION_VAR$ }
					state_region = { is_homeland = scope:$CULTURE_SCOPE$ }
					any_neighbouring_state = {
						has_variable = $REGION_VAR$
					}
				}
				set_variable = $REGION_VAR$
			}
		}

		# Count the region. Reset to 0, then increment per marked state.
		# `prev` from inside the iter = this `owner` block's scope = the country.
		set_variable = { name = $COUNT_VAR$ value = 0 }
		every_scope_state = {
			limit = { has_variable = $REGION_VAR$ }
			prev = { change_variable = { name = $COUNT_VAR$ add = 1 } }
		}
	}
	debug_log = "DECOL_DEBUG: _decolonization_preview_region completed (region var $REGION_VAR$, count var $COUNT_VAR$)"  # DEBUG_DECOLONIZATION_TRACE
}

form_decolonized_country = { # scope (THIS) is capital state of new country
	save_scope_as = new_country_capital
	owner = { save_scope_as = decolonizing_parent }
	# Mark the capital with a per-call cede variable. create_dynamic_country's
	# cede_state_trigger iterates every state in the world; using a variable
	# is the vanilla idiom (see 00_colonial_administration_buttons.txt). The
	# `this = scope:X` form is silently rejected by that trigger context.
	# Cleared in on_created below.
	scope:new_country_capital = { set_variable = decolonization_cede_marker }
	debug_log = "DECOL_DEBUG: form_decolonized_country entered, capital=[SCOPE.sState('new_country_capital').GetName]"  # DEBUG_DECOLONIZATION_TRACE

	# Culture selection: prefer a culture that has the capital as homeland;
	# fall back to largest non-owner-primary pop; last-resort use any culture
	# present in the state. This makes the call deterministic even when the
	# indigenous homeland culture has been displaced (a common Round Table case).
	random_scope_culture = {
		limit = { has_homeland = scope:new_country_capital }
		save_scope_as = new_country_culture
	}
	if = {
		limit = { NOT = { exists = scope:new_country_culture } }
		ordered_scope_pop = {
			limit = {
				NOT = { culture = { is_primary_culture_of = scope:decolonizing_parent } }
			}
			order_by = total_size
			position = 0
			culture = { save_scope_as = new_country_culture }
		}
	}
	if = {
		limit = { NOT = { exists = scope:new_country_culture } }
		random_scope_culture = {
			save_scope_as = new_country_culture
		}
	}
	# Trip wire — fires only if all three culture-selection branches failed.
	# Means the state has zero pops, which shouldn't happen given the eligibility
	# triggers; if it does, create_dynamic_country will reject and no country forms.
	if = { limit = { NOT = { exists = scope:new_country_culture } } debug_log = "DECOL_DEBUG: BUG — scope:new_country_culture UNSET after all 3 fallbacks; create_dynamic_country will reject" }  # DEBUG_DECOLONIZATION_TRACE

	# Recursive flood-fill: in addition to the capital, also cede every other
	# metropole-owned state that (a) is a homeland of the new country's primary
	# culture AND (b) is reachable from the capital via a chain of such states.
	# Each iteration of `while` expands one ring of neighbors; the limit fails
	# when no further unmarked metropole state qualifies, naturally terminating.
	# Without this, a released country would only ever get its single capital
	# state — Venezuela releasing as just Zulia rather than Zulia + Falcón +
	# the rest of north-coast Venezuela.
	while = {
		limit = {
			scope:decolonizing_parent = {
				any_scope_state = {
					NOT = { has_variable = decolonization_cede_marker }
					state_region = { is_homeland = scope:new_country_culture }
					any_neighbouring_state = {
						has_variable = decolonization_cede_marker
					}
				}
			}
		}
		scope:decolonizing_parent = {
			every_scope_state = {
				limit = {
					NOT = { has_variable = decolonization_cede_marker }
					state_region = { is_homeland = scope:new_country_culture }
					any_neighbouring_state = {
						has_variable = decolonization_cede_marker
					}
				}
				set_variable = decolonization_cede_marker
			}
		}
	}

	# as the metropole). The random_list wrap, by contrast, IS evaluated at

# Clean up state attached to the metropole, civil-rights movements, and
# qualifying colonial subjects when je_colonial_empire ends (complete or fail).
# The bar-band modifiers attached to the JE itself disappear with the JE; the
# items below survive the JE's lifetime and need explicit removal.
#
# Called by both on_complete and on_fail. The on_fail callsite preserves
# colonial_collapse_subject_unrest_modifier (applied separately, intentionally
# permanent) -- this helper does not touch that modifier.
#
# Scope: country (the metropole).
colonial_empire_je_cleanup_effect = {
	# Path-tracking and total-duration variables
	remove_variable = colonial_invest_months
	remove_variable = colonial_garrison_months
	remove_variable = colonial_assimilate_months
	remove_variable = colonial_solidified_months
	remove_variable = colonial_je_total_months
	remove_variable = colonial_at_100_months

	# Pressuring-great-powers roster slots (populated by the monthly pulse for
	# status_desc display). Surviving these past JE end would orphan the slot
	# references in the journal-history tooltip.
	remove_variable = colonial_condemner_idx
	remove_variable = colonial_condemner_rank_1
	remove_variable = colonial_condemner_rank_2
	remove_variable = colonial_condemner_rank_3
	remove_variable = colonial_condemner_rank_4
	remove_variable = colonial_condemner_rank_5
	remove_variable = colonial_condemner_rank_6
	remove_variable = colonial_condemner_rank_7

	# Country-scope policy modifiers from the three policy buttons. If any
	# button was active at end-of-JE, its modifier persisted after the JE
	# vanished; clear them so the country reverts to neutral. The Invest
	# program also has a dedicated budget-expense modifier refreshed each
	# pulse — clear it alongside the policy modifier or it would linger.
	remove_modifier = colonial_development_investment_modifier
	remove_modifier = colonial_development_investment_expense_modifier
	remove_modifier = colonial_military_garrison_modifier
	remove_modifier = colonial_cultural_assimilation_modifier

	# Civil-rights movement radicalism modifiers reapplied each pulse while
	# the JE was active. Remove from any civil-rights movement.
	every_political_movement = {
		limit = { is_political_movement_type = movement_civil_rights }
		remove_modifier = colonial_pressure_movement_radicalism
		remove_modifier = colonial_crumbling_movement_radicalism
	}

	# Subject-side LD pressure modifiers (bar-band driven). Remove from
	# every qualifying colonial subject; the failure spike modifier
	# (colonial_collapse_subject_unrest_modifier) is intentionally NOT in
	# this list.
	every_subject_or_below = {
		limit = { is_qualifying_colonial_subject = yes }
		remove_modifier = colonial_overlord_strained_subject_ld_modifier
		remove_modifier = colonial_overlord_crumbling_subject_ld_modifier
		remove_modifier = colonial_overlord_collapsing_subject_ld_modifier
	}
}
