fmc_on_weekly_pulse = {
	if = {
		limit = {
			NOT = {
				is_country_type = decentralized
			}
			#this = market.owner
		}
		fmc_update_price_stabiliser = yes
		fmc_country_update_construction_data = yes
		if = {
			limit = {
				is_ai = yes
			}
			fmc_ai_update_construction = yes
		}
	}
}


fmc_on_border_changes = {
	if = {
		limit = {
			NOT = {
				is_country_type = decentralized
			}
			#this = market.owner
		}
		fmc_country_update_sectors = yes
	}
}
