fmc_on_weekly_pulse = {
	if = {
		limit = {
			NOT = {
				is_country_type = decentralized
			}
			#this = market.owner
		}
		country_update_construction_data = yes
		queue_more_private_investment = yes
		if = {
			limit = {
				is_ai = yes
			}
			ai_update_construction = yes
		}
	}
}


fmc_on_border_changes = {
	if = {
		limit = {
			NOT = {
				is_country_type = decentralized
			}
			#this = market.owner
		}
		country_update_sectors = yes
	}
}
