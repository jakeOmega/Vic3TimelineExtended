te_construction_market_country_update_sites = {
	every_state = {
		if = {
			limit = {
				any_scope_building = {
					is_under_construction = yes
				}
			}
			te_construction_market_set_specified_level = {
				SPEC_LEVEL = 1
				SPEC_TYPE = te_construction_market_site
			}
		}
		te_construction_market_state_on_end = yes
	}
	if = {
		limit = {
			NOT = {
				any_scope_building = {
					is_building_type = te_construction_market_site
				}
			}
		}
		capital = {
			create_building = {
				building = te_construction_market_site
				level = 1
			}
		}
	}
}


te_construction_market_country_update_data = {
	te_construction_market_country_update_private_share = yes
	te_construction_market_apply_per_site_modifier = yes
}


te_construction_market_country_update_private_share = {
	remove_modifier = te_construction_market_private_share_modifier
	add_modifier = {
		name = te_construction_market_private_share_modifier
		multiplier = te_construction_market_private_share_target
	}
}


te_construction_market_apply_per_site_modifier = {
	every_scope_building = {
		limit = {
			is_building_type = te_construction_market_site
		}
		remove_modifier = te_construction_market_capacity_modifier
		if = {
			limit = {
				occupancy > 1
			}
			te_construction_market_reset_site = yes
		}
		# Per-site multiplier evaluated in building scope — understaffed
		# sites get a proportionally larger target so total throughput
		# tracks total_use_target regardless of staffing distribution.
		add_modifier = {
			name = te_construction_market_capacity_modifier
			multiplier = te_construction_market_employment_adjusted_per_site
		}
	}
}


te_construction_market_reset_site = {
	state = {
		remove_building = te_construction_market_site
		create_building = {
			building = te_construction_market_site
			level = 1
		}
	}
}
