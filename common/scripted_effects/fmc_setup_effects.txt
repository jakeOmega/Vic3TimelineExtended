initialize_gov_factories_variable = {
    if = {
	    limit = {
	        not = {
	            has_variable = num_gov_factories
	        }
	    }
	    set_variable = {
	        name = num_gov_factories
	        value = 0
	    }
	}
}


country_replace_sectors = {
	every_scope_state = {
		replace_building = {
			OLD_TYPE = building_construction_sector
			NEW_TYPE = building_construction_sector_mod
		}
	}
}

country_construction_setup = {
	initialize_gov_factories_variable = yes
	country_replace_sectors = yes
	if = {
		limit = {
			fmc_amount_of_sectors = 0
		}
		capital = {
			create_building = {
				building = building_construction_sector
				level = 1
			}
		}
