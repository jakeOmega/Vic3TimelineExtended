fmc_state_on_construction_start = {
	create_building = {
		building = building_construction_sector
		level = 1
	}
}

fmc_state_on_construction_end = {
	if = {
		limit = {
			NOT = {
				any_scope_building = {
					is_under_construction = yes
				}
			}
			any_scope_building = {
				type = bt:building_construction_sector
			}
			NOT = { is_capital = yes }
		}
		remove_building = building_construction_sector
	}
}
