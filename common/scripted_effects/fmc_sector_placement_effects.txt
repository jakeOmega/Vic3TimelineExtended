fmc_state_on_construction_start = {
	if = {
		limit = {
			NOT = {
				any_scope_building = {
					type = bt:fmc_building_construction_site
				}
			}
		}
		create_building = {
			building = fmc_building_construction_site
			level = 1
		}
	}

	owner = {
		if = {
			limit = {
				capital = {
					AND = {
						any_scope_building = {
							type = bt:fmc_building_construction_site
						}
						NOT = {
							any_scope_building = {
								is_under_construction = yes
							}
						}
					}
				}
			}
			capital = {
				remove_building = fmc_building_construction_site
			}
		}
	}


	# owner = {
	# 	fmc_update_construction_effect = yes
	# }
}

fmc_state_on_construction_end = {
	if = {
		limit = {
			NOT = {
				any_scope_building = {
					is_under_construction = yes
				}
			}
			any_scope_building = {
				type = bt:fmc_building_construction_site
			}
			NOT = {
				is_capital = yes				
			}
		}
		remove_building = fmc_building_construction_site
	}
	owner = {
		if = {
			limit = {
				NOT = {
					any_scope_building = {
						type = bt:fmc_building_construction_site
					}
				}
			}
			capital = {
				create_building = {
					building = fmc_building_construction_site
					level = 1
				}
			}
	
		}
	}
}
