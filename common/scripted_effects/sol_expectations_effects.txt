# SoL Expectations System - Monthly Update
# Scope: country
# Tracks a shift variable that converges expected SoL toward a target.
# Target = average_sol + country_sol_expectations_target_add.
# Each month, reads the gap (target - average_expected_sol) and nudges
# the shift to close it. At equilibrium, expected SoL ≈ target.
#   Example: average_sol=20, target_add=+2 → expected converges to 22
# Converges with ~5-year half-life, ~90% adapted in 20 years.
#
# Additional features:
# - Min/max clamp: shift is bounded by configurable floor/ceiling.
#   country_sol_expectations_shift_min_add raises the floor (welfare floors).
#   country_sol_expectations_shift_max_add lowers the ceiling.
# - Per-strata offsets: country_sol_expectations_upper/middle/lower_offset_add
#   apply direct (non-adaptive) shifts to specific class expectations.
#   Models class-relative comparison (aristocratic vs egalitarian societies).
# - The system applies 3 separate strata modifiers (upper, middle, lower),
#   each with multiplier = clamped shift + class offset. No shared "overall"
#   modifier; each class's expectation is fully explicit.

sol_expectations_monthly_update = {
	# Initialize shift variable if it doesn't exist yet (start at 0 = no adjustment)
	if = {
		limit = { NOT = { has_variable = sol_expectations_shift } }
		set_variable = { name = sol_expectations_shift value = 0 }
	}

	# Cache the gap BEFORE removing the modifier.
	# average_expected_sol won't recalculate mid-effect after modifier removal.
	# Gap = (average_sol + target_add) - average_expected_sol.
	# Gap > 0 means expectations are below target; gap < 0 means above.
	set_variable = { name = sol_expectations_gap_cached value = sol_expectations_gap_value }

	# Nudge the shift to close the gap: shift += gap * rate
	change_variable = {
		name = sol_expectations_shift
		add = {
			value = var:sol_expectations_gap_cached
			multiply = sol_expectations_adaptation_rate_value
		}
	}

	# Apply persistent monthly bias from modifiers (laws, techs, buildings, etc.)
	# Positive = expectations rise (people expect more), negative = lower expectations.
	change_variable = {
		name = sol_expectations_shift
		add = modifier:country_sol_expectations_shift_add
	}

	# Clamp shift to configured [min, max] bounds.
	# Default: [0, 50] (effectively unlimited). Welfare laws raise the floor;
	# people are unhappy starving regardless of how long it's persisted.
	set_variable = { name = sol_expectations_shift value = sol_expectations_shift_clamped_value }

	# Apply per-class shift modifiers (adaptive shift + class offset for each stratum).
	sol_expectations_apply_strata_shifts = yes
}

# ========== Utility Scripted Effects ==========
# For events, decisions, and other systems to call.

# Instantly adjust the expectations shift by a fixed amount.
# Scope: country
# Parameters:
#   AMOUNT — how much to add to the shift. Positive = raise expected SoL,
#            negative = lower it. Can be a script value or literal.
# Example: sol_expectations_instant_adjust = { AMOUNT = 2 }
#          sol_expectations_instant_adjust = { AMOUNT = -3 }
sol_expectations_instant_adjust = {
	if = {
		limit = { NOT = { has_variable = sol_expectations_shift } }
		set_variable = { name = sol_expectations_shift value = 0 }
	}

	change_variable = {
		name = sol_expectations_shift
		add = $AMOUNT$
	}

	# Re-apply modifier immediately
	sol_expectations_reapply_modifier = yes
}

# Instantly close a fraction of the remaining gap between expected and actual SoL.
# Scope: country
# Parameters:
#   FRACTION — 0 to 1 (0.5 = close half the gap, 1.0 = fully reset to equilibrium).
#              Values > 1 overshoot past equilibrium.
# Example: sol_expectations_close_gap = { FRACTION = 0.5 }
sol_expectations_close_gap = {
	if = {
		limit = { NOT = { has_variable = sol_expectations_shift } }
		set_variable = { name = sol_expectations_shift value = 0 }
	}

	# Cache the current gap (with our modifier still active)
	set_variable = { name = sol_expectations_gap_cached value = sol_expectations_gap_value }

	change_variable = {
		name = sol_expectations_shift
		add = {
			value = var:sol_expectations_gap_cached
			multiply = $FRACTION$
		}
	}

	# Re-apply modifier immediately
	sol_expectations_reapply_modifier = yes
}

# Fully reset expectations to match current SoL (close 100% of gap).
# Scope: country
# Example: sol_expectations_reset = yes
sol_expectations_reset = {
	if = {
		limit = { NOT = { has_variable = sol_expectations_shift } }
		set_variable = { name = sol_expectations_shift value = 0 }
	}

	# Cache the current gap and close it completely
	set_variable = { name = sol_expectations_gap_cached value = sol_expectations_gap_value }

	change_variable = {
		name = sol_expectations_shift
		add = var:sol_expectations_gap_cached
	}

	# Re-apply modifier immediately
	sol_expectations_reapply_modifier = yes
}

# Internal helper: clamp shift and re-apply all strata modifiers.
# Scope: country
sol_expectations_reapply_modifier = {
	# Clamp before reapplying
	set_variable = { name = sol_expectations_shift value = sol_expectations_shift_clamped_value }

	sol_expectations_apply_strata_shifts = yes
}

# Internal helper: remove and re-apply the 3 per-class strata shift modifiers.
# Each uses multiplier = clamped shift + class offset.
# Scope: country
sol_expectations_apply_strata_shifts = {
	# Upper strata
	if = {
		limit = { has_modifier = sol_expectations_upper_strata_shift }
		remove_modifier = sol_expectations_upper_strata_shift
	}
	if = {
		limit = {
			OR = {
				sol_expectations_upper_shift_value > 0.1
				sol_expectations_upper_shift_value < -0.1
			}
		}
		add_modifier = {
			name = sol_expectations_upper_strata_shift
			multiplier = sol_expectations_upper_shift_value
		}
	}

	# Middle strata
	if = {
		limit = { has_modifier = sol_expectations_middle_strata_shift }
		remove_modifier = sol_expectations_middle_strata_shift
	}
	if = {
		limit = {
			OR = {
				sol_expectations_middle_shift_value > 0.1
				sol_expectations_middle_shift_value < -0.1
			}
		}
		add_modifier = {
			name = sol_expectations_middle_strata_shift
			multiplier = sol_expectations_middle_shift_value
		}
	}

	# Lower strata
	if = {
		limit = { has_modifier = sol_expectations_lower_strata_shift }
		remove_modifier = sol_expectations_lower_strata_shift
	}
	if = {
		limit = {
			OR = {
				sol_expectations_lower_shift_value > 0.1
				sol_expectations_lower_shift_value < -0.1
			}
		}
		add_modifier = {
			name = sol_expectations_lower_strata_shift
			multiplier = sol_expectations_lower_shift_value
		}
	}
}
