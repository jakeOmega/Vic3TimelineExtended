### Heir Education - Scripted Effects
### Reworked: 5-tier system (terrible/poor/average/skilled/exceptional)
### Hidden intelligence (1-5) modifies gain amount per pulse tick
### 8% rebel child chance inverts ideology at resolution
### Adult heirs pre-initialized based on age

# ──────────────────────── INTELLIGENCE-MODIFIED GAIN EFFECTS ────────────────────────
# Called by monthly pulse when a focus's 3% chance succeeds.
# Intelligence modifies the AMOUNT gained per tick:
#   Low intelligence (1-2): 25% chance of wasted opportunity (no gain)
#   Medium intelligence (3): standard +1 gain
#   High intelligence (4-5): 30% chance of double gain (+2)

heir_ed_gain_admin = {
	# Intelligence modifies gain amount:
	#   High (4-5): 30% chance of double gain (+2)
	#   Medium (3): standard gain (+1)
	#   Low (1-2): 25% chance of wasted opportunity (+0)
	if = {
		limit = { var:heir_ed_intelligence >= 4 }
		random_list = {
			30 = {
				change_variable = { name = heir_ed_admin add = 2 }
				change_variable = { name = heir_ed_total add = 2 }
			}
			70 = {
				change_variable = { name = heir_ed_admin add = 1 }
				change_variable = { name = heir_ed_total add = 1 }
			}
		}
	}
	else_if = {
		limit = { var:heir_ed_intelligence <= 2 }
		random_list = {
			25 = { }
			75 = {
				change_variable = { name = heir_ed_admin add = 1 }
				change_variable = { name = heir_ed_total add = 1 }
			}
		}
	}
	else = {
		change_variable = { name = heir_ed_admin add = 1 }
		change_variable = { name = heir_ed_total add = 1 }
	}
	scope:journal_entry = {
		set_bar_progress = { name = heir_education_progress_bar value = root.var:heir_ed_total }
	}
	add_modifier = { name = heir_education_innovation_cost days = 30 is_decaying = yes }
	heir_education_admin_ig_reaction = yes
}

# ──────────────────────── ADULT HEIR INITIALIZATION ────────────────────────
# Pre-initializes investment variables for heirs who aren't newborns.
# Simulates passive development based on heir's age.
# Called from JE immediate block.

heir_education_init_adult = {
	heir ?= {
		# Age 15+: significant prior development
		if = {
			limit = { age >= 15 }
			root = {
				# Admin: 1-3 points
				random_list = {
					25 = { change_variable = { name = heir_ed_admin add = 1 } change_variable = { name = heir_ed_total add = 1 } }
					50 = { change_variable = { name = heir_ed_admin add = 2 } change_variable = { name = heir_ed_total add = 2 } }
					25 = { change_variable = { name = heir_ed_admin add = 3 } change_variable = { name = heir_ed_total add = 3 } }
				}
				# Diplo: 1-3 points
				random_list = {
					25 = { change_variable = { name = heir_ed_diplo add = 1 } change_variable = { name = heir_ed_total add = 1 } }
					50 = { change_variable = { name = heir_ed_diplo add = 2 } change_variable = { name = heir_ed_total add = 2 } }
					25 = { change_variable = { name = heir_ed_diplo add = 3 } change_variable = { name = heir_ed_total add = 3 } }
				}
				# Military: 1-3 points
				random_list = {
					25 = { change_variable = { name = heir_ed_military add = 1 } change_variable = { name = heir_ed_total add = 1 } }
					50 = { change_variable = { name = heir_ed_military add = 2 } change_variable = { name = heir_ed_total add = 2 } }
					25 = { change_variable = { name = heir_ed_military add = 3 } change_variable = { name = heir_ed_total add = 3 } }
				}
				# Random ideology lean
				random_list = {
					30 = { change_variable = { name = heir_ed_ideology add = 1 } change_variable = { name = heir_ed_total add = 1 } }
					40 = { }
					30 = { change_variable = { name = heir_ed_ideology add = -1 } change_variable = { name = heir_ed_total add = 1 } }
				}
				# Random IG lean
				random_list = {
					33 = { change_variable = { name = heir_ed_ig_radical add = 1 } change_variable = { name = heir_ed_total add = 1 } }
					34 = { change_variable = { name = heir_ed_ig_moderate add = 1 } change_variable = { name = heir_ed_total add = 1 } }
					33 = { change_variable = { name = heir_ed_ig_regressive add = 1 } change_variable = { name = heir_ed_total add = 1 } }
				}
			}
		}
		# Age 10-14: moderate prior development
		else_if = {
			limit = { age >= 10 }
			root = {
				random_list = {
					50 = { change_variable = { name = heir_ed_admin add = 1 } change_variable = { name = heir_ed_total add = 1 } }
					50 = { change_variable = { name = heir_ed_admin add = 2 } change_variable = { name = heir_ed_total add = 2 } }
				}
				random_list = {
					50 = { change_variable = { name = heir_ed_diplo add = 1 } change_variable = { name = heir_ed_total add = 1 } }
					50 = { change_variable = { name = heir_ed_diplo add = 2 } change_variable = { name = heir_ed_total add = 2 } }
				}
				random_list = {
					50 = { change_variable = { name = heir_ed_military add = 1 } change_variable = { name = heir_ed_total add = 1 } }
					50 = { change_variable = { name = heir_ed_military add = 2 } change_variable = { name = heir_ed_total add = 2 } }
				}
			}
		}
		# Age 5-9: minimal prior development
		else_if = {
			limit = { age >= 5 }
			root = {
				random_list = {
					60 = { }
					40 = { change_variable = { name = heir_ed_admin add = 1 } change_variable = { name = heir_ed_total add = 1 } }
				}
				random_list = {
					60 = { }
					40 = { change_variable = { name = heir_ed_diplo add = 1 } change_variable = { name = heir_ed_total add = 1 } }
				}
				random_list = {
					60 = { }
					40 = { change_variable = { name = heir_ed_military add = 1 } change_variable = { name = heir_ed_total add = 1 } }
				}
			}
		}
	}
}

# ──────────────────────── MAIN RESOLUTION ────────────────────────
# Called when the heir reaches adulthood or on JE timeout.
# 5-tier system: terrible / poor / average / skilled / exceptional
# Investment thresholds scaled for 3% monthly pulse:
#   0   = no investment -> heavily weighted toward terrible/poor
#   1-2 = light investment
#   3-4 = moderate investment
#   5-7 = heavy investment
#   8+  = extreme investment (sustained high-intelligence focus over many years)

heir_education_resolve_effect = {

		limit = { has_variable = heir_ed_admin var:heir_ed_admin >= 8 }

		}
	}
	else = {
		# Neutral ideology
		heir ?= {
			random_list = {
				50 = { set_ideology = ideology:ideology_moderate }
				20 = { set_ideology = ideology:ideology_reformer }
				20 = { set_ideology = ideology:ideology_traditionalist }
				10 = { set_ideology = ideology:ideology_liberal }
			}
		}
	}

	# ─── Resolve IG Alignment ───
	# Compares radical vs moderate vs regressive investment
	# Winner's magnitude determines alignment strength
	if = {
		limit = {
			var:heir_ed_ig_radical >= var:heir_ed_ig_moderate
			var:heir_ed_ig_radical >= var:heir_ed_ig_regressive
			var:heir_ed_ig_radical >= 1
		}
		# Radical wins
		if = {
			limit = { var:heir_ed_ig_radical >= 4 }
			heir ?= {
				random_list = {
					40 = { trigger = { owner = { exists = ig:ig_trade_unions } } set_interest_group = ig:ig_trade_unions }
					30 = { trigger = { owner = { exists = ig:ig_intelligentsia } } set_interest_group = ig:ig_intelligentsia }
					30 = { trigger = { owner = { exists = ig:ig_rural_folk } } set_interest_group = ig:ig_rural_folk }
				}
			}
		}
		else = {
			heir ?= {
				random_list = {
					35 = { trigger = { owner = { exists = ig:ig_trade_unions } } set_interest_group = ig:ig_trade_unions }
					35 = { trigger = { owner = { exists = ig:ig_intelligentsia } } set_interest_group = ig:ig_intelligentsia }
					30 = { trigger = { owner = { exists = ig:ig_rural_folk } } trigger = { owner = { exists = ig:ig_rural_folk } } set_interest_group = ig:ig_rural_folk }
				}
			}
		}
	}
	else_if = {
		limit = {
			var:heir_ed_ig_moderate >= var:heir_ed_ig_radical
			var:heir_ed_ig_moderate >= var:heir_ed_ig_regressive
			var:heir_ed_ig_moderate >= 1
		}
		# Moderate wins
		if = {
			limit = { var:heir_ed_ig_moderate >= 4 }
			heir ?= {
				random_list = {
					40 = { trigger = { owner = { exists = ig:ig_industrialists } } set_interest_group = ig:ig_industrialists }
					30 = { trigger = { owner = { exists = ig:ig_petty_bourgeoisie } } set_interest_group = ig:ig_petty_bourgeoisie }
					30 = { trigger = { owner = { exists = ig:ig_armed_forces } } set_interest_group = ig:ig_armed_forces }
				}
			}
		}
		else = {
			heir ?= {
				random_list = {
					35 = { trigger = { owner = { exists = ig:ig_industrialists } } set_interest_group = ig:ig_industrialists }
					35 = { trigger = { owner = { exists = ig:ig_petty_bourgeoisie } } set_interest_group = ig:ig_petty_bourgeoisie }
					30 = { trigger = { owner = { exists = ig:ig_armed_forces } } trigger = { owner = { exists = ig:ig_armed_forces } } set_interest_group = ig:ig_armed_forces }
				}
			}
		}
	}
	else_if = {
		limit = {
			var:heir_ed_ig_regressive >= var:heir_ed_ig_radical
			var:heir_ed_ig_regressive >= var:heir_ed_ig_moderate
			var:heir_ed_ig_regressive >= 1
		}
		# Regressive wins
		if = {
			limit = { var:heir_ed_ig_regressive >= 4 }
			heir ?= {
				random_list = {
					40 = { trigger = { owner = { exists = ig:ig_landowners } } set_interest_group = ig:ig_landowners }
					30 = { trigger = { owner = { exists = ig:ig_devout } } set_interest_group = ig:ig_devout }
					30 = { trigger = { owner = { exists = ig:ig_rural_folk } } set_interest_group = ig:ig_rural_folk }
				}
			}
		}
		else = {
			heir ?= {
				random_list = {
					35 = { trigger = { owner = { exists = ig:ig_landowners } } set_interest_group = ig:ig_landowners }
					35 = { trigger = { owner = { exists = ig:ig_devout } } set_interest_group = ig:ig_devout }
					30 = { trigger = { owner = { exists = ig:ig_rural_folk } } trigger = { owner = { exists = ig:ig_rural_folk } } set_interest_group = ig:ig_rural_folk }
				}
			}
		}
	}
	else = {
		# No clear winner - random IG
		heir ?= {
			random_list = {
				15 = { trigger = { owner = { exists = ig:ig_industrialists } } set_interest_group = ig:ig_industrialists }
				15 = { trigger = { owner = { exists = ig:ig_landowners } } set_interest_group = ig:ig_landowners }
				14 = { trigger = { owner = { exists = ig:ig_intelligentsia } } set_interest_group = ig:ig_intelligentsia }
				14 = { trigger = { owner = { exists = ig:ig_devout } } set_interest_group = ig:ig_devout }
				14 = { trigger = { owner = { exists = ig:ig_trade_unions } } set_interest_group = ig:ig_trade_unions }
				14 = { trigger = { owner = { exists = ig:ig_armed_forces } } set_interest_group = ig:ig_armed_forces }
				14 = { trigger = { owner = { exists = ig:ig_rural_folk } } trigger = { owner = { exists = ig:ig_rural_folk } } set_interest_group = ig:ig_rural_folk }
			}
		}
	}

	# ─── Mark heir as educated ───
	heir ?= {
		set_variable = received_education
		set_aptitude_trait_vars_effect = yes
	}
}

# Assigns aptitude traits to adult characters who lack them.
# 5-tier distribution for characters without formal education.
# Generals/admirals biased toward better military traits.

	# Administrative trait
	every_scope_character = {
		limit = {
			NOT = { age < define:NCharacters|ADULT_AGE }
			OR = {
				is_ruler = yes

			}
		}
		random_list = {

			35 = { add_trait = ruler_average_administrator }
			12 = { add_trait = ruler_skilled_administrator }

			OR = {

			NOT = { age < define:NCharacters|ADULT_AGE }

				is_ruler = yes
