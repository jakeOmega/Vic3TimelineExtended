### Heir Education - Scripted Effects

# ──────────────────────── MAIN RESOLUTION ────────────────────────
# Called when the heir reaches adulthood (JE complete) or on timeout.
# Applies traits, ideology, and IG to the heir based on accumulated investments.
# Investment thresholds scaled for monthly pulse system (~5% chance/month per focus):
#   0   = no investment -> weighted toward poor/average
#   1-2 = light investment
#   3-5 = moderate investment
#   6+  = heavy investment (sustained focus over many years)

heir_education_resolve_effect = {
	# ─── Resolve Administrative Trait ───
	if = {
		limit = { has_variable = heir_ed_admin var:heir_ed_admin >= 6 }
		heir ?= {
			random_list = {
				0 = { add_trait = ruler_poor_administrator }
				5 = { add_trait = ruler_average_administrator }
				35 = { add_trait = ruler_skilled_administrator }
				60 = { add_trait = ruler_exceptional_administrator }
			}
		}
	}
	else_if = {
		limit = { has_variable = heir_ed_admin var:heir_ed_admin >= 3 }
		heir ?= {
			random_list = {
				5 = { add_trait = ruler_poor_administrator }
				15 = { add_trait = ruler_average_administrator }
				45 = { add_trait = ruler_skilled_administrator }
				35 = { add_trait = ruler_exceptional_administrator }
			}
		}
	}
	else_if = {
		limit = { has_variable = heir_ed_admin var:heir_ed_admin >= 1 }
		heir ?= {
			random_list = {
				10 = { add_trait = ruler_poor_administrator }
				30 = { add_trait = ruler_average_administrator }
				40 = { add_trait = ruler_skilled_administrator }
				20 = { add_trait = ruler_exceptional_administrator }
			}
		}
	}
	else = {
		heir ?= {
			random_list = {
				25 = { add_trait = ruler_poor_administrator }
				45 = { add_trait = ruler_average_administrator }
				25 = { add_trait = ruler_skilled_administrator }
				5 = { add_trait = ruler_exceptional_administrator }
			}
		}
	}

	# ─── Resolve Diplomatic Trait ───
	if = {
		limit = { has_variable = heir_ed_diplo var:heir_ed_diplo >= 6 }
		heir ?= {
			random_list = {
				0 = { add_trait = ruler_poor_diplomat }
				5 = { add_trait = ruler_average_diplomat }
				35 = { add_trait = ruler_skilled_diplomat }
				60 = { add_trait = ruler_exceptional_diplomat }
			}
		}
	}
	else_if = {
		limit = { has_variable = heir_ed_diplo var:heir_ed_diplo >= 3 }
		heir ?= {
			random_list = {
				5 = { add_trait = ruler_poor_diplomat }
				15 = { add_trait = ruler_average_diplomat }
				45 = { add_trait = ruler_skilled_diplomat }
				35 = { add_trait = ruler_exceptional_diplomat }
			}
		}
	}
	else_if = {
		limit = { has_variable = heir_ed_diplo var:heir_ed_diplo >= 1 }
		heir ?= {
			random_list = {
				10 = { add_trait = ruler_poor_diplomat }
				30 = { add_trait = ruler_average_diplomat }
				40 = { add_trait = ruler_skilled_diplomat }
				20 = { add_trait = ruler_exceptional_diplomat }
			}
		}
	}
	else = {
		heir ?= {
			random_list = {
				25 = { add_trait = ruler_poor_diplomat }
				45 = { add_trait = ruler_average_diplomat }
				25 = { add_trait = ruler_skilled_diplomat }
				5 = { add_trait = ruler_exceptional_diplomat }
			}
		}
	}

	# ─── Resolve Military Trait ───
	if = {
		limit = { has_variable = heir_ed_military var:heir_ed_military >= 6 }
		heir ?= {
			random_list = {
				0 = { add_trait = ruler_poor_commander }
				5 = { add_trait = ruler_average_commander }
				35 = { add_trait = ruler_skilled_commander }
				60 = { add_trait = ruler_exceptional_commander }
			}
		}
	}
	else_if = {
		limit = { has_variable = heir_ed_military var:heir_ed_military >= 3 }
		heir ?= {
			random_list = {
				5 = { add_trait = ruler_poor_commander }
				15 = { add_trait = ruler_average_commander }
				45 = { add_trait = ruler_skilled_commander }
				35 = { add_trait = ruler_exceptional_commander }
			}
		}
	}
	else_if = {
		limit = { has_variable = heir_ed_military var:heir_ed_military >= 1 }
		heir ?= {
			random_list = {
				10 = { add_trait = ruler_poor_commander }
				30 = { add_trait = ruler_average_commander }
				40 = { add_trait = ruler_skilled_commander }
				20 = { add_trait = ruler_exceptional_commander }
			}
		}
	}
	else = {
		heir ?= {
			random_list = {
				25 = { add_trait = ruler_poor_commander }
				45 = { add_trait = ruler_average_commander }
				25 = { add_trait = ruler_skilled_commander }
				5 = { add_trait = ruler_exceptional_commander }
			}
		}
	}

	# ─── Resolve Ideology ───
	# heir_ed_ideology: negative = conservative, positive = progressive
	if = {
		limit = { has_variable = heir_ed_ideology var:heir_ed_ideology >= 4 }
		# Strong progressive push
		heir ?= {
			random_list = {
				35 = { set_ideology = ideology:ideology_reformer }
				25 = { set_ideology = ideology:ideology_liberal_leader }
				20 = { set_ideology = ideology:ideology_social_democrat }
				10 = { set_ideology = ideology:ideology_humanitarian }
				5 = { set_ideology = ideology:ideology_communist }
				3 = { set_ideology = ideology:ideology_vanguardist }
				2 = { }
			}
		}
	}
	else_if = {
		limit = { has_variable = heir_ed_ideology var:heir_ed_ideology >= 1 }
		# Mild progressive push
		heir ?= {
			random_list = {
				30 = { set_ideology = ideology:ideology_reformer }
				20 = { set_ideology = ideology:ideology_liberal_leader }
				15 = { set_ideology = ideology:ideology_humanitarian }
				5 = { set_ideology = ideology:ideology_social_democrat }
				30 = { }
			}
		}
	}
	else_if = {
		limit = { has_variable = heir_ed_ideology var:heir_ed_ideology <= -4 }
		# Strong conservative push
		heir ?= {
			random_list = {
				35 = { set_ideology = ideology:ideology_royalist }
				25 = { set_ideology = ideology:ideology_traditionalist }
				15 = { set_ideology = ideology:ideology_authoritarian }
				10 = { set_ideology = ideology:ideology_theocrat }
				8 = { set_ideology = ideology:ideology_ethno_nationalist }
				5 = { set_ideology = ideology:ideology_jingoist_leader }
				2 = { }
			}
		}
	}
	else_if = {
		limit = { has_variable = heir_ed_ideology var:heir_ed_ideology <= -1 }
		# Mild conservative push
		heir ?= {
			random_list = {
				30 = { set_ideology = ideology:ideology_royalist }
				20 = { set_ideology = ideology:ideology_traditionalist }
				10 = { set_ideology = ideology:ideology_authoritarian }
				5 = { set_ideology = ideology:ideology_theocrat }
				35 = { }
			}
		}
	}
	# else: no ideology investment, no change

	# ─── Resolve Interest Group ───
	# Three IG mentorship groups: radical, moderate, regressive
	# The group with the highest counter wins; magnitude affects probability

	# Find winning group
	# Radical wins: radical >= moderate AND radical >= regressive AND radical >= 1
	if = {
		limit = {
			var:heir_ed_ig_radical >= var:heir_ed_ig_moderate
			var:heir_ed_ig_radical >= var:heir_ed_ig_regressive
			var:heir_ed_ig_radical >= 4
		}
		heir ?= {
			random_list = {
				35 = { set_interest_group = owner.ig:ig_intelligentsia }
				25 = { set_interest_group = owner.ig:ig_rural_folk }
				15 = { set_interest_group = owner.ig:ig_trade_unions }
				25 = { }
			}
		}
	}
	else_if = {
		limit = {
			var:heir_ed_ig_radical >= var:heir_ed_ig_moderate
			var:heir_ed_ig_radical >= var:heir_ed_ig_regressive
			var:heir_ed_ig_radical >= 1
		}
		heir ?= {
			random_list = {
				25 = { set_interest_group = owner.ig:ig_intelligentsia }
				15 = { set_interest_group = owner.ig:ig_rural_folk }
				5 = { set_interest_group = owner.ig:ig_trade_unions }
				55 = { }
			}
		}
	}
	# Moderate wins: moderate > radical AND moderate >= regressive AND moderate >= 1
	else_if = {
		limit = {
			var:heir_ed_ig_moderate > var:heir_ed_ig_radical
			var:heir_ed_ig_moderate >= var:heir_ed_ig_regressive
			var:heir_ed_ig_moderate >= 4
		}
		heir ?= {
			random_list = {
				30 = { set_interest_group = owner.ig:ig_industrialists }
				25 = { set_interest_group = owner.ig:ig_petty_bourgeoisie }
				20 = { set_interest_group = owner.ig:ig_armed_forces }
				25 = { }
			}
		}
	}
	else_if = {
		limit = {
			var:heir_ed_ig_moderate > var:heir_ed_ig_radical
			var:heir_ed_ig_moderate >= var:heir_ed_ig_regressive
			var:heir_ed_ig_moderate >= 1
		}
		heir ?= {
			random_list = {
				20 = { set_interest_group = owner.ig:ig_industrialists }
				15 = { set_interest_group = owner.ig:ig_petty_bourgeoisie }
				10 = { set_interest_group = owner.ig:ig_armed_forces }
				55 = { }
			}
		}
	}
	# Regressive wins: regressive > radical AND regressive > moderate AND regressive >= 1
	else_if = {
		limit = {
			var:heir_ed_ig_regressive > var:heir_ed_ig_radical
			var:heir_ed_ig_regressive > var:heir_ed_ig_moderate
			var:heir_ed_ig_regressive >= 4
		}
		heir ?= {
			random_list = {
				35 = { set_interest_group = owner.ig:ig_landowners }
				30 = { set_interest_group = owner.ig:ig_devout }
				35 = { }
			}
		}
	}
	else_if = {
		limit = {
			var:heir_ed_ig_regressive > var:heir_ed_ig_radical
			var:heir_ed_ig_regressive > var:heir_ed_ig_moderate
			var:heir_ed_ig_regressive >= 1
		}
		heir ?= {
			random_list = {
				25 = { set_interest_group = owner.ig:ig_landowners }
				15 = { set_interest_group = owner.ig:ig_devout }
				60 = { }
			}
		}
	}
	else = {
		# No explicit IG mentorship: education type influences IG weakly
		if = {
			limit = {
				has_variable = heir_ed_admin
				has_variable = heir_ed_diplo
				has_variable = heir_ed_military
				var:heir_ed_admin > var:heir_ed_diplo
				var:heir_ed_admin > var:heir_ed_military
			}
			heir ?= {
				random_list = {
					25 = { set_interest_group = owner.ig:ig_industrialists }
					15 = { set_interest_group = owner.ig:ig_petty_bourgeoisie }
					60 = { }
				}
			}
		}
		else_if = {
			limit = {
				has_variable = heir_ed_diplo
				has_variable = heir_ed_admin
				has_variable = heir_ed_military
				var:heir_ed_diplo > var:heir_ed_admin
				var:heir_ed_diplo > var:heir_ed_military
			}
			heir ?= {
				random_list = {
					30 = { set_interest_group = owner.ig:ig_intelligentsia }
					10 = { set_interest_group = owner.ig:ig_landowners }
					60 = { }
				}
			}
		}
		else_if = {
			limit = {
				has_variable = heir_ed_military
				has_variable = heir_ed_admin
				has_variable = heir_ed_diplo
				var:heir_ed_military > var:heir_ed_admin
				var:heir_ed_military > var:heir_ed_diplo
			}
			heir ?= {
				random_list = {
					35 = { set_interest_group = owner.ig:ig_armed_forces }
					65 = { }
				}
			}
		}
	}

	# ─── Mark heir as educated ───
	heir ?= {
		set_variable = received_education
	}
}

# ──────────────────────── CLEANUP ────────────────────────

heir_education_cleanup_effect = {
	remove_variable = heir_ed_admin
	remove_variable = heir_ed_diplo
	remove_variable = heir_ed_military
	remove_variable = heir_ed_ideology
	remove_variable = heir_ed_ig_radical
	remove_variable = heir_ed_ig_moderate
	remove_variable = heir_ed_ig_regressive
	remove_variable = heir_ed_total
	# Focus variables
	if = { limit = { has_variable = heir_ed_focus_admin } remove_variable = heir_ed_focus_admin }
	if = { limit = { has_variable = heir_ed_focus_diplo } remove_variable = heir_ed_focus_diplo }
	if = { limit = { has_variable = heir_ed_focus_military } remove_variable = heir_ed_focus_military }
	if = { limit = { has_variable = heir_ed_focus_progressive } remove_variable = heir_ed_focus_progressive }
	if = { limit = { has_variable = heir_ed_focus_conservative } remove_variable = heir_ed_focus_conservative }
	if = { limit = { has_variable = heir_ed_focus_radical } remove_variable = heir_ed_focus_radical }
	if = { limit = { has_variable = heir_ed_focus_moderate } remove_variable = heir_ed_focus_moderate }
	if = { limit = { has_variable = heir_ed_focus_regressive } remove_variable = heir_ed_focus_regressive }
	# Modifiers
	if = { limit = { has_modifier = heir_education_innovation_cost } remove_modifier = heir_education_innovation_cost }
	if = { limit = { has_modifier = heir_education_grace_period } remove_modifier = heir_education_grace_period }
	if = { limit = { has_modifier = heir_education_event_cooldown } remove_modifier = heir_education_event_cooldown }
}

# ──────────────────────── NON-HEIR TRAIT ASSIGNMENT ────────────────────────
# Assigns aptitude traits to adult characters who lack them.
# Mostly gives average traits, with small chances for better/worse.
# Generals/admirals are biased toward better military traits.

assign_aptitude_traits_effect = {
	# Administrative trait
	every_scope_character = {
		limit = {
			NOT = { age < define:NCharacters|ADULT_AGE }
			OR = {
				is_ruler = yes
				has_role = politician
				has_role = agitator
				has_role = general
				has_role = admiral
			}
			NOR = {
				has_trait = ruler_poor_administrator
				has_trait = ruler_average_administrator
				has_trait = ruler_skilled_administrator
				has_trait = ruler_exceptional_administrator
			}
		}
		random_list = {
			15 = { add_trait = ruler_poor_administrator }
			55 = { add_trait = ruler_average_administrator }
			25 = { add_trait = ruler_skilled_administrator }
			5 = { add_trait = ruler_exceptional_administrator }
		}
	}

	# Diplomatic trait
	every_scope_character = {
		limit = {
			NOT = { age < define:NCharacters|ADULT_AGE }
			OR = {
				is_ruler = yes
				has_role = politician
				has_role = agitator
				has_role = general
				has_role = admiral
			}
			NOR = {
				has_trait = ruler_poor_diplomat
				has_trait = ruler_average_diplomat
				has_trait = ruler_skilled_diplomat
				has_trait = ruler_exceptional_diplomat
			}
		}
		random_list = {
			15 = { add_trait = ruler_poor_diplomat }
			55 = { add_trait = ruler_average_diplomat }
			25 = { add_trait = ruler_skilled_diplomat }
			5 = { add_trait = ruler_exceptional_diplomat }
		}
	}

	# Military trait - biased toward skilled/exceptional for generals/admirals
	every_scope_character = {
		limit = {
			NOT = { age < define:NCharacters|ADULT_AGE }
			OR = {
				is_ruler = yes
				has_role = politician
				has_role = agitator
				has_role = general
				has_role = admiral
			}
			NOR = {
				has_trait = ruler_poor_commander
				has_trait = ruler_average_commander
				has_trait = ruler_skilled_commander
				has_trait = ruler_exceptional_commander
			}
		}
		if = {
			limit = {
				OR = {
					has_role = general
					has_role = admiral
				}
			}
			random_list = {
				5 = { add_trait = ruler_poor_commander }
				30 = { add_trait = ruler_average_commander }
				45 = { add_trait = ruler_skilled_commander }
				20 = { add_trait = ruler_exceptional_commander }
			}
		}
		else = {
			random_list = {
				15 = { add_trait = ruler_poor_commander }
				55 = { add_trait = ruler_average_commander }
				25 = { add_trait = ruler_skilled_commander }
				5 = { add_trait = ruler_exceptional_commander }
			}
		}
	}
}
