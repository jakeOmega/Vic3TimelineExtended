# Minority Script Preservation - preserves minority writing systems
# alongside the reformed national script. Reduces assimilation pressure
# but calms separatist sentiment.
amendment_minority_script_preservation = {
	parent = law_multilingual_federalism

	allowed_laws = {
		law_state_led_language_reform
		law_civic_monolingualism
	}

	modifier = {
		state_assimilation_mult = -0.10
		country_support_separatism_resistance_mult = 0.10
		country_bureaucracy_investment_cost_factor_mult = 0.03
	}

	possible = {
		any_scope_state = {
			any_scope_culture = {
				NOT = { is_primary_culture_of = root }
			}
		}
	}

	would_sponsor = {
		OR = {
			is_interest_group_type = ig_intelligentsia
			AND = {
				exists = scope:approval
				scope:approval > 0
			}
		}
	}

	ai_will_revoke = {
		any_interest_group = {
			is_in_government = yes
			law_stance = {
				law = law_type:law_multilingual_federalism
				value < neutral
			}
		}
	}

	can_repeal = {
		legitimacy >= legitimacy_to_repeal_amendment
	}

	amendment_activism_multiplier = 0.25
}
