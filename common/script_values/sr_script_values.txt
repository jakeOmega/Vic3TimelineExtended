# ============================================================================
# Strategic Reserve — Script Values
# ============================================================================
# Weekly signed-flow bookkeeping. Positive configured rates store goods,
# negative configured rates withdraw them. The hub applies the chosen base rate
# up to its own throughput cap (10 goods/level/week), and generic building
# throughput modifiers scale the active market flow from there.
# Scope contract: these values are evaluated in country scope. Live hub data is
# bridged in from building scope by sr_refresh_hub_cache_effect, which writes
# sr_hub_level_cached and sr_hub_throughput_cached before these values are
# consumed by the JE, buttons, or hub-flow refresh helpers.
# ============================================================================

# Hub level — 0 if no hub exists.
sr_hub_level = {
	value = 0
	add = var:sr_hub_level_cached
}

sr_weekly_base_rate_cap = {
	value = 0
	add = modifier:country_sr_weekly_rate_cap_add
	min = 1
}

sr_neg_weekly_base_rate_cap = {
	value = sr_weekly_base_rate_cap
	multiply = -1
}

# Total throughput_add modifier on the hub, summing all contributions.
sr_hub_throughput = {
	value = 0
	add = var:sr_hub_throughput_cached
}

# Throughput scaling factor — clamped at 0.1x so a negative pile-on of
# debuffs can't flip the delta sign or zero out flow entirely.
sr_throughput_factor = {
	value = 1
	add = sr_hub_throughput
	min = 0.1
	max = 5
}

# Shared button step size. Tier 0 = 1, 1 = 10, 2 = 100, 3 = 1000, 4 = 10000.
sr_adjust_step_value = {
	value = 1
	if = {
		limit = { var:sr_adjust_step_tier = 0 }
		value = 1
	}
	else_if = {
		limit = { var:sr_adjust_step_tier = 1 }
		value = 10
	}
	else_if = {
		limit = { var:sr_adjust_step_tier = 2 }
		value = 100
	}
	else_if = {
		limit = { var:sr_adjust_step_tier = 3 }
		value = 1000
	}
	else = {
		value = 10000
	}
}

# --- GRAIN ---
sr_grain_decay_rate = {
	value = 0
	add = modifier:country_sr_grain_decay_add
	divide = 52 # convert from per-year to per-week decay
	min = 0
	max = 1
}

sr_grain_weekly_decay = {
	value = var:sr_grain_stored
	multiply = sr_grain_decay_rate
}

# Actual gross flow the hub must apply to achieve the configured target net change
# (includes decay replenishment). Positive = hub buying, negative = hub selling.
sr_grain_actual_rate = {
	value = 0
	if = {
		limit = { var:sr_hub_workforce_cached > 0.99 }
		add = var:sr_grain_rate
	}
	add = sr_grain_weekly_decay
}

sr_grain_actual_rate_base_applied = {
	value = sr_grain_actual_rate
	min = {
		value = 0
		subtract = sr_weekly_base_rate_cap
	}
	max = sr_weekly_base_rate_cap
}

sr_grain_weekly_delta = {
	value = 0
	add = sr_grain_actual_rate_base_applied
	subtract = sr_grain_weekly_decay
}

sr_grain_capacity = {
	value = 0
	add = modifier:country_sr_grain_capacity_add
}

sr_grain_good_mult = {
	value = 0
	if = {
		limit = { sr_grain_actual_rate_base_applied > 0 }
		add = sr_grain_actual_rate_base_applied
	}
	else = {
		subtract = sr_grain_actual_rate_base_applied
	}
	# Compensate for throughput modifiers, so total rate equals desired rate
	divide = sr_throughput_factor
}

sr_grain_max_withdrawable = {
	value = 0
	add = var:sr_grain_stored
	subtract = sr_grain_weekly_decay
	multiply = -1
}

sr_grain_max_storable = {
	value = 0
	add = sr_grain_capacity
	subtract = var:sr_grain_stored
	subtract = sr_grain_weekly_decay
}

sr_grain_sale_profit = {
	value = 0
	if = {
		limit = { sr_grain_actual_rate_base_applied < 0 }
		subtract = sr_grain_actual_rate_base_applied
		g:grain = {
			multiply = base_price
		}
		market = {
			mg:grain = {
				multiply = {
					value = 1
					add = market_goods_pricier
				}
			}
		}
	}
}
