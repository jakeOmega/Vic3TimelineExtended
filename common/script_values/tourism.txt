# --- INDIVIDUAL TOURISM OUTPUT MODIFIERS ---

# From the state's static, unchangeable appeal score (0-100)
# Assumes the score is stored in a state variable, e.g., via script from your dictionary
tourism_bonus_from_base_appeal = {
	value = base_tourism_appeal_value
}

city_size_rank = {
	if = {
		limit = {
			has_variable = state_city_size_rank
		}
		value = var:state_city_size_rank
	}
	else = {
		value = 0 # Default value if city size is not defined
	}
}

# Tiered bonus from Cities
tourism_bonus_from_urban_centers = {
	value = 0
	if = {
		limit = {
			city_size_rank < 21
			city_size_rank > 0
		}
		add = {
			value = 21
			subtract = city_size_rank
			multiply = 0.05
		}
	}
}

# Large bonus from Airports (building has a max level, so no DR needed)
tourism_bonus_from_airports = {
	value = 0
	if = {
		limit = { has_building = building_airport }
		add = {
			value = b:building_airport.level
			multiply = 15	# +15% per level
		}
	}
	divide = 100	# Normalize to a 0-100 scale
}

# Penalties from repressive laws
tourism_penalty_from_laws = {
	value = 0
	if = { limit = { owner = { has_law = law_type:law_secret_police } } subtract = 30 }
	if = { limit = { owner = { has_law = law_type:law_outlawed_dissent } } subtract = 30 }
	if = { limit = { owner = { has_law = law_type:law_ethnostate } } subtract = 20 }
	if = { limit = { owner = { has_law = law_type:law_racial_segregation } } subtract = 5 }
	if = { limit = { owner = { has_law = law_type:law_intrusive_surveillance } } subtract = 25 }
	if = { limit = { owner = { has_law = law_type:law_state_controlled_internet } } subtract = 5 }
	divide = 100	# Normalize to a 0-100 scale
}

# Bonuses from progressive laws
tourism_bonus_from_laws = {
	value = 0
	if = { limit = { owner = { has_law = law_type:law_guaranteed_liberties } } add = 10 }
	if = { limit = { owner = { has_law = law_type:law_multicultural } } add = 15 }
	if = { limit = { owner = { has_law = law_type:law_moderate_data_privacy } } add = 5 }
	if = { limit = { owner = { has_law = law_type:law_strong_privacy_rights } } add = 10 }
	divide = 100	# Normalize to a 0-100 scale
}

tourism_net_bonus_from_laws = {
	value = tourism_bonus_from_laws
	add = tourism_penalty_from_laws
}

# Penalty from instability
tourism_penalty_from_turmoil = {
	value = turmoil
	multiply = -1
}

# Penalty from environmental damage
tourism_penalty_from_pollution = {
	value = state_region.pollution_amount
	divide = 255
	multiply = -2
}
