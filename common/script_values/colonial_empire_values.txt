# ============================================================================
# COLONIAL EMPIRE SCRIPT VALUES
# Used by colonial_stability_bar monthly_progress and decolonization events
# ============================================================================

# Count of overseas colonial states (non-homeland, overseas, 100k+ pop, has unintegrated pops)
colonial_state_count = {
	value = 0
	every_scope_state = {
		limit = {
			is_overseas_colonial_state = yes
		}
		add = 1
	}
}

# Count of overseas colonial states with very low acceptance pops (< 20)
colonial_low_acceptance_count = {
	value = 0
	every_scope_state = {
		limit = {
			is_overseas_colonial_state = yes
			any_scope_pop = {
				pop_acceptance < 20
			}
		}
		add = 1
	}
}

# Count of overseas colonial states where ALL pops have high acceptance (>= 60)
colonial_high_acceptance_count = {
	value = 0
	every_scope_state = {
		limit = {
			NOT = { is_homeland_of_country_cultures = owner }
			state_population >= 100000
			is_overseas_from_capital = yes
			NOT = {
				any_scope_pop = {
					pop_acceptance < acceptance_status_4
				}
			}
		}
		add = 1
	}
}

# Count of great powers with anti-colonial stance (set via event 14)
colonial_gp_condemners_count = {
	value = 0
	every_country = {
		limit = {
			NOT = { this = ROOT }
			country_rank >= rank_value:great_power
			has_variable = gp_anti_colonial_stance
		}
		add = 1
	}
}

# Count of great powers with pro-colonial stance (set via event 14)
colonial_gp_supporters_count = {
	value = 0
	every_country = {
		limit = {
			NOT = { this = ROOT }
			country_rank >= rank_value:great_power
			has_variable = gp_pro_colonial_stance
		}
		add = 1
	}
}

# Bureaucracy cost of colonial maintenance - scaled by number of colonial states
colonial_maintenance_bureaucracy_cost = {
	value = colonial_state_count
	multiply = -0.01
	min = -0.25
}

# Colonial development investment cost - scales with colonial holdings
colonial_development_cost = {
	value = colonial_state_count
	multiply = 50
	add = 100
	min = 100
}

# Metropole state count: total states minus overseas colonial states.
# Floored at 1 to avoid divide-by-zero when an empire is wholly colonial.
metropole_state_count = {
	value = 0
	every_scope_state = {
		limit = {
			NOT = { is_overseas_colonial_state = yes }
		}
		add = 1
	}
	min = 1
}

# Overreach ratio: how much your colonial holdings exceed ~1.5x your metropole.
# 0 (or below) means no penalty; positive values feed the bar's overreach drain.
# A larger metropole tolerates a larger empire proportionally — France can hold
# more before overreach than Belgium, Belgium can hold its small empire freely.
colonial_overreach_ratio = {
	value = colonial_state_count
	divide = metropole_state_count
	subtract = 1.5
	min = 0
}

# Colonial GDP-scaled monthly expense for the Investment policy. Used as the
# multiplier on colonial_development_investment_expense_modifier so the cost
# appears as a line item in the country's budget panel (~0.5% of GDP/month).
# Positive — multiplier scales country_expenses_add = 1 into actual expenses.
colonial_invest_monthly_expense = {
	value = gdp
	multiply = 0.005
}

# One-shot startup cost for the Development Investment program, scaled to GDP
# so the gate matches the scale of the empire — ~2% of monthly GDP, applied
# once on activation. Mirrored by colonial_assimilate_startup_cost so the two
# program-startup costs read as parallel. Negative because used with add_treasury.
colonial_invest_startup_cost = {
	value = gdp
	multiply = -0.02
}

# One-shot startup cost for the Cultural Assimilation program. Parallel to
# colonial_invest_startup_cost — smaller because Assimilation already carries
# the heavy bureaucracy mult drag.
colonial_assimilate_startup_cost = {
	value = gdp
	multiply = -0.01
}

# Scope: state. SoL threshold below which the state qualifies as colonial-
# feeling regardless of cultural acceptance: 0.8 x owner's national average.
# Multiplicative scaling means a poorer metropole has a lower absolute floor
# (its colonies need less SoL to "feel poor"); a richer metropole has higher
# absolute tolerance but the same proportional sensitivity. Captures the
# Portugal-with-assimilatory-citizenship case: high acceptance but the
# colonies are still markedly poorer than the mainland.
colonial_state_sol_qualifying_floor = {
	value = owner.average_sol
	multiply = 0.8
}
