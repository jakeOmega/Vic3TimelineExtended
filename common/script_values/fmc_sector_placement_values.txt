fmc_buildings_under_construction = {
	value = 0
	every_scope_building = {
		limit = {
			is_under_construction = yes
		}
		add = 1
	}
}

fmc_amount_of_sectors = {
	value = 0
	every_scope_building = {
		limit = {
			type = bt:building_construction_sector
		}
		add = 1
	}
}

fmc_construction_per_sector = {
	value = total_construction_use_target
	if = {
		limit = {
			fmc_amount_of_sectors > 0
		}
		divide = fmc_amount_of_sectors
	}
	#subtract = 1
}
