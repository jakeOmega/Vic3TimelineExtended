# ==============================================================================
# Private Construction System — Script Values
# Scope: country unless noted.
# ==============================================================================

# ---- Construction price ----

# Baseline construction-services price used when the price smoother is
# uninitialised.  Expressed as a raw-goods price-index equivalent.
PC_BASE_CONSTRUCTION_PRICE = 500

# Current market price of construction goods, scaled to the baseline.
pc_construction_price = {
	value = this.market.mg:construction.market_goods_pricier
	add = 1
	multiply = PC_BASE_CONSTRUCTION_PRICE
}

# ---- Investment pool flows ----

# Amount the investment pool spent this week (gross minus net income).
pc_investment_spending = {
	value = investment_pool_gross_income
	subtract = investment_pool_net_income
}

# Construction units privately purchased this week.
pc_private_construction_bought = {
	value = pc_investment_spending
	divide = pc_construction_price
}

# Price-sensitivity factor: rises as construction goods get more expensive.
pc_investment_price_factor = {
	value = this.market.mg:construction.market_goods_pricier
	add = 2.25
	divide = 2.25
}

# ---- Current construction usage ----

# Total construction goods consumed across this country's market this week.
pc_total_construction_use = {
	value = this.market.mg:construction.market_goods_consumption
}

# Total construction spending in money terms.
pc_total_construction_spending = {
	value = pc_total_construction_use
	multiply = pc_construction_price
}

# Government share of construction spending.
pc_government_construction_spending = {
	value = pc_total_construction_spending
	subtract = pc_investment_spending
}

# Private construction units used.
pc_private_construction_use = {
	value = pc_investment_spending
	divide = pc_construction_price
}

# Government construction units used.
pc_government_construction_use = {
	value = pc_government_construction_spending
	divide = pc_construction_price
}

# ---- Target calculations ----

# Target investment pool size: enough to sustain 24 weeks of gross income.
pc_pool_target_size = {
	if = {
		limit = { investment_pool_gross_income > 0 }
		value = investment_pool_gross_income
	}
	else = {
		value = 100000
	}
	multiply = 24
}

# How much of this week's gross income the private sector should spend on
# construction.  Scales up when the pool is above target and down when below,
# creating a self-correcting feedback loop.
pc_private_spending_target = {
	value = investment_pool_gross_income
	multiply = {
		value = investment_pool
		subtract = pc_pool_target_size
		divide = pc_pool_target_size
		add = 1
	}
}

# Government construction units to purchase (zero if construction is paused
# or the target variable has not been initialised yet).
pc_government_use_target = {
	if = {
		limit = { has_variable = pc_government_target }
		if = {
			limit = { is_construction_paused = no }
			value = var:pc_government_target
		}
		else = {
			value = 0
		}
	}
	else = {
		value = 1
	}
}

# Private construction units target, capped at 1000 × the country's
# construction-speed multiplier.
pc_private_use_target = {
	value = pc_private_spending_target
	divide = var:pc_price_smoothed
	if = {
		limit = { investment_pool_gross_income > 100000000 }
		min = 1000
	}
	max = {
		value = 1000
		multiply = modifier:country_max_weekly_construction_progress_add
	}
}

# Combined private and government construction target this week.
pc_total_use_target = {
	value = pc_private_use_target
	add = pc_government_use_target
}

# Private share of total target as a percentage (used as modifier multiplier).
pc_private_share_pct = {
	if = {
		limit = { pc_total_use_target > 0 }
		value = pc_private_use_target
		divide = pc_total_use_target
	}
	else = {
		value = 1
	}
	multiply = 100
}

# Next value for the exponential moving average price smoother.
# Steps 1/10th of the gap between the current smoothed price and the live price.
pc_price_smoothed_next = {
	if = {
		limit = { NOT = { has_variable = pc_price_smoothed } }
		value = PC_BASE_CONSTRUCTION_PRICE
	}
	else_if = {
		limit = { var:pc_price_smoothed < pc_construction_price }
		value = var:pc_price_smoothed
		add = {
			value = pc_construction_price
			subtract = var:pc_price_smoothed
			divide = 10
		}
	}
	else = {
		value = var:pc_price_smoothed
		subtract = {
			value = var:pc_price_smoothed
			subtract = pc_construction_price
			divide = 10
		}
	}
}

# ---- Construction queue estimates ----

pc_queue_government_remaining = {
	value = construction_queue_government_duration
	multiply = pc_total_construction_use
	if = {
		limit = { construction_queue_government_duration > 52000 }
		value = 0
	}
}

pc_queue_private_remaining = {
	value = construction_queue_private_duration
	multiply = pc_total_construction_use
	if = {
		limit = { construction_queue_private_duration > 52000 }
		value = 0
	}
}

# ---- Hub placement helpers (country scope) ----

# Number of active building_private_construction_hub buildings
# across all of this country's states.
pc_hub_count = {
	value = 0
	every_scope_building = {
		limit = {
			type = bt:building_private_construction_hub
			occupancy > 0
		}
		add = 1
	}
}

# Target construction throughput per hub (used as the throughput modifier
# multiplier).
pc_throughput_per_hub = {
	value = pc_total_use_target
	if = {
		limit = { pc_hub_count > 0 }
		divide = pc_hub_count
	}
}

# ---- AI values ----

# How many government construction units the AI should purchase this week.
pc_ai_government_target = {
	if = {
		limit = {
			is_construction_paused = no
			construction_queue_num_queued_government_levels > 0
		}
		value = net_fixed_income
		min = 0
		divide = var:pc_price_smoothed
		multiply = {
			add = scaled_gold_reserves
			subtract = scaled_debt
			add = 1
		}
		min = {
			value = 10
			if = {
				limit = { bureaucracy < 0 }
				add = {
					value = income
					divide = {
						value = pc_construction_price
						min = 50
						max = 150
					}
					divide = 5
				}
			}
		}
		round_to = 1
		max = {
			value = construction_queue_num_queued_government_levels
			multiply = modifier:country_max_weekly_construction_progress_add
		}
	}
	else = {
		value = 0
	}
}
