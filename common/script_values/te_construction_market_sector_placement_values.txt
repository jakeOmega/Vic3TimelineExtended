te_construction_market_buildings_under_construction = {
	value = 0
	every_scope_building = {
		limit = {
			is_under_construction = yes
		}
		add = 1
	}
}

te_construction_market_num_sites = {
	value = 0
	every_scope_building = {
		limit = {
			is_building_type = te_construction_market_site
			occupancy > 0
		}
		add = 1
	}
}

te_construction_market_per_site = {
	value = te_construction_market_total_use_target
	if = {
		limit = {
			te_construction_market_num_sites > 0
		}
		divide = te_construction_market_num_sites
	}
}

# Per-site target adjusted for actual staffing of THIS site. Computed in
# building scope (reads occupancy). Used by te_construction_market_apply_per_site_modifier
# via a country-variable handoff — see that effect's comment for why a
# direct `multiplier = <this_script_value>` doesn't work.
#
# `subtract = 1` yields `per_site - 1` so the resulting modifier-value
# plus the engine's implicit +1 baseline lands exactly at `per_site`.
#
# Conservation: total throughput across sites = num_sites × per_site =
# total_use_target, regardless of staffing distribution. Inverse-scaling
# by staffing means an understaffed site contributes the same as a fully
# staffed one (its workers are individually more productive).
#
# Uses `occupancy` (filled slots / max slots, any pop type) rather than
# `has_employee_slots_filled = { pop_type = laborers ... }`. The latter
# was observed in-game (Yemen save) to under-report even at 100% laborer
# staffing: 30/30 laborers + "Fully Employed", yet the trigger returned
# below the 20% threshold and the else branch fired, inflating throughput
# to +1005%. `occupancy` reads the engine's slot-fill ratio directly.
te_construction_market_employment_adjusted_per_site = {
	value = owner.te_construction_market_per_site
	# Floor staffing at 0.1 to avoid divide-by-zero (matches the staircase's
	# lowest non-zero bucket). A 0-9% staffed site receives a 10× per-site
	# target — large but bounded.
	if = {
		limit = { occupancy >= 0.1 }
		divide = te_construction_market_employee_slots_filled
	}
	else = {
		multiply = 10
	}
	subtract = 1
}

# Building-scope staircase approximating fill fraction. 10 buckets of +0.1,
# so a fully-staffed site reads 1.0 and an empty one reads 0.0.
# Uses `occupancy` (any pop type) rather than `has_employee_slots_filled`
# (laborer-specific) for the reason documented above.
te_construction_market_employee_slots_filled = {
	value = 0
	if = { limit = { occupancy >= 0.1 } add = 0.1 }
	if = { limit = { occupancy >= 0.2 } add = 0.1 }
	if = { limit = { occupancy >= 0.3 } add = 0.1 }
	if = { limit = { occupancy >= 0.4 } add = 0.1 }
	if = { limit = { occupancy >= 0.5 } add = 0.1 }
	if = { limit = { occupancy >= 0.6 } add = 0.1 }
	if = { limit = { occupancy >= 0.7 } add = 0.1 }
	if = { limit = { occupancy >= 0.8 } add = 0.1 }
	if = { limit = { occupancy >= 0.9 } add = 0.1 }
	if = { limit = { occupancy >= 1 } add = 0.1 }
}
