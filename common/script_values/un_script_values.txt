# ---- Member Statistics ----

un_member_count = {
	value = 0
	every_country = {
		limit = { has_modifier = un_member_modifier }
		add = 1
	}
}

un_independent_country_count = {
	value = 0
	every_country = {
		limit = { is_subject = no }
		add = 1
	}
}

un_gp_member_count = {
	value = 0
	every_country = {
		limit = {
			has_modifier = un_member_modifier
			country_rank >= rank_value:great_power
		}
		add = 1
	}
}

un_total_gp_count = {
	value = 0
	every_country = {
		limit = { country_rank >= rank_value:great_power }
		add = 1
	}
}

# ---- Economic & Population Share ----

un_member_gdp_share = {
	value = 0
	every_country = {
		limit = { has_modifier = un_member_modifier }
		add = gdp
	}
	multiply = 100
	divide = {
		value = global_gdp_script_value
		min = 1
	}
}

un_member_pop_share = {
	value = 0
	every_country = {
		limit = { has_modifier = un_member_modifier }
		add = total_population
	}
	multiply = 100
	divide = {
		value = global_population
		min = 1
	}
}

# ---- Authority Drift Factors ----

un_gp_with_humanitarian_regs = {
	value = 0
	every_country = {
		limit = {
			has_modifier = un_member_modifier
			country_rank >= rank_value:great_power
			has_law = law_type:law_humanitarian_regulations
		}
		add = 1
	}
}

un_member_wars_count = {
	value = 0
	every_country = {
		limit = {
			has_modifier = un_member_modifier
			is_at_war = yes
			any_country = {
				has_war_with = PREV
				has_modifier = un_member_modifier
			}
		}
		add = 1
	}
}

# ---- Per-Country Drift Component Script Values (for loc display) ----

un_authority_drift_base = {
	value = 0
	if = {
		limit = { var:un_authority > 55 }
		add = -0.25
	}
	else_if = {
		limit = { var:un_authority < 45 }
		add = 0.25
	}
}

un_authority_drift_democracy = {
	value = 0
	every_country = {
		limit = {
			country_rank >= rank_value:great_power
			has_law = law_type:law_humanitarian_regulations
		}
		add = 0.1
	}
	every_country = {
		limit = {
			country_rank >= rank_value:great_power
			has_law = law_type:law_limited_war
		}
		add = 0.2
	}
}

un_authority_drift_champion = {
	value = 0
	every_country = {
		limit = { je:je_united_nations = { has_modifier = un_champion_order_cost } }
		add = 0.3
		if = {
			limit = { global_country_ranking < 3 }
			add = 0.2
		}
	}
}

un_authority_drift_undermine = {
	value = 0
	every_country = {
		limit = { je:je_united_nations = { has_modifier = un_undermine_order_cost } }
		add = -0.3
		if = {
			limit = { global_country_ranking < 3 }
			add = -0.2
		}
	}
}

un_authority_drift_wars = {
	value = 0
	every_country = {
		limit = {
			is_at_war = yes
			any_country = {
				has_war_with = ROOT
				has_modifier = un_member_modifier
			}
		}
		add = -1
	}
}

un_authority_drift_total = {
	value = un_authority_drift_base
	add = un_authority_drift_democracy
	add = un_authority_drift_champion
	add = un_authority_drift_undermine
	add = un_authority_drift_wars
}

# ---- Financial Cost / Benefit Scaling ----

# Expense for contributing to a UN program (~0.1% of GDP)
un_program_expense_value = {
	value = gdp
	multiply = 0.001
	divide = 52 # weekly cost
}

# Scaled expense for humanitarian aid (accounts for multiple ongoing programs)
un_aid_total_expense_value = {
	value = un_program_expense_value
	if = {
		limit = {
			has_variable = un_aid_contribution_count
			var:un_aid_contribution_count >= 2
		}
		multiply = var:un_aid_contribution_count
	}
}

# Scaled expense for peacekeeping (accounts for multiple ongoing operations)
un_peacekeeping_total_expense_value = {
	value = un_program_expense_value
	if = {
		limit = {
			has_variable = un_peacekeeping_contribution_count
			var:un_peacekeeping_contribution_count >= 2
		}
		multiply = var:un_peacekeeping_contribution_count
	}
}

			}
		}
		add = 1
	}
}

# GDP-weighted peacekeeping benefit scale
# Total contributor GDP / recipient GDP, normalised
# Evaluated in the recipient country's scope
un_peacekeeping_benefit_scale = {
	value = 0
	every_country = {
		limit = {
			OR = {
				has_modifier = un_peacekeeping_contributor_modifier
				has_modifier = un_peacekeeping_deployed_modifier
			}
		}
		add = gdp
	}
	divide = {
		value = gdp
		min = 100000
	}
	multiply = 0.5
	min = 0.5
	max = 5.0
}

# GDP-weighted humanitarian/development aid benefit scale
# Evaluated in the recipient country's scope
