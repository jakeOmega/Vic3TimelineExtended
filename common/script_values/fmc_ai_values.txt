fmc_ai_construction_purchase_target = {
	if = {
		limit = {
			is_construction_paused = no
			construction_queue_num_queued_government_levels > 0
		}
		value = net_fixed_income
		min = 0
		divide = var:fmc_price_stabiliser
		multiply = {
			add = scaled_gold_reserves
			subtract = scaled_debt
			add = 1

			# Bankruptcy-avoidance brake: when reserves are thin under shock
			# (war or financial-cycle stress), cut spending PROPORTIONALLY
			# rather than waiting for debt to spike. Linear sliding scale —
			# avoids cliff effects that would cascade economic shocks.
			#
			# scaled_gold_reserves runs ~0 (empty) to ~1 (well-stocked).
			# Subtracting (target - reserves) when reserves < target eats
			# into the multiplier; the higher the target (war/banking stress),
			# the deeper the cut while reserves are thin.

			# Wartime: target adequate reserves of ~0.5 scaled
			if = {
				limit = {
					is_at_war = yes
					scaled_gold_reserves < 0.5
				}
				subtract = {
					value = 0.5
					subtract = scaled_gold_reserves
				}
			}

			# Banking cycle stress (high bubble pressure or active crash):
			# target ~0.3 scaled — modest extra caution
			if = {
				limit = {
					var:bubble_pressure >= 50
					scaled_gold_reserves < 0.3
				}
				subtract = {
					value = 0.3
					subtract = scaled_gold_reserves
				}
			}

			# Floor at 0.1× — never fully halt; AI should keep some momentum
			min = 0.1
		}
		min = {
			value = 10
			if = {
				limit = {
					bureaucracy < 0
				}
				add = {
					value = income
					divide = {
						value = fmc_construction_price
						min = 50
						max = 150
					}
					divide = 5
				}
			}
		}
		round_to = 1
		max = {
			value = construction_queue_num_queued_government_levels
			multiply = modifier:country_max_weekly_construction_progress_add
		}
	}
	else = {
		value = 0
	}
}
