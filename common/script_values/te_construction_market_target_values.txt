te_construction_market_private_spending_target = {
	value = investment_pool_gross_income
	multiply = {
		value = investment_pool
		subtract = te_construction_market_investment_pool_size_target
		divide = te_construction_market_investment_pool_size_target
		add = 1
	}
}


te_construction_market_public_use_target = {
	if = {
		limit = {
			has_variable = te_construction_market_public_target
		}
		if = {
			limit = {
				is_construction_paused = no
			}
			value = var:te_construction_market_public_target
		}
		else = {
			value = 0
		}

	}
	else = {
		value = 1
	}
}

te_construction_market_investment_pool_size_target = {
	if = {
		limit = {
			investment_pool_gross_income > 0
		}
		value = investment_pool_gross_income
	}
	else = {
		value = 100000
	}
	multiply = 24
}

te_construction_market_investment_pool_x_1000 = {
	value = investment_pool
	multiply = 1000
}

te_construction_market_private_use_target = {
	value = te_construction_market_private_spending_target
	divide = var:te_construction_market_price_stabilizer
	if = {
		limit = {
			investment_pool_gross_income > 100000000
		}
		min = 1000
	}
	max = {
		value = 1000
		multiply = modifier:country_max_weekly_construction_progress_add
	}
}


te_construction_market_total_use_target = {
	value = te_construction_market_private_use_target
	add = te_construction_market_public_use_target
}

te_construction_market_private_share_target = {
	if = {
		limit = {
			te_construction_market_total_use_target > 0
		}
		value = te_construction_market_private_use_target
		divide = te_construction_market_total_use_target
	}
	else = {
		value = 1
	}
	multiply = 100
}

te_construction_market_price_stabilizer_next_value = {
	if = {
		limit = {
			NOT = {
				has_variable = te_construction_market_price_stabilizer
			}
		}
		value = TE_CONSTRUCTION_MARKET_SERVICES_PRICE
	}
	else = {
		if = {
			limit = {
				var:te_construction_market_price_stabilizer < te_construction_market_price
			}
			value = var:te_construction_market_price_stabilizer
			add = {
				value = te_construction_market_price
				subtract = var:te_construction_market_price_stabilizer
				divide = 10
			}
		}
		else = {
			value = var:te_construction_market_price_stabilizer
			subtract = {
				value = var:te_construction_market_price_stabilizer
				subtract = te_construction_market_price
				divide = 10
			}
		}
	}
}
