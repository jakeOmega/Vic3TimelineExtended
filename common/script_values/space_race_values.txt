# Monthly progress value (unified for both approaches)
# Base progress from space program PM, techs, buildings, companies, partnerships, AND funding
# All contribute via country_space_race_progress_add modifier (visible in tooltip breakdown)
# Funding: sr_funding_progress modifier on each JE (0.5 per funding level, stacks across JEs)
# Approach: ambitious gives country_space_race_progress_mult = 0.5 per JE
sr_progress = {
	value = modifier:country_space_race_progress_add
	# Approach multiplier (ambitious JEs contribute +50% each via modifier)
	multiply = {
		value = 1
		add = modifier:country_space_race_progress_mult
	}
	min = 0.5
}

# Failure risk weight for random_list roll
# Base risk set per-milestone via sr_base_risk proxy variable
# Safe approach JEs reduce risk via country_space_race_risk_mult = -0.5 each
sr_failure_weight = {
	value = var:sr_base_risk
	multiply = {
		value = 1
		add = modifier:country_space_race_risk_mult
	}
	min = 1
	max = 50
}

# Success weight = 100 - failure weight
sr_success_weight = {
	value = 100
	subtract = sr_failure_weight
	min = 50
}

sr_current_milestone_progress = {
	value = 0
	if = {
		limit = { has_variable = sr_milestone_progress }
		add = var:sr_milestone_progress
	}
}

# Normalized 0-100 percentage for progress bar display
# goal_add_value is locked at JE activation, so dynamic goals break the bar
# This SV computes (progress / current_dynamic_goal) * 100
# Returns 100 when colonization is fully complete
sr_progress_solar_colonization_percent = {
	value = 0
	if = {
		limit = { has_variable = sr_completed_solar_colonization }
		add = 100
	}
	else = {
		if = {
			limit = { has_variable = sr_progress_solar_colonization }
			add = var:sr_progress_solar_colonization
		}
		divide = {
			value = sr_solar_colonization_goal
			min = 1
		}
		multiply = 100
		min = 0
		max = 100
	}
}
