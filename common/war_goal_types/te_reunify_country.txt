# ============================================================================
# REUNIFY COUNTRY -- generic same-culture annexation wargoal.
#
# Companion to vanilla `unification` (locked to formables via
# `validate_sole_formation_candidate`) and `annex_country` (no thematic
# gate). Selectable on any country pair where:
#   - the unifier shares a primary culture with the target
#   - the unifier holds claims on at least 2/3 of the target's home states
#     (irredentist_pressure_score >= 8 in our value file)
#   - any ideology driver is active (tilt-aware AI bias; not a hard lock)
#   - the unifier's bloc leader has approved (or the unifier IS the leader
#     / has no bloc)
#
# Vanilla's infamy formula already discounts primary-culture-homeland
# states (vanilla_diplomacy_reference.md §5.3) -- we don't double-tax by
# halving on top. The wargoal inherits annex_country's infamy structure
# and lets the engine apply the cultural discount.
#
# See /home/jakef/.claude/plans/can-we-make-a-sparkling-pine.md.
# ============================================================================

te_reunify_country = {
	icon = "gfx/interface/icons/war_goals/unification.dds"

	kind = annex_country

	settings = {
		require_target_be_part_of_war
		annexes_entire_country
		skip_build_list

		# Validation
		validate_conflicts_war_goals_all
		validate_conflicts_annex_country
	}

	execution_priority = 100

	contestion_type = control_target_country_capital

	target_type = country

	possible = {
		# scopes: root = holder, creator_country, diplomatic_play, target_country, target_state, stakeholder, target_region, article_options
		scope:creator_country = {
			is_irredentist_candidate_for = yes
			irredentist_unifier_can_act = yes
			irredentist_bloc_check_war = yes
			irredentist_pressure_score >= 8
		}

		# Don't surface against the formables that have their own
		# unification plays -- vanilla `unification` is the right tool there.
		NOR = {
			scope:target_country = { has_journal_entry = je_reunify_china }
			scope:target_country = { has_journal_entry = je_german_unification }
			scope:target_country = { has_journal_entry = je_risorgimento }
		}
	}

	valid = {
		# in addition to the basic engine validation
	}

	maneuvers = {
		value = 0

		add = {
			desc = "MANEUVERS_BASE_VALUE"
			value = 10
		}

		scope:target_country = {
			multiply = {
				desc = "MANEUVERS_TARGET_COUNTRY_POPULATION_FACTOR"
				value = total_population
				divide = define:NDiplomacy|SWAY_OFFER_WARGOAL_MANEUVERS_COST_POPULATION_SCALING_FACTOR
				multiply = define:NDiplomacy|SWAY_OFFER_WARGOAL_MANEUVERS_COST_POPULATION_SCALING_MULTIPLIER
				max = define:NDiplomacy|SWAY_OFFER_WARGOAL_MANEUVERS_COST_POPULATION_SCALING_MULTIPLIER_MAX_TOTAL
			}
		}
	}

	infamy = {
		value = 0

		add = {
			desc = "INFAMY_BASE_VALUE"
			value = 5
		}

		scope:target_country = {
			multiply = {
				desc = "INFAMY_TARGET_COUNTRY_POPULATION_FACTOR"
				add = {
					every_scope_state = {
						add = {
							value = state_population
							divide = define:NDiplomacy|WAR_GOAL_INFAMY_POPULATION_SCALING_FACTOR
							multiply = define:NDiplomacy|WAR_GOAL_INFAMY_POPULATION_SCALING_MULTIPLIER
							multiply = {
								value = 1
								if = {
									limit = { is_incorporated = no }
									add = define:NDiplomacy|WAR_GOAL_INFAMY_UNINCORPORATED_FACTOR
								}
							}
							max = define:NDiplomacy|WAR_GOAL_INFAMY_POPULATION_SCALING_MULTIPLIER_MAX_PER_STATE
						}
					}
				}
				add = 1
				max = define:NDiplomacy|WAR_GOAL_INFAMY_POPULATION_SCALING_MULTIPLIER_MAX_TOTAL
			}
		}

		# Mirror the unrecognized-target multiplier from vanilla annex_country.
		multiply = {
			desc = "country_infamy_generation_against_unrecognized_mult"
			value = 1
			if = {
				limit = {
					NOR = {
						is_country_type = unrecognized
						is_country_type = decentralized
					}
					scope:target_country = {
						OR = {
							is_country_type = unrecognized
							is_country_type = decentralized
						}
					}
				}
				add = modifier:country_infamy_generation_against_unrecognized_mult
			}
		}

		# Interest-tier multiplier as in vanilla annex_country -- a hegemonic
		# unifier in the target's region pays less infamy.
		multiply = {
			desc = "concept_interest_tier"
			value = 1

			if = {
				limit = {
					NOT = {
						has_strategic_region_interest_tier = {
							strategic_region = scope:target_region
							value >= 2
						}
					}
				}
				add = interest_tier_effect_infamy_cost_increase_high_value
			}

			if = {
				limit = {
					has_strategic_region_interest_tier = {
						strategic_region = scope:target_region
						value >= 2
					}
					NOT = {
						has_strategic_region_interest_tier = {
							strategic_region = scope:target_region
							value >= 3
						}
					}
				}
				add = interest_tier_effect_infamy_cost_increase_medium_value
			}

			if = {
				limit = {
					has_strategic_region_interest_tier = {
						strategic_region = scope:target_region
						value >= 5
					}
				}
				add = interest_tier_effect_infamy_cost_decrease_medium_value
			}
		}

		min = {
			desc = "INFAMY_MINIMUM_VALUE"
			value = 5
		}
	}

	on_enforced = {
		# Engine handles annexation per `kind = annex_country`. We follow up
		# with a third-party-interest notification: every country with
		# Engaged-or-higher interest in a strategic region the absorbed
		# country was in gets irredentism.6 with the warlike flavor.
		#
		# scopes here: root = holder, creator_country, target_country
		scope:creator_country = {
			save_scope_as = irr_unifier
		}
		scope:target_country ?= {
			save_scope_as = irr_annexed_country
		}
		te_irredentism_notify_third_parties_effect = { PEACEFUL = no }
	}

	ai = {
		is_significant_demand = yes
	}
}
