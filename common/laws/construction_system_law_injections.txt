# ====================================================================================
# CONSTRUCTION SYSTEM - Economic Law Allocation Overrides
# ====================================================================================
# Injects country_private_construction_allocation_mult into each economic law
# to counteract vanilla law effects and replace them with the dynamic system.
# These values cancel out vanilla allocation effects so the dynamic ratio controls everything.

INJECT:law_traditionalism = {
	modifier = {
		country_private_construction_allocation_mult = -0.25
	}
}

INJECT:law_interventionism = {
	modifier = {
		country_private_construction_allocation_mult = -0.50
		country_building_airport_require_subsidies_bool = yes
		country_building_highway_require_subsidies_bool = yes
	}
}
