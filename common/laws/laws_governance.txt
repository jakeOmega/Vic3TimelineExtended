law_neocameralism = {
	group = lawgroup_governance_principles
	icon = "gfx/interface/icons/law_icons/laissez_faire.dds"

	progressiveness = 0

	unlocking_laws = {
		law_laissez_faire
		law_interventionism
	}

	unlocking_technologies = {
		marketing_research
	}

	can_enact = {
		NOT = { has_government_type = gov_chartered_company }
	}

	can_impose = {
		OR = {
			AND = {
				is_in_same_power_bloc = scope:target_country
				AND = {
					modifier:country_can_impose_same_lawgroup_governance_principles_in_power_bloc_bool = yes
					has_law = scope:law
				}
			}
			can_impose_law_default = yes
		}
	}

	modifier = {
		country_legitimacy_headofstate_add = 20
		country_legitimacy_ideological_incoherence_mult = 0.1
		interest_group_ig_industrialists_pol_str_mult = 0.25
		state_political_strength_from_wealth_mult = 1.0
		country_company_throughput_bonus_add = 0.1
		building_minimum_wage_mult = -0.1
	}

	ai_will_do = {
		exists = ruler
		ruler = {
			has_ideology = ideology:ideology_corporate
		}
	}

	ai_enact_weight_modifier = {
		value = 0

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
	}

	ai_impose_chance = {
		value = 0

		if = {
			limit = {
				has_law = law_type:law_neocameralism
				has_strategy = ai_strategy_reactionary_agenda
			}

			add = base_impose_law_weight
			if = {
				limit = {
					scope:target_country = { has_law = law_type:law_council_republic }
					NOT = { has_strategy = ai_strategy_egalitarian_agenda }
				}
					multiply = 5
			}
			else = {
				multiply = 0.5
			}
		}
	}
}
