nuke_diplo_action = {
    groups= {
		general
	}
	requires_approval = no

	state_selection = second_required
    second_state_list = second_country
	
	show_confirmation_box = yes

	possible = {
        has_variable = nuclear_weapon_stockpile
        var:nuclear_weapon_stockpile > 0
    }

	second_state_trigger = {
        owner = {
            has_war_with = scope:country
        }
    }
    
	accept_effect = {
        change_infamy = 25
        change_variable = {
            name = nuclear_weapon_stockpile
            subtract = 1
        }
        scope:second_state = {
            state_region = {
                add_devastation = 50
            }
            kill_population_percent_in_state = {
                percent = 0.05
            }
            add_modifier = {
                name = nuclear_strike_aftermath
                months = short_modifier_time
                is_decaying = yes
            }
        }
        scope:target_country = {
            every_scope_war = {
                add_war_war_support = {
                    target = PREV
                    value = -25
                }
            }
        }
    }

	is_hostile = yes
}
