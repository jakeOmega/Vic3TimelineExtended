nuke_diplo_action = {
    groups= {
		general
	}
	requires_approval = no

	state_selection = second_required
    second_state_list = second_country
	
	show_confirmation_box = yes

    potential = {
        has_modifier = nuclear_power
    }

	possible = {
        var:nuclear_weapon_stockpile > 0
        OR = {
            NOT = {
                OR = {
                    has_law = law_type:law_humanitarian_regulations
                    has_law = law_type:law_limited_war
                }
            }
            any_scope_state = {
                OR = {
                    has_modifier = nuclear_strike_aftermath
                    has_modifier = tactical_nuke_aftermath
                }
            }
        }
    }

	second_state_trigger = {
        owner = {
            has_war_with = scope:country
        }
    }
    
	accept_effect = {
        save_scope_as = attacking_country
        scope:second_state = {
            save_scope_as = target_state
            nuclear_first_stike = yes
        }
    }

	is_hostile = yes

    ai = {
		evaluation_chance = {
            value = 0.05 # base 5% chance to evaluate — nuclear use is a grave decision
            if = {
                limit = {
                    ruler = {
                        OR = {
                            has_trait = reckless
                            has_trait = wrathful
                            has_trait = war_criminal
                        }
                    }
                }
                multiply = 3
            }
            if = {
                limit = {
                    ruler = { has_trait = ruthless }
                    NOT = { ruler = { has_trait = cautious } }
                }
                multiply = 2
            }
            if = {
                limit = {
                    ruler = {
                        OR = {
                            has_trait = cautious
                            has_trait = honorable
                        }
                    }
                }
                multiply = 0 # cautious or honorable leaders never initiate nuclear strikes
            }
            if = {
                limit = {
                    any_scope_state = {
                        has_modifier = nuclear_strike_aftermath
                    }
                }
                multiply = 5 # retaliation after being nuked is strongly favored
            }
            if = {
                limit = {
                    scope:target_country = { has_modifier = nuclear_power }
                }
                multiply = 0.1 # extreme reluctance against nuclear-armed targets (MAD)
            }
        }

		will_select_as_second_state = {
            always = yes
        }
	
		will_propose = {
            has_law = law_type:law_total_war
            has_diplomatic_pact = {
                type = rivalry
                who = scope:target_country
            }
            NOT = { infamy > 75 }
            # Must have incorporated territory threatened by enemy wargoals
            any_scope_state = {
                is_target_of_wargoal = scope:target_country
                is_incorporated = yes
            }
            OR = {
                # Case 1: Target is NOT nuclear-armed — normal first strike logic
                NOT = {
                    scope:target_country = {
                        has_modifier = nuclear_power
                    }
                }
                # Case 2: Target IS nuclear-armed — requires BOTH losing core territory AND radical leader
                AND = {
                    scope:target_country = { has_modifier = nuclear_power }
                    # Must have core territory actually occupied or nearly lost
                    any_scope_state = {
                        is_incorporated = yes
                        is_on_front = yes
                    }
                    # AND leader must be aggressive/unstable
                    OR = {
                        ruler = {
                            has_trait = reckless
                            NOT = { has_trait = honorable }
                        }
                        ruler = {
                            has_trait = wrathful
                            NOT = { has_trait = honorable }
                        }
                        ruler = {
                            has_trait = war_criminal
                        }
                    }
                }
                # Case 3: Already been nuked — retaliation regardless of enemy's nuclear status
                any_scope_state = {
                    OR = {
                        has_modifier = nuclear_strike_aftermath
                        has_modifier = tactical_nuke_aftermath
                    }
                }
            }
        }

		propose_score = {
            scope:second_state = {
                value = nuclear_industrial_strike_target_score
            }
        }
	}
}


tactical_nuke_diplo_action = {
    groups= {
		general
	}
	requires_approval = no

	state_selection = second_required
    second_state_list = second_country
	
	show_confirmation_box = yes

    potential = {
        has_modifier = nuclear_power
    }

	possible = {
        has_technology_researched = tactical_nuclear_weapons
        has_variable = nuclear_weapon_stockpile
        var:nuclear_weapon_stockpile > 0
        OR = {
            NOT = {
                has_law = law_type:law_limited_war
            }
            any_scope_state = {
                OR = {
                    has_modifier = nuclear_strike_aftermath
                    has_modifier = tactical_nuke_aftermath
                }
            }
        }
    }

	second_state_trigger = {
        owner = {
            has_war_with = scope:country
            has_building = building_barrack
        }
    }
    
	accept_effect = {
        save_scope_as = attacking_country
        change_variable = {
            name = nuclear_weapon_stockpile
            subtract = 1
        }
        scope:second_state = {
            save_scope_as = target_state
        }
        random_list = {
            0 = {
                modifier = nuclear_strike_success_chance
                change_infamy = 10
                scope:second_state = {
                    nuclear_tactical_strike = yes
                }
            }
            0 = {
                modifier = nuclear_strike_fail_chance
                scope:target_country = {
                    trigger_event = {id = nuclear_weapon_events.7 popup = yes }
                    change_relations = {
                        country = ROOT
                        value = -5
                    }
                }
                trigger_event = {id = nuclear_weapon_events.8 popup = yes }
            }
        }
    }

	is_hostile = yes

    ai = {
		evaluation_chance = {
            value = 0.5 # tactical nukes considered more readily than strategic, but still not routine
            if = {
                limit = {
                    ruler = {
                        OR = {
                            has_trait = reckless
                            has_trait = wrathful
                            has_trait = war_criminal
                        }
                    }
                }
                multiply = 2
            }
            if = {
                limit = {
                    ruler = {
                        OR = {
                            has_trait = cautious
                            has_trait = honorable
                        }
                    }
                }
                multiply = 0.25 # cautious/honorable leaders very reluctant even for tactical use
            }
            if = {
                limit = {
                    scope:target_country = { has_modifier = nuclear_power }
                }
                multiply = 0.1 # extreme reluctance against nuclear-armed targets
            }
            if = {
                limit = {
                    any_scope_state = {
                        OR = {
                            has_modifier = nuclear_strike_aftermath
                            has_modifier = tactical_nuke_aftermath
                        }
                    }
                }
                multiply = 3 # retaliation after being nuked
            }
        }

		will_select_as_second_state = {
            OR = {
                has_building = building_barrack
                has_building = building_naval_base
                has_building = building_conscription_center
            }
        }

        will_propose = {
            has_law = law_type:law_total_war
            has_diplomatic_pact = {
                type = rivalry
                who = scope:target_country
            }
            NOT = { infamy > 75 }
            any_scope_state = {
                is_target_of_wargoal = scope:target_country
                is_incorporated = yes
            }
            OR = {
                # Case 1: Target is NOT nuclear-armed — use tactical nukes more freely
                NOT = {
                    scope:target_country = { has_modifier = nuclear_power }
                }
                # Case 2: Target IS nuclear-armed — require occupied core territory AND aggressive leader
                AND = {
                    scope:target_country = { has_modifier = nuclear_power }
                    any_scope_state = {
                        is_incorporated = yes
                        is_on_front = yes
                    }
                    OR = {
                        ruler = {
                            has_trait = reckless
                            NOT = { has_trait = honorable }
                        }
                        ruler = {
                            has_trait = wrathful
                            NOT = { has_trait = honorable }
                        }
                        ruler = { has_trait = war_criminal }
                        ruler = { has_trait = ruthless NOT = { has_trait = cautious } }
                    }
                }
                # Case 3: Already been nuked — retaliation
                any_scope_state = {
                    OR = {
                        has_modifier = nuclear_strike_aftermath
                        has_modifier = tactical_nuke_aftermath
                    }
                }
            }
        }
	
		propose_score = {
            value = 0
            scope:second_state = {
                if = {
                    limit = {
                        has_building = building_barrack
                    }
                    add = {
                        value = b:building_barrack.level
                        divide = 100
                    }
                }
                if = {
                    limit = {
                        has_building = building_naval_base
                    }
                    add = {
                        value = b:building_naval_base.level
                        divide = 100
                    }
                }
                if = {
                    limit = {
                        has_building = building_conscription_center
                    }
                    add = {
                        value = b:building_conscription_center.level
                        divide = 100
                    }
                }
                if = {
                    limit = { owner = { has_modifier = nuclear_power }}
                    divide = 1000 # near-zero priority against nuclear-armed targets
                }
            }
        }
	}
}
