# ============================================================================
# COVERT WARFARE — Covert Operation Diplomatic Actions
# ============================================================================
# These diplomatic actions allow a country to launch covert operations
# against a target country. The operations persist as tracked variables and
# their effects are applied via the JE monthly pulse.
# ============================================================================

# ---- PEACETIME: Election Interference ----
# Undermine a rival's political stability through covert influence campaigns
covert_election_interference_action = {
	groups = {
		general
	}
	requires_approval = no
	show_confirmation_box = yes
	is_hostile = yes
	should_notify_third_parties = no

	potential = {
		has_game_rule = covert_warfare_enabled
		has_technology_researched = television
		NOT = { has_variable = iw_op_election_interference }
	}

	possible = {
		custom_tooltip = {
			text = iw_has_available_slot_tt
			covert_operations_available_slots >= 1
		}
		custom_tooltip = {
			text = iw_not_at_war_tt
			is_at_war = no
		}
		has_diplomatic_pact = { who = scope:target_country type = rivalry }
		custom_tooltip = {
			text = iw_funding_not_zero_tt
			has_variable = iw_funding_level
			var:iw_funding_level >= 1
		}
	}

	accept_effect = {
		set_variable = iw_op_election_interference
		set_variable = { name = iw_target_election_interference value = scope:target_country }
		if = {
			limit = { NOT = { has_variable = iw_active_operations } }
			set_variable = { name = iw_active_operations value = 0 }
		}
		change_variable = { name = iw_active_operations add = 1 }
		post_notification = iw_operation_launched
	}

	ai = {
		evaluation_chance = {
			value = 0.1
		}

		will_propose = {
			has_diplomatic_pact = { who = scope:target_country type = rivalry }
			covert_operations_available_slots >= 1
			is_at_war = no
			has_variable = iw_funding_level
			var:iw_funding_level >= 1
		}

		propose_score = {
			value = 10
		}
	}
}
