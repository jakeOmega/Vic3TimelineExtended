			modifier:power_bloc_leader_can_peacefully_annex_bool = yes
		}
	}

	potential = {
		scope:target_country = {
			is_direct_subject_of = root
		}
		is_revolutionary = no
	}

	possible = {
		scope:target_country = {
			NOT = { has_truce_with = ROOT }
		}
		custom_tooltip = {
			text = DIPLO_ACTION_GDP_REQUIREMENT
			scope:target_country = {
				is_direct_subject_of = root
				OR = {
					gdp < peaceful_annex_gdp_requirement
					peaceful_annex_gdp_requirement < 0
				}
			}
		}
		scope:target_country = {
			OR = {
				has_diplomatic_pact = { who = root type = puppet }
				has_diplomatic_pact = { who = root type = colony }

			liberty_desire < 25
		}
		custom_tooltip = {
			text = DIPLO_ACTION_RELATIONS_CORDIAL_OR_GREATER
			scope:target_country.relations:root >= relations_threshold:cordial
		}

		scope:target_country = { is_diplomatic_play_committed_participant = no }

	accept_effect = {
		annex = scope:target_country
	}

	is_hostile = no

	ai = {

			value = 1
		}

		evaluation_chance = {
			value = 100
		}
	}

colonial_culture_change = {
	groups= {
		overlord
	}
	show_in_lens = no

	texture = "gfx/interface/icons/diplomatic_action_icons/force_state_religion.dds"

	selectable = {
		always = yes
	}

	potential = {
		scope:target_country = {
			is_direct_subject_of = ROOT
			OR = {
				has_government_type = gov_chartered_company
				has_government_type = gov_colonial_administration
			}
		}
	}

	possible = {
		OR = {
			scope:target_country = {
				any_primary_culture = {
					NOT = { is_primary_culture_of = ROOT }
				}
			}
			any_primary_culture = {
				NOT = { is_primary_culture_of = scope:target_country }
			}
		}

		prestige >= {
			value = scope:target_country.prestige
			multiply = 3
		}
	}

	accept_effect = {
		every_primary_culture = {
			limit = {
				NOT = { is_primary_culture_of = scope:target_country }
			}
			scope:target_country = {
				add_primary_culture = PREV
			}
		}

		scope:target_country = {
			every_primary_culture = {
				limit = {
					NOT = { is_primary_culture_of = ROOT }
				}
				scope:target_country = {
					remove_primary_culture = PREV
				}
			}
		}

		if = {
			limit = {
				NOT = { scope:target_country.religion = ROOT.religion }
			}
			scope:target_country = { set_state_religion = ROOT.religion }
		}

		scope:target_country = { add_liberty_desire = 20 }
	}

	ai = {
		evaluation_chance = {
			# scope:target_country is unbound in evaluation_chance (proposer scope only);
			# per-target attitude gating belongs in will_propose / propose_score instead.
			value = 0.25
			if = {
				limit = { country_rank >= rank_value:great_power }
				add = 0.25
			}
		}

		will_propose = {
			scope:target_country = {
				is_direct_subject_of = ROOT
				liberty_desire < 50
			}
			NOT = { # Don't break Canadian unification
				scope:target_country = { c:HBC ?= THIS }
				ROOT = {c:GBR ?= THIS }
			}
			OR = {
				has_attitude = { who = scope:target_country attitude = domineering }
				has_attitude = { who = scope:target_country attitude = protective }
			}
		}

		propose_score = {
			value = 10
			if = {
				limit = { prestige >= { value = scope:target_country.prestige multiply = 5 } }
				add = 10
			}
			if = {
				limit = {
					has_attitude = { who = scope:target_country attitude = domineering }
				}
				add = 15
			}
			if = {
				limit = { scope:target_country = { liberty_desire < 25 } }
				add = 10
			}
		}
	}
}

diplomatic_alignment = {
	groups= {
		general
	}

	requires_approval = no
	show_confirmation_box = no

	selectable = {
		NOT = { is_country_type = decentralized }
		is_power_bloc_leader = yes
		power_bloc ?= {
			modifier:power_bloc_can_use_diplomatic_alignment_bool = yes
		}
	}

	potential = {
		scope:target_country = {
			NOT = { is_country_type = decentralized }
		}
	}

	possible = {
		always = yes
	}

	pact = {
		cost = 200
		is_two_sided_pact = no
		show_in_outliner = yes

		monthly_effect = {
			every_diplomatically_relevant_country = {
				limit = {
					NOT = { THIS = scope:target_country }
					scope:target_country = {
						has_diplomatic_relevance = PREV
						NOT = {
							scope:target_country.relations:PREV = ROOT.relations:PREV
						}
					}
				}
				save_scope_as = third_part_country
				scope:target_country = {
					if = {
						limit = {
							relations:PREV > ROOT.relations:PREV
						}
						change_relations = {
							country = PREV
							value = -1
						}
					}
					else = {
						change_relations = {
							country = PREV
							value = 1
						}
					}
				}
			}
		}

		requirement_to_maintain = {
			trigger = {
				has_diplomatic_relevance = scope:target_country
			}
		}

		requirement_to_maintain = {
			trigger = {
				custom_tooltip = {
					text = "DIPLOMATIC_PACT_REQUIREMENT_AT_PEACE"
					NOT = { has_war_with = scope:target_country }
				}
			}
		}

		requirement_to_maintain = {
			trigger = {
				custom_tooltip = {
					text = "DIPLOMATIC_PACT_REQUIREMENT_NOT_DECENTRALIZED"
					NOT = { is_country_type = decentralized }
					scope:target_country = {
						NOT = { is_country_type = decentralized }
					}
				}
			}
		}

		requirement_to_maintain = {
			trigger = {
				NOT = { has_diplomats_expelled = scope:target_country }
			}
		}

		requirement_to_maintain = {
			trigger = {
				NOT = { scope:target_country = { has_diplomats_expelled = root } }
			}
		}
	}

	ai = {
		max_influence_spending_fraction = 0.25

		evaluation_chance = {
			value = 0.25
		}

		propose_score = {
			value = 5
		}

		will_propose = {
			OR = {
				has_attitude = {
					who = scope:target_country
					attitude = cooperative
				}
				has_attitude = {
					who = scope:target_country
					attitude = conciliatory
				}
				has_attitude = {
					who = scope:target_country
					attitude = genial
				}
				has_attitude = {
					who = scope:target_country
					attitude = protective
				}
			}
		}

		will_break = {
			NOR = {
				has_attitude = {
					who = scope:target_country
					attitude = genial
				}
				has_attitude = {
					who = scope:target_country
					attitude = conciliatory
				}
				has_attitude = {
					who = scope:target_country
					attitude = cooperative
				}
				has_attitude = {

# ============================================================================
# VOLUNTARY UNION
#
# Peaceful counterpart to the te_reunify_country wargoal: a same-culture
# country with high relations and a strong claims footprint can propose a
# voluntary merger to a smaller / weaker same-culture neighbor. Engine
# annexes on accept (per `accept_effect = { annex = scope:target_country }`,
# same primitive vanilla annex_subject_peaceful uses).
#
# Tilt-aware: any active driver can use the action, but AI propose_score
# and target accept_score lean heavily on irredentist_war_tilt --
# warlike-tilt unifiers rarely propose, peaceful-tilt targets are more
# willing to accept. Anschluss-style outcomes (warlike unifier with
# extremely amicable target) remain reachable but rare.
#
# Bloc gate: voluntary_union_eligible already requires both sides have an
# active driver and the target is not a subject; irredentist_bloc_check_peace
# layered on top requires both bloc leaders' approval (Phase 5).
#
# See /home/jakef/.claude/plans/can-we-make-a-sparkling-pine.md.
# ============================================================================
voluntary_union = {
	groups = { general }
	requires_approval = yes
	show_confirmation_box = yes
	is_hostile = no
	show_in_lens = no   # avoid auto-lookup of gfx/interface/icons/lens_toolbar_icons/voluntary_union.dds

	texture = "gfx/interface/icons/diplomatic_action_icons/force_become_subject.dds"

	selectable = {
		# Activator gate intentionally absent -- voluntary union is open to
		# any sovereign nation post-decolonization. AI scoring (tilt-driven
		# propose_score / accept_score) decides whether anyone actually uses it.
		has_technology_researched = decolonization
	}

	potential = {
		scope:target_country = {
			NOT = { is_country_type = decentralized }
		}
	}

	possible = {
		is_irredentist_candidate_for = yes
		voluntary_union_eligible = yes
		irredentist_bloc_check_peace = yes
		is_diplomatic_play_committed_participant = no
		scope:target_country = {
			is_diplomatic_play_committed_participant = no
			NOT = { has_modifier = irr_rejected_union_modifier }
		}
		NOT = { has_truce_with = scope:target_country }
	}

	accept_effect = {
		# Notify interested third parties BEFORE the annex -- the scripted
		# effect needs scope:target_country (still valid here) to walk its
		# strategic regions. Set up the scope aliases the effect expects.
		save_scope_as = irr_unifier
		scope:target_country = {
			save_scope_as = irr_annexed_country
		}
		te_irredentism_notify_third_parties_effect = { PEACEFUL = yes }
		annex = scope:target_country
	}

	ai = {
		# In `accept_score`: ROOT = recipient (the country being asked to dissolve
		# itself); `scope:actor` = proposer. In `propose_score`: ROOT = proposer;
		# `scope:target_country` = recipient.
		#
		# Vanilla convention (see common/diplomatic_actions/04_trade_states.txt):
		# every `add` block carries a `desc =` so the in-game tooltip shows the
		# full breakdown. Without `desc`, the term is invisible -- a +40 final
		# number with no breakdown is the worst possible UX.

		evaluation_chance = {
			# scope:target_country is unbound here (evaluation_chance runs in proposer scope only);
			# per-target relations gating belongs in propose_score, which already does it via the
			# relations bands below.
			value = 0.05
		}

		accept_score = {
			value = 0

			# Voluntary annexation = the country ceasing to exist. The default
			# stance is a strong NO; positive factors must accumulate to flip it.
			add = {
				desc = "VOLUNTARY_UNION_ACCEPT_BASELINE"
				value = -60
			}

			# === Hard nopes ===

			if = {
				limit = {
					has_diplomatic_pact = { who = scope:actor type = rivalry }
				}
				add = {
					desc = "VOLUNTARY_UNION_ACCEPT_RIVALRY"
					value = -200
				}
			}

			# === Relations bands (only the matched band fires) ===

			if = {
				limit = { scope:actor.relations:root >= relations_threshold:friendly }
				add = {
					desc = "VOLUNTARY_UNION_ACCEPT_REL_FRIENDLY"
					value = 70
				}
			}
			else_if = {
				limit = { scope:actor.relations:root >= relations_threshold:amicable }
				add = {
					desc = "VOLUNTARY_UNION_ACCEPT_REL_AMICABLE"
					value = 45
				}
			}
			else_if = {
				limit = { scope:actor.relations:root >= relations_threshold:cordial }
				add = {
					desc = "VOLUNTARY_UNION_ACCEPT_REL_CORDIAL"
					value = 20
				}
			}
			else_if = {
				limit = { scope:actor.relations:root <= relations_threshold:hostile }
				add = {
					desc = "VOLUNTARY_UNION_ACCEPT_REL_HOSTILE"
					value = -150
				}
			}
			else_if = {
				limit = { scope:actor.relations:root <= relations_threshold:cold }
				add = {
					desc = "VOLUNTARY_UNION_ACCEPT_REL_COLD"
					value = -75
				}
			}
			else_if = {
				limit = { scope:actor.relations:root <= relations_threshold:poor }
				add = {
					desc = "VOLUNTARY_UNION_ACCEPT_REL_POOR"
					value = -40
				}
			}

			# === Attitudes (AI-only; mutually exclusive) ===

			if = {
				limit = { has_attitude = { who = scope:actor attitude = belligerent } }
				add = {
					desc = "VOLUNTARY_UNION_ACCEPT_ATT_BELLIGERENT"
					value = -100
				}
			}
			else_if = {
				limit = { has_attitude = { who = scope:actor attitude = antagonistic } }
				add = {
					desc = "VOLUNTARY_UNION_ACCEPT_ATT_ANTAGONISTIC"
					value = -75
				}
			}
			else_if = {
				limit = { has_attitude = { who = scope:actor attitude = wary } }
				add = {
					desc = "VOLUNTARY_UNION_ACCEPT_ATT_WARY"
					value = -10
				}
			}
			else_if = {
				limit = { has_attitude = { who = scope:actor attitude = cooperative } }
				add = {
					desc = "VOLUNTARY_UNION_ACCEPT_ATT_COOPERATIVE"
					value = 35
				}
			}
			else_if = {
				limit = { has_attitude = { who = scope:actor attitude = genial } }
				add = {
					desc = "VOLUNTARY_UNION_ACCEPT_ATT_GENIAL"
					value = 50
				}
			}
			else_if = {
				limit = { has_attitude = { who = scope:actor attitude = protective } }
				add = {
					desc = "VOLUNTARY_UNION_ACCEPT_ATT_PROTECTIVE"
					value = 45
				}
			}

			# === Recipient's own ideological tilt ===
			# (Tilt as a script value reads scope:target_country which equals
			# ROOT here; the laws/IG checks evaluate correctly on ROOT.)

			if = {
				limit = {
					any_interest_group = {
						is_in_government = yes
						OR = {
							leader = { has_ideology = ideology:ideology_pacifist }
							leader = { has_ideology = ideology:ideology_humanitarian }
							leader = { has_ideology = ideology:ideology_anarchist }
						}
					}
				}
				add = {
					desc = "VOLUNTARY_UNION_ACCEPT_PEACEFUL_IGS"
					value = 25
				}
			}
			if = {
				limit = {
					any_interest_group = {
						is_in_government = yes
						OR = {
							leader = { has_ideology = ideology:ideology_jingoist_leader }
							leader = { has_ideology = ideology:ideology_fascist }
							leader = { has_ideology = ideology:ideology_ethno_nationalist }
						}
					}
				}
				add = {
					desc = "VOLUNTARY_UNION_ACCEPT_NATIONALIST_IGS"
					value = -30
				}
			}
			if = {
				limit = {
					ig:ig_armed_forces ?= { is_in_government = yes }
				}
				add = {
					desc = "VOLUNTARY_UNION_ACCEPT_ARMED_FORCES_OPPOSE"
					value = -15
				}
			}

			# === Bloc-context boosts ===

			if = {
				limit = {
					is_in_same_power_bloc = scope:actor
					scope:actor = { is_power_bloc_leader = yes }
				}
				add = {
					desc = "VOLUNTARY_UNION_ACCEPT_OUR_BLOC_LEADER"
					value = 25
				}
			}
			else_if = {
				limit = {
					is_in_same_power_bloc = scope:actor
					power_bloc ?= { has_identity = identity:identity_diplomatic }
				}
				add = {
					desc = "VOLUNTARY_UNION_ACCEPT_DIPLOMATIC_FRAMEWORK"
					value = 15
				}
			}

			# === Prior signals ===

			# Recipient said "open to it" via Event 4A -- they already nodded yes.
			if = {
				limit = { has_modifier = irr_open_to_union_modifier }
				add = {
					desc = "VOLUNTARY_UNION_ACCEPT_PRIOR_OPENNESS"
					value = 50
				}
			}
			# Standing cultural-ties bond from Event 1B.
			if = {
				limit = { has_modifier = irr_cultural_ties_modifier }
				add = {
					desc = "VOLUNTARY_UNION_ACCEPT_CULTURAL_TIES"
					value = 20
				}
			}

			# === Protection appeal: a small kin-state joining a powerful protector ===
			# voluntary_union_eligible already requires the proposer outrank the
			# target; this rewards a *wide* gap -- the in-fiction draw of dissolving
			# into a strong same-culture great/major power for security. ROOT is the
			# recipient here, so NOT = { country_rank >= rank_value:major_power }
			# means the target itself sits below major-power rank.
			if = {
				limit = {
					scope:actor = { country_rank >= rank_value:great_power }
					NOT = { country_rank >= rank_value:major_power }
				}
				add = {
					desc = "VOLUNTARY_UNION_ACCEPT_PROTECTION_GREAT"
					value = 20
				}
			}
			else_if = {
				limit = {
					scope:actor = { country_rank >= rank_value:major_power }
					NOT = { country_rank >= rank_value:major_power }
				}
				add = {
					desc = "VOLUNTARY_UNION_ACCEPT_PROTECTION_MAJOR"
					value = 10
				}
			}
		}

		will_propose = {
			voluntary_union_eligible = yes
			irredentist_bloc_check_peace = yes
			# Don't propose to a country that fundamentally hates us. AI score
			# would discount it anyway, but spare them the spam.
			NOT = {
				has_diplomatic_pact = { who = scope:target_country type = rivalry }
			}
			scope:target_country.relations:root >= relations_threshold:cordial
		}

		propose_score = {
			value = 0

			# Baseline interest -- the system is designed to be a viable choice
			# for ideologically peaceful unifiers, not a constant background poll.
			add = {
				desc = "VOLUNTARY_UNION_PROPOSE_BASELINE"
				value = 5
			}

			# === Pressure score: stronger candidate = more interest ===
			if = {
				limit = { irredentist_pressure_score >= 6 }
				add = {
					desc = "VOLUNTARY_UNION_PROPOSE_PRESSURE_HIGH"
					value = 5
				}
			}
			if = {
				limit = { irredentist_pressure_score >= 10 }
				add = {
					desc = "VOLUNTARY_UNION_PROPOSE_PRESSURE_MAX"
					value = 5
				}
			}

			# === Tilt: peaceful tilt prefers this path; warlike tilt discounts ===
			if = {
				limit = { irredentist_war_tilt >= 3 }
				add = {
					desc = "VOLUNTARY_UNION_PROPOSE_TILT_WARLIKE"
					value = -8
				}
			}
			else_if = {
				limit = { irredentist_war_tilt >= 1 }
				add = {
					desc = "VOLUNTARY_UNION_PROPOSE_TILT_LEAN_WARLIKE"
					value = -3
				}
			}
			else_if = {
				limit = { irredentist_war_tilt <= -2 }
				add = {
					desc = "VOLUNTARY_UNION_PROPOSE_TILT_PEACEFUL"
					value = 8
				}
			}

			# Target signaled openness via Event 4A -- the moment is now.
			if = {
				limit = {
					scope:target_country = {
						has_modifier = irr_open_to_union_modifier
					}
				}
				add = {
					desc = "VOLUNTARY_UNION_PROPOSE_TARGET_OPEN"
					value = 10
				}
			}

			# Relations softer than amicable -- discount even though
			# voluntary_union_eligible already requires amicable+ to propose.
			if = {
				limit = { scope:target_country.relations:root < relations_threshold:friendly }
				add = {
					desc = "VOLUNTARY_UNION_PROPOSE_RELATIONS_NOT_FRIENDLY"
					value = -3
				}
			}
		}
	}
}
