# Government-flavored dynamic names for the seven new formable countries.
#
# All entries use is_main_tag_only = yes (only the formed tag uses these,
# not subject states with the same tag).
#
# Trigger pattern follows vanilla USA (00_dynamic_country_names.txt:2785):
#   - The default / republic entry guards against communist + technocracy
#     to avoid double-matches against the more specific entries.
#   - Each ideology trigger uses the canonical coa_def_*_flag_trigger.
#
# When no entry's trigger matches, the engine falls back to the tag's
# localization key (e.g. AFU:0 "African Union").
#
# BHT and IDN entries here APPEND to vanilla's. Vanilla BHT has one
# priority-1 "Dominion of India" entry gated on subject status — independent
# BHT falls through to vanilla's tag name; our priority-0 entries fire only
# when independent and a flavor trigger matches. Vanilla IDN has two
# priority-0 monarchy entries (Majapahit, Srivijaya) gated on Javan/Malay/
# Sumatran ruler culture — our entries are mutually exclusive with those by
# trigger conditions (no monarchy entry from us; the vanilla flavor stays).

# === African Union ===

AFU = {
	dynamic_country_name = { # Default / republic / federation
		name = dyn_c_african_union
		adjective = AFU_ADJ
		is_main_tag_only = yes
		priority = 0
		trigger = {
			coa_def_republic_flag_trigger = yes
			NOT = { coa_def_communist_flag_trigger = yes }
			NOT = { coa_def_technocracy_flag_trigger = yes }
		}
	}
	dynamic_country_name = { # Communist
		name = dyn_c_pan_african_socialist_federation
		adjective = AFU_ADJ
		is_main_tag_only = yes
		priority = 0
		trigger = {

		}
	}
	dynamic_country_name = { # Fascist

		is_main_tag_only = yes
		priority = 0
		trigger = {
			coa_def_fascist_flag_trigger = yes
		}
	}
	dynamic_country_name = { # Solomonic flavor — Ethiopian-led monarchy
		name = dyn_c_solomonic_african_empire
		adjective = AFU_ADJ
		is_main_tag_only = yes

		trigger = {
			coa_def_monarchy_flag_trigger = yes

						has_culture = cu:amhara

		}
	}
