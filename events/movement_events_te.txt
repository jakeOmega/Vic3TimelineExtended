# Event 1: Civil Rights March
# A mass march by discriminated communities demanding equal rights.
movement_events_te.1 = {
	type = country_event
	placement = root

	event_image = { texture = "gfx/event_pictures/civil_rights_peaceful_march.dds" }

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	title = movement_events_te.1.t
	desc = movement_events_te.1.d
	flavor = movement_events_te.1.f

	duration = 3
	cooldown = { days = normal_modifier_time }

	trigger = {
		has_active_civil_rights_movement = yes
		has_technology_researched = civil_rights_movement
		any_scope_state = {
			is_incorporated = yes
			any_scope_pop = {
				pop_acceptance < acceptance_status_4
			}
		}
	}

	cancellation_trigger = {
		NOT = {
			has_active_civil_rights_movement = yes
		}
	}

	immediate = {
		save_civil_rights_movement_scope = yes
		random_scope_state = {
			limit = {
				is_incorporated = yes
				any_scope_pop = {
					pop_acceptance < acceptance_status_4
				}
			}
			save_scope_as = march_state
		}
	}

	option = {
		name = movement_events_te.1.a
		default_option = yes
		ai_chance = {
			base = 2
			modifier = {
				trigger = { has_technology_researched = social_justice_movements }
				add = 2
			}
		}
		# Let them march - democratic legitimacy, moderate outcome
		add_modifier = {
			name = civil_rights_march_momentum_modifier
			days = normal_modifier_time
			is_decaying = yes
		}
		scope:civil_rights_movement = {
			add_modifier = {
				name = civil_rights_march_calming
				days = normal_modifier_time
				is_decaying = yes
			}
		}
		add_loyalists = { value = small_radicals strata = lower }
		add_radicals = { value = very_small_radicals strata = upper }
	}

	option = {
		name = movement_events_te.1.b
		ai_chance = {
			base = 1
			modifier = {
				trigger = {
					has_severe_discriminatory_minority_law = yes
				}
				add = 2
			}
		}
		# Disperse the crowds - order maintained, popular anger
		add_modifier = {
			name = civil_rights_suppression_modifier
			days = normal_modifier_time
			is_decaying = yes
		}
		add_radicals = { value = medium_radicals strata = lower }
		add_radicals = { value = small_radicals strata = middle }
		add_loyalists = { value = small_radicals strata = upper }
	}

	option = {
		name = movement_events_te.1.c
		trigger = {
			has_active_civil_rights_movement = yes
		}
		ai_chance = {
			base = 0
			modifier = {
				trigger = { has_technology_researched = social_justice_movements }
				add = 3
			}
		}
		# Government endorses the march - radical political shift
		random_political_movement = {
			limit = { is_political_movement_type = movement_civil_rights }
			add_modifier = {
				name = civil_rights_government_endorsement_modifier
				days = normal_modifier_time
				is_decaying = yes
			}
		}
		ig:ig_intelligentsia ?= {
			leader = {
				set_ideology = ideology:ideology_multicultural
			}
		}
		add_loyalists = { value = medium_radicals strata = lower }
		add_loyalists = { value = small_radicals pop_type = academics }
		add_radicals = { value = medium_radicals strata = upper }
	}
}

# Event 4: International Pressure on Human Rights
# Foreign nations criticize our treatment of minorities, demanding reform.
movement_events_te.4 = {
	type = country_event
	placement = root

	event_image = { texture = "gfx/event_pictures/civil_rights_peaceful_march.dds" }

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	title = movement_events_te.4.t
	desc = movement_events_te.4.d
	flavor = movement_events_te.4.f

	duration = 3
	cooldown = { days = long_modifier_time }

	trigger = {
		has_active_civil_rights_movement = yes
		OR = {
			NOT = { 
				has_progressive_minority_law = yes
			}
			has_law = law_type:law_active_persecution
			has_law = law_type:law_no_womens_rights
		}
		any_country = {
			NOT = { this = root }
			country_rank >= rank_value:great_power
			has_technology_researched = civil_rights_movement
		}
	}

	cancellation_trigger = {
		NOT = {
			has_active_civil_rights_movement = yes
		}
	}

	immediate = {
		random_country = {
			limit = {
				NOT = { this = root }
				country_rank >= rank_value:great_power
				has_technology_researched = civil_rights_movement
			}
			save_scope_as = criticizing_country
		}
		save_civil_rights_movement_scope = yes
	}

	option = {
		name = movement_events_te.4.a
		default_option = yes
		ai_chance = {
			base = 2
			modifier = {
				trigger = { country_rank >= rank_value:great_power }
				add = 2
			}
		}
		# Reject foreign interference - sovereignty preserved, reputation damaged
		change_relations = {
			country = scope:criticizing_country
			value = -30
		}
		create_diplomatic_catalyst = { type = catalyst_event_negative target = scope:criticizing_country }
		add_lobby_appeasement_from_diplomacy_unidirectional = {
			FIRST = root
			SECOND = scope:criticizing_country
			PRO_AMOUNT = 1
			ANTI_AMOUNT = -1
			FACTOR = appeasement_relations_decreased
		}
		add_radicals = { value = small_radicals pop_type = academics }
		add_loyalists = { value = small_radicals strata = upper }
		scope:civil_rights_movement = {
			add_modifier = {
				name = international_criticism_movement_radicalism
				days = normal_modifier_time
				is_decaying = yes
			}
		}
	}

	option = {
		name = movement_events_te.4.b
		ai_chance = {
			base = 1
			modifier = {
				trigger = { has_technology_researched = social_justice_movements }
				add = 3
			}
		}
		# Promise reforms - diplomatic goodwill, domestic nationalist backlash
		add_modifier = {
			name = human_rights_reform_pledge_modifier
			days = normal_modifier_time
			is_decaying = yes
		}
		change_relations = {
			country = scope:criticizing_country
			value = 20
		}
		create_diplomatic_catalyst = { type = catalyst_event_positive target = scope:criticizing_country }
		add_loyalists = { value = small_radicals pop_type = academics }
		add_radicals = { value = small_radicals strata = upper }
		scope:civil_rights_movement = {
			add_modifier = {
				name = international_criticism_promises
				days = normal_modifier_time
				is_decaying = yes
			}
		}
		add_lobby_appeasement_from_diplomacy_unidirectional = {
			FIRST = root
			SECOND = scope:criticizing_country
			PRO_AMOUNT = 2
			ANTI_AMOUNT = -2
			FACTOR = appeasement_relations_decreased
		}
	}

	option = {
		name = movement_events_te.4.c
		ai_chance = {
			base = 1
		}
		# Deflect with counter-accusations - risky diplomacy
		change_relations = {
			country = scope:criticizing_country
			value = -50
		}
		create_diplomatic_catalyst = { type = catalyst_diplomatic_incident target = scope:criticizing_country }
		add_loyalists = { value = small_radicals strata = upper }
	}
}
