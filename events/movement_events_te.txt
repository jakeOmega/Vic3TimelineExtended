# Event 1: Civil Rights March
# A mass march by discriminated communities demanding equal rights.
movement_events_te.1 = {
	type = country_event
	placement = root

	event_image = {
		video = "unspecific_ruler_speaking_to_people"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	title = movement_events_te.1.t
	desc = movement_events_te.1.d
	flavor = movement_events_te.1.f

	duration = 3
	cooldown = { days = normal_modifier_time }

	trigger = {
		any_political_movement = {
			is_political_movement_type = movement_civil_rights
		}
		has_technology_researched = civil_rights_movement
		any_scope_state = {
			is_incorporated = yes
			any_scope_pop = {
				pop_acceptance < acceptance_status_4
			}
		}
	}

	cancellation_trigger = {
		NOT = {
			any_political_movement = {
				is_political_movement_type = movement_civil_rights
			}
		}
	}

	immediate = {
		random_political_movement = {
			limit = { is_political_movement_type = movement_civil_rights }
			save_scope_as = civil_rights_movement
		}
		random_scope_state = {
			limit = {
				is_incorporated = yes
				any_scope_pop = {
					pop_acceptance < acceptance_status_4
				}
			}
			save_scope_as = march_state
		}
	}

	option = {
		name = movement_events_te.1.a
		default_option = yes
		ai_chance = {
			base = 2
			modifier = {
				trigger = { has_technology_researched = social_justice_movements }
				add = 2
			}
		}
		# Let them march — democratic legitimacy, moderate outcome
		add_modifier = {
			name = civil_rights_march_momentum_modifier
			days = normal_modifier_time
			is_decaying = yes
		}
		add_loyalists = { value = small_radicals strata = lower }
		add_radicals = { value = very_small_radicals strata = upper }
	}

	option = {
		name = movement_events_te.1.b
		ai_chance = {
			base = 1
			modifier = {
				trigger = {
					OR = {
						has_law = law_type:law_minority_rights_violent_hostility
						has_law = law_type:law_minority_rights_ghettoization
					}
				}
				add = 2
			}
		}
		# Disperse the crowds — order maintained, popular anger
		add_modifier = {
			name = civil_rights_suppression_modifier
			days = normal_modifier_time
			is_decaying = yes
		}
		add_radicals = { value = medium_radicals strata = lower }
		add_radicals = { value = small_radicals strata = middle }
		add_loyalists = { value = small_radicals strata = upper }
	}

	option = {
		name = movement_events_te.1.c
		ai_chance = {
			base = 0
			modifier = {
				trigger = { has_technology_researched = social_justice_movements }
				add = 3
			}
		}
		# Government endorses the march — radical political shift
		add_modifier = {
			name = civil_rights_government_endorsement_modifier
			days = normal_modifier_time
			is_decaying = yes
		}
		ig:ig_intelligentsia ?= {
			leader = {
				set_ideology = ideology:ideology_multicultural
			}
		}
		add_loyalists = { value = medium_radicals strata = lower }
		add_loyalists = { value = small_radicals pop_type = academics }
		add_radicals = { value = medium_radicals strata = upper }
	}
}
