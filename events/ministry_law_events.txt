ministry_law_events.1 = {
	type = country_event
	placement = root

	title = ministry_law_events.1.t
	desc = ministry_law_events.1.d
	flavor = ministry_law_events.1.f

	category = enactment

	event_image = {
		video = "europenorthamerica_london_center"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_default.dds"

	duration = 3

	cooldown = { days = normal_modifier_time }

	trigger = {
		is_enacting_law_type_group = law_group_ministries
		NOT = {
			is_enacting_law = law_type:law_no_ministry_of_war
			is_enacting_law = law_type:law_no_ministry_of_foreign_affairs
			is_enacting_law = law_type:law_no_ministry_of_commerce
			is_enacting_law = law_type:law_no_ministry_of_culture
			is_enacting_law = law_type:law_no_ministry_of_labor
			is_enacting_law = law_type:law_no_ministry_of_the_environment
			is_enacting_law = law_type:law_no_ministry_of_intelligence_and_security
			is_enacting_law = law_type:law_no_ministry_of_refugee_affairs
			is_enacting_law = law_type:law_no_ministry_of_propaganda
			is_enacting_law = law_type:law_no_ministry_of_science
			is_enacting_law = law_type:law_no_ministry_of_thought_control
			is_enacting_law = law_type:law_no_ministry_of_consumer_protection
			is_enacting_law = law_type:law_no_ministry_of_urban_planning
			is_enacting_law = law_type:law_no_ministry_of_religion
			is_enacting_law = law_type:law_no_ministry_of_international_aid
		}
		NOR = {
			has_modifier = ministry_staffing_advance
			has_modifier = ministry_staffing_cautious
		}
	}

	immediate = {
		currently_enacting_law = {
			save_scope_as = current_law_scope
		}
	}

	cancellation_trigger = {
		NOT = { currently_enacting_law = scope:current_law_scope }
	}

	option = { # Recruit aggressively — fill every desk
		name = ministry_law_events.1.a
		default_option = yes
		add_enactment_modifier = {
			name = ministry_staffing_advance

		add_radicals = {
			value = small_radicals
			strata = middle
		}

	option = { # Staff it gradually to ensure quality

			name = ministry_staffing_cautious
		}

			is_enacting_law = law_type:law_no_ministry_of_culture
			is_enacting_law = law_type:law_no_ministry_of_labor
			is_enacting_law = law_type:law_no_ministry_of_the_environment
			is_enacting_law = law_type:law_no_ministry_of_intelligence_and_security
			is_enacting_law = law_type:law_no_ministry_of_refugee_affairs
			is_enacting_law = law_type:law_no_ministry_of_propaganda
			is_enacting_law = law_type:law_no_ministry_of_science
			is_enacting_law = law_type:law_no_ministry_of_thought_control
			is_enacting_law = law_type:law_no_ministry_of_consumer_protection
			is_enacting_law = law_type:law_no_ministry_of_urban_planning
			is_enacting_law = law_type:law_no_ministry_of_religion
			is_enacting_law = law_type:law_no_ministry_of_international_aid
		}
		NOR = {
			has_modifier = ministry_budget_advance
			has_modifier = ministry_budget_cautious
		}
		any_interest_group = {
			ig_counts_as_marginal = no
			law_stance = {
				law = owner.currently_enacting_law.type
				value < neutral
			}
		}
	}

	immediate = {
		currently_enacting_law = {
			save_scope_as = current_law_scope
		}
		random_interest_group = {
			limit = {
				ig_counts_as_marginal = no
				law_stance = {
					law = owner.currently_enacting_law.type
					value < neutral
				}
			}
			save_scope_as = opposing_ig
		}
	}

	cancellation_trigger = {
		NOT = { currently_enacting_law = scope:current_law_scope }
	}

	option = { # Allocate the full budget — half measures fail
		name = ministry_law_events.2.a
		default_option = yes
		add_enactment_modifier = {
			name = ministry_budget_advance
		}
		add_radicals = {
			value = small_radicals
			strata = rich
		}
	}

	option = { # Promise a lean operation to win over the skeptics
		name = ministry_law_events.2.b
		add_enactment_modifier = {
			name = ministry_budget_cautious
		}
		scope:opposing_ig = {
			add_modifier = {
				name = ministry_budget_advance
				days = normal_modifier_time
			}
		}
	}
}

		}
		NOR = {
			has_modifier = ministry_labor_struggle_advance

	}

	immediate = {
		currently_enacting_law = {

		}
		ig:ig_trade_unions ?= {
			save_scope_as = trade_unions_ig
		}

			save_scope_as = industrialists_ig
		}
	}

	cancellation_trigger = {
		NOT = { currently_enacting_law = scope:current_law_scope }
	}

	option = { # The ministry will serve the national interest

			name = ministry_labor_struggle_advance
		}
		add_radicals = {
			value = small_radicals
		}
	}

	option = { # Proceed carefully to minimize opposition
		name = ministry_law_events.9.b
		add_enactment_modifier = {
			name = ministry_labor_struggle_cautious
		}
		add_loyalists = {
			value = small_radicals
		}
	}

	option = { # Invite the key stakeholders to shape the ministry
		name = ministry_law_events.9.c
		trigger = {
			OR = {
				AND = {
					is_enacting_law = law_type:law_pro_labor_ministry_of_labor
					ig:ig_trade_unions ?= {
						ig_counts_as_marginal = no
					}
				}
				AND = {
					is_enacting_law = law_type:law_pro_capital_ministry_of_labor
					ig:ig_industrialists ?= {
						ig_counts_as_marginal = no
					}
				}
			}
		}
		highlighted_option = yes
		add_enactment_modifier = {
			name = ministry_labor_struggle_advance
		}
		if = {
			limit = { is_enacting_law = law_type:law_pro_labor_ministry_of_labor }
			scope:trade_unions_ig = {
				add_modifier = {
					name = ministry_labor_struggle_advance
					days = normal_modifier_time
				}
			}
			scope:industrialists_ig = {
				add_modifier = {
					name = ministry_labor_struggle_cautious
					days = normal_modifier_time
				}
			}
		}
		if = {
			limit = { is_enacting_law = law_type:law_pro_capital_ministry_of_labor }
			scope:industrialists_ig = {
				add_modifier = {
					name = ministry_labor_struggle_advance
					days = normal_modifier_time
				}
			}
			scope:trade_unions_ig = {
				add_modifier = {
					name = ministry_labor_struggle_cautious
					days = normal_modifier_time
				}
			}
