	category = enactment

	event_image = { texture = "gfx/event_pictures/ministry_establishment.dds" }

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_default.dds"

	duration = 3

	cooldown = { days = normal_modifier_time }

	trigger = {
		is_enacting_any_ministry_law = yes
		NOR = {
			has_modifier = ministry_staffing_advance
			has_modifier = ministry_staffing_cautious
		}
	}

	immediate = {
		currently_enacting_law = {
			save_scope_as = current_law_scope
		}
	}

	cancellation_trigger = {
		NOT = { currently_enacting_law = scope:current_law_scope }
	}

	option = { # Recruit aggressively - fill every desk
		name = ministry_law_events.1.a
		default_option = yes
		add_enactment_modifier = {
			name = ministry_staffing_advance

			value = small_radicals

		}

	option = { # Staff it gradually to ensure quality

			name = ministry_staffing_cautious
		}
		# Domain side-effect
		add_modifier = { name = te_legitimacy_boost days = normal_modifier_time }
	}
}

# ======================================================================
#  2 – The Budget Debate  (Generic – any ministry being created)
# ======================================================================
ministry_law_events.2 = {
	type = country_event
	placement = root

	title = ministry_law_events.2.t
	desc = ministry_law_events.2.d
	flavor = ministry_law_events.2.f

	category = enactment

	event_image = { texture = "gfx/event_pictures/ministry_establishment.dds" }

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_trade.dds"

	duration = 3

	cooldown = { days = normal_modifier_time }

	trigger = {
		is_enacting_any_ministry_law = yes
		NOR = {
			has_modifier = ministry_budget_advance
			has_modifier = ministry_budget_cautious
		}
		any_interest_group = {
			ig_counts_as_marginal = no
			law_stance = {
				law = owner.currently_enacting_law.type
				value < neutral
			}
		}
	}

	immediate = {
		currently_enacting_law = {
			save_scope_as = current_law_scope
		}
		random_interest_group = {
			limit = {
				ig_counts_as_marginal = no
				law_stance = {
					law = owner.currently_enacting_law.type
					value < neutral
				}
			}
			save_scope_as = opposing_ig
		}
	}

	cancellation_trigger = {
		NOT = { currently_enacting_law = scope:current_law_scope }
	}

	option = { # Allocate the full budget - half measures fail
		name = ministry_law_events.2.a
		default_option = yes
		add_enactment_modifier = {
			name = ministry_budget_advance
		}
		# Domain side-effect
		add_modifier = { name = te_authority_boost days = normal_modifier_time }
	}

	option = { # Promise a lean operation to win over the skeptics
		name = ministry_law_events.2.b
		add_enactment_modifier = {
			name = ministry_budget_cautious
		}
		scope:opposing_ig = {
			add_modifier = {
				name = ig_approval_positive_modifier
				days = normal_modifier_time
				is_decaying = yes
			}
		}
	}
}

			name = ministry_labor_struggle_cautious
		}
		add_loyalists = {
			value = small_radicals
		}
	}

	option = { # Invite the key stakeholders to shape the ministry
		name = ministry_law_events.9.c
		trigger = {
			OR = {
				AND = {
					is_enacting_law = law_type:law_pro_labor_ministry_of_labor
					ig:ig_trade_unions ?= {
						ig_counts_as_marginal = no
					}
				}
				AND = {
					is_enacting_law = law_type:law_pro_capital_ministry_of_labor
					ig:ig_industrialists ?= {
						ig_counts_as_marginal = no
					}
				}
			}
		}
		highlighted_option = yes
		add_enactment_modifier = {
			name = ministry_labor_struggle_advance
		}
		if = {
			limit = { is_enacting_law = law_type:law_pro_labor_ministry_of_labor }

				add_modifier = {
					name = ig_approval_positive_modifier
