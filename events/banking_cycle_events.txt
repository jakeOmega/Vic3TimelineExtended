namespace = banking_cycle_events

# ============================================================================
# BANKING CYCLE EVENTS
# Random shocks and policy dilemmas that inject uncertainty into the cycle.
# Fired from je_banking_cycle monthly pulse with cooldown.
#
# EVENT ROSTER (~20 events):
# --- EARLY ERA (no extra tech gate beyond stock_exchange) ---
#  1. Speculative Mania       - expansion+, bubble building
#  2. Foreign Capital Inflow   - stable+, not at war
#  3. Credit Crunch Fears      - stagnation/downturn
#  4. Country Bank Run         - downturn/panic, RARE, high impact
#  5. Railway Bond Fever       - expansion+, railroads tech
#  6. Harvest Failure & Credit - any phase, agrarian economy
#
# --- MID ERA (central_banking / investment_banks / mutual_funds) ---
#  7. Bank Consolidation Wave  - boom/frenzy, central_banking
#  8. Novel Financial Instrument - expansion+, investment_banks
#  9. Asset Price Uncertainty  - expansion/boom, bubble >= 30, central_banking
# 10. Sovereign Debt Scare    - stagnation/downturn/panic, central_banking
# 11. Insurance Sector Wobble - expansion+, mutual_funds
# 12. Gold Standard Pressure  - any phase, international_exchange_standards
# 13. Industrial Overinvestment - boom/frenzy, corporate_management
#
# --- LATE ERA (keynesian_economics / modern_financial_instruments / consumer_credit) ---
# 14. Housing Price Spiral     - boom/frenzy, consumer_credit, high impact
# 15. Currency Crisis          - downturn+, modern_financial_instruments
# 16. Central Bank Independence Debate - stable+, keynesian_economics
# 17. Consumer Credit Boom    - expansion+, consumer_credit
# 18. Derivatives Scandal     - boom/frenzy, modern_financial_instruments, RARE
# 19. Sovereign Wealth Fund Proposal - expansion+, keynesian_economics, law gate
# 20. Shadow Banking Discovery - boom/frenzy, modern_financial_instruments, RARE, high impact
# ============================================================================
