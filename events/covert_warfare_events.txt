namespace = covert_warfare

# ============================================================================
# EVENT 1: Covert Operation Detected (Attacker's Perspective)
# ============================================================================
# Fired when a monthly detection check succeeds. A random active operation
# pact is burned (removed). The attacker gains infamy and the target
# country is notified.
# ============================================================================

covert_warfare.1 = {
	type = country_event
	placement = root

	title = covert_warfare.1.t
	desc = covert_warfare.1.d
	flavor = covert_warfare.1.f

	event_image = {
		video = "event_newspaper"
	}
	icon = "gfx/interface/icons/event_icons/event_default.dds"

	duration = 3

	trigger = {
		covert_operations_active >= 1
	}

	immediate = {
		# Pick a random active operation pact and save its target
		random_scope_diplomatic_pact = {
			limit = {
				OR = {
					is_diplomatic_action_type = covert_election_interference_action
					is_diplomatic_action_type = covert_financial_subversion_action
					is_diplomatic_action_type = covert_infrastructure_sabotage_action
					is_diplomatic_action_type = covert_comms_disruption_action
					is_diplomatic_action_type = covert_industrial_espionage_action
					is_diplomatic_action_type = covert_military_espionage_action
					is_diplomatic_action_type = covert_influence_campaign_action
					is_diplomatic_action_type = covert_ideological_subversion_action
					is_diplomatic_action_type = covert_destabilization_action
				}
			}
			save_scope_as = burned_pact
			every_participant = {
				limit = { NOT = { this = ROOT } }
				save_scope_as = detected_by_country
			}
		}
	}

	option = {
		name = covert_warfare.1.a
		default_option = yes

		# Infamy from being caught
		change_infamy = covert_ops_detected_infamy

		# Cancel the burned operation pact
		if = {
			limit = {
				exists = scope:detected_by_country
				exists = scope:burned_pact
			}
			# Remove the pact — check each type
			if = {
				limit = { scope:burned_pact = { is_diplomatic_action_type = covert_election_interference_action } }
				remove_diplomatic_pact = { country = scope:detected_by_country type = covert_election_interference_action }
			}
			else_if = {
				limit = { scope:burned_pact = { is_diplomatic_action_type = covert_financial_subversion_action } }
				remove_diplomatic_pact = { country = scope:detected_by_country type = covert_financial_subversion_action }
			}
			else_if = {
				limit = { scope:burned_pact = { is_diplomatic_action_type = covert_infrastructure_sabotage_action } }
				remove_diplomatic_pact = { country = scope:detected_by_country type = covert_infrastructure_sabotage_action }
			}
			else_if = {
				limit = { scope:burned_pact = { is_diplomatic_action_type = covert_comms_disruption_action } }
				remove_diplomatic_pact = { country = scope:detected_by_country type = covert_comms_disruption_action }
			}
			else_if = {
				limit = { scope:burned_pact = { is_diplomatic_action_type = covert_industrial_espionage_action } }
				remove_diplomatic_pact = { country = scope:detected_by_country type = covert_industrial_espionage_action }
			}
			else_if = {
				limit = { scope:burned_pact = { is_diplomatic_action_type = covert_military_espionage_action } }
				remove_diplomatic_pact = { country = scope:detected_by_country type = covert_military_espionage_action }
			}
			else_if = {
				limit = { scope:burned_pact = { is_diplomatic_action_type = covert_influence_campaign_action } }
				remove_diplomatic_pact = { country = scope:detected_by_country type = covert_influence_campaign_action }
			}
			else_if = {
				limit = { scope:burned_pact = { is_diplomatic_action_type = covert_ideological_subversion_action } }
				remove_diplomatic_pact = { country = scope:detected_by_country type = covert_ideological_subversion_action }
			}
			else_if = {
				limit = { scope:burned_pact = { is_diplomatic_action_type = covert_destabilization_action } }
				remove_diplomatic_pact = { country = scope:detected_by_country type = covert_destabilization_action }
			}

			# Notify the target
			scope:detected_by_country = {
				trigger_event = { id = covert_warfare.2 }
			}
		}

		ai_chance = {
			base = 10
		}
	}

	option = {
		name = covert_warfare.1.b

		# Deny and double down — same mechanical effect but flavorful
		change_infamy = covert_ops_detected_infamy

		# Cancel the burned operation pact
		if = {
			limit = {
				exists = scope:detected_by_country
				exists = scope:burned_pact
			}
			if = {
				limit = { scope:burned_pact = { is_diplomatic_action_type = covert_election_interference_action } }
				remove_diplomatic_pact = { country = scope:detected_by_country type = covert_election_interference_action }
			}
			else_if = {
				limit = { scope:burned_pact = { is_diplomatic_action_type = covert_financial_subversion_action } }
				remove_diplomatic_pact = { country = scope:detected_by_country type = covert_financial_subversion_action }
			}
			else_if = {
				limit = { scope:burned_pact = { is_diplomatic_action_type = covert_infrastructure_sabotage_action } }
				remove_diplomatic_pact = { country = scope:detected_by_country type = covert_infrastructure_sabotage_action }
			}
			else_if = {
				limit = { scope:burned_pact = { is_diplomatic_action_type = covert_comms_disruption_action } }
				remove_diplomatic_pact = { country = scope:detected_by_country type = covert_comms_disruption_action }
			}
			else_if = {
				limit = { scope:burned_pact = { is_diplomatic_action_type = covert_industrial_espionage_action } }
				remove_diplomatic_pact = { country = scope:detected_by_country type = covert_industrial_espionage_action }
			}
			else_if = {
				limit = { scope:burned_pact = { is_diplomatic_action_type = covert_military_espionage_action } }
				remove_diplomatic_pact = { country = scope:detected_by_country type = covert_military_espionage_action }
			}
			else_if = {
				limit = { scope:burned_pact = { is_diplomatic_action_type = covert_influence_campaign_action } }
				remove_diplomatic_pact = { country = scope:detected_by_country type = covert_influence_campaign_action }
			}
			else_if = {
				limit = { scope:burned_pact = { is_diplomatic_action_type = covert_ideological_subversion_action } }
				remove_diplomatic_pact = { country = scope:detected_by_country type = covert_ideological_subversion_action }
			}
			else_if = {
				limit = { scope:burned_pact = { is_diplomatic_action_type = covert_destabilization_action } }
				remove_diplomatic_pact = { country = scope:detected_by_country type = covert_destabilization_action }
			}

			# Notify the target
			scope:detected_by_country = {
				trigger_event = { id = covert_warfare.2 }
			}
		}

		ai_chance = {
			base = 5
		}
	}
}

# ============================================================================
# EVENT 2: Foreign Covert Operation Detected (Defender's Perspective)
# ============================================================================
# Fired on the target country when they detect that someone is running
# operations against them. Gives them options to respond.
# ============================================================================

covert_warfare.2 = {
	type = country_event
	placement = root

	title = covert_warfare.2.t
	desc = covert_warfare.2.d
	flavor = covert_warfare.2.f

	event_image = {
		video = "event_newspaper"
	}
	icon = "gfx/interface/icons/event_icons/event_default.dds"

	duration = 3

	trigger = {
		# Only fire when someone has run an operation on us
		OR = {
			has_modifier = covert_election_interference
			has_modifier = covert_financial_subversion
			has_modifier = covert_infrastructure_sabotage
			has_modifier = covert_comms_disruption
			has_modifier = covert_influence_campaign
		}
		NOT = { has_variable = iw_defender_event_cooldown }
	}

	immediate = {
		set_variable = { name = iw_defender_event_cooldown value = yes }
	}

	option = {
		name = covert_warfare.2.a
		default_option = yes

		# Harden defenses: temporary IC boost
		add_modifier = {
			name = intelligence_capacity_defense
			months = normal_modifier_time
			is_decaying = yes
			multiplier = 3
		}

		ai_chance = {
			base = 10
		}
	}

	option = {
		name = covert_warfare.2.b

		# Retaliate: reduce infamy — use the detection to gain diplomatic credibility
		change_infamy = -3

		ai_chance = {
			base = 5
		}
	}
}
