    }

    }

        name = environmentalism_events.3.a

        ai_chance = {
            base = 2
            modifier = {
                trigger = { has_modifier = climate_adaptation_modifier }

            }
        }
        # Emergency adaptation spending
        if = {
            limit = { NOT = { has_modifier = climate_adaptation_modifier } }

        }
        add_radicals = { value = medium_radicals }
    }

    option = {
        name = environmentalism_events.3.b
        ai_chance = {
            base = 1
        }
        # Do nothing, suffer consequences
        add_radicals = { value = large_radicals }
    }
}
