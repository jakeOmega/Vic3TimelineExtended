# Event 7: Industrial Pollution Scandal (random event, early game)
environmentalism_events.7 = {
    type = country_event
    placement = root

    event_image = {
        video = "unspecific_factory"

    flavor = environmentalism_events.7.f

    duration = 3

    trigger = {
        has_global_variable = greenhouse_gas_emissions
        NOT = { has_modifier = global_warming }
        any_scope_state = {
            is_incorporated = yes

            }

    }

    immediate = {
    }

    option = {
        name = environmentalism_events.7.a

    }

    option = {
        name = environmentalism_events.7.b

            base = 1

                trigger = { temperature_anomaly_display >= 0.3 }
                add = 2
            }

        add_modifier = { name = climate_adaptation_modifier months = 24 }
    }
}
