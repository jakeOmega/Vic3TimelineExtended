# Event 1: Colonial Independence Movement Rising
decolonization_events.1 = {
	type = country_event
	placement = root

	event_image = {
		texture = "gfx/event_pictures/colonial_protest.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_protest.dds"

	title = decolonization_events.1.t
	desc = decolonization_events.1.d
	flavor = decolonization_events.1.f

	duration = 3

	cooldown = { days = long_modifier_time }

	trigger = {
		NOR = {
			has_modifier = decolonization_negotiations_modifier
			has_modifier = colonial_crackdown_modifier
			has_modifier = peaceful_decolonization_modifier
		}
		has_technology_researched = decolonization
		any_subject_or_below = {
			liberty_desire >= 50
		}
	}

	immediate = {
		random_subject_or_below = {
			limit = { liberty_desire >= 50 }
			save_scope_as = restless_subject
		}
	}

	# Option A: Negotiate for autonomy
	option = {
		name = decolonization_events.1.a
		default_option = yes
		ai_chance = {
			base = 3
			modifier = {
				trigger = {
					ig:ig_intelligentsia ?= {
						leader = { has_ideology = ideology:ideology_anti_colonialist }
					}
				}
				add = 3
			}
			modifier = {
				trigger = { colonial_gp_condemners_count >= 2 }
				add = 3
			}
		}
		add_modifier = {
			name = decolonization_negotiations_modifier
			days = normal_modifier_time
			is_decaying = yes
		}
		scope:restless_subject = {
			add_liberty_desire = -15
			post_notification = colonial_negotiations_started_notice
		}
		# Intelligentsia approve of diplomacy, landowners see weakness
		ig_approval_effect = { IG = ig_intelligentsia MODIFIER = ig_approval_positive_modifier DAYS = short_modifier_time }
		ig_approval_effect = { IG = ig_landowners MODIFIER = ig_approval_negative_modifier DAYS = short_modifier_time }
		add_modifier = { name = colonial_stability_negative_event days = 90 }
		# Colonial lobby appeasement: negotiation is a concession
		add_colonial_lobby_appeasement = {
			FIRST = root
			SECOND = scope:restless_subject
			PRO_AMOUNT = -1
			ANTI_AMOUNT = 1
			FACTOR = appeasement_colonial_concession
		}
	}

	# Option B: Crack down on the movement
	option = {
		name = decolonization_events.1.b
		ai_chance = {
			base = 1
			modifier = {
				trigger = { country_rank >= rank_value:great_power }
				add = 1
			}
		}
		add_modifier = {
			name = colonial_crackdown_modifier
			days = normal_modifier_time
			is_decaying = yes
		}
		scope:restless_subject = {
			add_liberty_desire = 20
			post_notification = colonial_crackdown_begun_notice
		}
		add_radicals = { value = medium_radicals pop_type = academics }
		# Armed forces approve, intelligentsia horrified
		ig_approval_effect = { IG = ig_armed_forces MODIFIER = ig_approval_positive_modifier DAYS = short_modifier_time }
		ig_approval_effect = { IG = ig_intelligentsia MODIFIER = ig_approval_negative_modifier DAYS = short_modifier_time }
		add_modifier = { name = colonial_stability_positive_event months = 2 }
		# Colonial lobby appeasement: crackdown pleases colonial lobby
		add_colonial_lobby_appeasement = {
			FIRST = root
			SECOND = scope:restless_subject
			PRO_AMOUNT = 1
			ANTI_AMOUNT = -1
			FACTOR = appeasement_colonial_crackdown
		}
	}

	# Option C: Grant independence outright
	option = {
		name = decolonization_events.1.c
		ai_chance = {
			base = 2
			modifier = {
				trigger = {
					scope:restless_subject = { liberty_desire >= 75 }
				}
				add = 5
			}
			modifier = {
				trigger = { NOT = { country_rank >= rank_value:great_power } }
				add = 5
			}
			modifier = {
				trigger = { colonial_gp_condemners_count >= 3 }
				add = 5
			}
		}
		scope:restless_subject = {
			post_notification = colonial_independence_granted_notice
			make_independent = yes
		}
		add_modifier = {
			name = peaceful_decolonization_modifier
			days = long_modifier_time
			is_decaying = yes
		}
		change_relations = {
			country = scope:restless_subject
			value = 30
		}
		create_diplomatic_catalyst = { type = catalyst_event_positive target = scope:restless_subject }
		add_lobby_appeasement_from_diplomacy_unidirectional = {
			FIRST = root
			SECOND = scope:restless_subject
			PRO_AMOUNT = 1
			ANTI_AMOUNT = -1
			FACTOR = appeasement_relations_increased
		}
		# Colonial lobby: granting independence is a major concession
		add_colonial_lobby_appeasement = {
			FIRST = root
			SECOND = scope:restless_subject
			PRO_AMOUNT = -2
			ANTI_AMOUNT = 2
			FACTOR = appeasement_colonial_concession
		}
		# Prestige boost from moral leadership, but landowners furious
		ig_approval_effect = { IG = ig_landowners MODIFIER = ig_approval_very_negative_modifier DAYS = short_modifier_time }
		add_modifier = { name = colonial_stability_positive_event days = 150 }
	}

	# Option D: Neocolonialism - release but extract economic concessions
	option = {
		name = decolonization_events.1.d_neo
		trigger = {
			has_law = law_type:law_neocolonialism
		}
		highlighted_option = yes
		ai_chance = {
			base = 4
		}
		scope:restless_subject = {
			post_notification = colonial_neocolonial_terms_notice
			make_independent = yes
			add_modifier = {
				name = neocolonial_dependency_imposed_modifier
				days = long_modifier_time
				is_decaying = yes
			}
		}
		add_modifier = {
			name = neocolonial_economic_extraction_modifier
			days = long_modifier_time
			is_decaying = yes
		}
		change_relations = {
			country = scope:restless_subject
			value = -10
		}
		add_lobby_appeasement_from_diplomacy_unidirectional = {
			FIRST = root
			SECOND = scope:restless_subject
			PRO_AMOUNT = -1
			ANTI_AMOUNT = 1
			FACTOR = appeasement_relations_decreased
		}
		# Colonial lobby: neocolonialism is a mixed signal (less harsh than crack down, not independence)
		add_colonial_lobby_appeasement = {
			FIRST = root
			SECOND = scope:restless_subject
			PRO_AMOUNT = -1
			ANTI_AMOUNT = 1
			FACTOR = appeasement_colonial_concession
		}
		# Industrialists love this, intelligentsia see through it
		ig_approval_effect = { IG = ig_industrialists MODIFIER = ig_approval_positive_modifier DAYS = short_modifier_time }
		ig_approval_effect = { IG = ig_intelligentsia MODIFIER = ig_approval_negative_modifier DAYS = short_modifier_time }
		add_modifier = { name = colonial_stability_positive_event days = 90 }
	}
}

# Event 4: Cultural Reclamation in Former Colony (fires for newly independent countries)
decolonization_events.4 = {
	type = country_event
	placement = root

	event_image = {
		texture = "gfx/event_pictures/political_rally.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_protest.dds"

	title = decolonization_events.4.t
	desc = decolonization_events.4.d
	flavor = decolonization_events.4.f

	duration = 3

	cooldown = { days = long_modifier_time }

	trigger = {
		NOT = { has_modifier = cultural_reclamation_modifier }
		has_technology_researched = decolonization
		is_subject = no
		NOT = {
			any_scope_state = {
				is_homeland_of_country_cultures = ROOT
			}
		}
	}

	immediate = {
	}

	# Option A: Embrace indigenous culture
	option = {
		name = decolonization_events.4.a
		default_option = yes
		ai_chance = { base = 5 }
		every_primary_culture = {
			ROOT = {
				remove_primary_culture = PREV
			}
		}
		capital = {
			every_scope_culture = {
				limit = {
					has_homeland = PREV
				}
				ROOT = {
					add_primary_culture = PREV
				}
			}
		}
		add_modifier = {
			name = cultural_reclamation_modifier
			days = long_modifier_time
			is_decaying = yes
		}
		add_loyalists = { value = medium_radicals strata = lower }
		add_loyalists = { value = small_radicals pop_type = academics }
		# Upper strata who benefited from colonial culture resist
		add_radicals = { value = small_radicals strata = upper }
	}

	# Option B: Maintain colonial-era identity
	option = {
		name = decolonization_events.4.b
		ai_chance = { base = 1 }
		add_radicals = { value = medium_radicals strata = lower }
		add_loyalists = { value = small_radicals strata = upper }
		# Intelligentsia split - colonial education was their path to power
		add_loyalists = { value = very_small_radicals pop_type = academics }
	}
}

	trigger = {
		has_technology_researched = decolonization
		has_variable = recently_formed_country
		is_subject = no
		NOT = { has_modifier = partition_territorial_claims_modifier }
		NOT = { has_modifier = partition_accepted_borders_modifier }
		# Must border another country that shares cultural heritage
		any_scope_state = {
			any_neighbouring_state = {
				owner = {
					NOT = { this = root }
					is_country_alive = yes
					NOT = { is_country_type = decentralized }
					OR = {
						# Another newly formed country with shared heritage
						AND = {
							has_variable = recently_formed_country
							any_primary_culture = {
								shares_heritage_trait_group_with_any_primary_culture = root
							}
						}
						# Or colonial state with our culture's homeland
						AND = {
							any_scope_state = {
								is_overseas_colonial_state = yes
								any_scope_culture = {
									shares_heritage_trait_group_with_any_primary_culture = root
								}
							}
						}
					}
				}
			}
		}
	}

	immediate = {
		# Find a neighboring country with shared heritage
		random_scope_state = {
			limit = {
				any_neighbouring_state = {
					owner = {
						NOT = { this = root }
						is_country_alive = yes
						NOT = { is_country_type = decentralized }
						OR = {
							AND = {
								has_variable = recently_formed_country
								any_primary_culture = {
									shares_heritage_trait_group_with_any_primary_culture = root
								}
							}
							AND = {
								any_scope_state = {
									is_overseas_colonial_state = yes
									any_scope_culture = {
										shares_heritage_trait_group_with_any_primary_culture = root
									}
								}
							}
						}
					}
				}
			}
			random_neighbouring_state = {
				limit = {
					owner = {
						NOT = { this = root }
						is_country_alive = yes
						NOT = { is_country_type = decentralized }
					}
				}
				owner = {
					save_scope_as = partition_neighbor
				}
			}
		}
	}

	# Option A: Demand unification: press for incorporating neighboring peoples
	option = {
		name = decolonization_events.16.a
		default_option = yes
		ai_chance = {
			base = 3
			modifier = {
				trigger = {
					exists = scope:partition_neighbor
					scope:partition_neighbor = { has_variable = recently_formed_country }
				}
				add = 3
			}
		}
		add_modifier = {
			name = partition_territorial_claims_modifier
			days = long_modifier_time
			is_decaying = yes
		}
		if = {
			limit = { exists = scope:partition_neighbor }

				country = scope:partition_neighbor
				value = -40
			}
		}
		add_loyalists = { value = medium_radicals strata = lower }
		add_radicals = { value = small_radicals strata = upper }
		ig_approval_effect = { IG = ig_armed_forces MODIFIER = ig_approval_positive_modifier DAYS = short_modifier_time }
	}

	# Option B: Propose federation with our neighbor
	option = {
		name = decolonization_events.16.b
		trigger = {
			exists = scope:partition_neighbor

# ============================================================================
# Event 19: The [Region] Crisis: Suez Crisis Model
# Fires for great powers when a recently-independent former colony
# nationalizes assets. The GP can intervene militarily, apply sanctions,
# or accept the new status quo. Other GPs then react.
# ============================================================================
decolonization_events.19 = {
	type = country_event
	placement = root

	event_image = {
		texture = "gfx/event_pictures/military_parade.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_military.dds"

	title = decolonization_events.19.t
	desc = decolonization_events.19.d
	flavor = decolonization_events.19.f

	duration = 3

	cooldown = { days = long_modifier_time }

	trigger = {
		has_technology_researched = decolonization
		country_rank >= rank_value:great_power
		NOR = {
			has_modifier = gp_military_intervention_modifier
			has_modifier = gp_economic_sanctions_modifier
			has_variable = gp_accepted_nationalization
		}
		# A former colony has nationalized assets
		any_country = {
			NOT = { this = root }
			has_variable = former_overlord
			var:former_overlord = root
			is_subject = no
			has_modifier = independence_nationalization_modifier
		}
	}

	immediate = {
		random_country = {
			limit = {
				NOT = { this = root }
				has_variable = former_overlord
				var:former_overlord = root
				is_subject = no
				has_modifier = independence_nationalization_modifier
			}
			save_scope_as = nationalizing_country
		}
	}

	# Option A: Military intervention: send in the troops
	option = {
		name = decolonization_events.19.a
		ai_chance = {
			base = 1
			modifier = {
				trigger = {
					ig:ig_armed_forces ?= {
						is_in_government = yes
					}
				}
				add = 2
			}
			# Less likely if many GPs condemn colonialism
			modifier = {
				trigger = { colonial_gp_condemners_count >= 3 }
				add = -5
			}
		}
		add_modifier = {
			name = gp_military_intervention_modifier
			days = long_modifier_time
			is_decaying = yes
		}
		# The target suffers
		scope:nationalizing_country = {
			add_modifier = {
				name = gp_intervention_target_modifier
				days = long_modifier_time
				is_decaying = yes
			}
		}
		change_relations = {
			country = scope:nationalizing_country
			value = -100
		}
		change_infamy = 15
		# Domestic reaction to imperial adventure
		ig_approval_effect = { IG = ig_armed_forces MODIFIER = ig_approval_positive_modifier DAYS = short_modifier_time }
		ig_approval_effect = { IG = ig_landowners MODIFIER = ig_approval_positive_modifier DAYS = short_modifier_time }
		ig_approval_effect = { IG = ig_intelligentsia MODIFIER = ig_approval_very_negative_modifier DAYS = normal_modifier_time }
		ig_approval_effect = { IG = ig_trade_unions MODIFIER = ig_approval_very_negative_modifier DAYS = short_modifier_time }
		add_radicals = { value = large_radicals pop_type = academics }
		# Trigger GP response event for all other great powers
		every_country = {
			limit = {
				NOT = { this = root }
				country_rank >= rank_value:great_power
				has_technology_researched = decolonization
			}
			trigger_event = { id = decolonization_events.20 }
		}
	}

	# Option B: Economic sanctions: punish without war
	option = {
		name = decolonization_events.19.b
		default_option = yes
		ai_chance = {
			base = 3
			modifier = {
				trigger = { has_variable = gp_pro_colonial_stance }
				add = 3
			}
		}
		add_modifier = {
			name = gp_economic_sanctions_modifier
			days = long_modifier_time
			is_decaying = yes
		}
		change_relations = {
			country = scope:nationalizing_country
			value = -40
		}
		change_infamy = 5
		ig_approval_effect = { IG = ig_industrialists MODIFIER = ig_approval_positive_modifier DAYS = short_modifier_time }
		ig_approval_effect = { IG = ig_intelligentsia MODIFIER = ig_approval_negative_modifier DAYS = short_modifier_time }
	}

	# Option C: Accept the new order: swallow the humiliation
	option = {
		name = decolonization_events.19.c
		ai_chance = {
			base = 2
			modifier = {
				trigger = { has_variable = gp_anti_colonial_stance }
				add = 5
			}
			modifier = {
				trigger = { colonial_gp_condemners_count >= 3 }
				add = 5
			}
		}
		set_variable = { name = gp_accepted_nationalization days = long_modifier_time }
		add_modifier = {
			name = peaceful_decolonization_modifier
			days = normal_modifier_time
			is_decaying = yes
		}
		change_relations = {
			country = scope:nationalizing_country
			value = 20
		}
		add_radicals = { value = small_radicals strata = upper }
		ig_approval_effect = { IG = ig_landowners MODIFIER = ig_approval_very_negative_modifier DAYS = short_modifier_time }
		ig_approval_effect = { IG = ig_intelligentsia MODIFIER = ig_approval_positive_modifier DAYS = short_modifier_time }
	}
}

# Event 20: International Condemnation of Intervention
# Fires for other GPs when a great power intervenes militarily in a former colony.
# Models the international reaction to the Suez Crisis / similar interventions.
decolonization_events.20 = {
	type = country_event
	placement = root

	event_image = {
		texture = "gfx/event_pictures/diplomatic_meeting.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	title = decolonization_events.20.t
	desc = decolonization_events.20.d
	flavor = decolonization_events.20.f

	duration = 3

	trigger = {
		# Always fires when triggered directly by event 19
		always = yes
	}

	immediate = {
		# The intervening power is whoever triggered this via event 19
		# We need to find them
		random_country = {
			limit = {
				NOT = { this = root }
				country_rank >= rank_value:great_power
				has_modifier = gp_military_intervention_modifier
			}
			save_scope_as = intervening_power
		}
		random_country = {
			limit = {
				NOT = { this = root }
				has_modifier = gp_intervention_target_modifier
			}
			save_scope_as = intervention_victim
		}
	}

	# Option A: Condemn the intervention: demand withdrawal
	option = {
		name = decolonization_events.20.a
		default_option = yes
		ai_chance = {
			base = 5
			modifier = {
				trigger = { has_variable = gp_anti_colonial_stance }
				add = 10
			}
			modifier = {
				trigger = {
					exists = scope:intervening_power
					scope:intervening_power = {
						relations:root < relations_threshold:amicable
					}
				}
				add = 5
			}
		}
		if = {
			limit = { exists = scope:intervening_power }
			change_relations = {
				country = scope:intervening_power
				value = -40
			}
			# The condemning power's moral authority
			scope:intervening_power = {
				add_modifier = {
					name = gp_intervention_condemned_modifier
					days = normal_modifier_time
					is_decaying = yes
				}
			}
		}
		if = {
			limit = { exists = scope:intervention_victim }
			change_relations = {
				country = scope:intervention_victim
				value = 30
			}
		}
		add_modifier = {
			name = gp_moral_authority_modifier
			days = normal_modifier_time
			is_decaying = yes
		}
		ig_approval_effect = { IG = ig_intelligentsia MODIFIER = ig_approval_positive_modifier DAYS = short_modifier_time }
		ig_approval_effect = { IG = ig_trade_unions MODIFIER = ig_approval_positive_modifier DAYS = short_modifier_time }
	}

	# Option B: Support the intervention: old empires must be defended
	option = {
		name = decolonization_events.20.b
		ai_chance = {
			base = 1
			modifier = {
				trigger = { has_variable = gp_pro_colonial_stance }
				add = 5
			}
			modifier = {
				trigger = {
					exists = scope:intervening_power
					scope:intervening_power = {
						relations:root >= relations_threshold:amicable
					}
				}
				add = 5
			}
		}
		if = {
			limit = { exists = scope:intervening_power }
			change_relations = {
				country = scope:intervening_power
				value = 30
			}
		}
		if = {
			limit = { exists = scope:intervention_victim }
			change_relations = {
				country = scope:intervention_victim
				value = -30
			}
		}
		ig_approval_effect = { IG = ig_landowners MODIFIER = ig_approval_positive_modifier DAYS = short_modifier_time }
		ig_approval_effect = { IG = ig_industrialists MODIFIER = ig_approval_positive_modifier DAYS = short_modifier_time }
		ig_approval_effect = { IG = ig_intelligentsia MODIFIER = ig_approval_negative_modifier DAYS = short_modifier_time }
		add_radicals = { value = small_radicals pop_type = academics }
	}

	# Option C: Remain neutral: this is not our affair
	option = {
		name = decolonization_events.20.c
		ai_chance = {
			base = 2
		}
		add_loyalists = { value = very_small_radicals }
	}
}
