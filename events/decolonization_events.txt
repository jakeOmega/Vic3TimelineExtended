# Event 1: Colonial Independence Movement Rising
decolonization_events.1 = {
	type = country_event
	placement = root

	event_image = {
		texture = "gfx/event_pictures/colonial_protest.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_protest.dds"

	title = decolonization_events.1.t
	desc = decolonization_events.1.d
	flavor = decolonization_events.1.f

	duration = 3

	cooldown = { days = long_modifier_time }

	trigger = {
		NOR = {
			has_modifier = decolonization_negotiations_modifier
			has_modifier = colonial_crackdown_modifier
			has_modifier = peaceful_decolonization_modifier
		}
		has_technology_researched = decolonization
		any_subject_or_below = {
			liberty_desire >= 50
		}
	}

	immediate = {
		random_subject_or_below = {
			limit = { liberty_desire >= 50 }
			save_scope_as = restless_subject
		}
	}

	# Option A: Negotiate for autonomy
	option = {
		name = decolonization_events.1.a
		default_option = yes
		ai_chance = {
			base = 1
			modifier = {
				trigger = {
					ig:ig_intelligentsia ?= {
						leader = { has_ideology = ideology:ideology_anti_colonialist }
					}
				}
				add = 3
			}
		}
		add_modifier = {
			name = decolonization_negotiations_modifier
			days = normal_modifier_time
			is_decaying = yes
		}
		scope:restless_subject = {
			add_liberty_desire = -15
			post_notification = colonial_negotiations_started_notice
		}
		# Intelligentsia approve of diplomacy, landowners see weakness
		every_interest_group = {
			limit = { is_interest_group_type = ig_intelligentsia }
			add_modifier = { name = ig_approval_positive_modifier days = short_modifier_time }
		}
		every_interest_group = {
			limit = { is_interest_group_type = ig_landowners }
			add_modifier = { name = ig_approval_negative_modifier days = short_modifier_time }
		}
		add_modifier = { name = colonial_stability_negative_event days = 90 }
	}

	# Option B: Crack down on the movement
	option = {
		name = decolonization_events.1.b
		ai_chance = {
			base = 2
		}
		add_modifier = {
			name = colonial_crackdown_modifier
			days = normal_modifier_time
			is_decaying = yes
		}
		scope:restless_subject = {
			add_liberty_desire = 20
			post_notification = colonial_crackdown_begun_notice
		}
		add_radicals = { value = medium_radicals pop_type = academics }
		# Armed forces approve, intelligentsia horrified
		every_interest_group = {
			limit = { is_interest_group_type = ig_armed_forces }
			add_modifier = { name = ig_approval_positive_modifier days = short_modifier_time }
		}
		every_interest_group = {
			limit = { is_interest_group_type = ig_intelligentsia }
			add_modifier = { name = ig_approval_negative_modifier days = short_modifier_time }
		}
		add_modifier = { name = colonial_stability_positive_event months = 2 }
	}

	# Option C: Grant independence outright
	option = {
		name = decolonization_events.1.c
		ai_chance = {
			base = 1
			modifier = {
				trigger = {
					scope:restless_subject = { liberty_desire >= 75 }
				}
				add = 3
			}
		}
		scope:restless_subject = {
			post_notification = colonial_independence_granted_notice
			make_independent = yes
		}
		add_modifier = {
			name = peaceful_decolonization_modifier
			days = long_modifier_time
			is_decaying = yes
		}
		change_relations = {
			country = scope:restless_subject
			value = 30
		}
		add_lobby_appeasement_from_diplomacy_unidirectional = {
			FIRST = root
			SECOND = scope:restless_subject
			PRO_AMOUNT = 1
			ANTI_AMOUNT = -1
			FACTOR = appeasement_relations_increased
		}
		# Prestige boost from moral leadership, but landowners furious
		every_interest_group = {
			limit = { is_interest_group_type = ig_landowners }
			add_modifier = { name = ig_approval_very_negative_modifier days = short_modifier_time }
		}
		add_modifier = { name = colonial_stability_positive_event days = 150 }
	}

	# Option D: Neocolonialism - release but extract economic concessions
	option = {
		name = decolonization_events.1.d_neo
		trigger = {
			has_law = law_type:law_neocolonialism
		}
		ai_chance = {
			base = 4
		}
		scope:restless_subject = {
			post_notification = colonial_neocolonial_terms_notice
			make_independent = yes
			add_modifier = {
				name = neocolonial_dependency_imposed_modifier
				days = long_modifier_time
				is_decaying = yes
			}
		}
		add_modifier = {
			name = neocolonial_economic_extraction_modifier
			days = long_modifier_time
			is_decaying = yes
		}
		change_relations = {
			country = scope:restless_subject
			value = -10
		}
		add_lobby_appeasement_from_diplomacy_unidirectional = {
			FIRST = root
			SECOND = scope:restless_subject
			PRO_AMOUNT = -1
			ANTI_AMOUNT = 1
			FACTOR = appeasement_relations_decreased
		}
		# Industrialists love this, intelligentsia see through it
		every_interest_group = {
			limit = { is_interest_group_type = ig_industrialists }
			add_modifier = { name = ig_approval_positive_modifier days = short_modifier_time }
		}
		every_interest_group = {
			limit = { is_interest_group_type = ig_intelligentsia }
			add_modifier = { name = ig_approval_negative_modifier days = short_modifier_time }
		}
		add_modifier = { name = colonial_stability_positive_event days = 90 }
	}
}

# Event 4: Cultural Reclamation in Former Colony (fires for newly independent countries)
decolonization_events.4 = {
	type = country_event
	placement = root

	event_image = {
		texture = "gfx/event_pictures/independence_celebration.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_protest.dds"

	title = decolonization_events.4.t
	desc = decolonization_events.4.d
	flavor = decolonization_events.4.f

	duration = 3

	cooldown = { days = long_modifier_time }

	trigger = {
		NOT = { has_modifier = cultural_reclamation_modifier }
		has_technology_researched = decolonization
		is_subject = no
		NOT = {
			any_scope_state = {
				is_homeland_of_country_cultures = ROOT
			}
		}
	}

	immediate = {
	}

	# Option A: Embrace indigenous culture
	option = {
		name = decolonization_events.4.a
		default_option = yes
		ai_chance = { base = 5 }
		every_primary_culture = {
			ROOT = {
				remove_primary_culture = PREV
			}
		}
		capital = {
			every_scope_culture = {
				limit = {
					has_homeland = PREV
				}
				ROOT = {
					add_primary_culture = PREV
				}
			}
		}
		add_modifier = {
			name = cultural_reclamation_modifier
			days = long_modifier_time
			is_decaying = yes
		}
		add_loyalists = { value = medium_radicals strata = lower }
		add_loyalists = { value = small_radicals pop_type = academics }
		# Upper strata who benefited from colonial culture resist
		add_radicals = { value = small_radicals strata = upper }
	}

	# Option B: Maintain colonial-era identity
	option = {
		name = decolonization_events.4.b
		ai_chance = { base = 1 }
		add_radicals = { value = medium_radicals strata = lower }
		add_loyalists = { value = small_radicals strata = upper }
		# Intelligentsia split - colonial education was their path to power
		add_loyalists = { value = very_small_radicals pop_type = academics }
	}
}

		every_interest_group = {
			limit = { is_interest_group_type = ig_intelligentsia }
			add_modifier = { name = ig_approval_positive_modifier days = short_modifier_time }
		}
		add_modifier = { name = colonial_stability_positive_event days = 150 }
	}

	# Option B: Suppress with military force
	option = {
		name = decolonization_events.13.b
		ai_chance = {
			base = 2
		}
		add_modifier = {
			name = colonial_crackdown_modifier
			days = long_modifier_time
			is_decaying = yes
		}
		scope:uprising_colony = {
			add_radicals_in_state = { value = large_radicals }
		}
		add_radicals = { value = medium_radicals pop_type = academics }
		# Armed forces relish the fight, everyone else suffers
		every_interest_group = {
			limit = { is_interest_group_type = ig_armed_forces }
			add_modifier = { name = ig_approval_positive_modifier days = short_modifier_time }

	}

	# Option C: Neocolonialism - release with economic extraction
	option = {

		trigger = {
			has_law = law_type:law_neocolonialism
		}

			base = 3

# ============================================================================
# EVENT 14: GREAT POWER COLONIAL POLICY STANCE
# Fires for great powers, allowing them to choose a colonial policy stance.
# This is the mechanism by which GP decolonization pressure becomes a CHOICE.
# ============================================================================

# Event 14: Colonial Policy Review (fires for GPs)
decolonization_events.14 = {
	type = country_event
	placement = root

	event_image = {
		texture = "gfx/event_pictures/diplomatic_meeting.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_diplomacy.dds"

	title = decolonization_events.14.t
	desc = decolonization_events.14.d
	flavor = decolonization_events.14.f

	duration = 3

	trigger = {
		has_technology_researched = decolonization
		country_rank >= rank_value:great_power
		NOR = {
			has_variable = gp_colonial_policy_cooldown
		}
		any_country = {
			NOT = { this = root }
			any_scope_state = {
				is_overseas_colonial_state = yes
			}
		}
	}

	immediate = {
		set_variable = { name = gp_colonial_policy_cooldown days = 1095 }
	}

	# Option A: Condemn colonialism
	option = {
		name = decolonization_events.14.a
		default_option = yes
		ai_chance = {
			base = 3
			modifier = {
				trigger = {
					NOT = {
						any_scope_state = {
							is_overseas_colonial_state = yes
						}
					}
				}
				add = 5
			}
			modifier = {
				trigger = {
					any_scope_state = {
						is_overseas_colonial_state = yes
					}
				}
				add = -5
			}
		}
		set_variable = gp_anti_colonial_stance
		if = {
			limit = { has_variable = gp_pro_colonial_stance }
			remove_variable = gp_pro_colonial_stance
		}
		add_modifier = {
			name = gp_anti_colonial_policy_modifier
			days = 1095
			is_decaying = yes
		}
		# Intelligentsia and trade unions approve - progressive stance
		every_interest_group = {
			limit = { is_interest_group_type = ig_intelligentsia }
			add_modifier = { name = ig_approval_positive_modifier days = short_modifier_time }
		}
		every_interest_group = {
			limit = { is_interest_group_type = ig_trade_unions }
			add_modifier = { name = ig_approval_positive_modifier days = short_modifier_time }
		}
		# Landowners and industrialists lose colonial markets
		every_interest_group = {
			limit = { is_interest_group_type = ig_landowners }
			add_modifier = { name = ig_approval_negative_modifier days = short_modifier_time }
		}
		every_interest_group = {
			limit = { is_interest_group_type = ig_industrialists }
			add_modifier = { name = ig_approval_negative_modifier days = short_modifier_time }
		}
	}

	# Option B: Remain neutral
	option = {
		name = decolonization_events.14.b
		ai_chance = {
			base = 2
		}
		if = {
			limit = { has_variable = gp_anti_colonial_stance }
			remove_variable = gp_anti_colonial_stance
		}
		if = {
			limit = { has_variable = gp_pro_colonial_stance }
			remove_variable = gp_pro_colonial_stance
		}
		# Cautious approach - no strong reactions either way
		add_loyalists = { value = very_small_radicals }
	}

	# Option C: Support colonial interests
	option = {
		name = decolonization_events.14.c
		ai_chance = {
			base = 1
			modifier = {
				trigger = {
					any_scope_state = {
						is_overseas_colonial_state = yes
					}
				}
				add = 5
			}
			modifier = {
				trigger = {
					any_country = {
						NOT = { this = root }
						relations:root >= relations_threshold:amicable
						any_scope_state = {
							is_overseas_colonial_state = yes
						}
					}
				}
				add = 3
			}
		}
		set_variable = gp_pro_colonial_stance
		if = {
			limit = { has_variable = gp_anti_colonial_stance }
			remove_variable = gp_anti_colonial_stance
		}
		add_modifier = {
			name = gp_pro_colonial_policy_modifier
			days = 1095
			is_decaying = yes
		}
		# Landowners and industrialists approve - colonial markets
		every_interest_group = {
			limit = { is_interest_group_type = ig_landowners }
			add_modifier = { name = ig_approval_positive_modifier days = short_modifier_time }
		}
		every_interest_group = {
			limit = { is_interest_group_type = ig_industrialists }
			add_modifier = { name = ig_approval_positive_modifier days = short_modifier_time }
		}
		# Intelligentsia and trade unions dismayed
		every_interest_group = {
			limit = { is_interest_group_type = ig_intelligentsia }
			add_modifier = { name = ig_approval_negative_modifier days = short_modifier_time }
		}
		every_interest_group = {
			limit = { is_interest_group_type = ig_trade_unions }
			add_modifier = { name = ig_approval_negative_modifier days = short_modifier_time }
		}
		add_radicals = { value = small_radicals pop_type = academics }
	}
}
