# Event 1: Colonial Independence Movement Rising
decolonization_events.1 = {
	type = country_event
	placement = root

	event_image = {
		video = "unspecific_ruler_speaking_to_people"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_protest.dds"

	title = decolonization_events.1.t
	desc = decolonization_events.1.d
	flavor = decolonization_events.1.f

	duration = 3

	cooldown = { days = long_modifier_time }

	trigger = {
		NOR = {
			has_modifier = decolonization_negotiations_modifier
			has_modifier = colonial_crackdown_modifier
			has_modifier = peaceful_decolonization_modifier
		}
		has_technology_researched = decolonization
		any_subject_or_below = {
			liberty_desire >= 50
		}
	}

	immediate = {
		random_subject_or_below = {
			limit = { liberty_desire >= 50 }
			save_scope_as = restless_subject
		}
	}

	option = {
		name = decolonization_events.1.a
		default_option = yes
		ai_chance = {
			base = 1
			modifier = {
				trigger = {
					ig:ig_intelligentsia ?= {
						leader = { has_ideology = ideology:ideology_anti_colonialist }
					}
				}
				add = 3
			}
		}
		# Begin negotiations for autonomy — reduce liberty desire
		add_modifier = {
			name = decolonization_negotiations_modifier
			days = normal_modifier_time
			is_decaying = yes
		}
		scope:restless_subject = {
			add_liberty_desire = -15
		}
		# Stability cost of concessions
		add_modifier = { name = colonial_stability_negative_event months = 3 }
	}

	option = {
		name = decolonization_events.1.b
		ai_chance = {
			base = 2
		}
		# Crack down on the movement
		add_modifier = {
			name = colonial_crackdown_modifier
			days = normal_modifier_time
			is_decaying = yes
		}
		scope:restless_subject = {
			add_liberty_desire = 20
		}
		add_radicals = { value = medium_radicals pop_type = academics }
		# Stability from showing strength
		add_modifier = { name = colonial_stability_positive_event months = 2 }
	}

	option = {
		name = decolonization_events.1.c
		ai_chance = {
			base = 1
			modifier = {
				trigger = {
					scope:restless_subject = { liberty_desire >= 75 }
				}
				add = 3
			}
		}
		# Grant them independence
		scope:restless_subject = {
			make_independent = yes
		}
		add_modifier = {
			name = peaceful_decolonization_modifier
			days = long_modifier_time
			is_decaying = yes
		}
		# Stability from releasing pressure
		add_modifier = { name = colonial_stability_positive_event months = 5 }
	}
}

# Event 4: Cultural Reclamation in Former Colony (fires for newly independent countries)
decolonization_events.4 = {
	type = country_event
	placement = root

	event_image = {
		video = "unspecific_ruler_speaking_to_people"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_protest.dds"

	title = decolonization_events.4.t
	desc = decolonization_events.4.d
	flavor = decolonization_events.4.f

	duration = 3

	cooldown = { days = long_modifier_time }

	trigger = {
		NOT = { has_modifier = cultural_reclamation_modifier }
		has_technology_researched = decolonization
		is_subject = no
		# Country has no homeland for its primary culture — likely a former colony
		NOT = {
			any_scope_state = {
				is_homeland_of_country_cultures = ROOT
			}
		}
	}

	immediate = {
	}

	option = {
		name = decolonization_events.4.a
		default_option = yes
		ai_chance = { base = 5 }
		# Embrace the local culture (ties to adopt_local_culture decision)
		# Remove colonial cultures, adopt homeland culture
		every_primary_culture = {
			ROOT = {
				remove_primary_culture = PREV
			}
		}
		capital = {
			every_scope_culture = {
				limit = {
					has_homeland = PREV
				}
				ROOT = {
					add_primary_culture = PREV
				}
			}
		}
		add_modifier = {
			name = cultural_reclamation_modifier
			days = long_modifier_time
			is_decaying = yes
		}
		add_loyalists = { value = medium_radicals }
	}

	option = {
		name = decolonization_events.4.b
		ai_chance = { base = 1 }
		# Maintain the colonial-era cultural identity
		add_radicals = { value = medium_radicals strata = lower }
		add_loyalists = { value = small_radicals strata = upper }
	}
}

# Event 10: Colonial Integration Milestone
decolonization_events.10 = {
	type = country_event
	placement = root

	event_image = {
		video = "unspecific_signed_treaty"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_diplomacy.dds"

	title = decolonization_events.10.t
	desc = decolonization_events.10.d
	flavor = decolonization_events.10.f

	duration = 3

	cooldown = { days = normal_modifier_time }

	trigger = {
		NOT = { has_modifier = colonial_integration_progress_modifier }
		has_technology_researched = decolonization
		# Has at least one overseas colonial state with high acceptance (ready to integrate)
		any_scope_state = {
			NOT = { is_homeland_of_country_cultures = ROOT }
			state_population >= 100000
			is_overseas_from_capital = yes
			NOT = {
				any_scope_pop = {
					pop_acceptance < acceptance_status_4
				}
			}
		}
	}

	immediate = {
		random_scope_state = {
			limit = {
				NOT = { is_homeland_of_country_cultures = ROOT }
				state_population >= 100000
				is_overseas_from_capital = yes
				NOT = {
					any_scope_pop = {
						pop_acceptance < acceptance_status_4
					}
				}
			}
			save_scope_as = integrating_colony
		}
	}

	option = {
		name = decolonization_events.10.a
		default_option = yes
		ai_chance = { base = 5 }
		# Celebrate the integration — grant full homeland status
		scope:integrating_colony = {
			state_region = {
				save_scope_as = colony_state_region
			}
		}
		ROOT = {
			every_primary_culture = {
				add_homeland = scope:colony_state_region
			}
		}
		add_modifier = {
			name = colonial_integration_progress_modifier
			days = normal_modifier_time
			is_decaying = yes
		}
		add_modifier = { name = colonial_stability_positive_event months = 5 }
		add_loyalists = { value = medium_radicals }
		# Integration process causes some domestic disruption
		add_radicals = { value = small_radicals strata = upper }
	}

	option = {
		name = decolonization_events.10.b
		ai_chance = { base = 1 }
		# Maintain the current arrangement — don't rush
		add_loyalists = { value = small_radicals }
		add_modifier = { name = colonial_stability_positive_event months = 2 }
	}
}

# ============================================================================
# EVENT 14: GREAT POWER COLONIAL POLICY STANCE
# Fires for great powers, allowing them to choose a colonial policy stance.
# This is the mechanism by which GP decolonization pressure becomes a CHOICE.
# ============================================================================

# Event 14: Colonial Policy Review (fires for GPs)
decolonization_events.14 = {
	type = country_event
	placement = root

	event_image = {
		video = "unspecific_signed_treaty"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_diplomacy.dds"

	title = decolonization_events.14.t
	desc = decolonization_events.14.d
	flavor = decolonization_events.14.f

	duration = 3

	trigger = {
		has_technology_researched = decolonization
		country_rank >= rank_value:great_power
		# Cooldown — don't fire if we already chose recently
		NOR = {
			has_variable = gp_colonial_policy_cooldown
		}
		# There exists at least one country with overseas colonies
		any_country = {
			NOT = { this = root }
			any_scope_state = {
				is_overseas_colonial_state = yes
			}
		}
	}

	immediate = {
		set_variable = { name = gp_colonial_policy_cooldown days = 1095 }
	}

	option = {
		name = decolonization_events.14.a
		default_option = yes
		ai_chance = {
			base = 3
			# More likely if we have NO colonial holdings ourselves
			modifier = {
				trigger = {
					NOT = {
						any_scope_state = {
							is_overseas_colonial_state = yes
						}
					}
				}
				add = 5
			}
			# Less likely if we have colonies of our own (hypocrisy)
			modifier = {
				trigger = {
					any_scope_state = {
						is_overseas_colonial_state = yes
					}
				}
				add = -5
			}
		}
		# Condemn colonialism
		set_variable = gp_anti_colonial_stance
		if = {
			limit = { has_variable = gp_pro_colonial_stance }
			remove_variable = gp_pro_colonial_stance
		}
		add_modifier = {
			name = gp_anti_colonial_policy_modifier
			days = 1095
			is_decaying = yes
		}
	}

	option = {
		name = decolonization_events.14.b
		ai_chance = {
			base = 2
		}
		# Remain neutral — no active stance
		if = {
			limit = { has_variable = gp_anti_colonial_stance }
			remove_variable = gp_anti_colonial_stance
		}
		if = {
			limit = { has_variable = gp_pro_colonial_stance }
			remove_variable = gp_pro_colonial_stance
		}
	}

	option = {
		name = decolonization_events.14.c
		ai_chance = {
			base = 1
			# More likely if we have colonial holdings ourselves
			modifier = {
				trigger = {
					any_scope_state = {
						is_overseas_colonial_state = yes
					}
				}
				add = 5
			}
			# More likely if we're allied with a colonial power
			modifier = {
				trigger = {
					any_country = {
						NOT = { this = root }
						relations:root >= relations_threshold:amicable
						any_scope_state = {
							is_overseas_colonial_state = yes
						}
					}
				}
				add = 3
			}
		}
		# Support colonial interests
		set_variable = gp_pro_colonial_stance
		if = {
			limit = { has_variable = gp_anti_colonial_stance }
			remove_variable = gp_anti_colonial_stance
		}
		add_modifier = {
			name = gp_pro_colonial_policy_modifier
			days = 1095
			is_decaying = yes
		}
	}
}
