
namespace = te_construction_market_events

#create a daily pulse event
te_construction_market_events.1 = {
	type = country_event
	hidden = yes

	trigger = {
		OR = {
			NOT = {
				has_global_variable = te_construction_market_last_pulse_date
			}
			te_construction_market_days_since_last_pulse >= 1
		}
	}

	immediate = {

		if = {
			limit = {
				NOT = {
					has_global_variable = te_construction_market_pulse_count
				}
			}
			set_global_variable = {
				name = te_construction_market_pulse_count
				value = 0
			}
		}
		change_global_variable = {
			name = te_construction_market_pulse_count
			add = 1
		}

		set_global_variable = {
			name = te_construction_market_last_pulse_date
			value = game_date
		}
		random_country = {
			trigger_event = {
				id = te_construction_market_events.2
				days = 1
			}
			trigger_event = {
				id = te_construction_market_events.2
				days = 2
			}
		}
		random_country = {
			trigger_event = {
				id = te_construction_market_events.2
				days = 1
			}
			trigger_event = {
				id = te_construction_market_events.2
				days = 2
			}
		}
		if = {
			limit = {
				te_construction_market_pulse_count_modulo_7 = 0
			}
			every_country = {
				trigger_event = {
					id = te_construction_market_events.3
				}
			}
		}
	}
}

#helper event to trigger the main event
te_construction_market_events.2 = {
	type = country_event
	hidden = yes

	immediate = {
		trigger_event = {
			id = te_construction_market_events.1
		}
	}
}
