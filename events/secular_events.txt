		}

			base = 1
		}
		add_loyalists = {
			value = very_small_radicals
			interest_group = ig:ig_devout
		}
		add_radicals = {
			value = small_radicals
			pop_type = academics
		}
	}

	# Use it to push secularism
	option = {
		name = secular_events.1.c
		ai_chance = {
			base = 2
		}
		add_modifier = {
			name = secular_momentum_modifier
			days = normal_modifier_time
			is_decaying = yes

			limit = {
				OR = {
					is_political_movement_type = movement_religious_majority

			add_modifier = {
				name = secular_momentum_religious_movement_modifier

			}
		}

# Event 4: State Religion Debate
# Pressure mounts to formally disestablish the state church.
secular_events.4 = {
	type = country_event
	placement = root

	event_image = {
		texture = "gfx/event_pictures/parliamentary_debate.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	title = secular_events.4.t
	desc = secular_events.4.d
	flavor = secular_events.4.f

	duration = 3
	cooldown = { days = long_modifier_time }

	trigger = {
		has_technology_researched = decline_of_organized_religion
		has_law = law_type:law_ministry_of_religion
		NOT = { has_global_variable = state_religion_debate_happened }
	}

	immediate = {
		set_global_variable = state_religion_debate_happened
		random_interest_group = {
			limit = {
				is_interest_group_type = ig_devout
			}
			save_scope_as = devout_ig
		}
	}

	# Support disestablishment
	option = {
		name = secular_events.4.a
		default_option = yes
		ai_chance = {
			base = 2
		}
		add_modifier = {
			name = disestablishment_push_modifier
			days = normal_modifier_time
			is_decaying = yes
		}
		every_political_movement = {
			limit = {
				OR = {
					is_political_movement_type = movement_religious_majority
					is_political_movement_type = movement_religious_minority
				}
			}
			add_modifier = {
				name = disestablishment_religious_movement_modifier
				days = normal_modifier_time
				is_decaying = yes
			}
		}
		scope:devout_ig = {
			add_modifier = {
				name = disestablishment_anger_modifier
				days = normal_modifier_time
				is_decaying = yes
			}
		}
		add_loyalists = {
			value = small_radicals
			pop_type = academics
		}
	}

	# Defend the state church
	option = {
		name = secular_events.4.b
		ai_chance = {
			base = 2
			modifier = {
				trigger = {
					ig:ig_devout ?= { is_in_government = yes }
				}
				add = 3
			}
		}
		add_loyalists = {
			value = small_radicals
			interest_group = ig:ig_devout
		}
		add_radicals = {
			value = small_radicals
			pop_type = academics
		}
	}

	# Form a committee to study the question
	option = {
		name = secular_events.4.c
		ai_chance = {
			base = 3
		}
		add_radicals = {
			value = very_small_radicals
			pop_type = academics
		}
		add_radicals = {
			value = very_small_radicals
			interest_group = ig:ig_devout
		}
	}
}
