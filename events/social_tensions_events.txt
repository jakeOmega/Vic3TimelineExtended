# Event 1: Domestic Terror Attack
# A bombing or mass-casualty attack by radical elements. Tied to terrorism_and_anti_terrorism tech era.
social_tensions_events.1 = {
	type = country_event
	placement = ROOT

	title = social_tensions_events.1.t
	desc = {
		first_valid = {
			triggered_desc = { desc = social_tensions_events.1.d_cyber trigger = { has_technology_researched = cyber_warfare } }
			triggered_desc = { desc = social_tensions_events.1.d_modern trigger = { has_technology_researched = terrorism_and_anti_terrorism } }
			desc = social_tensions_events.1.d_base
		}
	}
	flavor = social_tensions_events.1.f

	event_image = { texture = "gfx/event_pictures/terrorist_attack.dds" }
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_fire.dds"

	duration = 3
	cooldown = { days = long_modifier_time }

	trigger = {
		country_rank >= rank_value:major_power
		OR = {
			has_technology_researched = dynamite
			has_technology_researched = terrorism_and_anti_terrorism
		}
		NOR = {
			has_modifier = terror_crackdown_modifier
			has_modifier = terror_resilience_modifier
			has_modifier = terror_surveillance_modifier
			has_modifier = terror_preempted_modifier
		}
		# Strong intelligence ministry (level 5) prevents most attacks
		NOT = {
			has_intelligence_ministry_5 = yes
		}
		# More likely in countries with political turmoil
		OR = {
			radical_fraction = { value >= 0.1 }
			is_at_war = yes
			any_political_movement = { political_movement_radicalism >= 25 }
		}
	}

	immediate = {
		random_scope_state = {
			limit = {
				is_incorporated = yes
				state_population > 100000
			}
			save_scope_as = terror_target_state
		}
	}

	option = { # Security crackdown
		name = social_tensions_events.1.a
		default_option = yes
		add_modifier = { name = terror_crackdown_modifier days = long_modifier_time is_decaying = yes }
		add_radicals = { value = small_radicals strata = lower }
		add_loyalists = { value = small_radicals pop_type = officers }
		ai_chance = {
			base = 5
			modifier = { trigger = { has_law = law_type:law_militarized_police } add = 5 }
		}
	}

	option = { # Rally national resilience
		name = social_tensions_events.1.b
		add_modifier = { name = terror_resilience_modifier days = normal_modifier_time is_decaying = yes }
		add_loyalists = { value = medium_radicals }
		ai_chance = {
			base = 5
			modifier = { trigger = { has_law = law_type:law_dedicated_police } add = 3 }
		}
	}

	option = { # Expand surveillance
		name = social_tensions_events.1.c
		trigger = {
			has_technology_researched = automated_surveillance
		}
		highlighted_option = yes
		custom_tooltip = social_tensions_surveillance_tech_tt
		add_modifier = { name = terror_surveillance_modifier days = long_modifier_time is_decaying = yes }
		add_radicals = { value = very_small_radicals strata = lower }
		add_loyalists = { value = small_radicals strata = upper }
		ai_chance = {
			base = 3
			modifier = { trigger = { has_law = law_type:law_intrusive_surveillance } add = 5 }
		}
	}

	option = { # Intelligence-led preemption - best outcome, requires strong institutions
		name = social_tensions_events.1.d
		trigger = {
			has_intelligence_ministry_3 = yes
		}
		highlighted_option = yes
		custom_tooltip = social_tensions_intel_ministry_3_tt
		add_modifier = { name = terror_preempted_modifier days = long_modifier_time is_decaying = yes }
		add_loyalists = { value = small_radicals }
		ai_chance = {
			base = 8
		}

		ig_approval_effect = { IG = ig_armed_forces MODIFIER = ig_approval_positive_modifier DAYS = normal_modifier_time }
		ai_chance = {
			base = 3
			modifier = { trigger = { has_law = law_type:law_militarized_police } add = 5 }
			modifier = { trigger = { has_law = law_type:law_punishment_focused_criminal_justice } add = 3 }
		}
	}

	option = { # Reform policing practices
		name = social_tensions_events.5.b
		default_option = yes
		add_modifier = { name = police_reform_modifier days = long_modifier_time is_decaying = yes }
