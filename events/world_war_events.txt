# Event 1: Ideological Confrontation â€” rival ideological bloc demands concessions

	placement = root

	icon = "gfx/interface/icons/event_icons/event_diplomacy.dds"

	desc = world_war_events.1.d
	flavor = world_war_events.1.f

	duration = 3

	trigger = {
		NOT = { has_variable = ww_active_phase }
		ww_ideological_tension >= 30
		any_rival_country = {
			country_rank >= rank_value:great_power

		random_country = {
			limit = {
				has_diplomatic_pact = { who = ROOT type = rivalry }

			}
		}

			country = scope:ideological_rival
			value = -30
		}

			value = medium_radicals

# WAR OUTBREAK EVENT (5) â€” Triggers the actual world war
# Fired from the on_action when tension reaches critical
# ============================================================================

world_war_events.5 = {
	type = country_event
	placement = root

	event_image = { video = "europenorthamerica_before_the_battle" }
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_military.dds"

	title = world_war_events.5.t
	desc = world_war_events.5.d
	flavor = world_war_events.5.f

	duration = 3

	trigger = {
		NOT = { has_variable = ww_active_phase }
		world_war_conditions_met = yes
		ww_ideological_tension >= 70
	}

	cooldown = { days = long_modifier_time }

	immediate = {
		# Find the primary rival for the war
		random_country = {
			limit = {
				has_diplomatic_pact = { who = ROOT type = rivalry }
				country_rank >= rank_value:great_power
				ROOT = { country_has_opposed_ideology = { TARGET = prev } }
			}
			save_scope_as = war_target
		}
		# Mark both sides
		set_variable = ww_active_phase
		set_variable = ww_aggressor_side
		set_variable = { name = ww_years_elapsed value = 0 }
		set_variable = { name = ww_months_counter value = 0 }
		scope:war_target = {
			set_variable = ww_active_phase
			set_variable = ww_defender_side
			set_variable = { name = ww_years_elapsed value = 0 }
			set_variable = { name = ww_months_counter value = 0 }
		}
		# Mark ideologically aligned great powers
		every_country = {
			limit = {
				country_rank >= rank_value:great_power
				NOT = { this = root }
				NOT = { this = scope:war_target }
				root = { country_has_opposed_ideology = { TARGET = prev } }
			}
			set_variable = ww_active_phase
			set_variable = ww_defender_side
		}
		every_country = {
			limit = {
				country_rank >= rank_value:great_power
				NOT = { this = root }
				NOT = { this = scope:war_target }
				NOT = { has_variable = ww_defender_side }
				scope:war_target = { country_has_opposed_ideology = { TARGET = prev } }
			}
			set_variable = ww_active_phase
			set_variable = ww_aggressor_side
		}
	}

	# Only option: war has begun
	option = {
		name = world_war_events.5.a
		default_option = yes
		ai_chance = { base = 10 }
		# Create the diplomatic play against the rival
		create_diplomatic_play = {
			target_country = scope:war_target
			type = dp_contain_threat
			escalation = 90
		}
		# Remove leadup modifiers and apply war modifiers
		if = {
			limit = { has_modifier = ww_rising_tensions_modifier }
			remove_modifier = ww_rising_tensions_modifier
		}
		if = {
			limit = { has_modifier = ww_appeasement_modifier }
			remove_modifier = ww_appeasement_modifier
		}
		# National rally effect
		add_loyalists = {
			value = large_radicals
			pop_type = soldiers
		}
		add_loyalists = {
			value = medium_radicals
			strata = lower
		}
	}
}

	title = world_war_events.20.t
	desc = world_war_events.20.d
	flavor = world_war_events.20.f

	duration = 3

	trigger = {
		has_variable = ww_active_phase
		is_at_war = no
		country_rank >= rank_value:major_power
		NOT = { has_modifier = ww_late_entry_cooldown }
		any_country = {
			has_variable = ww_aggressor_side
			is_at_war = yes
		}
	}

	cooldown = { days = normal_modifier_time }

	immediate = {
		# Find the main aggressor still at war
		random_country = {
			limit = {
				has_variable = ww_aggressor_side
				is_at_war = yes
				country_rank >= rank_value:great_power
			}
			save_scope_as = ww_main_aggressor
		}
		# Find a defender we could ally with
		random_country = {
			limit = {
				has_variable = ww_defender_side
				is_at_war = yes
				country_rank >= rank_value:great_power
			}
			save_scope_as = ww_main_defender
		}
	}

	# Option A: Join the war against the aggressors â€” full commitment
	option = {
		name = world_war_events.20.a
		default_option = yes
		ai_chance = {
			base = 2
			modifier = {
				trigger = {
					country_is_democratic = yes
					scope:ww_main_aggressor = { country_is_fascist = yes }
				}
				add = 5
			}
			modifier = {
				trigger = {
					country_is_communist = yes
					scope:ww_main_aggressor = { country_is_fascist = yes }
				}
				add = 5
			}
			modifier = {
				trigger = { country_rank >= rank_value:great_power }
				add = 3
			}
		}
		# Join the existing war on the defender's side
		scope:ww_main_defender = {
			every_scope_war = {
				limit = {
					is_war_participant = scope:ww_main_aggressor
				}
				join_war = {
					target = root
					side = scope:ww_main_defender
				}
			}
		}
		set_variable = ww_defender_side
		add_modifier = {
			name = ww_fresh_forces_modifier
			days = normal_modifier_time
			is_decaying = yes
		}
		if = {
			limit = { country_is_democratic = yes }
			add_modifier = {
				name = ww_arsenal_of_democracy_modifier
				days = normal_modifier_time
				is_decaying = yes
			}
		}
		add_loyalists = {
			value = large_radicals
			pop_type = soldiers
		}
		add_modifier = {
			name = ww_late_entry_cooldown
			days = 365
			is_decaying = yes
		}
	}

	# Option B: Provide lend-lease support only â€” stay out of direct combat
	option = {
		name = world_war_events.20.b
		ai_chance = {
			base = 3
			modifier = {
				trigger = { country_rank < rank_value:great_power }
				add = 3
			}
		}
		add_modifier = {
			name = ww_lend_lease_modifier
			days = long_modifier_time
			is_decaying = yes
		}
		ig_industrialists_approval_positive = yes
		add_modifier = {

		}
	}
