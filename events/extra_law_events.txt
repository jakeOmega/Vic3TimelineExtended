# ======================================================================
#  1 – The Bankers’ Lobby  (Financial Regulation)
# ======================================================================
extra_law_events.1 = {
	type = country_event
	placement = root

	title = extra_law_events.1.t
	desc = extra_law_events.1.d
	flavor = extra_law_events.1.f

	category = enactment

	event_image = { texture = "gfx/event_pictures/parliamentary_debate_heated.dds" }

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_trade.dds"

	duration = 3

	cooldown = { days = normal_modifier_time }

	trigger = {
		OR = {
			is_enacting_law = law_type:law_prudential_narrow_banking
			is_enacting_law = law_type:law_central_bank_independence
			is_enacting_law = law_type:law_directed_credit_development_banks
		}
		NOR = {
			has_modifier = finreg_bankers_lobby_advance
			has_modifier = finreg_bankers_lobby_cautious
		}
		any_interest_group = {
			ig_counts_as_marginal = no
			law_stance = {
				law = owner.currently_enacting_law.type
				value < neutral
			}
		}
	}

	immediate = {
		currently_enacting_law = {
			save_scope_as = current_law_scope
		}
		random_interest_group = {
			limit = {
				ig_counts_as_marginal = no
				law_stance = {
					law = owner.currently_enacting_law.type
					value < neutral
				}
			}
			save_scope_as = opposing_ig
		}
	}

	cancellation_trigger = {
		NOT = { currently_enacting_law = scope:current_law_scope }
	}

	option = { # Push through despite the lobby
		name = extra_law_events.1.a
		default_option = yes
		add_enactment_modifier = {
			name = finreg_bankers_lobby_advance
		}
		scope:opposing_ig = {
			add_modifier = {
				name = ig_approval_negative_modifier
				days = normal_modifier_time
			}
		}
	}

	option = { # Hear their concerns to build broader support
		name = extra_law_events.1.b
		add_enactment_modifier = {
			name = finreg_bankers_lobby_cautious
		}
		add_loyalists = {
			value = small_radicals
			strata = upper
		}
	}
}

#   societal upheaval caused by laws like communal child-rearing,
#   state atheism, or collectivized agriculture in a non-
#   revolutionary polity.
#
#   Severity scales with government type:
#     Enhanced  (20yr) – conservative monarchy/theocracy with
#                         autocratic or oligarchic representation
#     Standard  (10yr) – most other governments
#     Reduced   (5yr)  – technocracy (efficient top-down change)
#
#   Law-specific effects target the groups most affected:
#     Collectivized agriculture → farmers, peasants (naturally
#       scales with agrarian pop share: industrial countries
#       with few farmers get proportionally less backlash)
#     Command economy → capitalists, shopkeepers; lighter under
#       technocracy
#     State atheism → clergymen, devout IG
#     Communal child rearing → broadest backlash across society
#     Cooperative ownership → capitalists, shopkeepers
#
#   Option A: Stand firm: governance-scaled modifier + law-specific
#             IG and pop type reactions
#   Option B: Reverse course: law reverts to a moderate alternative
# ======================================================================

	type = country_event
	placement = root

			triggered_desc = {
				desc = extra_law_events.36.d_atheism
				trigger = { var:radical_law_backlash_type = 2 }
			}
			triggered_desc = {
				desc = extra_law_events.36.d_cooperative
				trigger = { var:radical_law_backlash_type = 3 }
			}
			triggered_desc = {
				desc = extra_law_events.36.d_command
				trigger = { var:radical_law_backlash_type = 4 }
			}

				desc = extra_law_events.36.d_collectivized

			add_modifier = {
				name = radical_law_backlash
				is_decaying = yes
				days = normal_modifier_time
			}

		else = {

				value = very_large_radicals
				strata = upper
			}
			add_radicals = {
				value = large_radicals
				strata = middle
			}
			add_radicals = {
				value = medium_radicals
				pop_type = clergymen
			}
		}

		# State atheism (type 2): devout are primary opponents
		if = {
			limit = { var:radical_law_backlash_type = 2 }
			ig:ig_devout ?= {
				add_modifier = {
					name = ig_approval_very_negative_modifier
					days = long_modifier_time
				}

			ig:ig_landowners ?= {

			ig:ig_industrialists ?= {
				add_modifier = {
					name = ig_approval_very_negative_modifier
					days = normal_modifier_time
				}
			}
			ig:ig_petty_bourgeoisie ?= {

				pop_type = shopkeepers
			}

				value = large_radicals
				strata = upper
			}

			ig:ig_industrialists ?= {
				add_modifier = {
					name = ig_approval_very_negative_modifier
					days = normal_modifier_time
				}
			}
			add_radicals = {
				value = very_large_radicals
				pop_type = capitalists
			}
			add_radicals = {
				value = large_radicals
				pop_type = shopkeepers
			}
			# Technocracy: lighter broad resistance (technocrats planned this)
			if = {
				limit = { has_law = law_type:law_technocracy }
				add_radicals = {

					strata = upper
