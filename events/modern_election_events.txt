# =============================================================
# 1. VIRAL FIRESTORM
# A party's social media post goes massively viral, dominating the campaign
# =============================================================
modern_election_events.1 = {
	type = country_event
	placement = ROOT

	title = modern_election_events.1.t
	desc = modern_election_events.1.d
	flavor = modern_election_events.1.f

	event_image = { texture = "gfx/event_pictures/social_media_crisis.dds" }
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_election.dds"

	duration = 3
	cooldown = { days = very_long_modifier_time }

	trigger = {
		in_election_campaign = yes
		NOT = { has_variable = election_event_cooldown }
		NOT = { has_variable = mute_election_events }
		NOT = { has_law_or_variant = law_type:law_single_party_state }
		has_technology_researched = social_media
		any_interest_group = {
			has_party = yes
			is_in_government = no
		}
	}

	immediate = {
		set_variable = { name = election_event_cooldown months = election_event_cooldown_months }
		random_interest_group = {
			limit = { has_party = yes is_in_government = yes }
			party = { save_scope_as = govt_party }
		}
		random_interest_group = {
			limit = { has_party = yes is_in_government = no }
			party = { save_scope_as = opp_party }
		}
	}

	cancellation_trigger = { in_election_campaign = no }

	option = { # Amplify the message
		name = modern_election_events.1.a
		default_option = yes
		if = {
			limit = { exists = scope:govt_party }
			scope:govt_party = { add_momentum = momentum_medium }
		}
		if = {
			limit = { exists = scope:opp_party }
			scope:opp_party = { add_momentum = momentum_small_decrease }
		}
		ai_chance = { base = 5 }
	}

	option = { # Let the opposition have their moment
		name = modern_election_events.1.b
		if = {
			limit = { exists = scope:opp_party }
			scope:opp_party = { add_momentum = momentum_medium }
		}
		add_loyalists = { value = small_radicals }
		ai_chance = { base = 3 }
	}

	option = { # Call for digital restraint
		name = modern_election_events.1.c
		add_modifier = { name = me_digital_decorum_modifier days = short_modifier_time }
		ai_chance = { base = 2 }
	}
}
