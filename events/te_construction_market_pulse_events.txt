# Construction-market subsystem.
# Based on the third-party "Free Market Construction" mod. See README.md.

namespace = te_construction_market_pulse_events

# Yearly pulse entry point — guards on a "days since last pulse" check so that
# only one country's on_yearly_pulse_country actually advances the global tick.
# Schedules a couple of relay copies in random countries to give the engine a
# chance to fire the body even if the original origin country is decentralized
# or otherwise skipped. Every 7th tick fans out to every country to run the
# weekly pulse body.
te_construction_market_pulse_events.1 = {
	type = country_event
	hidden = yes

	trigger = {
		OR = {
			NOT = {
				has_global_variable = te_construction_market_last_pulse_date
			}
			te_construction_market_days_since_last_pulse >= 1
		}
	}

	immediate = {

		if = {
			limit = {
				NOT = {
					has_global_variable = te_construction_market_pulse_count
				}
			}
			set_global_variable = {
				name = te_construction_market_pulse_count
				value = 0
			}
		}
		change_global_variable = {
			name = te_construction_market_pulse_count
			add = 1
		}

		set_global_variable = {
			name = te_construction_market_last_pulse_date
			value = game_date
		}
		random_country = {
			trigger_event = {
				id = te_construction_market_pulse_events.2
				days = 1
			}
			trigger_event = {
				id = te_construction_market_pulse_events.2
				days = 2
			}
		}
		random_country = {
			trigger_event = {
				id = te_construction_market_pulse_events.2
				days = 1
			}
			trigger_event = {
				id = te_construction_market_pulse_events.2
				days = 2
			}
		}
		if = {
			limit = {
				te_construction_market_pulse_count_modulo_7 = 0
			}
			every_country = {
				trigger_event = {
					id = te_construction_market_pulse_events.3
				}
			}
		}
	}
}

# Relay event — bounces back into pulse_events.1 from a different country/day
# offset to make sure the daily-pulse body has a chance to run even when the
# origin country is decentralized or otherwise skips the on_yearly_pulse hook.
te_construction_market_pulse_events.2 = {
	type = country_event
	hidden = yes

	immediate = {
		trigger_event = {
			id = te_construction_market_pulse_events.1
		}
	}
}

# Weekly pulse body — runs every ~7 daily ticks for every country. Recomputes
# the price stabilizer, the public/private split modifier, and the AI buy
# target.
te_construction_market_pulse_events.3 = {
	type = country_event
	hidden = yes

	immediate = {
		te_construction_market_on_weekly_pulse = yes
	}
}
