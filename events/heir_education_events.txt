namespace = heir_education_events

# ─── Event 1: A Promising Pupil ───
# Fires when the heir shows notable talent. Gives a bonus investment point.
heir_education_events.1 = {
	type = country_event
	placement = ROOT

	title = heir_education_events.1.t
	desc = heir_education_events.1.d
	flavor = heir_education_events.1.f

	event_image = { video = "europenorthamerica_capitalists_meeting" }
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	duration = 3

	trigger = {
		has_law = law_type:law_monarchy
		exists = heir
		has_variable = heir_ed_total
		var:heir_ed_total >= 2
	}

	immediate = {
	}

	# Option A: "Remarkable progress!"
	# Grants a bonus point in whichever trait focus is active (or admin by default)
	option = {
		name = heir_education_events.1.a
		default_option = yes
		if = {
			limit = { has_variable = heir_ed_focus_military }
			custom_tooltip = {
				text = HEIR_ED_MILITARY_PLUS_1
				change_variable = { name = heir_ed_military add = 1 }
			}
		}
		else_if = {
			limit = { has_variable = heir_ed_focus_diplo }
			custom_tooltip = {
				text = HEIR_ED_DIPLO_PLUS_1
				change_variable = { name = heir_ed_diplo add = 1 }
			}
		}
		else = {
			custom_tooltip = {
				text = HEIR_ED_ADMIN_PLUS_1
				change_variable = { name = heir_ed_admin add = 1 }
			}
		}
		custom_tooltip = {
			text = HEIR_ED_TOTAL_PLUS_1
			change_variable = { name = heir_ed_total add = 1 }
		}
		ai_chance = { base = 10 }
	}

	# Option B: "We should broaden the curriculum."
	# Grants a point in each trait category but costs innovation and authority
	option = {
		name = heir_education_events.1.b
		custom_tooltip = {
			text = HEIR_ED_ADMIN_PLUS_1
			change_variable = { name = heir_ed_admin add = 1 }
		}
		custom_tooltip = {
			text = HEIR_ED_DIPLO_PLUS_1
			change_variable = { name = heir_ed_diplo add = 1 }
		}
		custom_tooltip = {
			text = HEIR_ED_MILITARY_PLUS_1
			change_variable = { name = heir_ed_military add = 1 }
		}
		custom_tooltip = {
			text = HEIR_ED_TOTAL_PLUS_3
			change_variable = { name = heir_ed_total add = 3 }
		}
		add_modifier = { name = heir_education_innovation_cost days = 180 is_decaying = yes }
		add_radicals = {
			value = very_small_radicals
			strata = upper
		}
		ai_chance = { base = 5 }
	}
}

# ─── Event 2: A Difficult Student ───
# The heir resists their education. Player chooses response.
heir_education_events.2 = {
	type = country_event
	placement = ROOT

	title = heir_education_events.2.t
	desc = heir_education_events.2.d
	flavor = heir_education_events.2.f

	event_image = { video = "votp_conspiring" }
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	duration = 3

	trigger = {
		has_law = law_type:law_monarchy
		exists = heir
		has_variable = heir_ed_total
		var:heir_ed_total >= 1
	}

	immediate = {
	}

	# Option A: "Discipline builds character."
	# Persevere with current education: keeps progress but creates unrest
	option = {
		name = heir_education_events.2.a
		default_option = yes
		custom_tooltip = {
			text = HEIR_ED_TOTAL_PLUS_1
			change_variable = { name = heir_ed_total add = 1 }
		}
		if = {
			limit = { has_variable = heir_ed_focus_military }
			custom_tooltip = {
				text = HEIR_ED_MILITARY_PLUS_1
				change_variable = { name = heir_ed_military add = 1 }
			}
		}
		else_if = {
			limit = { has_variable = heir_ed_focus_diplo }
			custom_tooltip = {
				text = HEIR_ED_DIPLO_PLUS_1
				change_variable = { name = heir_ed_diplo add = 1 }
			}
		}
		else_if = {
			limit = { has_variable = heir_ed_focus_admin }
			custom_tooltip = {
				text = HEIR_ED_ADMIN_PLUS_1
				change_variable = { name = heir_ed_admin add = 1 }
			}
		}
		add_radicals = {
			value = small_radicals
			strata = lower
		}
		ai_chance = { base = 8 }
	}

	# Option B: "Perhaps a lighter touch."
	# Ease off: lose progress but gain stability
	option = {
		name = heir_education_events.2.b
		custom_tooltip = {
			text = HEIR_ED_TOTAL_MINUS_1
			if = {
				limit = { has_variable = heir_ed_total var:heir_ed_total >= 1 }
				change_variable = { name = heir_ed_total add = -1 }
			}
		}
		add_loyalists = {
			value = small_radicals
			strata = lower
		}
		ai_chance = { base = 5 }
	}

	# Option C: "Let the child find their own path."
	# Random bonus in an unexpected area, but the lack of structure costs innovation
	option = {
		name = heir_education_events.2.c
		custom_tooltip = {
			text = HEIR_ED_RANDOM_PLUS_1
			random_list = {
				33 = { change_variable = { name = heir_ed_admin add = 1 } }
				33 = { change_variable = { name = heir_ed_diplo add = 1 } }
				34 = { change_variable = { name = heir_ed_military add = 1 } }
			}
		}
		custom_tooltip = {
			text = HEIR_ED_TOTAL_PLUS_1
			change_variable = { name = heir_ed_total add = 1 }
		}
		add_modifier = { name = heir_education_innovation_cost days = 60 is_decaying = yes }
		ai_chance = { base = 3 }
	}
}

# ─── Event 3: Foreign Correspondence ───
# An allied nation offers educational exchange. Requires an alliance.
heir_education_events.3 = {
	type = country_event
	placement = ROOT

	title = heir_education_events.3.t
	desc = heir_education_events.3.d
	flavor = heir_education_events.3.f

	event_image = { video = "unspecific_signed_contract" }
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_map.dds"

	duration = 3

	trigger = {
		has_law = law_type:law_monarchy
		exists = heir
		has_variable = heir_ed_total
		var:heir_ed_total >= 3
		any_country = {
			has_treaty_alliance_with = { TARGET = ROOT }
		}
	}

	immediate = {
		random_country = {
			limit = {
				has_treaty_alliance_with = { TARGET = ROOT }
			}
			save_scope_as = ally_country
		}
	}

	# Option A: "Send the heir abroad."
	# Big diplo bonus + progressive shift, but costs relations and innovation
	option = {
		name = heir_education_events.3.a
		default_option = yes
		custom_tooltip = {
			text = HEIR_ED_DIPLO_PLUS_2
			change_variable = { name = heir_ed_diplo add = 2 }
		}
		custom_tooltip = {
			text = HEIR_ED_TOTAL_PLUS_2
			change_variable = { name = heir_ed_total add = 2 }
		}
		# Exposure to foreign ideas pushes slightly progressive
		custom_tooltip = {
			text = HEIR_ED_IDEOLOGY_PROGRESSIVE
			change_variable = { name = heir_ed_ideology add = 1 }
		}
		add_modifier = { name = heir_education_innovation_cost days = 120 is_decaying = yes }
		add_radicals = {
			value = very_small_radicals
			strata = upper
		}
		ai_chance = { base = 8 }
	}

	# Option B: "The heir should remain here."
	# No diplo bonus, but gains a point in current focus and conservative stability
	option = {
		name = heir_education_events.3.b
		if = {
			limit = { has_variable = heir_ed_focus_military }
			custom_tooltip = {
				text = HEIR_ED_MILITARY_PLUS_1
				change_variable = { name = heir_ed_military add = 1 }
			}
		}
		else_if = {
			limit = { has_variable = heir_ed_focus_admin }
			custom_tooltip = {
				text = HEIR_ED_ADMIN_PLUS_1
				change_variable = { name = heir_ed_admin add = 1 }
			}
		}
		else = {
			custom_tooltip = {
				text = HEIR_ED_ADMIN_PLUS_1
				change_variable = { name = heir_ed_admin add = 1 }
			}
		}
		custom_tooltip = {
			text = HEIR_ED_TOTAL_PLUS_1
			change_variable = { name = heir_ed_total add = 1 }
		}
		custom_tooltip = {
			text = HEIR_ED_IDEOLOGY_CONSERVATIVE
			change_variable = { name = heir_ed_ideology add = -1 }
		}
		add_loyalists = {
			value = small_radicals
			strata = upper
		}
		ai_chance = { base = 5 }
	}
}
