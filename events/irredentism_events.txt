# ============================================================================
# IRREDENTIST PRESSURE — same-culture-with-claims event suite
#
# Dispatched from on_monthly_pulse_country via irredentism_events_on_action
# in common/on_actions/extra_on_actions.txt. All events fire under any
# active driver; AI weights tilt-aware via irredentist_war_tilt.
#
# Event 1  "The Diaspora Calls"          (unifier, primary)
# Event 2  "[Other] Demands Our Lands"   (target, follow-up to Event 1 option C)
# Event 3  "An Opportunity for Union"    (unifier, peaceful proposal)
# Event 4  "A Sister Nation Asks Us"     (target, follow-up to Event 3 option A)
#
# Events 5a/5b (bloc-leader approval) live in Phase 5.
#
# See /home/jakef/.claude/plans/can-we-make-a-sparkling-pine.md.
# ============================================================================

# ============================================================================
# Event 1: "The Diaspora Calls"
# Fires on a unifier with at least one strong candidate (pressure score >= 4).
# Picks a same-culture target the unifier holds claims on, lets the player
# choose how to push: stoke nationalism, fund cultural ties, demand a state
# back, or bide time.
# ============================================================================
irredentism.1 = {
	type = country_event
	placement = root

	event_image = { video = "europenorthamerica_native_american" }

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	title = irredentism.1.t
	desc = irredentism.1.d
	flavor = irredentism.1.f

	duration = 3

	cooldown = { years = 8 }

	trigger = {
		# System gates
		has_technology_researched = decolonization
		irredentist_unifier_can_act = yes
		NOT = { is_country_type = decentralized }
		# At least one valid candidate exists: shares a primary culture, owns a
		# claimed home state still populated by our co-ethnics, and is reachable.
		any_country = {
			NOT = { THIS = ROOT }
			NOT = { is_country_type = decentralized }
			irr_target_same_nation = yes
			any_scope_state = {
				irr_claimed_populated_home_state = yes
			}
			irr_target_reachable = yes
		}
	}

	immediate = {
		save_scope_as = irr_unifier
		# Pick a same-culture, claim-holding, reachable candidate whose claimed
		# home state still holds our co-ethnics (mirrors the trigger gate above).
		random_country = {
			limit = {
				NOT = { THIS = ROOT }
				NOT = { is_country_type = decentralized }
				irr_target_same_nation = yes
				any_scope_state = {
					irr_claimed_populated_home_state = yes
				}
				irr_target_reachable = yes
			}
			save_scope_as = target_country
		}
		# Save the specific culture shared with the target -- the diaspora the
		# event text refers to. random_primary_culture so the rendered culture is
		# always one that actually drove the match (a multi-culture unifier may
		# share only one of its primaries with the target).
		random_primary_culture = {
			limit = { is_primary_culture_of = scope:target_country }
			save_scope_as = irr_shared_culture
		}
		# Pick the specific state we'd demand back if the player picks option C --
		# a claimed home state still populated by our co-ethnics.
		scope:target_country = {
			random_scope_state = {
				limit = {
					irr_claimed_populated_home_state = yes
				}
				save_scope_as = irr_demanded_state
			}
		}
		# Also pick an unclaimed home-state for the "stoke" option, if any. Save
		# the state_region too so Option A can add the claim at region scope --
		# the vanilla idiom (s:STATE_X = { add_claim }) that renders in the
		# option tooltip, rather than a nested state->region scope hop.
		scope:target_country = {
			random_scope_state = {
				limit = {
					is_homeland_of_country_cultures = ROOT
					NOT = { has_claim_by = ROOT }
				}
				state_region = { save_scope_as = irr_unclaimed_home_region }
			}
		}
	}

	# --- Option A: Stoke irredentism ---
	# Only offered when we found a co-ethnic homeland the target owns that we
	# don't already claim -- otherwise "stoke for more claims" has nothing to
	# claim and the option would read as a broken promise.
	option = {
		name = irredentism.1.a
		trigger = { exists = scope:irr_unclaimed_home_region }
		ai_chance = {
			base = 1
			modifier = {
				trigger = { irredentist_war_tilt >= 2 }
				add = 4
			}
			modifier = {
				trigger = { irredentist_war_tilt >= 5 }
				add = 4
			}
			modifier = {
				trigger = { irredentist_war_tilt < 0 }
				add = -1
			}
		}
		# Claim the unclaimed home state (region scope so the tooltip renders the
		# gained claim). Existence is guaranteed by the option trigger above.
		scope:irr_unclaimed_home_region = { add_claim = ROOT }
		# Agitation/claim-laying is a modest provocation -- much lighter than
		# actually opening a play (Option C's path), so a small infamy hit.
		change_infamy = 1
		change_relations = {
			country = scope:target_country
			value = -10
		}
		# Armed forces approve, intelligentsia/peaceful IGs disapprove.
		ig:ig_armed_forces ?= {
			add_modifier = {
				name = ig_approval_positive_modifier
				days = normal_modifier_time
			}
		}
	}

	# --- Option B: Fund cultural ties ---
	option = {
		name = irredentism.1.b
		ai_chance = {
			base = 2
			modifier = {
				trigger = { irredentist_war_tilt <= -2 }
				add = 4
			}
			modifier = {
				trigger = { irredentist_war_tilt >= 3 }
				add = -2
			}
		}
		change_relations = {
			country = scope:target_country
			value = 15
		}
		# Spell out the payoff: the cultural-ties bond makes them likelier to
		# accept a future voluntary union.
		custom_tooltip = irredentism.1.b.tt
		# Lasting goodwill -- biases voluntary_union AI in their direction.
		add_modifier = {
			name = irr_cultural_ties_modifier
			days = 1825   # 5 years
			is_decaying = yes
		}
		scope:target_country = {
			add_modifier = {
				name = irr_cultural_ties_modifier
				days = 1825
				is_decaying = yes
			}
		}
	}

	# --- Option C: Demand return of [state] ---
	option = {
		name = irredentism.1.c
		ai_chance = {
			base = 0
			modifier = {
				trigger = { irredentist_war_tilt >= 1 }
				add = 2
			}
			modifier = {
				trigger = { irredentist_pressure_score >= 8 }
				add = 3
			}
			modifier = {
				trigger = {
					is_diplomatic_play_committed_participant = yes
				}
				add = -100   # Don't pile a play on top of an existing one
			}
		}
		trigger = {
			# Can only demand a state if we have one to demand AND we're
			# not already in a play.
			exists = scope:irr_demanded_state
			is_diplomatic_play_committed_participant = no
			scope:target_country = { is_diplomatic_play_committed_participant = no }
		}
		# The demand is a diplomatic note, not yet a play: it triggers a decision
		# on the target (irredentism.2). If they REFUSE, that event opens the
		# return-state diplomatic play; if they CEDE, the state transfers
		# peacefully. The custom_tooltip makes this two-step flow explicit.
		custom_tooltip = irredentism.1.c.tt
		scope:target_country = {
			trigger_event = { id = irredentism.2 days = 5 }
		}
	}

	# --- Option D: Bide our time ---
	# Default because Option A can be hidden (no unclaimed home state); biding is
	# the neutral, always-available timeout fallback.
	option = {
		name = irredentism.1.bide
		default_option = yes
		ai_chance = {
			base = 1
			modifier = {
				# AI biders mostly when they have no driver beyond a low tilt
				trigger = {
					irredentist_war_tilt >= -1
					irredentist_war_tilt <= 1
				}
				add = 1
			}
		}
		# Memory: future event firing slightly biased to escalation.
		set_variable = {
			name = irr_bided_against
			value = scope:target_country
			days = 3650   # 10 years
		}
	}
}

# ============================================================================
# Event 2: "[Demander] Demands [State]"
# Fires on the target after Event 1 option C delivers the demand (no play yet).
# The target decides: CEDE the state peacefully (set_state_owner) or REFUSE,
# which is what actually opens the return-state diplomatic play (initiated by
# scope:irr_unifier). AI fold/defy weighting reads relative army strength via
# irr_demander_army_ratio. scope:irr_unifier / scope:irr_demanded_state
# propagate from Event 1 (same pattern as Event 3 -> Event 4).
# ============================================================================
irredentism.2 = {
	type = country_event
	placement = root

	event_image = { video = "europenorthamerica_rich_and_poor" }

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	title = irredentism.2.t
	desc = irredentism.2.d
	flavor = irredentism.2.f

	duration = 3

	immediate = {
		save_scope_as = irr_target
	}

	# --- Option A: Refuse the demand -- they must take it by force ---
	option = {
		name = irredentism.2.a
		default_option = yes
		ai_chance = {
			base = 3
			# Warlike governments dig in.
			modifier = {
				trigger = { irredentist_war_tilt >= 1 }
				add = 2
			}
			# Confident when we clearly outmatch the demander.
			modifier = {
				trigger = { irr_demander_army_ratio <= 0.5 }
				add = 3
			}
			# Reluctant to defy a power that badly outguns us.
			modifier = {
				trigger = { irr_demander_army_ratio >= 2 }
				add = -2
			}
		}
		add_modifier = {
			name = irr_defiance_modifier
			days = 1825
			is_decaying = yes
		}
		ig:ig_armed_forces ?= {
			add_modifier = {
				name = ig_approval_positive_modifier
				days = normal_modifier_time
			}
		}
		# Refusal is what triggers force: the demander opens the return-state
		# diplomatic play (and bears its infamy as the initiator). custom_tooltip
		# because the play-creation runs inside scope:irr_unifier and the engine
		# won't reliably render a clear consequence line otherwise.
		custom_tooltip = irredentism.2.a.tt
		scope:irr_unifier = {
			create_diplomatic_play = {
				target_country = ROOT
				type = dp_return_state
				target_state = scope:irr_demanded_state
			}
		}
	}

	# --- Option B: Cede the state peacefully ---
	option = {
		name = irredentism.2.b
		ai_chance = {
			base = 1
			# Peaceful governments prefer to yield rather than fight kin.
			modifier = {
				trigger = { irredentist_war_tilt <= -1 }
				add = 2
			}
			# Fold when meaningfully outmatched...
			modifier = {
				trigger = { irr_demander_army_ratio >= 2 }
				add = 4
			}
			# ...and fold hard when hopelessly so.
			modifier = {
				trigger = { irr_demander_army_ratio >= 4 }
				add = 4
			}
			# Don't capitulate to someone weaker than us.
			modifier = {
				trigger = { irr_demander_army_ratio <= 0.5 }
				add = -3
			}
		}
		# Peaceful transfer of the demanded state to the demander -- no war.
		scope:irr_demanded_state = {
			set_state_owner = scope:irr_unifier
		}
		# De-escalation goodwill toward the power we accommodated.
		change_relations = {
			country = scope:irr_unifier
			value = 10
		}
	}
}

# ============================================================================
# Event 3: "An Opportunity for Union"
# Fires on a unifier when voluntary_union_eligible (high relations + mutual
# active driver). Player can propose voluntary union (surfaces the action via
# diplomatic catalyst on the target) or withdraw to bank the goodwill.
# ============================================================================
irredentism.3 = {
	type = country_event
	placement = root

	event_image = { video = "africa_diplomats_negotiating" }

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	title = irredentism.3.t
	desc = irredentism.3.d
	flavor = irredentism.3.f

	duration = 3

	cooldown = { years = 6 }

	trigger = {
		has_technology_researched = decolonization
		irredentist_unifier_can_act = yes
		NOT = { is_country_type = decentralized }
		any_country = {
			NOT = { THIS = ROOT }
			irr_target_same_nation = yes
			ROOT.relations:THIS >= relations_threshold:amicable
			any_scope_state = {
				irr_claimed_populated_home_state = yes
			}
			irr_target_reachable = yes
		}
	}

	immediate = {
		save_scope_as = irr_unifier
		random_country = {
			limit = {
				NOT = { THIS = ROOT }
				irr_target_same_nation = yes
				ROOT.relations:THIS >= relations_threshold:amicable
				any_scope_state = {
					irr_claimed_populated_home_state = yes
				}
				irr_target_reachable = yes
			}
			save_scope_as = target_country
		}
	}

	# --- Option A: Propose voluntary union ---
	option = {
		name = irredentism.3.a
		default_option = yes
		ai_chance = {
			base = 1
			modifier = {
				trigger = { irredentist_war_tilt <= -2 }
				add = 4
			}
			modifier = {
				trigger = { irredentist_war_tilt >= 3 }
				add = -2
			}
		}
		# Surface the proposal narratively; the diplomatic action itself
		# becomes available in the lens once Phase 4 ships voluntary_union.
		create_diplomatic_catalyst = {
			type = catalyst_event_positive
			target = scope:target_country
		}
		scope:target_country = {
			trigger_event = { id = irredentism.4 days = 5 }
		}
		add_modifier = {
			name = irr_offered_union_modifier
			days = 1825
			is_decaying = yes
		}
	}

	# --- Option B: Withdraw -- bank goodwill instead ---
	option = {
		name = irredentism.3.b
		ai_chance = {
			base = 1
			modifier = {
				trigger = { irredentist_war_tilt >= 2 }
				add = 1
			}
		}
		change_relations = {
			country = scope:target_country
			value = 5
		}
	}
}

# ============================================================================
# Event 4: "A Sister Nation Asks Us to Join"
# Fires on the target after Event 3's "Propose union" option. Target chooses
# whether to accept or reject (the diplomatic action itself flows through
# voluntary_union; Event 4 is the narrative surface for the choice).
# ============================================================================
irredentism.4 = {
	type = country_event
	placement = root

	event_image = { video = "africa_diplomats_negotiating" }

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	title = irredentism.4.t
	desc = irredentism.4.d
	flavor = irredentism.4.f

	duration = 3

	immediate = {
		save_scope_as = irr_target
	}

	# --- Option A: We are open to it ---
	option = {
		name = irredentism.4.a
		default_option = yes
		ai_chance = {
			base = 1
			modifier = {
				trigger = { irredentist_war_tilt <= -2 }
				add = 4
			}
			modifier = {
				# Event 4 fires on the target; we want target's relations with
				# the unifier (carried forward from Event 3 as scope:irr_unifier).
				trigger = {
					exists = scope:irr_unifier
					scope:irr_unifier.relations:root >= relations_threshold:friendly
				}
				add = 2
			}
		}
		# Modifier biases voluntary_union acceptance for 5y.
		add_modifier = {
			name = irr_open_to_union_modifier
			days = 1825
			is_decaying = yes
		}
	}

	# --- Option B: We will remain a free nation ---
	option = {
		name = irredentism.4.b
		ai_chance = {
			base = 3
			modifier = {
				trigger = { irredentist_war_tilt >= 1 }
				add = 3
			}
		}
		add_modifier = {
			name = irr_rejected_union_modifier
			days = 1825
			is_decaying = yes
		}
		ig:ig_armed_forces ?= {
			add_modifier = {
				name = ig_approval_positive_modifier
				days = normal_modifier_time
			}
		}
	}
}

# ============================================================================
# Event 5a: "Our Client Asks Us to Bless a War"
# Fires on the unifier's bloc leader when a non-leader bloc member is a
# strong irredentist-pressure candidate but lacks war-path approval.
# Endorse → sets var on requester unblocking irredentist_bloc_check_war.
# Veto → cooldown modifier blocking re-attempt for 5y.
# ============================================================================
irredentism.7 = {
	type = country_event
	placement = root

	event_image = { video = "africa_diplomats_negotiating" }

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	title = irredentism.7.t
	desc = irredentism.7.d
	flavor = irredentism.7.f

	duration = 3

	cooldown = { years = 5 }

	# Inherited scopes (irr_requester, target_country) are saved by the
	# on-action 3 days before fire. Either may have invalidated by now
	# (annexation, decentralization). Suppress the event in that case --
	# vanilla event-target accessor errors otherwise cascade through
	# every option's ai_chance evaluation.
	trigger = {
		exists = scope:irr_requester
		exists = scope:target_country
		scope:irr_requester = { NOT = { is_country_type = decentralized } }
		scope:target_country = { NOT = { is_country_type = decentralized } }
	}

	immediate = {
		save_scope_as = irr_bloc_leader
	}

	# --- Endorse the war ---
	option = {
		name = irredentism.7.a
		default_option = yes
		ai_chance = {
			base = 1
			# Ideology match between leader and requester is the largest pull.
			modifier = {
				trigger = {
					scope:irr_requester = {
						any_interest_group = {
							is_in_government = yes
							OR = {
								leader = { has_ideology = ideology:ideology_jingoist_leader }
								leader = { has_ideology = ideology:ideology_fascist }
								leader = { has_ideology = ideology:ideology_communist }
								leader = { has_ideology = ideology:ideology_vanguardist }
								leader = { has_ideology = ideology:ideology_ethno_nationalist }
							}
						}
					}
					ROOT = {
						any_interest_group = {
							is_in_government = yes
							OR = {
								leader = { has_ideology = ideology:ideology_jingoist_leader }
								leader = { has_ideology = ideology:ideology_fascist }
								leader = { has_ideology = ideology:ideology_communist }
								leader = { has_ideology = ideology:ideology_vanguardist }
								leader = { has_ideology = ideology:ideology_ethno_nationalist }
							}
						}
					}
				}
				add = 4
			}
			# Bloc strength advantage over target's bloc.
			modifier = {
				trigger = {
					OR = {
						scope:target_country = { is_in_power_bloc = no }
						AND = {
							scope:target_country = { is_in_power_bloc = yes }
							scope:target_country.power_bloc ?= {
								any_power_bloc_member = {
									country_rank < ROOT.country_rank
								}
							}
						}
					}
				}
				add = 3
			}
			# Target is unrecognized -- low cost, easy approval.
			modifier = {
				trigger = { scope:target_country = { is_country_type = unrecognized } }
				add = 2
			}
			# Leader's own infamy is high -- reluctance to add inherited cost.
			modifier = {
				trigger = { infamy >= infamy_threshold:notorious }
				add = -3
			}
			# Target is the leader's own ally / co-bloc member.
			# Alliances live in treaty articles (Sphere of Influence rework),
			# not in has_diplomatic_pact -- iterate ROOT's treaties.
			modifier = {
				trigger = {
					OR = {
						any_scope_treaty = {
							binds = scope:target_country
							any_scope_article = {
								has_type = alliance
							}
						}
						is_in_same_power_bloc = scope:target_country
					}
				}
				add = -5
			}
		}
		# Set approval var on the requester for 5 years. Use null-safe `?=`
		# in case the requester was annexed between on-action and fire.
		scope:irr_requester ?= {
			set_variable = {
				name = irredentist_war_approval
				value = scope:target_country
				days = 1825
			}
		}
		# Trigger a notification event on the requester so they know they can act.
		scope:irr_requester ?= {
			trigger_event = { id = irredentism.1 days = 7 }
		}
	}

	# --- Veto the war ---
	option = {
		name = irredentism.7.b
		ai_chance = {
			base = 2
			modifier = {
				trigger = { infamy >= infamy_threshold:infamous }
				add = 2
			}
			modifier = {
				trigger = {
					any_scope_treaty = {
						binds = scope:target_country
						any_scope_article = {
							has_type = alliance
						}
					}
				}
				add = 5
			}
			modifier = {
				trigger = {
					scope:target_country = { country_rank >= rank_value:major_power }
				}
				add = 2
			}
		}
		# Cool-down on the requester so the AI doesn't spam asks (null-safe).
		scope:irr_requester ?= {
			add_modifier = {
				name = irr_war_vetoed_modifier
				days = 1825
				is_decaying = yes
			}
		}
		if = {
			limit = { exists = scope:irr_requester }
			change_relations = {
				country = scope:irr_requester
				value = -10
			}
		}
	}
}

# ============================================================================
# Event 5b: "A Sister Asks to Be Absorbed"
# Fires on the target's bloc leader when a non-leader bloc member is the
# target of a voluntary union proposal. Bless → unblocks
# irredentist_bloc_check_peace; Forbid → modifier blocks for 5y.
# ============================================================================
irredentism.8 = {
	type = country_event
	placement = root

	event_image = { video = "africa_diplomats_negotiating" }

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	title = irredentism.8.t
	desc = irredentism.8.d
	flavor = irredentism.8.f

	duration = 3

	cooldown = { years = 5 }

	# Inherited scopes (irr_proposed_target, irr_unifier) saved by the on-action
	# 3 days before fire. Either may have invalidated -- suppress event then.
	trigger = {
		exists = scope:irr_proposed_target
		exists = scope:irr_unifier
		scope:irr_proposed_target = { NOT = { is_country_type = decentralized } }
		scope:irr_unifier = { NOT = { is_country_type = decentralized } }
	}

	immediate = {
		save_scope_as = irr_bloc_leader
	}

	# --- Bless the union ---
	option = {
		name = irredentism.8.a
		default_option = yes
		ai_chance = {
			base = 1
			# Ideology match makes blessing easier.
			modifier = {
				trigger = {
					ROOT = {
						any_interest_group = {
							is_in_government = yes
							OR = {
								leader = { has_ideology = ideology:ideology_market_liberal }
								leader = { has_ideology = ideology:ideology_humanitarian }
								leader = { has_ideology = ideology:ideology_pacifist }
							}
						}
					}
				}
				add = 3
			}
			# Amicable+ relations with the unifier.
			modifier = {
				trigger = {
					scope:irr_unifier.relations:root >= relations_threshold:amicable
				}
				add = 3
			}
			# Target is a peripheral / small contributor to the bloc.
			modifier = {
				trigger = {
					scope:irr_proposed_target = { country_rank <= rank_value:minor_power }
				}
				add = 2
			}
		}
		scope:irr_proposed_target = {
			set_variable = {
				name = irredentist_peace_approval
				value = scope:irr_unifier
				days = 1825
			}
		}
		# When the unifier is also in OUR bloc (we're its leader too),
		# the same blessing covers them. Without this, two non-leader members
		# of the same bloc could never pass irredentist_bloc_check_peace
		# even after their shared leader said yes.
		if = {
			limit = {
				scope:irr_unifier ?= {
					is_in_same_power_bloc = ROOT
				}
			}
			scope:irr_unifier ?= {
				set_variable = {
					name = irredentist_peace_approval
					value = scope:irr_proposed_target
					days = 1825
				}
			}
		}
	}

	# --- Forbid the union ---
	option = {
		name = irredentism.8.b
		ai_chance = {
			base = 2
			modifier = {
				trigger = {
					scope:irr_unifier.relations:root <= relations_threshold:cold
				}
				add = 4
			}
			modifier = {
				trigger = {
					scope:irr_proposed_target = { country_rank >= rank_value:major_power }
				}
				add = 3
			}
		}
		scope:irr_proposed_target = {
			add_modifier = {
				name = irr_union_forbidden_modifier
				days = 1825
				is_decaying = yes
			}
		}
	}
}

# ============================================================================
# Event 6: "[Annexed] absorbed by [Unifier]" — third-party notification
# Fires on countries with Engaged-or-higher interest in the annexed country's
# strategic regions. Uses var:irr_notify_peaceful_flag (set by the dispatching
# scripted effect) to choose warlike vs peaceful flavor text.
# ============================================================================
irredentism.6 = {
	type = country_event
	placement = scope:irr_unifier

	event_image = { video = "africa_diplomats_negotiating" }

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	title = irredentism.6.t
	desc = {
		# Pick desc by the peaceful flag the dispatching effect set on us.
		first_valid = {
			triggered_desc = {
				trigger = {
					has_variable = irr_notify_peaceful_flag
					var:irr_notify_peaceful_flag = yes
				}
				desc = irredentism.6.d.peaceful
			}
			triggered_desc = { desc = irredentism.6.d.warlike }
		}
	}
	flavor = irredentism.6.f

	duration = 3

	# Per-target cooldown handles dedup if multiple regions overlap.
	cooldown = { years = 1 }

	trigger = {
		exists = scope:irr_annexed_country
		exists = scope:irr_unifier
	}

	immediate = {
		save_scope_as = irr_observer
	}

	option = {
		name = irredentism.6.a
		default_option = yes
		# Mirror the engine's own diplomatic-incident relations effect
		# directionally: observers slightly cool toward the unifier when
		# the annexation was warlike.
		if = {
			limit = {
				has_variable = irr_notify_peaceful_flag
				var:irr_notify_peaceful_flag = no
			}
			change_relations = {
				country = scope:irr_unifier
				value = -5
			}
		}
		remove_variable = irr_notify_peaceful_flag
	}
}
