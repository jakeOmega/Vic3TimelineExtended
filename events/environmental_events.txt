# Event 1: Pollution Incident
environmental_events.1 = {
	type = country_event
	placement = root

	event_image = {
		texture = "gfx/event_pictures/industrial_pollution.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	title = environmental_events.1.t
	desc = environmental_events.1.d
	flavor = environmental_events.1.f

	duration = 3
	cooldown = { days = normal_modifier_time }

	trigger = {
		OR = {
			has_technology_researched = environmental_movement
			has_technology_researched = pollution_control
		}
		NOT = { has_law = law_type:law_ministry_of_the_environment }
	}

	immediate = {
	}

	# Impose strict regulations
	option = {
		name = environmental_events.1.a
		default_option = yes
		ai_chance = {
			base = 2
		}
		add_modifier = {
			name = pollution_regulations_modifier
			days = normal_modifier_time
			is_decaying = yes
		}
		add_radicals = {
			value = very_small_radicals
			strata = upper
		}
	}

	# Protect industry
	option = {
		name = environmental_events.1.b
		ai_chance = {
			base = 2
			modifier = {
				trigger = { has_law = law_type:law_laissez_faire }
				add = 2
			}
		}
		add_modifier = {
			name = industry_protected_modifier
			days = normal_modifier_time
			is_decaying = yes
		}
		add_radicals = {
			value = very_small_radicals
			pop_type = academics
		}
	}

	# Investigate
	option = {
		name = environmental_events.1.c
		ai_chance = {
			base = 3
		}
		add_radicals = {
			value = very_small_radicals
			pop_type = academics
		}
		add_loyalists = {
			value = very_small_radicals
			strata = upper
		}
	}
}

# Event 4: Oil Spill Disaster
# A catastrophic oil spill devastates a coastal region.
environmental_events.4 = {
	type = country_event
	placement = root

	event_image = {
		texture = "gfx/event_pictures/industrial_pollution.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	title = environmental_events.4.t
	desc = environmental_events.4.d
	flavor = environmental_events.4.f

	duration = 3
	cooldown = { days = long_modifier_time }

	trigger = {
		OR = {
			has_technology_researched = environmental_movement
			has_technology_researched = pollution_control
		}
		any_scope_state = {
			is_coastal = yes
		}
	}

	immediate = {
		random_scope_state = {
			limit = { is_coastal = yes }
			save_scope_as = spill_state
		}
	}

	# Massive cleanup effort - costly but popular
	option = {
		name = environmental_events.4.a
		default_option = yes
		ai_chance = {
			base = 2
		}
		add_modifier = {
			name = oil_spill_cleanup_modifier
			days = normal_modifier_time
			is_decaying = yes
		}
		add_treasury = {
			value = yearly_gross_income
			multiply = -0.15
		}
		scope:spill_state = {
			add_loyalists_in_state = {
				value = small_radicals
			}
		}
	}

	# Hold industry accountable
	option = {
		name = environmental_events.4.b
		ai_chance = {
			base = 2
		}
		add_modifier = {
			name = industry_accountability_modifier
			days = normal_modifier_time
			is_decaying = yes
		}
		add_radicals = {
			value = very_small_radicals
			strata = upper
		}
	}

	# Minimize the damage publicly
	option = {
		name = environmental_events.4.c
		ai_chance = {
			base = 1
		}
		scope:spill_state = {
			add_radicals_in_state = {
				value = medium_radicals
			}
		}
		add_loyalists = {
			value = very_small_radicals
			strata = upper
		}
	}
}
