	type = country_event
	placement = root

	event_image = { texture = "gfx/event_pictures/valles_marineris.dds" }
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_industry.dds"

	title = space_race_colony_events.1.t
	desc = space_race_colony_events.1.d
	flavor = space_race_colony_events.1.f

	duration = 3
	trigger = { }
	immediate = {
		create_character = {
			save_scope_as = flavor_speaker
			on_created = { place_character_in_void = 6 }
		}
	}

	option = {
		name = space_race_colony_events.1.a
		default_option = yes
		ai_chance = { base = 5 }
		# Geological research base in the canyon
		je:je_space_race_solar_colonization = { add_modifier = { name = sr_colony_valles_marineris_research } }
		add_loyalists = { value = medium_radicals pop_type = academics }

	option = {

		add_loyalists = { value = small_radicals }
