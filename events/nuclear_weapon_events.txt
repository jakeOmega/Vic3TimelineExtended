# ============================================================================
# EVENT 17: First Nuclear Weapon Used in Combat (Attacker)
# ============================================================================
# Fired when any country successfully uses a nuclear weapon in combat for the
# first time in world history. Fires only on the attacker, alongside the
# standard strike events. A world-historical moment.
# ============================================================================

nuclear_weapon_events.17 = {
	type = country_event
	placement = root

	event_image = { texture = "gfx/event_pictures/nuclear_false_alarm_panic.dds" }

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/mushroom_cloud.dds"

	title = nuclear_weapon_events.17.t
	desc = nuclear_weapon_events.17.d
	flavor = nuclear_weapon_events.17.f

	duration = 3

	trigger = {
		# triggered by nuclear strike scripted effects
	}

	immediate = {
	}

	# "A grave responsibility."
	option = {
		name = nuclear_weapon_events.17.a
		default_option = yes
		add_modifier = { name = nuclear_event_prestige multiplier = 100 days = normal_modifier_time is_decaying = yes }
		add_radicals = {
			value = very_small_radicals
		}
		ai_chance = { base = 5 }
	}

	# "History will judge us."
	option = {
		name = nuclear_weapon_events.17.b
		add_modifier = { name = nuclear_event_prestige multiplier = 50 days = normal_modifier_time is_decaying = yes }
		every_interest_group = {
			limit = { is_interest_group_type = ig_intelligentsia }
			add_modifier = { name = ig_approval_positive_modifier days = normal_modifier_time is_decaying = yes }
		}
		ai_chance = {
			base = 3
			modifier = { trigger = { ruler_is_cautious = yes } add = 5 }
		}
	}

	# Aggressive rulers: "Let this be a warning."
	option = {
		name = nuclear_weapon_events.17.c
		trigger = { ruler_is_aggressive = yes }
		highlighted_option = yes
		add_modifier = { name = nuclear_event_prestige multiplier = 200 days = normal_modifier_time is_decaying = yes }
		change_infamy = 10
		every_interest_group = {
			limit = { is_interest_group_type = ig_armed_forces }
			add_modifier = { name = ig_approval_positive_modifier days = normal_modifier_time is_decaying = yes }
		}
		add_radicals = {
			value = small_radicals
			interest_group = ig:ig_intelligentsia
		}
		ai_chance = {
			base = 1
			modifier = { trigger = { ruler_is_aggressive = yes } add = 10 }
		}
	}
}

# ============================================================================
# EVENT 18: Nuclear Proliferation Alert
# ============================================================================
# Fired on all countries (except the proliferating one) when a non-first
# country reaches 75% progress toward its first nuclear weapon.
# scope:proliferating_country = the country nearing completion.
# ============================================================================

nuclear_weapon_events.18 = {
	type = country_event
	placement = root

	event_image = { texture = "gfx/event_pictures/tactical_nuclear_strike.dds" }

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	title = nuclear_weapon_events.18.t
	desc = nuclear_weapon_events.18.d
	flavor = nuclear_weapon_events.18.f

	duration = 3

	trigger = {
		# triggered by JE pulse
	}

	immediate = {
	}

	# Nuclear power: "Monitor the situation."
	option = {
		name = nuclear_weapon_events.18.a
		trigger = { has_modifier = nuclear_power }
		default_option = yes
		add_modifier = { name = nuclear_event_prestige multiplier = 10 days = normal_modifier_time is_decaying = yes }
		ai_chance = { base = 5 }
	}

	# Has a nuclear program but no nukes yet: "We cannot fall behind."
	option = {
		name = nuclear_weapon_events.18.b
		trigger = {
			NOT = { has_modifier = nuclear_power }
			has_variable = nuclear_weapon_program_progress
			has_variable = nuclear_weapons_program_funding
		}
		highlighted_option = yes
		custom_tooltip = {
			text = NW_PROGRESS_ADD_5_TT
			change_variable = { name = nuclear_weapon_program_progress add = 5 }
		}
		ai_chance = { base = 7 }
	}

	# Anyone: "Push for non-proliferation."
	option = {
		name = nuclear_weapon_events.18.c
		default_option = yes
		if = {
			limit = { exists = scope:proliferating_country }
			change_relations = {
				country = scope:proliferating_country
				value = -10
			}
		}
		add_modifier = { name = nuclear_event_prestige multiplier = 5 days = normal_modifier_time is_decaying = yes }
		ai_chance = {
			base = 3
			modifier = { trigger = { ruler_is_cautious = yes } add = 5 }
		}
	}
}
