namespace = nuclear_weapon_events

nuclear_weapon_events.1 = {
    type = country_event
    placement = root

    event_image = {
 		texture = "gfx/event_pictures/mushroom_cloud.dds"
 	}

 	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

 	icon = "gfx/interface/icons/event_icons/event_fire.dds"

 	title = nuclear_weapon_events.1.t
 	desc = nuclear_weapon_events.1.d
 	flavor = nuclear_weapon_events.1.f

    duration = 3

 	trigger = {
 		# triggered by nuclear strike
 	}
	
 	immediate = {
		post_notification = industrial_nuke_succeeds_defender
 	}

 	option = {
 		name = nuclear_weapon_events.1.a
 		default_option = yes
		ai_chance = {
			base = 1
		}
 	}
	
	option = {
		name = nuclear_weapon_events.1.b
		trigger = {
			owner = {
				has_modifier = nuclear_power
				var:nuclear_weapon_stockpile > 0
			}
		}
		custom_tooltip = nuclear_retaliation_single_tt

		ai_chance = {
			base = 10
		}
		scope:attacking_country = {
			ordered_scope_state = {
				order_by = nuclear_industrial_strike_target_score
				position = 0
				nuclear_response_strike = yes
			}
		}
	}

	option = {
		name = nuclear_weapon_events.1.c
		trigger = {
			owner = {
				has_modifier = nuclear_power
				var:nuclear_weapon_stockpile > 2
			}
		}
		custom_tooltip = nuclear_retaliation_full_tt

		ai_chance = {
			base = 20
		}

		scope:attacking_country = {
			ordered_scope_state = {
				order_by = nuclear_industrial_strike_target_score
				position = 0
				nuclear_response_strike = yes
			}

			ordered_scope_state = {
				order_by = nuclear_industrial_strike_target_score
				position = 1
				nuclear_response_strike = yes
			}

			ordered_scope_state = {
				order_by = nuclear_industrial_strike_target_score
				position = 2
				nuclear_response_strike = yes
			}
		}
	}
}

				desc = nuclear_weapon_events.9.d_world_first
			}
			triggered_desc = {
				trigger = { always = yes }
				desc = nuclear_weapon_events.9.d
			}
		}
	}
	flavor = {
		first_valid = {
			triggered_desc = {
				trigger = { scope:first_nuke_country = { has_variable = is_world_first_nuclear_power } }
				desc = nuclear_weapon_events.9.f_world_first
			}
			triggered_desc = {
				trigger = { always = yes }
				desc = nuclear_weapon_events.9.f
			}
		}
	}

    duration = 3

 	trigger = {
 		# triggered by nuclear program
 	}
	
 	immediate = {
 		
 	}

	# World-first: shock and existential dread
	option = {
		name = nuclear_weapon_events.9.a_world_first
		trigger = { scope:first_nuke_country = { has_variable = is_world_first_nuclear_power } }
		default_option = yes
		add_radicals = {
			value = very_small_radicals
		}
	}

	# Subsequent nuclear powers: concern, but not shock
 	option = {
 		name = nuclear_weapon_events.9.a
		trigger = { NOT = { scope:first_nuke_country = { has_variable = is_world_first_nuclear_power } } }
 		default_option = yes
 	}
}
