# Event 1: Arms Race Escalation (rival powers building up military)
international_relations_events.1 = {
	type = country_event
	placement = root

	event_image = { texture = "gfx/event_pictures/arms_race_factories.dds" }

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_military.dds"

	title = international_relations_events.1.t
	desc = international_relations_events.1.d
	flavor = international_relations_events.1.f

	duration = 3

	cooldown = { days = normal_modifier_time }

	trigger = {
		NOR = {
			has_modifier = arms_race_escalation_modifier
			has_modifier = diplomatic_overture_modifier
			has_modifier = diplomatic_overture_rebuffed_modifier
			has_modifier = arms_race_ignored_modifier
		}
		has_technology_researched = radar
		country_rank >= rank_value:major_power
		any_rival_country = {
			country_rank >= rank_value:major_power
		}
	}

	immediate = {
		random_country = {
			limit = {
				has_diplomatic_pact = { who = ROOT type = rivalry }
				country_rank >= rank_value:major_power
			}
			save_scope_as = arms_race_rival
		}
	}

	option = {
		name = international_relations_events.1.a
		default_option = yes
		ai_chance = {
			base = 3
			modifier = {
				trigger = { country_rank >= rank_value:great_power }
				add = 2
			}
		}
		# Match their buildup - military boost but expensive
		add_modifier = {
			name = arms_race_escalation_modifier
			days = normal_modifier_time
			is_decaying = yes
		}
		change_relations = {
			country = scope:arms_race_rival
			value = -10
		}
		add_lobby_appeasement_from_diplomacy_unidirectional = {
			FIRST = root
			SECOND = scope:arms_race_rival
			PRO_AMOUNT = -1
			ANTI_AMOUNT = 1
			FACTOR = appeasement_relations_decreased
		}
		scope:arms_race_rival = {
			post_notification = arms_race_pressure
			add_modifier = {
				name = arms_race_rival_pressure_modifier
				days = normal_modifier_time
				is_decaying = yes
			}
		}
		# Armed forces approve, but raises military costs
		ig_approval_effect = { IG = ig_armed_forces MODIFIER = ig_approval_positive_modifier DAYS = normal_modifier_time }
	}

	option = {
		name = international_relations_events.1.b
		ai_chance = {
			base = 2
			modifier = {
				trigger = {
					has_attitude = { who = scope:arms_race_rival attitude = cautious }
				}
				add = 1
			}
		}
		# Seek diplomatic channels - risky, may be rebuffed
		random_list = {
			60 = {
				# Diplomacy succeeds
				add_modifier = {
					name = diplomatic_overture_modifier
					days = normal_modifier_time
					is_decaying = yes
				}
				change_relations = {
					country = scope:arms_race_rival
					value = 10
				}
				add_lobby_appeasement_from_diplomacy_unidirectional = {
					FIRST = root
					SECOND = scope:arms_race_rival
					PRO_AMOUNT = 1
					ANTI_AMOUNT = -1
					FACTOR = appeasement_relations_increased
				}
			}
			40 = {
				# Rival rebuffs us - humiliation
				add_modifier = {
					name = diplomatic_overture_rebuffed_modifier
					days = normal_modifier_time
					is_decaying = yes
				}
				change_relations = {
					country = scope:arms_race_rival
					value = -15
				}
				add_lobby_appeasement_from_diplomacy_unidirectional = {
					FIRST = root
					SECOND = scope:arms_race_rival
					PRO_AMOUNT = -1
					ANTI_AMOUNT = 1
					FACTOR = appeasement_relations_decreased
				}
			}
		}
		# Armed forces disapprove of diplomatic approach
		ig_approval_effect = { IG = ig_armed_forces MODIFIER = ig_approval_negative_modifier DAYS = normal_modifier_time }
	}

	option = {
		name = international_relations_events.1.c
		ai_chance = {
			base = 1
			modifier = {
				trigger = { country_rank < rank_value:great_power }
				add = 2
			}
		}
		# Ignore the threat - save money on military costs, but weaker and lose face
		add_modifier = {
			name = arms_race_ignored_modifier
			days = normal_modifier_time
			is_decaying = yes
		}
		change_relations = {
			country = scope:arms_race_rival
			value = -20
		}
		add_lobby_appeasement_from_diplomacy_unidirectional = {
			FIRST = root
			SECOND = scope:arms_race_rival
			PRO_AMOUNT = -1
			ANTI_AMOUNT = 1
			FACTOR = appeasement_relations_decreased
		}
		ig_approval_effect = { IG = ig_armed_forces MODIFIER = ig_approval_very_negative_modifier DAYS = normal_modifier_time }
	}
}
