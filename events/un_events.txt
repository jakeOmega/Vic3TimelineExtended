# Event 1: The United Nations Charter - Formation Event (fires once)
un_events.1 = {
	type = country_event
	placement = root

	event_image = {
		texture = "gfx/event_pictures/un_charter.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_map.dds"

	title = un_events.1.t
	desc = {
		first_valid = {
			triggered_desc = { desc = un_events.1.d_council trigger = { has_law = law_type:law_council_republic } }
			triggered_desc = { desc = un_events.1.d_authoritarian trigger = { has_law = law_type:law_single_party_state } }
			triggered_desc = { desc = un_events.1.d_theocratic trigger = { has_law = law_type:law_theocracy } }
			desc = un_events.1.d
		}
	}
	flavor = un_events.1.f

	duration = 3

	trigger = {
		# Fired globally by on_action when 3+ great powers have the tech
		NOT = { has_variable = un_charter_signed }
		NOT = { has_modifier = un_member_modifier }
		is_subject = no
		NOT = { is_country_type = decentralized }
	}

	immediate = {
		set_variable = { name = un_charter_signed value = 1 }
	}

	# Option A: Sign the charter as a founding member
	option = {
		name = un_events.1.a
		default_option = yes
		ai_chance = {
			base = 8
			modifier = {
				trigger = { has_law = law_type:law_isolationism }
				add = -6
			}
		}
		set_variable = { name = can_build_un_hq value = 1 }
		# First signatory hosts UN Headquarters
		if = {
			limit = { NOT = { has_global_variable = un_hq_country } }
			set_global_variable = { name = un_hq_country value = ROOT }
			add_modifier = { name = un_headquarters_modifier }
		}
		# Great powers get Security Council permanent seats
		if = {
			limit = { country_rank >= rank_value:great_power }
			add_modifier = { name = un_security_council_modifier }
		}
		add_modifier = { name = un_member_modifier }
		add_modifier = { name = un_founding_member_modifier }
		ig_approval_effect = { IG = ig_intelligentsia MODIFIER = ig_approval_positive_modifier DAYS = normal_modifier_time }
	}

	# Option B: Observe but don't join yet
	option = {
		name = un_events.1.b
		ai_chance = {
			base = 2
			modifier = {
				trigger = { has_law = law_type:law_isolationism }
				add = 5
			}
		}
		add_modifier = {
			name = un_observer_status_modifier
			days = long_modifier_time
			is_decaying = yes
		}
	}

	# Option C: Reject this internationalist nonsense
	option = {
		name = un_events.1.c
		ai_chance = {
			base = 1
			modifier = {
				trigger = { has_law = law_type:law_total_war }
				add = 3
			}
		}
		add_modifier = {
			name = un_rejectionist_modifier
			days = normal_modifier_time
			is_decaying = yes
		}
		ig_approval_effect = { IG = ig_armed_forces MODIFIER = ig_approval_positive_modifier DAYS = normal_modifier_time }
	}
}

# Event 2: General Assembly Crisis - A rival's aggression is condemned
un_events.2 = {
	type = country_event
	placement = root

	event_image = {
		texture = "gfx/event_pictures/un_assembly.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_map.dds"

	title = un_events.2.t
	desc = un_events.2.d
	flavor = un_events.2.f

	duration = 3

	cooldown = { days = long_modifier_time }

	trigger = {
		has_modifier = un_member_modifier
		is_at_war = no
		any_country = {
			has_diplomatic_pact = { who = ROOT type = rivalry }
			is_at_war = yes
			is_diplomatic_play_initiator = yes
		}
		NOT = { has_modifier = un_ga_resolution_modifier }
		NOT = { has_global_variable = un_vote_active }
	}

	immediate = {
		random_country = {
			limit = {
				has_diplomatic_pact = { who = ROOT type = rivalry }
				is_at_war = yes
				is_diplomatic_play_initiator = yes
			}
			save_scope_as = condemned_country
		}
		random_country = {
			limit = {
				NOT = { this = ROOT }
				has_war_with = scope:condemned_country
				is_diplomatic_play_initiator = no
			}
			save_scope_as = defender_country
		}
		# Reserve the vote slot immediately so no other country can start one
		set_global_variable = un_vote_active
	}

	# Option A: Propose condemnation resolution to the General Assembly
	option = {
		name = un_events.2.a
		default_option = yes
		ai_chance = {
			base = 5
			modifier = {
				trigger = { country_rank >= rank_value:great_power }
				add = 3
			}
		}
		custom_tooltip = un_vote_initiated_tt
		# Set up global vote
		set_global_variable = un_vote_topic_condemn
		set_global_variable = { name = un_vote_support value = 1 }
		set_global_variable = { name = un_vote_oppose value = 0 }
		set_global_variable = { name = un_vote_net value = 1 }
		save_scope_as = un_vote_proposer
		scope:condemned_country = { save_scope_as = un_vote_target }
		set_global_variable = { name = un_vote_proposer_country value = ROOT }
		set_global_variable = { name = un_vote_target_country value = scope:condemned_country }
		# Proposer takes a personal stance
		add_modifier = {
			name = un_ga_resolution_modifier
			days = normal_modifier_time
			is_decaying = yes
		}
		# Trigger vote for all other UN members
		hidden_effect = {
			every_country = {
				limit = {
					has_modifier = un_member_modifier
					NOT = { this = ROOT }
				}
				trigger_event = { id = un_vote.1 days = 30 }
			}
			# Schedule results event
			trigger_event = { id = un_vote.2 days = 365 }
		}
	}

	# Option B: Vote in favor but don't lead
	option = {
		name = un_events.2.b
		ai_chance = {
			base = 5
		}
		add_modifier = {
			name = un_ga_resolution_modifier
			days = normal_modifier_time
			is_decaying = yes
		}
		custom_tooltip = un_authority_change_plus_2
		change_variable = { name = un_authority add = 2 }
		change_relations = { country = scope:condemned_country value = -5 }
		add_lobby_appeasement_from_diplomacy_unidirectional = {
			FIRST = root
			SECOND = scope:condemned_country
			PRO_AMOUNT = 1
			ANTI_AMOUNT = -1
			FACTOR = appeasement_relations_decreased
		}
		change_relations = { country = scope:defender_country value = 5 }
		add_lobby_appeasement_from_diplomacy_unidirectional = {
			FIRST = root
			SECOND = scope:defender_country
			PRO_AMOUNT = 1
			ANTI_AMOUNT = -1
			FACTOR = appeasement_relations_increased
		}
		# Release the vote slot since we didn't start one
		remove_global_variable = un_vote_active
	}

	# Option C: Abstain - avoid entanglements
	option = {
		name = un_events.2.c
		ai_chance = {
			base = 2
			modifier = {
				trigger = { has_law = law_type:law_isolationism }
				add = 5
			}
		}
		custom_tooltip = un_authority_change_minus_2
		change_variable = { name = un_authority add = -2 }
		ig_approval_effect = { IG = ig_armed_forces MODIFIER = ig_approval_positive_modifier DAYS = short_modifier_time }
		# Release the vote slot since we didn't start one
		remove_global_variable = un_vote_active
	}
}

# Event 7: Humanitarian Crisis Response - famine, refugees, or disaster
un_events.7 = {
	type = country_event
	placement = root

	event_image = {
		texture = "gfx/event_pictures/humanitarian_crisis.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_default.dds"

	title = un_events.7.t
	desc = un_events.7.d
	flavor = un_events.7.f

	duration = 3

	cooldown = { days = long_modifier_time }

	trigger = {
		has_modifier = un_member_modifier
		country_rank >= rank_value:major_power
		NOT = { has_modifier = un_humanitarian_aid_modifier }
		NOT = { has_modifier = un_humanitarian_token_modifier }
		any_country = {
			NOT = { this = ROOT }
			any_scope_state = {
				average_sol < 8
			}
		}
	}

	immediate = {
		random_country = {
			limit = {
				NOT = { this = ROOT }
				any_scope_state = {
					average_sol < 8
				}
			}
			save_scope_as = crisis_country
		}
	}

	# Option A: Major humanitarian aid - expensive but builds authority
	option = {
		name = un_events.7.a
		default_option = yes
		ai_chance = {
			base = 4
			modifier = {
				trigger = { country_rank >= rank_value:great_power }
				add = 3
			}
		}
		add_modifier = {
			name = un_humanitarian_aid_modifier
			days = normal_modifier_time
			is_decaying = yes
		}
		# GDP-scaled expense
		add_modifier = {
			name = un_humanitarian_aid_cost
			multiplier = un_program_expense_value
		}
		custom_tooltip = un_authority_change_plus_5
		change_variable = { name = un_authority add = 5 }
		change_relations = { country = scope:crisis_country value = 30 }
		ig_approval_effect = { IG = ig_intelligentsia MODIFIER = ig_approval_positive_modifier DAYS = normal_modifier_time }
		ig_approval_effect = { IG = ig_industrialists MODIFIER = ig_approval_negative_modifier DAYS = short_modifier_time }
		# GDP-scaled benefit to crisis country: min(giver, receiver) / receiver * 2
		set_variable = { name = temp_giver_gdp value = gdp }
		scope:crisis_country = {
			add_modifier = {
				name = un_humanitarian_aid_received_modifier
				multiplier = {
					value = prev.var:temp_giver_gdp
					divide = {
						value = gdp
						min = 100000
					}
					multiply = 2
					min = 0.5
					max = 3.0
				}
				days = normal_modifier_time
				is_decaying = yes
			}
		}
		# Notify the crisis country of incoming aid
		save_scope_as = aid_donor
		scope:crisis_country = {
			trigger_event = { id = un_events.101 days = 14 }
		}
	}

	# Option B: Token contribution
	option = {
		name = un_events.7.b
		ai_chance = {
			base = 5
		}
		add_modifier = {
			name = un_humanitarian_token_modifier
			days = normal_modifier_time
			is_decaying = yes
		}
		custom_tooltip = un_authority_change_plus_1
		change_variable = { name = un_authority add = 1 }
		change_relations = { country = scope:crisis_country value = 10 }
		# GDP-scaled token benefit to crisis country
		set_variable = { name = temp_giver_gdp value = gdp }
		scope:crisis_country = {
			post_notification = un_humanitarian_token_received
			add_modifier = {
				name = un_humanitarian_token_received_modifier
				multiplier = {
					value = prev.var:temp_giver_gdp
					divide = {
						value = gdp
						min = 100000
					}
					min = 0.3
					max = 1.0
				}
				days = normal_modifier_time
				is_decaying = yes
			}
		}
	}

	# Option C: It's not our problem
	option = {
		name = un_events.7.c
		ai_chance = {
			base = 2
			modifier = {
				trigger = { has_law = law_type:law_isolationism }
				add = 4
			}
			# AI less likely to refuse at high authority (consequences are worse)
			modifier = {
				trigger = { var:un_authority >= 60 country_rank >= rank_value:great_power }
				add = -3
			}
		}
		custom_tooltip = un_authority_change_minus_3
		change_variable = { name = un_authority add = -3 }
		change_relations = { country = scope:crisis_country value = -10 }
		add_loyalists = {
			value = very_small_radicals
			strata = upper
		}
		ig_approval_effect = { IG = ig_intelligentsia MODIFIER = ig_approval_negative_modifier DAYS = short_modifier_time }
		# At moderate authority, GPs refusing aid face diplomatic consequences
		if = {
			limit = {
				var:un_authority >= 40
				country_rank >= rank_value:major_power
			}
			custom_tooltip = un_refuse_aid_diplomatic_penalty_tt
			add_modifier = {
				name = un_refused_aid_penalty_modifier
				days = normal_modifier_time
				is_decaying = yes
			}
		}
		# At high authority, GPs refusing aid face severe diplomatic and domestic penalties
		if = {
			limit = {
				var:un_authority >= 70
				country_rank >= rank_value:great_power
			}
			custom_tooltip = un_refuse_aid_severe_penalty_tt
			change_variable = { name = un_authority add = -5 }
			add_radicals = {
				value = small_radicals
				strata = lower
			}
			ig_approval_effect = { IG = ig_trade_unions MODIFIER = ig_approval_negative_modifier DAYS = normal_modifier_time }
		}
	}
}
