# Event 1: Women in the Workforce
# A wave of women entering previously male-dominated professions
society_technology_events.1 = {
	type = country_event
	placement = root

	event_image = { video = "unspecific_ruler_speaking_to_people" }
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	title = society_technology_events.1.t
	desc = society_technology_events.1.d
	flavor = society_technology_events.1.f

	duration = 3
	cooldown = { days = long_modifier_time }

	trigger = {
		has_technology_researched = second_wave_feminism
		NOT = { has_modifier = ste_women_workforce }
		NOT = { has_modifier = ste_women_workforce_blocked }
		OR = {
			has_law = law_type:law_women_in_the_workplace
			has_law = law_type:law_women_own_property
		}
	}

	immediate = {
	}

	option = {
		name = society_technology_events.1.a
		default_option = yes
		ai_chance = {
			base = 3
			modifier = {
				trigger = { has_law = law_type:law_women_in_the_workplace }

			}
		}
		# Embrace integration — productivity boost, traditionalist anger

			name = ste_women_workforce

	option = {

			base = 2

				trigger = {
