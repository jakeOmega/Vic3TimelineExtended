# Event 1: Whistleblower Revelation
surveillance_events.1 = {
	type = country_event
	placement = root

	event_image = {
		texture = "gfx/event_pictures/surveillance_debate.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	title = surveillance_events.1.t
	desc = surveillance_events.1.d
	flavor = surveillance_events.1.f

	duration = 3
	cooldown = { days = normal_modifier_time }

	trigger = {
		OR = {
			has_technology_researched = automated_surveillance
			has_technology_researched = cybersecurity
		}
		NOT = { has_law = law_type:law_strong_privacy_rights }
	}

	immediate = {
	}

	# Protect the whistleblower - reform
	option = {
		name = surveillance_events.1.a
		default_option = yes
		ai_chance = {
			base = 2
		}
		add_modifier = {
			name = whistleblower_reform_modifier
			days = normal_modifier_time
			is_decaying = yes
		}
		add_radicals = {
			value = very_small_radicals
			strata = upper
		}
	}

	# Prosecute the whistleblower
	option = {
		name = surveillance_events.1.b
		ai_chance = {
			base = 2
		}
		add_modifier = {
			name = whistleblower_prosecuted_modifier
			days = normal_modifier_time
			is_decaying = yes
		}
		add_radicals = {
			value = very_small_radicals
			pop_type = academics
		}
	}

	# Suppress the story
	option = {
		name = surveillance_events.1.c
		ai_chance = {
			base = 1
		}
		add_radicals = {
			value = small_radicals
			pop_type = academics
		}		add_loyalists = {
			value = very_small_radicals
			strata = upper
		}	}
}

	placement = root

	event_image = {
		texture = "gfx/event_pictures/surveillance_debate.dds"
	}

	icon = "gfx/interface/icons/event_icons/event_science.dds"

	title = surveillance_events.4.t
	desc = surveillance_events.4.d
	flavor = surveillance_events.4.f

	duration = 3
	cooldown = { days = long_modifier_time }

	trigger = {
		has_technology_researched = automated_surveillance
		has_technology_researched = cybersecurity
		NOT = { has_law = law_type:law_strong_privacy_rights }
		NOT = { has_global_variable = ai_predictive_policing_debate_happened }
	}

	immediate = {
		set_global_variable = ai_predictive_policing_debate_happened
	}

	# Deploy predictive policing
	option = {
		name = surveillance_events.4.a
		ai_chance = {
			base = 2
		}
		add_modifier = {
			name = predictive_policing_deployed_modifier
			days = normal_modifier_time
			is_decaying = yes
		}
		add_radicals = {
			value = small_radicals
			strata = lower
		}
	}

	# Only for intelligence, not domestic policing
	option = {
		name = surveillance_events.4.b
		default_option = yes
		ai_chance = {
			base = 3
		}
		add_modifier = {
			name = predictive_intelligence_only_modifier
			days = normal_modifier_time
			is_decaying = yes
		}
	}

	# Ban AI policing systems
	option = {
		name = surveillance_events.4.c
		ai_chance = {
			base = 1
			modifier = {
				trigger = {
					has_law = law_type:law_moderate_data_privacy
				}
				add = 2
			}
		}
		add_modifier = {
			name = ai_policing_banned_modifier
			days = normal_modifier_time
			is_decaying = yes
		}
		add_loyalists = {
			value = small_radicals
			pop_type = academics
		}
	}
}
