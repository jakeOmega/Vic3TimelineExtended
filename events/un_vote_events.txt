namespace = un_vote

# ============================================================================
# UN GENERAL ASSEMBLY VOTING EVENTS
# Generic two-phase vote system for UN resolutions.
#
# Phase 1: Proposer event (existing un_events) triggers un_vote.1 for all
#          UN members via every_country. Global variables track votes.
# Phase 2: After 90 days, un_vote.2 fires for the proposer with results.
#
# Vote topics (identified by global flag variables):
#   un_vote_topic_condemn                - Condemn aggressor (event 2)
#   un_vote_topic_human_rights           - Human Rights Declaration (event 3)
#   un_vote_topic_reform                 - UN Institutional Reform (event 6)
#   un_vote_topic_heritage               - Cultural Heritage Program (event 9)
#   un_vote_topic_decolonization         - Decolonization Resolution (event 12)
#   un_vote_topic_npt                    - Non-Proliferation Treaty (event 14)
#   un_vote_topic_pandemic               - Global Pandemic Response (event 16)
#   un_vote_topic_climate                - Climate Accord (event 17)
#   un_vote_topic_refugee                - International Refugee Crisis (event 18)
#   un_vote_topic_space                  - International Space Cooperation (event 19)
#   un_vote_topic_peacekeeping_request   - Peacekeeping deployment (request button)
#   un_vote_topic_aid_request            - Humanitarian aid (request button)
#   un_vote_topic_sanctions              - International sanctions (sanctions button)
#   un_vote_topic_law_of_sea             - Convention on the Law of the Sea (event 21)
#   un_vote_topic_icc                    - International Criminal Court (event 22)
#
# Vote tracking global variables:
#   un_vote_active              - flag: vote in progress
#   un_vote_support             - count of yes votes
#   un_vote_oppose              - count of no votes
#   un_vote_net                 - net votes (support - oppose), starts at 1
#   un_vote_proposer_country    - scope: country that proposed
#   un_vote_target_country      - scope: target country (if applicable)
# ============================================================================

	}

	immediate = {
		set_variable = un_voted_this_round
		# Resolve proposer/target from global vars if not already in scope chain
		# (needed when vote is triggered from scripted buttons rather than events)
		if = {
			limit = {
				NOT = { exists = scope:un_vote_proposer }
				exists = global_var:un_vote_proposer_country
			}
			global_var:un_vote_proposer_country = { save_scope_as = un_vote_proposer }
		}
		if = {
			limit = {
				NOT = { exists = scope:un_vote_target }
				exists = global_var:un_vote_target_country
			}
			global_var:un_vote_target_country = { save_scope_as = un_vote_target }
		}
	}

	# Option A: Vote in favor
	option = {
		# Target-perspective override: voting "in favor" of a condemnation /
		# sanctions resolution against ROOT means accepting the verdict.
		name = {
			trigger = { scope:un_vote_target ?= this }
			text = un_vote.1.a_target
		}
		name = {
			trigger = { always = yes }
			text = un_vote.1.a
		}
		default_option = yes

		# Treaty-topic preview: if this resolution passes and we comply
		# (the default Option A path in un_vote.3), the engine-generated
		# modifier tooltip surfaces what we're agreeing to before we
		# commit. show_as_tooltip renders the add_modifier without
		# executing — the actual application happens later in
		# un_vote.3 option A via un_vote_apply_treaty_modifier.
		show_as_tooltip = {
			if = {
				limit = { has_global_variable = un_vote_topic_human_rights }
				je:je_united_nations ?= { add_modifier = { name = un_human_rights_declaration_modifier } }
			}
			if = {
				limit = { has_global_variable = un_vote_topic_npt }
				je:je_united_nations ?= { add_modifier = { name = un_nonproliferation_modifier } }
			}
			if = {
				limit = { has_global_variable = un_vote_topic_climate }
				je:je_united_nations ?= { add_modifier = { name = un_climate_binding_modifier } }
			}
			if = {
				limit = { has_global_variable = un_vote_topic_reform }
				add_modifier = {
					name = un_reform_champion_modifier
					days = normal_modifier_time
					is_decaying = yes
				}
			}
			if = {
				limit = { has_global_variable = un_vote_topic_heritage }
				je:je_united_nations ?= { add_modifier = { name = un_heritage_program_modifier } }
			}
			if = {
				limit = { has_global_variable = un_vote_topic_decolonization }
				add_modifier = {
					name = un_decolonization_champion_modifier
					days = normal_modifier_time
					is_decaying = yes
				}

			if = {

				je:je_united_nations ?= { add_modifier = { name = un_pandemic_cooperation_modifier } }
			}
			if = {
				limit = { has_global_variable = un_vote_topic_refugee }
				je:je_united_nations ?= { add_modifier = { name = un_refugee_program_modifier } }
			}
			if = {
				limit = { has_global_variable = un_vote_topic_space }
				je:je_united_nations ?= { add_modifier = { name = un_space_partnership_modifier } }
			}
			if = {
				limit = { has_global_variable = un_vote_topic_law_of_sea }
				add_modifier = { name = un_law_of_sea_modifier }
			}
			if = {
				limit = {
					has_global_variable = un_vote_topic_icc
					NOT = { has_global_variable = un_vote_veto_cast }
				}
				add_modifier = { name = un_icc_member_modifier }
			}
		}

		custom_tooltip = {
			text = UN_VOTE_IN_FAVOR_TT
			change_global_variable = { name = un_vote_support add = 1 }
			change_global_variable = { name = un_vote_net add = 1 }
			set_variable = { name = un_vote_choice value = 1 }
		}

		# Target-perspective acceptance: voting in favor of a resolution
		# targeting yourself is "performative contrition." Pre-applies the
		# un_self_corrected_modifier (which un_vote.2 reads to skip the standard
		# un_condemned / un_sanctions_target modifier on resolution pass), sets
		# a global flag that nudges other AI voters toward yes ("permission"),
		# and produces an asymmetric IG reaction set instead of the standard
		# topic-conditional one.
		if = {
			limit = { scope:un_vote_target ?= this }
			custom_tooltip = UN_VOTE_ACCEPT_TARGET_TT
			hidden_effect = {
				# Skips un_condemned / un_sanctions_target in un_vote.2
				set_variable = un_target_self_corrected
				# Other AI voters' Option A reads this to lean yes (target gave permission)
				set_global_variable = un_target_accepted
			}

			add_modifier = {
				name = un_self_corrected_modifier
				days = long_modifier_time
				is_decaying = yes
			}

			# Small reconciliation gesture toward proposer (less than the standard
			# +15 that other voters give — the proposer just tried to condemn us)
			if = {
				limit = { exists = scope:un_vote_proposer }
				change_relations = { country = scope:un_vote_proposer value = 10 }
				create_diplomatic_catalyst = { type = catalyst_event_positive target = scope:un_vote_proposer }
			}

			# Domestic IG reactions to public capitulation
			# Armed forces strongly disapprove (longer than the standard short_modifier_time)
			if = {
				limit = { ig:ig_armed_forces ?= { is_marginal = no } }
				ig:ig_armed_forces = { add_modifier = { name = ig_approval_negative_modifier days = normal_modifier_time is_decaying = yes } }
			}
			# Petty bourgeoisie nationalist sentiment disapproves
			if = {
				limit = { ig:ig_petty_bourgeoisie ?= { is_marginal = no } }
				ig:ig_petty_bourgeoisie = { add_modifier = { name = ig_approval_negative_modifier days = short_modifier_time is_decaying = yes } }
			}
			# Intelligentsia approves of international norm-following
			if = {
				limit = { ig:ig_intelligentsia ?= { is_marginal = no } }
				ig:ig_intelligentsia = { add_modifier = { name = ig_approval_positive_modifier days = short_modifier_time is_decaying = yes } }
			}
		}
		else = {
			# Standard non-target voter path

			# Relations boost with proposer + diplomatic catalyst
			if = {
				limit = { exists = scope:un_vote_proposer }
				change_relations = { country = scope:un_vote_proposer value = 15 }
				create_diplomatic_catalyst = { type = catalyst_event_positive target = scope:un_vote_proposer }
			}

			# Relations damage with vote target on hostile resolutions
			# (mirrors the NO-vote +15-with-target solidarity bonus). Voting yes
			# on a condemn/sanctions/icc/expulsion vote is a public hostile act
			# against the target — it shouldn't be a free vote. Skip when voter
			# == target (the target-perspective branch above handles that case).
			if = {
				limit = {
					exists = scope:un_vote_target
					NOT = { scope:un_vote_target ?= this }
					OR = {
						has_global_variable = un_vote_topic_condemn
						has_global_variable = un_vote_topic_sanctions
						has_global_variable = un_vote_topic_icc
						has_global_variable = un_vote_topic_expulsion
					}
				}
				change_relations = { country = scope:un_vote_target value = -15 }
				create_diplomatic_catalyst = { type = catalyst_event_negative target = scope:un_vote_target }
			}

			# IG reactions to voting yes
			# Intelligentsia generally approves of international cooperation
			if = {
				limit = {
					ig:ig_intelligentsia ?= { is_marginal = no }
				}
				ig:ig_intelligentsia = { add_modifier = { name = ig_approval_positive_modifier days = short_modifier_time is_decaying = yes } }
			}
			# Armed forces disapprove of cooperative/disarmament votes
			if = {
				limit = {
					OR = {
						has_global_variable = un_vote_topic_npt
						has_global_variable = un_vote_topic_peacekeeping_request
						has_global_variable = un_vote_topic_icc
					}
					ig:ig_armed_forces ?= { is_marginal = no }
				}
				ig:ig_armed_forces = { add_modifier = { name = ig_approval_negative_modifier days = short_modifier_time is_decaying = yes } }
			}
			# Industrialists disapprove of binding climate/pandemic regulation
			if = {
				limit = {
					OR = {
						has_global_variable = un_vote_topic_climate
						has_global_variable = un_vote_topic_pandemic
					}
					ig:ig_industrialists ?= { is_marginal = no }
				}
				ig:ig_industrialists = { add_modifier = { name = ig_approval_negative_modifier days = short_modifier_time is_decaying = yes } }
			}
		}

		ai_chance = {
			base = 40

			# ================================================================
			# DIPLOMATIC RELATIONSHIP MODIFIERS
			# ================================================================

			# Allies of proposer lean support
			modifier = {
				trigger = {
					exists = scope:un_vote_proposer
					has_treaty_alliance_with = { TARGET = scope:un_vote_proposer }

				add = 20
			}

			modifier = {
				trigger = {
					exists = scope:un_vote_proposer
					is_in_same_power_bloc = scope:un_vote_proposer

				add = 15
			}
			# Diplomatically relevant proposer: pay more attention
			modifier = {
				trigger = {
					exists = scope:un_vote_proposer
					has_diplomatic_relevance = scope:un_vote_proposer
				}
				add = 5
			}
			# Rivals of proposer lean oppose
			modifier = {
				trigger = {
					exists = scope:un_vote_proposer
					has_diplomatic_pact = {
						who = scope:un_vote_proposer
						type = rivalry
					}
				}
				add = -30
			}
			# Great powers more opinionated
			modifier = {
				trigger = { country_rank >= rank_value:great_power }
				factor = 1.5
			}
			# Target accepted the resolution: lowers political cost for other
			# voters to support it ("permission" effect). Gated on NOT-target so
			# the target's own ai_chance doesn't double-dip its own gesture.
			# Note: vote events fire concurrently; only AI evaluated AFTER the
			# target's tick will see the flag — partial coverage by design.
			modifier = {
				trigger = {
					has_global_variable = un_target_accepted
					NOT = { scope:un_vote_target ?= this }
				}
				add = 15
			}

			# ================================================================
			# INFAMY-BURNING INCENTIVE: countries with high infamy support
			# humanitarian votes to rehabilitate their image
			# ================================================================
			modifier = {
				trigger = {
					infamy >= 25
					OR = {
						has_global_variable = un_vote_topic_aid_request
						has_global_variable = un_vote_topic_peacekeeping_request
						has_global_variable = un_vote_topic_human_rights
						has_global_variable = un_vote_topic_refugee
						has_global_variable = un_vote_topic_pandemic
					}
				}
				add = 25
			}
			modifier = {
				trigger = {
					infamy >= 50
					OR = {
						has_global_variable = un_vote_topic_aid_request
						has_global_variable = un_vote_topic_peacekeeping_request
						has_global_variable = un_vote_topic_human_rights
						has_global_variable = un_vote_topic_refugee
						has_global_variable = un_vote_topic_pandemic
					}
				}
				add = 15
			}

			# ================================================================
			# DIMINISHING RETURNS: each existing commitment reduces enthusiasm
			# for pledging more (stacking penalty per program funded)
			# ================================================================
			modifier = {
				trigger = {
					has_global_variable = un_vote_topic_aid_request
					has_variable = un_aid_contribution_count
					var:un_aid_contribution_count >= 1
				}
				add = -15
			}
			modifier = {
				trigger = {
					has_global_variable = un_vote_topic_aid_request
					has_variable = un_aid_contribution_count
					var:un_aid_contribution_count >= 2
				}
				add = -15
			}
			modifier = {
				trigger = {
					has_global_variable = un_vote_topic_aid_request
					has_variable = un_aid_contribution_count
					var:un_aid_contribution_count >= 3
				}
				add = -15
			}
			modifier = {
				trigger = {
					has_global_variable = un_vote_topic_peacekeeping_request
					has_variable = un_peacekeeping_contribution_count
					var:un_peacekeeping_contribution_count >= 1
				}
				add = -15
			}
			modifier = {
				trigger = {
					has_global_variable = un_vote_topic_peacekeeping_request
					has_variable = un_peacekeeping_contribution_count
					var:un_peacekeeping_contribution_count >= 2
				}
				add = -15
			}
			modifier = {
				trigger = {
					has_global_variable = un_vote_topic_peacekeeping_request
					has_variable = un_peacekeeping_contribution_count
					var:un_peacekeeping_contribution_count >= 3
				}
				add = -15
			}

			# Power bloc leaders are more willing to fund aid/peacekeeping (leverage gain)
			modifier = {
				trigger = {
					OR = {
						has_global_variable = un_vote_topic_aid_request
						has_global_variable = un_vote_topic_peacekeeping_request
					}
					is_power_bloc_leader = yes
				}
				add = 20
			}

			# ================================================================
			# TOPIC: Condemn aggressor
			# ================================================================
			# Support if target is our rival
			modifier = {
				trigger = {
					has_global_variable = un_vote_topic_condemn
					exists = scope:un_vote_target
					has_diplomatic_pact = {
						who = scope:un_vote_target
						type = rivalry
					}
				}
				add = 30
			}
			# Oppose if target is our ally
			modifier = {
				trigger = {
					has_global_variable = un_vote_topic_condemn
					exists = scope:un_vote_target
					has_treaty_alliance_with = { TARGET = scope:un_vote_target }
				}
				add = -40
			}
			# Same power bloc as target: protect bloc member
			modifier = {
				trigger = {
					has_global_variable = un_vote_topic_condemn
					exists = scope:un_vote_target
					is_in_same_power_bloc = scope:un_vote_target
				}
				add = -20
			}
			# Peaceful nations support condemnation
			modifier = {
				trigger = {
					has_global_variable = un_vote_topic_condemn
					has_law = law_type:law_humanitarian_regulations
				}
				add = 15
			}

			# ================================================================
			# TOPIC: Human Rights
			# ================================================================
			modifier = {
				trigger = {
					has_global_variable = un_vote_topic_human_rights
					has_law = law_type:law_multicultural
				}
				add = 20
			}
			modifier = {
				trigger = {
					has_global_variable = un_vote_topic_human_rights
					has_law = law_type:law_universal_suffrage
				}
				add = 15
			}

			limit = {
				exists = scope:un_vote_target
				NOT = { scope:un_vote_target ?= this }
				OR = {
					has_global_variable = un_vote_topic_condemn
					has_global_variable = un_vote_topic_sanctions
				}
			}
			change_relations = { country = scope:un_vote_target value = 20 }
			create_diplomatic_catalyst = { type = catalyst_event_positive target = scope:un_vote_target }
		}

		# IG reactions: armed forces approve assertive sovereignty, intelligentsia disapproves
		if = {
			limit = { ig:ig_armed_forces ?= { is_marginal = no } }
			ig:ig_armed_forces = { add_modifier = { name = ig_approval_positive_modifier days = short_modifier_time is_decaying = yes } }
		}
		if = {
			limit = { ig:ig_intelligentsia ?= { is_marginal = no } }
			ig:ig_intelligentsia = { add_modifier = { name = ig_approval_negative_modifier days = short_modifier_time is_decaying = yes } }
		}

		ai_chance = {
			base = 0

			# Only consider vetoing when the resolution would otherwise pass.
			# (un_veto_would_change_outcome checks net >= 2, meaning even our "no"
			# alone wouldn't sink it — so a veto is needed to actually block.)
			modifier = {
				trigger = { un_veto_would_change_outcome = yes }
				add = 30
			}

			# Strongly veto rivals' proposals
			modifier = {
				trigger = {
					exists = scope:un_vote_proposer
					has_diplomatic_pact = { who = scope:un_vote_proposer type = rivalry }
				}
				add = 35
			}

			modifier = {
				trigger = {
					OR = {
						has_global_variable = un_vote_topic_sanctions
						has_global_variable = un_vote_topic_condemn
					}
					exists = scope:un_vote_target
					has_treaty_alliance_with = { TARGET = scope:un_vote_target }
				}
				add = 40
			}
			modifier = {
				trigger = {
					OR = {
						has_global_variable = un_vote_topic_sanctions
						has_global_variable = un_vote_topic_condemn
					}
					exists = scope:un_vote_target
					is_in_same_power_bloc = scope:un_vote_target
				}
				add = 25
			}

			# ICC referrals targeting US or our allies are existential — strongly veto
			modifier = {
				trigger = {
					has_global_variable = un_vote_topic_icc
					OR = {
						scope:un_vote_target ?= this
						AND = {
							exists = scope:un_vote_target
							has_treaty_alliance_with = { TARGET = scope:un_vote_target }
						}
					}
				}
				add = 50
			}

			# Reform that would weaken SC privileges (i.e., proposer is non-permanent)
			modifier = {
				trigger = {
					has_global_variable = un_vote_topic_reform
					exists = scope:un_vote_proposer
					NOT = { scope:un_vote_proposer = { is_un_permanent_member = yes } }
				}
				add = 30
			}

			# Peacekeeping deployment requested by an aligned target — let it go
			# (we want our friends helped); but veto when requested by a rival's bloc
			modifier = {
				trigger = {
					has_global_variable = un_vote_topic_peacekeeping_request
					exists = scope:un_vote_proposer
					NOT = { is_in_same_power_bloc = scope:un_vote_proposer }
					NOT = { has_treaty_alliance_with = { TARGET = scope:un_vote_proposer } }
				}
				add = 10
			}

			# Costs / restraints — discourage vetoing when:
			# Our UN authority is already weak (cost is steep)
			modifier = {
				trigger = {
					has_variable = un_authority
					var:un_authority < 30
				}
				add = -25
			}
			# We've recently vetoed (cool down)
			modifier = {
				trigger = { has_modifier = un_veto_isolation_modifier }
				add = -20
			}
			# Our infamy is already high and this topic would add more
			modifier = {
				trigger = {
					infamy >= 25
					OR = {
						has_global_variable = un_vote_topic_icc
						has_global_variable = un_vote_topic_condemn
						has_global_variable = un_vote_topic_peacekeeping_request
					}
				}
				add = -15
			}
		}
	}

			}

			# ---- Space: UNOOSA established ----
			if = {
				limit = { has_global_variable = un_vote_topic_space }
				custom_tooltip = un_vote_space_passed_tt
				change_variable = { name = un_authority add = 5 }
				if = {
					limit = { NOT = { has_global_variable = un_agency_unoosa } }
					set_global_variable = un_agency_unoosa
				}
			}

			# ---- Peacekeeping Request: Deploy peacekeepers to requesting country ----
			# Vetoed: deploy a non-coercive observer mission instead — the
			# proposer gets a small benefit but no contributors are levied.
			if = {
				limit = {
					has_global_variable = un_vote_topic_peacekeeping_request
					NOT = { has_global_variable = un_vote_veto_cast }
				}
				custom_tooltip = un_vote_peacekeeping_request_passed_tt
				change_variable = { name = un_authority add = 5 }
				# Apply scaled peacekeeping benefit to the requesting country
				if = {
					limit = { exists = global_var:un_vote_proposer_country }
					global_var:un_vote_proposer_country = {
						add_modifier = {
							name = un_peacekeeping_received_modifier
							multiplier = un_peacekeeping_benefit_scale
							days = long_modifier_time
							is_decaying = yes
						}
					}
				}
				# Apply costs to all countries that pledged contributions
				hidden_effect = {
					every_country = {
						limit = {
							je:je_united_nations ?= { has_modifier = un_peacekeeping_contributor_modifier }
						}
						# Remove old cost to re-apply with updated count-based scaling
						je:je_united_nations ?= { if = { limit = { has_modifier = un_peacekeeping_contributor_cost } remove_modifier = un_peacekeeping_contributor_cost } }
						je:je_united_nations ?= { add_modifier = {
							name = un_peacekeeping_contributor_cost
							multiplier = un_peacekeeping_total_expense_value
						} }
						add_modifier = {
							name = un_peacekeeping_cooldown
							days = long_modifier_time
							is_decaying = yes
						}
					}
				}
			}
			else_if = {
				limit = { has_global_variable = un_vote_topic_peacekeeping_request }
				custom_tooltip = un_vote_peacekeeping_request_passed_vetoed_tt
				change_variable = { name = un_authority add = 2 }
				# Observer mission to the proposer only — no enforcement, no levies
				if = {
					limit = { exists = global_var:un_vote_proposer_country }
					global_var:un_vote_proposer_country = {
						add_modifier = {
							name = un_observer_mission_modifier
							days = normal_modifier_time
							is_decaying = yes
						}
					}
				}
				# Drop pledged contributors (the binding deployment was vetoed)
				hidden_effect = {
					every_country = {
						limit = {
							je:je_united_nations ?= { has_modifier = un_peacekeeping_contributor_modifier }
							NOT = { je:je_united_nations ?= { has_modifier = un_peacekeeping_contributor_cost } }
						}
						je:je_united_nations ?= { remove_modifier = un_peacekeeping_contributor_modifier }
					}
				}
			}

			# ---- Aid Request: Deliver humanitarian aid to requesting country ----
			if = {
				limit = { has_global_variable = un_vote_topic_aid_request }
				custom_tooltip = un_vote_aid_request_passed_tt
				change_variable = { name = un_authority add = 5 }
				# Apply scaled aid benefit to the requesting country
				if = {
					limit = { exists = global_var:un_vote_proposer_country }
					global_var:un_vote_proposer_country = {
						add_modifier = {
							name = un_aid_received_modifier
							multiplier = un_aid_benefit_scale
							days = long_modifier_time
							is_decaying = yes
						}
					}
				}
				# Apply costs to all countries that pledged aid
				hidden_effect = {
					every_country = {
						limit = { has_modifier = un_humanitarian_aid_modifier }
						# Remove old cost to re-apply with updated count-based scaling
						if = { limit = { has_modifier = un_humanitarian_aid_cost } remove_modifier = un_humanitarian_aid_cost }
						add_modifier = {
							name = un_humanitarian_aid_cost
							multiplier = un_aid_total_expense_value
							days = long_modifier_time
							is_decaying = yes
						}
					}
				}
			}

			# ---- Sanctions: Apply sanctions to the target country ----
			# If the target voted to accept the resolution, they already bear
			# un_self_corrected_modifier; skip un_sanctions_target_modifier so
			# the player isn't double-penalized.
			# Vetoed: voluntary partial sanctions — apply un_sanctions_partial_modifier
			# to the target instead of the full sanctions package.
			if = {
				limit = {
					has_global_variable = un_vote_topic_sanctions
					NOT = { has_global_variable = un_vote_veto_cast }
				}
				custom_tooltip = un_vote_sanctions_passed_tt
				change_variable = { name = un_authority add = 5 }
				# Proposer becomes sanctions enforcer
				add_modifier = { name = un_sanctions_enforcer_modifier }
				# Apply sanctions to target
				if = {
					limit = { exists = scope:un_vote_target }
					scope:un_vote_target = {
						if = {
							limit = { NOT = { has_variable = un_target_self_corrected } }
							add_modifier = { name = un_sanctions_target_modifier }
						}
					}
					change_relations = { country = scope:un_vote_target value = -30 }
				}
			}
			else_if = {
				limit = { has_global_variable = un_vote_topic_sanctions }
				custom_tooltip = un_vote_sanctions_passed_vetoed_tt
				change_variable = { name = un_authority add = 2 }
				if = {
					limit = { exists = scope:un_vote_target }
					scope:un_vote_target = {
						if = {
							limit = { NOT = { has_variable = un_target_self_corrected } }
							add_modifier = { name = un_sanctions_partial_modifier }
						}
					}
					change_relations = { country = scope:un_vote_target value = -10 }
				}
			}

			# ---- Law of the Sea: Maritime cooperation framework ----
			if = {
				limit = { has_global_variable = un_vote_topic_law_of_sea }
				custom_tooltip = un_vote_law_of_sea_passed_tt
				change_variable = { name = un_authority add = 5 }
				set_global_variable = un_agency_itlos
			}

			# ---- International Criminal Court: Global criminal justice ----
			# Vetoed: no agency established, just a symbolic censure (proposer
			# gets un_court_contested_modifier as a small accountability gesture).
			if = {
				limit = {
					has_global_variable = un_vote_topic_icc
					NOT = { has_global_variable = un_vote_veto_cast }
				}
				custom_tooltip = un_vote_icc_passed_tt
				change_variable = { name = un_authority add = 5 }
				set_global_variable = un_agency_icc
			}
			else_if = {
				limit = { has_global_variable = un_vote_topic_icc }
				custom_tooltip = un_vote_icc_passed_vetoed_tt
				change_variable = { name = un_authority add = 2 }
				add_modifier = {
					name = un_court_contested_modifier
					days = normal_modifier_time
					is_decaying = yes
				}
			}

			# Influence reward for leading a successful resolution.
			# Skipped for vetoed binding resolutions — only the graduated form
			# went through, so the proposer didn't fully succeed.
			if = {
				limit = { NOT = { has_global_variable = un_vote_veto_cast } }
				add_modifier = {
					name = un_vote_success_reward
					days = normal_modifier_time
					is_decaying = yes
				}
			}

			# Infamy reduction: successfully leading a humanitarian resolution rehabilitates image
			if = {
				limit = {
					OR = {
						has_global_variable = un_vote_topic_human_rights
						has_global_variable = un_vote_topic_refugee
						has_global_variable = un_vote_topic_pandemic
						has_global_variable = un_vote_topic_aid_request
						has_global_variable = un_vote_topic_peacekeeping_request
					}
					infamy > 0
				}
				change_infamy = -3
			}

			# Power bloc leverage: successful proposer gains leverage over bloc members who voted
			if = {
				limit = { is_power_bloc_leader = yes }
				power_bloc ?= {
					every_power_bloc_member = {
						limit = {
							je:je_united_nations ?= { has_modifier = un_member_modifier }
							NOT = { THIS = ROOT }
						}
						root = {
							power_bloc ?= {
								add_leverage = { target = prev value = 3 }
							}
						}
					}
				}
			}
		}

		# ---- RESOLUTION FAILED ----
		if = {
			limit = {
				NOT = { global_var:un_vote_net >= 1 }
				NOT = { has_global_variable = un_vote_topic_expulsion }
			}
			custom_tooltip = un_vote_failed_tt
			change_variable = { name = un_authority add = -3 }
			add_modifier = {
				name = un_vote_failure_penalty
				days = normal_modifier_time
				is_decaying = yes
			}
			# Embarrassment: failed humanitarian resolution costs credibility
			if = {
				limit = {
					OR = {
						has_global_variable = un_vote_topic_human_rights
						has_global_variable = un_vote_topic_refugee
						has_global_variable = un_vote_topic_pandemic
					}
				}
				change_infamy = 2
			}
			# Power bloc leverage loss: failed proposer loses credibility with bloc
			if = {
				limit = { is_power_bloc_leader = yes }
				power_bloc ?= {
					every_power_bloc_member = {
						limit = {
							je:je_united_nations ?= { has_modifier = un_member_modifier }
							NOT = { THIS = ROOT }
						}
						root = {
							power_bloc ?= {
								add_leverage = { target = prev value = -3 }
							}
						}
					}
				}
			}
			# Revoke unfulfilled pledges from voters
			hidden_effect = {
				if = {
					limit = { has_global_variable = un_vote_topic_peacekeeping_request }
					every_country = {
						limit = {
							je:je_united_nations ?= { has_modifier = un_peacekeeping_contributor_modifier }
							NOT = { je:je_united_nations ?= { has_modifier = un_peacekeeping_contributor_cost } }
						}
						je:je_united_nations ?= { remove_modifier = un_peacekeeping_contributor_modifier }
					}
				}
				if = {
					limit = { has_global_variable = un_vote_topic_aid_request }
					every_country = {
						limit = {
							has_modifier = un_humanitarian_aid_modifier
							NOT = { has_modifier = un_humanitarian_aid_cost }
						}
						remove_modifier = un_humanitarian_aid_modifier
					}
				}
			}
		}

		# ---- Notify all other UN members of the result ----
		hidden_effect = {
			every_country = {
				limit = {
					je:je_united_nations ?= { has_modifier = un_member_modifier }
					NOT = { THIS = ROOT }
				}
				trigger_event = { id = un_vote.3 }
			}
		}

		# ---- Set per-topic 5-year cooldown before clearing vote state ----
		hidden_effect = {
			if = { limit = { has_global_variable = un_vote_topic_condemn } set_global_variable = { name = un_vote_cooldown_condemn days = normal_modifier_time } }
			if = { limit = { has_global_variable = un_vote_topic_human_rights } set_global_variable = { name = un_vote_cooldown_human_rights days = normal_modifier_time } }
			if = { limit = { has_global_variable = un_vote_topic_npt } set_global_variable = { name = un_vote_cooldown_npt days = normal_modifier_time } }
			if = { limit = { has_global_variable = un_vote_topic_climate } set_global_variable = { name = un_vote_cooldown_climate days = normal_modifier_time } }
			if = { limit = { has_global_variable = un_vote_topic_reform } set_global_variable = { name = un_vote_cooldown_reform days = normal_modifier_time } }
			if = { limit = { has_global_variable = un_vote_topic_heritage } set_global_variable = { name = un_vote_cooldown_heritage days = normal_modifier_time } }
			if = { limit = { has_global_variable = un_vote_topic_decolonization } set_global_variable = { name = un_vote_cooldown_decolonization days = normal_modifier_time } }
			if = { limit = { has_global_variable = un_vote_topic_pandemic } set_global_variable = { name = un_vote_cooldown_pandemic days = normal_modifier_time } }
			if = { limit = { has_global_variable = un_vote_topic_refugee } set_global_variable = { name = un_vote_cooldown_refugee days = normal_modifier_time } }
			if = { limit = { has_global_variable = un_vote_topic_space } set_global_variable = { name = un_vote_cooldown_space days = normal_modifier_time } }
			if = { limit = { has_global_variable = un_vote_topic_peacekeeping_request } set_global_variable = { name = un_vote_cooldown_peacekeeping_request days = normal_modifier_time } }
			if = { limit = { has_global_variable = un_vote_topic_aid_request } set_global_variable = { name = un_vote_cooldown_aid_request days = normal_modifier_time } }
			if = { limit = { has_global_variable = un_vote_topic_sanctions } set_global_variable = { name = un_vote_cooldown_sanctions days = normal_modifier_time } }
			if = { limit = { has_global_variable = un_vote_topic_law_of_sea } set_global_variable = { name = un_vote_cooldown_law_of_sea days = normal_modifier_time } }
			if = { limit = { has_global_variable = un_vote_topic_icc } set_global_variable = { name = un_vote_cooldown_icc days = normal_modifier_time } }
			if = { limit = { has_global_variable = un_vote_topic_expulsion } set_global_variable = { name = un_vote_cooldown_expulsion days = long_modifier_time } }
		}

		# ---- Cleanup: clear all vote state ----
		hidden_effect = {
			remove_global_variable = un_vote_active
			if = { limit = { has_global_variable = un_vote_topic_condemn } remove_global_variable = un_vote_topic_condemn }
			if = { limit = { has_global_variable = un_vote_topic_human_rights } remove_global_variable = un_vote_topic_human_rights }
			if = { limit = { has_global_variable = un_vote_topic_npt } remove_global_variable = un_vote_topic_npt }
			if = { limit = { has_global_variable = un_vote_topic_climate } remove_global_variable = un_vote_topic_climate }
			if = { limit = { has_global_variable = un_vote_topic_reform } remove_global_variable = un_vote_topic_reform }
			if = { limit = { has_global_variable = un_vote_topic_heritage } remove_global_variable = un_vote_topic_heritage }
			if = { limit = { has_global_variable = un_vote_topic_decolonization } remove_global_variable = un_vote_topic_decolonization }
			if = { limit = { has_global_variable = un_vote_topic_pandemic } remove_global_variable = un_vote_topic_pandemic }
			if = { limit = { has_global_variable = un_vote_topic_refugee } remove_global_variable = un_vote_topic_refugee }
			if = { limit = { has_global_variable = un_vote_topic_space } remove_global_variable = un_vote_topic_space }
			if = { limit = { has_global_variable = un_vote_topic_peacekeeping_request } remove_global_variable = un_vote_topic_peacekeeping_request }
			if = { limit = { has_global_variable = un_vote_topic_aid_request } remove_global_variable = un_vote_topic_aid_request }
			if = { limit = { has_global_variable = un_vote_topic_sanctions } remove_global_variable = un_vote_topic_sanctions }
			if = { limit = { has_global_variable = un_vote_topic_law_of_sea } remove_global_variable = un_vote_topic_law_of_sea }
			if = { limit = { has_global_variable = un_vote_topic_icc } remove_global_variable = un_vote_topic_icc }
			if = { limit = { has_global_variable = un_vote_topic_expulsion } remove_global_variable = un_vote_topic_expulsion }
			if = { limit = { has_global_variable = un_vote_proposer_country } remove_global_variable = un_vote_proposer_country }
			if = { limit = { has_global_variable = un_vote_target_country } remove_global_variable = un_vote_target_country }
			if = { limit = { has_global_variable = un_vote_promoted_country } remove_global_variable = un_vote_promoted_country }
			if = { limit = { has_global_variable = un_target_accepted } remove_global_variable = un_target_accepted }
			if = { limit = { has_global_variable = un_vote_veto_cast } remove_global_variable = un_vote_veto_cast }
			if = { limit = { has_global_variable = un_vote_vetoer_country } remove_global_variable = un_vote_vetoer_country }
			# Clear the target's self-correction flag (consumed by the pass branches above)
			if = {
				limit = { exists = scope:un_vote_target }
				scope:un_vote_target = {
					if = { limit = { has_variable = un_target_self_corrected } remove_variable = un_target_self_corrected }
				}
			}
			if = { limit = { has_global_variable = un_vote_net } remove_global_variable = un_vote_net }
			if = { limit = { has_global_variable = un_vote_support } remove_global_variable = un_vote_support }
			if = { limit = { has_global_variable = un_vote_oppose } remove_global_variable = un_vote_oppose }
			# Clear per-country voting flags
			every_country = {
				limit = { has_variable = un_voted_this_round }
				remove_variable = un_voted_this_round
			}
			# Clear proposer's own vote choice (non-proposers cleaned in un_vote.3)
			if = { limit = { has_variable = un_vote_choice } remove_variable = un_vote_choice }
		}

		ai_chance = { base = 100 }
	}
}

# ---- Phase 3: Notification to all other UN members ----
un_vote.3 = {
	type = country_event
	placement = ROOT

	event_image = { texture = "gfx/event_pictures/un_assembly_hall_vote.dds" }

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_map.dds"

	title = un_vote.3.t

	desc = {
		first_valid = {
			triggered_desc = {
				trigger = { has_variable = un_notif_vote_passed }
				desc = un_vote.3.d_passed
			}
			desc = un_vote.3.d_failed
		}
	}

	duration = 3

	trigger = {
		je:je_united_nations ?= { has_modifier = un_member_modifier }
	}

	immediate = {
		# Save vote results as country variables (globals may be cleaned up before desc evaluates)
		set_variable = {
			name = un_display_support
			value = global_var:un_vote_support
		}
		set_variable = {
			name = un_display_oppose
			value = global_var:un_vote_oppose
		}
		# Expulsion uses 2/3 supermajority; other topics use simple majority
		if = {
			limit = {
				has_global_variable = un_vote_topic_expulsion
				un_vote_expulsion_passed >= 0
			}
			set_variable = un_notif_vote_passed
		}
		else_if = {
			limit = {
				NOT = { has_global_variable = un_vote_topic_expulsion }
				global_var:un_vote_net >= 1
			}
			set_variable = un_notif_vote_passed
		}
		# Resolve proposer/target scopes from global vars
		if = {
			limit = { exists = global_var:un_vote_proposer_country }
			global_var:un_vote_proposer_country = { save_scope_as = un_vote_proposer }
		}
		if = {
			limit = { exists = global_var:un_vote_target_country }
			global_var:un_vote_target_country = { save_scope_as = un_vote_target }
		}

		# Save veto status locally so we don't depend on globals surviving
		if = { limit = { has_global_variable = un_vote_veto_cast } set_variable = un_vote_was_vetoed }

		# Save vote topics locally so we don't depend on globals surviving
		if = { limit = { has_global_variable = un_vote_topic_condemn } set_variable = un_topic_condemn }
		if = { limit = { has_global_variable = un_vote_topic_human_rights } set_variable = un_topic_human_rights }
		if = { limit = { has_global_variable = un_vote_topic_npt } set_variable = un_topic_npt }
		if = { limit = { has_global_variable = un_vote_topic_climate } set_variable = un_topic_climate }
		if = { limit = { has_global_variable = un_vote_topic_reform } set_variable = un_topic_reform }
		if = { limit = { has_global_variable = un_vote_topic_heritage } set_variable = un_topic_heritage }
		if = { limit = { has_global_variable = un_vote_topic_decolonization } set_variable = un_topic_decolonization }
		if = { limit = { has_global_variable = un_vote_topic_pandemic } set_variable = un_topic_pandemic }
		if = { limit = { has_global_variable = un_vote_topic_refugee } set_variable = un_topic_refugee }
		if = { limit = { has_global_variable = un_vote_topic_space } set_variable = un_topic_space }
		if = { limit = { has_global_variable = un_vote_topic_peacekeeping_request } set_variable = un_topic_peacekeeping_request }
		if = { limit = { has_global_variable = un_vote_topic_aid_request } set_variable = un_topic_aid_request }
		if = { limit = { has_global_variable = un_vote_topic_sanctions } set_variable = un_topic_sanctions }
		if = { limit = { has_global_variable = un_vote_topic_law_of_sea } set_variable = un_topic_law_of_sea }
		if = { limit = { has_global_variable = un_vote_topic_icc } set_variable = un_topic_icc }
		if = { limit = { has_global_variable = un_vote_topic_expulsion } set_variable = un_topic_expulsion }
	}

	# Option A: Catch-all "acknowledge" — vote failed, you voted yes, or you
	# joined the UN after vote.1 fired and never had a chance to vote. In the
	# last case the country still gets bound by the passed resolution (default
	# acceptance) since the apply blocks below already gate on the passed flag.
	option = {
		name = un_vote.3.a
		default_option = yes
		trigger = {
			OR = {
				NOT = { has_variable = un_notif_vote_passed }
				NOT = { has_variable = un_vote_choice }
				AND = {
					has_variable = un_vote_choice
					var:un_vote_choice = 1
				}
			}
		}

		# Apply the persistent treaty/declaration modifier directly in the
		# option block so the engine renders each add_modifier in the option
		# tooltip — players can see exactly what they're ratifying before
		# clicking acknowledge. (Previously these were buried inside
		# un_vote_apply_resolution_compliance, where the tooltip nesting
		# made the modifier name and fields easy to miss.)
		if = {
			limit = { has_variable = un_notif_vote_passed }
			un_vote_apply_treaty_modifier = yes
			un_vote_apply_resolution_compliance = yes
		}

		# Clean up notification variables
		hidden_effect = {
			un_vote_notification_cleanup = yes
		}
		ai_chance = { base = 100 }
	}

	# Option B: Vote passed AND you voted no - accept the resolution anyway
	option = {
		name = un_vote.3.b
		trigger = {
			has_variable = un_notif_vote_passed
			has_variable = un_vote_choice
			var:un_vote_choice = 2
		}

		# Same effects as yes voters - you accept the majority decision
		un_vote_apply_treaty_modifier = yes
		un_vote_apply_resolution_compliance = yes

		# Clean up notification variables
		hidden_effect = {
			un_vote_notification_cleanup = yes
		}
		ai_chance = { base = 50 }
	}

	# Option C: Vote passed AND you voted no - refuse to comply
	option = {
		name = un_vote.3.c
		trigger = {
			has_variable = un_notif_vote_passed
			has_variable = un_vote_choice
			var:un_vote_choice = 2
		}

		# Diplomatic penalties for defying the international community
		if = {
			limit = { exists = scope:un_vote_proposer }
			change_relations = { country = scope:un_vote_proposer value = -20 }
			create_diplomatic_catalyst = { type = catalyst_event_negative target = scope:un_vote_proposer }
		}
		change_infamy = 3

		# Power bloc leverage: defying bloc decision
		if = {
			limit = {
				exists = scope:un_vote_proposer
				scope:un_vote_proposer = { is_power_bloc_leader = yes }
				is_in_same_power_bloc = scope:un_vote_proposer
			}
			scope:un_vote_proposer = {
				power_bloc ?= {
					add_leverage = { target = root value = -15 }
				}
			}
		}

		# Clean up notification variables
		hidden_effect = {
			un_vote_notification_cleanup = yes
		}
		ai_chance = {
			base = 20
			# Countries that are rivals of proposer more likely to refuse
			modifier = {
				trigger = {
					exists = scope:un_vote_proposer
					has_diplomatic_pact = {
						who = scope:un_vote_proposer
						type = rivalry
					}
				}
				add = 40
			}
			# Great powers refuse more — sovereignty concerns
			modifier = {
				trigger = { country_rank >= rank_value:great_power }
				add = 20
			}
			# NPT: nuclear powers refuse rather than disarm
			modifier = {
				trigger = {
					has_variable = un_topic_npt
					has_variable = nuclear_weapon_stockpile
					var:nuclear_weapon_stockpile >= 1
				}
				add = 50
			}
			# Climate: heavy industry nations refuse binding targets
			modifier = {
				trigger = {
					has_variable = un_topic_climate
					ig:ig_industrialists ?= {
						is_powerful = yes
					}
				}
				add = 25
			}
		}
	}
