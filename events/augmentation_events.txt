# Event 1: Augmentation Divide
augmentation_events.1 = {
	type = country_event
	placement = root

	event_image = { texture = "gfx/event_pictures/augmentation_divide.dds" }

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	title = augmentation_events.1.t
	desc = augmentation_events.1.d
	flavor = augmentation_events.1.f

	duration = 3
	cooldown = { days = normal_modifier_time }

	trigger = {
		has_augmentation_tech = yes
	}

	immediate = {
	}

	# Subsidize augmentation for all
	option = {
		name = augmentation_events.1.a
		default_option = yes
		ai_chance = {
			base = 2
		}
		add_modifier = {
			name = augmentation_subsidies_modifier
			days = normal_modifier_time
			is_decaying = yes
		}
		add_treasury = {
			value = yearly_gross_income
			multiply = -0.05
		}
	}

	# Leave it to the market
	option = {
		name = augmentation_events.1.b
		ai_chance = {
			base = 3
		}
		add_radicals = {
			value = very_small_radicals
			strata = lower
		}
	}

	# Ban non-medical augmentation
	option = {
		name = augmentation_events.1.c
		ai_chance = {
			base = 1
			modifier = {
				trigger = {
					OR = {
						has_law = law_type:law_human_purity
						has_law = law_type:law_no_augmentation
					}
				}
				add = 4
			}
		}
		add_modifier = {
			name = augmentation_ban_modifier
			days = normal_modifier_time
			is_decaying = yes
		}
	}
}

# Event 4: Religious Condemnation
# Religious leaders declare augmentation an affront to divine creation.
augmentation_events.4 = {
	type = country_event
	placement = root

	event_image = { texture = "gfx/event_pictures/augmented_criminal.dds" }

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	title = augmentation_events.4.t
	desc = augmentation_events.4.d
	flavor = augmentation_events.4.f

	duration = 3
	cooldown = { days = long_modifier_time }

	trigger = {
		has_augmentation_tech = yes
		any_interest_group = {
			is_interest_group_type = ig_devout
			is_powerful = yes
		}
	}

	immediate = {
		random_interest_group = {
			limit = {
				is_interest_group_type = ig_devout
				is_powerful = yes
			}
			save_scope_as = devout_ig
		}
	}

	# Dismiss religious concerns
	option = {
		name = augmentation_events.4.a
		default_option = yes
		ai_chance = {
			base = 2
		}
		scope:devout_ig = {
			add_modifier = {
				name = augmentation_religious_anger_modifier
				days = normal_modifier_time
				is_decaying = yes
			}
		}
		add_loyalists = {
			value = very_small_radicals
			pop_type = academics
		}
	}

	# Establish an ethics board with religious input
	option = {
		name = augmentation_events.4.b
		ai_chance = {
			base = 3
		}
		add_modifier = {
			name = augmentation_ethics_board_modifier
			days = normal_modifier_time
			is_decaying = yes
		}
		add_loyalists = {
			value = very_small_radicals
			interest_group = ig:ig_devout
		}
	}

	# Side with the devout - restrict augmentation
	option = {
		name = augmentation_events.4.c
		ai_chance = {
			base = 1
			modifier = {
				trigger = {
					ig:ig_devout ?= { is_in_government = yes }
				}
				add = 3
			}
		}
		scope:devout_ig = {
			add_modifier = {
				name = augmentation_devout_victory_modifier
				days = normal_modifier_time
				is_decaying = yes
			}
		}
		add_radicals = {
			value = small_radicals
			pop_type = academics
		}
	}
}
